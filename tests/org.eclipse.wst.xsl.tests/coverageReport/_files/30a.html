640        public ITextHover getTextHover(ISourceViewer sourceViewer, String contentType, int stateMask) {
641                ITextHover textHover = null;
642 
643                /*
644                 * Returns a default problem, annotation, and best match hover
645                 * depending on stateMask
646                 */
647                TextHoverManager.TextHoverDescriptor[] hoverDescs = SSEUIPlugin.getDefault().getTextHoverManager().getTextHovers();
648                int i = 0;
649                while (i < hoverDescs.length && textHover == null) {
650                        if (hoverDescs[i].isEnabled() && computeStateMask(hoverDescs[i].getModifierString()) == stateMask) {
651                                String hoverType = hoverDescs[i].getId();
652                                if (TextHoverManager.PROBLEM_HOVER.equalsIgnoreCase(hoverType))
653                                        textHover = new ProblemAnnotationHoverProcessor();
654                                else if (TextHoverManager.ANNOTATION_HOVER.equalsIgnoreCase(hoverType))
655                                        textHover = new AnnotationHoverProcessor();
656                                else if (TextHoverManager.COMBINATION_HOVER.equalsIgnoreCase(hoverType))
657                                        textHover = new BestMatchHover(createDocumentationHovers(contentType));
658                                else if (TextHoverManager.DOCUMENTATION_HOVER.equalsIgnoreCase(hoverType)) {
659                                        ITextHover[] hovers = createDocumentationHovers(contentType);
660                                        if (hovers.length > 0) {
661                                                textHover = hovers[0];
662                                        }
663                                }
664                        }
665                        i++;
666                }
667                return textHover;
668        }
