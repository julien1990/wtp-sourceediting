333        /**
334         * @param tags
335         * @param priorities
336         */
337        private void loadTagsAndPrioritiesFrom(String tagString, String priorityString) {
338                String[] tags = StringUtils.unpack(tagString);
339 
340                StringTokenizer toker = null;
341                List list = new ArrayList();
342 
343                toker = new StringTokenizer(priorityString, ","); //$NON-NLS-1$
344                while (toker.hasMoreTokens()) {
345                        Integer number = null;
346                        try {
347                                number = Integer.valueOf(toker.nextToken());
348                        }
349                        catch (NumberFormatException e) {
350                                number = new Integer(IMarker.PRIORITY_NORMAL);
351                        }
352                        list.add(number);
353                }
354                Integer[] priorities = (Integer[]) list.toArray(new Integer[0]);
355 
356                fTaskTags = new TaskTag[Math.min(tags.length, priorities.length)];
357                for (int i = 0; i < fTaskTags.length; i++) {
358                        fTaskTags[i] = new TaskTag(tags[i], priorities[i].intValue());
359                }
360        }
