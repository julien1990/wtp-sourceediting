514        private void removeDocumentSelectionMediator(IDocumentSelectionMediator mediator) {
515                if (fMediators != null && mediator != null) {
516                        // if its not in the array, we'll ignore the request
517                        if (Utilities.contains(fMediators, mediator)) {
518                                int oldSize = fMediators.length;
519                                int newSize = oldSize - 1;
520                                IDocumentSelectionMediator[] newMediators = new IDocumentSelectionMediator[newSize];
521                                int index = 0;
522                                for (int i = 0; i < oldSize; i++) {
523                                        if (fMediators[i] == mediator) { // ignore
524                                        }
525                                        else {
526                                                // copy old to new if its not the one we are removing
527                                                newMediators[index++] = fMediators[i];
528                                        }
529                                }
530                                // now that we have a new array, let's switch it for the old
531                                // one
532                                fMediators = newMediators;
533                        }
534                }
