487        /**
488         * Returns the partition containing the given character position of the
489         * given document. The document has previously been connected to the
490         * partitioner. If the checkBetween parameter is true, an offset between a
491         * start and end tag will return a zero-length region.
492         */
493        private void internalGetPartition(int offset, boolean checkBetween) {
494                if (fStructuredDocument == null) {
495                        throw new IllegalStateException("document partitioner is not connected"); //$NON-NLS-1$
496                }
497 
498                boolean partitionFound = false;
499                int docLength = fStructuredDocument.getLength();
500                // get document region type and map to partition type :
501                // Note: a partion can be smaller than a flatnode, if that flatnode
502                // contains a region container.
503                // That's why we need to get "relevent region".
504                IStructuredDocumentRegion structuredDocumentRegion = fStructuredDocument.getRegionAtCharacterOffset(offset);
505                // flatNode is null if empty document
506                // this is king of a "normal case" for empty document
507                if (structuredDocumentRegion == null) {
508                        if (docLength == 0) {
509                                /*
510                                 * In order to prevent infinite error loops, this partition
511                                 * must never have a zero length unless the document is also
512                                 * zero length
513                                 */
514                                setInternalPartition(offset, 0, getDefaultPartitionType());
515                                partitionFound = true;
516                        }
517                        else {
518                                /*
519                                 * This case is "unusual". When would region be null, and
520                                 * document longer than 0. I think this means something's "out
521                                 * of sync". And we may want to "flag" that fact and just
522                                 * return one big region of 'unknown', instead of one
523                                 * character at a time.
524                                 */
525                                setInternalPartition(offset, 1, getUnknown());
526                                partitionFound = true;
527                        }
528                }        
529                else if (checkBetween) {
530                        // dmw: minimizes out to the first if test above
531                        //                        if (structuredDocumentRegion == null && docLength == 0) {
532                        //                                // known special case for an empty document
533                        //                                setInternalPartition(offset, 0, getDefault());
534                        //                                partitionFound = true;
535                        //                        }
536                        //                        else
537                        if (structuredDocumentRegion.getStartOffset() == offset) {
538                                IStructuredDocumentRegion previousStructuredDocumentRegion = structuredDocumentRegion.getPrevious();
539                                if (previousStructuredDocumentRegion != null) {
540                                        ITextRegion next = structuredDocumentRegion.getRegionAtCharacterOffset(offset);
541                                        ITextRegion previousStart = previousStructuredDocumentRegion.getRegionAtCharacterOffset(previousStructuredDocumentRegion.getStartOffset());
542                                        partitionFound = doParserSpecificCheck(offset, partitionFound, structuredDocumentRegion, previousStructuredDocumentRegion, next, previousStart);
543                                }
544                        }
545                }
546 
547                if (!partitionFound && structuredDocumentRegion != null) {
548                        ITextRegion resultRegion = structuredDocumentRegion.getRegionAtCharacterOffset(offset);
549                        partitionFound = isDocumentRegionBasedPartition(structuredDocumentRegion, resultRegion, offset);
550                        if (!partitionFound) {
551                                // Note: this new logic doesn't handle container regions
552                                // inside of
553                                // container regions ... may need to make this first clause
554                                // a recursive method
555                                if (resultRegion != null && resultRegion instanceof ITextRegionContainer) {
556                                        ITextRegionContainer containerRegion = (ITextRegionContainer) resultRegion;
557                                        // then need to "drill down" for relevent region and
558                                        // relevent offset
559                                        ITextRegion deepRegion = containerRegion.getRegionAtCharacterOffset(offset);
560                                        int endOffset = containerRegion.getEndOffset(deepRegion);
561                                        String type = getPartitionType(deepRegion, endOffset);
562                                        setInternalPartition(offset, endOffset - offset, type);
563                                } else {
564                                        if (resultRegion != null) {
565                                                String type = getPartitionType(resultRegion, offset);
566                                                setInternalPartition(offset, structuredDocumentRegion.getEndOffset(resultRegion) - offset, type);
567                                        } else {
568                                                // can happen at EOF
569                                                // https://bugs.eclipse.org/bugs/show_bug.cgi?id=224886
570                                                // The unknown type was causing problems with content assist in JSP documents
571                                                setInternalPartition(offset, 1, getDefaultPartitionType());
572                                        }
573                                }
574                        }
575                }
576        }
