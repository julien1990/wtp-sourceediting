383                 */
384                protected boolean validateData() {
385                
386                        String result = null;
387                        if (key == null || !key.equals(keyField.getText()) || type != getKeyType())
388                        {
389                                // get system catalog
390                                ICatalog systemCatalog = null;                        
391                                ICatalog defaultCatalog = XMLCorePlugin.getDefault().getDefaultXMLCatalog();
392                                INextCatalog[] nextCatalogs = defaultCatalog.getNextCatalogs();
393                                for (int i = 0; i < nextCatalogs.length; i++) {
394                                        INextCatalog catalog = nextCatalogs[i];
395                                        ICatalog referencedCatalog = catalog.getReferencedCatalog();
396                                        if (referencedCatalog != null) {
397                                                if (XMLCorePlugin.SYSTEM_CATALOG_ID.equals(referencedCatalog.getId())) {
398                                                        systemCatalog = referencedCatalog;
399                                                }
400                                        }
401                                }
402                                
403                                try {
404                                        switch( getKeyType() )
405                                        {
406                                        case ICatalogEntry.ENTRY_TYPE_PUBLIC:
407                                                result = catalog.resolvePublic(keyField.getText(), null);                
408                                                if (result == null) {
409                                                        result = systemCatalog.resolvePublic(keyField.getText(), null);
410                                                }
411                                                break;
412                                        case ICatalogEntry.ENTRY_TYPE_SYSTEM:
413                                                result = catalog.resolveSystem(keyField.getText());
414                                                if (result == null) {
415                                                        result = systemCatalog.resolveSystem(keyField.getText());
416                                                }
417                                                break;
418                                        case ICatalogEntry.ENTRY_TYPE_URI:
419                                                result = catalog.resolveURI(keyField.getText());
420                                                if (result == null) {
421                                                        result = systemCatalog.resolveURI(keyField.getText());
422                                                }
423                                                break;
424                                        }
425                                }
426                                catch (Exception e) {
427                                }
428                        }
429                        
430                        return (result == null);
431                }
