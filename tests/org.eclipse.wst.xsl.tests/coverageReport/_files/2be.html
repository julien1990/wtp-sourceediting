444        private IStatus validatePositiveNumber(String number) {
445                StatusInfo status = new StatusInfo();
446                if (number.length() == 0) {
447                        status.setError(SSEUIMessages.StructuredTextEditorPreferencePage_37);
448                }
449                else {
450                        try {
451                                int value = Integer.parseInt(number);
452                                if (value < 0)
453                                        status.setError(number + SSEUIMessages.StructuredTextEditorPreferencePage_38);
454                        }
455                        catch (NumberFormatException e) {
456                                status.setError(number + SSEUIMessages.StructuredTextEditorPreferencePage_38);
457                        }
458                }
459                return status;
460        }
