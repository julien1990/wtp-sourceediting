495    CMNodeList list = allGroup.getChildNodes();
496    for (int j = list.getLength() - 1; j >= 0; j--)
497    {

500      {  
501        map.put(node, new ItemCount());
502      }  
503    }    
504    int validitionCount = 0;
505    for (Iterator i = elementContent.iterator(); i.hasNext(); validitionCount++)
506    {
507      Object o = i.next();        
508      if (comparator.isElement(o))
509      {              
510        // test to see if the element is listed in the all group
511        //
512        CMNode matchingCMNode = null;
513        for (int j = list.getLength() - 1; j >= 0; j--)
514        {
515          CMNode node = list.item(j);
516          if (comparator.matches(o, node))

518            matchingCMNode = node;
519            break;
520          }             
521        }                              
522        if (matchingCMNode == null)
523        {     
524          isPartiallyValid = false;
525          isValid = false;
526          break;
527        }
528        else
529        {  
530          // test to see that the element occurs only once
531          //
532          ItemCount itemCount = (ItemCount)map.get(matchingCMNode);
533          if (itemCount != null)
534          {  
535            if (itemCount.count > 0)
536            {
537              // we don't want to allow too many elements!

539              isPartiallyValid = false;
540              break;
541            }  
542            else
543            {
544              itemCount.count++;
545            }  
546          }
547        }  
548      }  
549    }
550    if (isValid)

552      for (int j = list.getLength() - 1; j >= 0; j--)
553      {
554        CMNode node = list.item(j);

559//          System.out.print("content " + content.getNodeName() + " " + content.getMinOccur());
560          if (itemCount.count < content.getMinOccur())

563            break;
564          }  
565        }  
566      }
567    }
568    if (result instanceof ElementPathRecordingResult && isPartiallyValid)
569    {
570      ((ElementPathRecordingResult)result).setPartialValidationCount(validitionCount);
571    }  
572    result.isValid = isValid;
573  }
