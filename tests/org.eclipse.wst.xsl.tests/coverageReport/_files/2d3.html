608        // see ActionDescriptor#createExtension
609        private static IAction newInstance(final IConfigurationElement elm, final String cla) {
610                final Object[] result = new Object[1];
611                // If plugin has been loaded create extension.
612                // Otherwise, show busy cursor then create extension.
613                Bundle bundle= Platform.getBundle(elm.getDeclaringExtension().getNamespace());
614                if (bundle != null && bundle.getState() == Bundle.ACTIVE) {
615                        try {
616                                result[0] = elm.createExecutableExtension(cla);
617                        }
618                        catch (Exception e) {
619                                // catch and log ANY exception from extension point
620                                handleCreateExecutableException(result, e, elm.getAttribute(cla));
621                        }
622                }
623                else {
624                        BusyIndicator.showWhile(null, new Runnable() {
625                                public void run() {
626                                        try {
627                                                result[0] = elm.createExecutableExtension(cla);
628                                        }
629                                        catch (Exception e) {
630                                                // catch and log ANY exception from extension point
631                                                handleCreateExecutableException(result, e, elm.getAttribute(cla));
632                                        }
633                                }
634                        });
635                }
636                if ((result[0] != null) && (result[0] instanceof IAction)) {
637                        return (IAction)result[0];
638                }
639                return null;
640    }

643                Logger.logException("Unable to create proxy action for " + classname, e);
644                result[0] = null;
