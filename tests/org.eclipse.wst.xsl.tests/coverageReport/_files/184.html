606        public void insertNodesAtIndex(Node parent, List list, int index, boolean format) {
607                NodeList nodeList = parent.getChildNodes();
608                if (index == -1) {
609                        index = nodeList.getLength();
610                }
611                Node refChild = (index < nodeList.getLength()) ? nodeList.item(index) : null;
612 
613                // here we consider the case where the previous node is a 'white
614                // space' Text node
615                // we should really do the insert before this node
616                //
617                int prevIndex = index - 1;
618                Node prevChild = (prevIndex < nodeList.getLength()) ? nodeList.item(prevIndex) : null;
619                if (isWhitespaceTextNode(prevChild)) {
620                        refChild = prevChild;
621                }
622 
623                for (Iterator i = list.iterator(); i.hasNext();) {
624                        Node newNode = (Node) i.next();
625 
626                        if (newNode.getNodeType() == Node.ATTRIBUTE_NODE) {
627                                Element parentElement = (Element) parent;
628                                parentElement.setAttributeNode((Attr) newNode);
629                        }
630                        else {
631                                parent.insertBefore(newNode, refChild);
632                        }
633                }
634 
635                boolean formatDeep = false;
636                for (Iterator i = list.iterator(); i.hasNext();) {
637                        Node newNode = (Node) i.next();
638                        if (newNode.getNodeType() == Node.ELEMENT_NODE) {
639                                formatDeep = true;
640                        }
641 
642                        if (format) {
643                                reformat(newNode, formatDeep);
644                        }
645                }
646 
647                setViewerSelection(list);
648        }
