607        public StyleRange[] lineGetStyle(int eventLineOffset, int eventLineLength) {
608                StyleRange[] eventStyles = EMPTY_STYLE_RANGE;
609                try {
610                        if (getDocument() == null || eventLineLength == 0) {
611                                // getDocument() == null
612                                // during initialization, this is sometimes called before our
613                                // structured
614                                // is set, in which case we set styles to be the empty style
615                                // range
616                                // (event.styles can not be null)
617 
618                                // eventLineLength == 0
619                                // we sometimes get odd requests from the very last CRLF in
620                                // the
621                                // document
622                                // it has no length, and there is no node for it!
623                                eventStyles = EMPTY_STYLE_RANGE;
624                        }
625                        else {
626                                /*
627                                 * LineStyleProviders work using absolute document offsets. To
628                                 * support visible regions, adjust the requested range up to
629                                 * the full document offsets.
630                                 */
631                                IRegion styleRegion = getDocumentRangeFromWidgetRange(eventLineOffset, eventLineLength);
632                                if (styleRegion != null) {
633                                        int start = styleRegion.getOffset();
634                                        int length = styleRegion.getLength();
635 
636                                        ITypedRegion[] partitions = TextUtilities.computePartitioning(getDocument(), fPartitioning, start, length, false);
637                                        eventStyles = prepareStyleRangesArray(partitions, start, length);
638 
639                                        /*
640                                         * If there is a subtext offset, the style ranges must be
641                                         * adjusted to the expected offsets just check if
642                                         * eventLineOffset is different than start then adjust,
643                                         * otherwise u can leave it alone unless there is special
644                                         * handling for itextviewerextension5?
645                                         */
646                                        if (start != eventLineOffset) {
647                                                int offset = 0;
648                                                // figure out visible region to use for adjustment
649                                                // only adjust if need to
650                                                if (!(getTextViewer() instanceof ITextViewerExtension5)) {
651                                                        IRegion vr = getTextViewer().getVisibleRegion();
652                                                        if (vr != null) {
653                                                                offset = vr.getOffset();
654                                                        }
655                                                }
656                                                adjust(eventStyles, -offset);
657                                        }
658 
659                                        eventStyles = limitSize(eventStyles);
660                                        
661                                        // for debugging only
662                                        if (DEBUG) {
663                                                if (!valid(eventStyles, eventLineOffset, eventLineLength)) {
664                                                        Logger.log(Logger.WARNING, "Highlighter::lineGetStyle found invalid styles at offset " + eventLineOffset); //$NON-NLS-1$
665                                                }
666                                        }
667                                }
668 
669                        }
670 
671                }
672                catch (Exception e) {
673                        // if ANY exception occurs during highlighting,
674                        // just return "no highlighting"
675                        eventStyles = EMPTY_STYLE_RANGE;
676                        if (Debug.syntaxHighlighting) {
677                                System.out.println("Exception during highlighting!"); //$NON-NLS-1$
678                        }
679                }
680 
681                return eventStyles;
682        }

684        /**
685         * This method is to centralize the logic in limiting the overall number of style ranges
686         * that make it to the styled text widget. 
687         * 
688         * Too many styles sent to StyledText results in apparent, but not real, 
689         * hangs of Eclipse Display thread. See  
690         * https://bugs.eclipse.org/bugs/show_bug.cgi?id=108806
691         * 
692         * @param eventStyles
693         * @return
694         */
695         private StyleRange[] limitSize(StyleRange[] eventStyles) {
696                
697                // quick return with same object if not modification needed
698                if (eventStyles.length < MAX_NUMBER_STYLES) {
699                        return eventStyles;
700                }
701                else {
702                        // we could just take the easy way out and truncate, but will 
703                        // be much better appearing if both the start of the line and the 
704                        // end of the line are displayed with styles. Since these are both
705                        // the parts of the line a user is likely to look at. The middle of the 
706                        // line will still be "plain". Presumably, the user would re-format the 
707                        // file to avoid long lines, so unlikely to see the middle. 
708                        StyleRange[] newRanges = new StyleRange[LEFT_STYLES_SIZE + RIGHT_STYLES_SIZE + MIDDLE_STYLES_SIZE]; 
709                        System.arraycopy(eventStyles, 0, newRanges, 0, LEFT_STYLES_SIZE);
710                        //
711                        // do end, before we do middle
712                        System.arraycopy(eventStyles, eventStyles.length-RIGHT_STYLES_SIZE, newRanges, LEFT_STYLES_SIZE + MIDDLE_STYLES_SIZE, RIGHT_STYLES_SIZE);
713                        //
714                        // technically, we should compute the exact middle as one big style range, 
715                        // with default colors and styles, so if someone does actually type or work with 
716                        // documnet as is, will still be correct. 
717                        //
718                        StyleRange allBlank = new StyleRange();
719                        StyleRange lastKnown = newRanges[LEFT_STYLES_SIZE - 1];
720                        allBlank.start = lastKnown.start + lastKnown.length;
721                        StyleRange nextKnown = newRanges[LEFT_STYLES_SIZE + MIDDLE_STYLES_SIZE + 1];
722                        allBlank.length = nextKnown.start - allBlank.start;
723                        newRanges[LEFT_STYLES_SIZE] = allBlank;
724                        return newRanges;
725                }

772                }

775                }

778                // stack
779                LineStyleProvider currentLineStyleProvider = null;
780                boolean handled = false;
781                for (int i = 0; i < partitions.length; i++) {
782                        ITypedRegion currentPartition = partitions[i];
783                        currentLineStyleProvider = getProviderFor(currentPartition);
784                        currentLineStyleProvider.init(getDocument(), this);
785                        handled = currentLineStyleProvider.prepareRegions(currentPartition, currentPartition.getOffset(), currentPartition.getLength(), fHoldStyleResults);
