615                                         *

617                                         * with no save for files with no structured model
618                                         * associated with them (text files, javascript files,
619                                         * etc) in web project
620                                         */
621                                        bufferManager.disconnect(location, LocationKind.IFILE, getProgressMonitor());
622                                }
623                        }
624                }
625                catch (CoreException e) {
626                        Logger.logException("Error getting model for " + file.getPath(), e); //$NON-NLS-1$
627                }
628                return model;
