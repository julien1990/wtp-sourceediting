519        /**

521         * It will try to find a sibling node before this node first. If there is
522         * no sibling node before this node, it will try to find a sibling node

524         * already indented from its parent. If yes, this node's indentation will

526         * will be used. If this node is null or it's a document node or it's a

528         * string will be returned as the indentation.
529         */
530        protected String getSiblingIndent(Node node) {
531                String result = EMPTY_STRING;
532 
533                if ((node != null) && (node.getNodeType() != Node.DOCUMENT_NODE) && (node.getParentNode() != null) && (node.getParentNode().getNodeType() != Node.DOCUMENT_NODE)) {
534                        // find the text node before the previous non-text sibling
535                        // if that's not found, we will try the text node before the next
536                        // non-text sibling
537                        IDOMNode sibling = (IDOMNode) node.getPreviousSibling();
538                        while ((sibling != null) && (sibling.getNodeType() == Node.TEXT_NODE || sibling.getNodeType() == Node.COMMENT_NODE)) {
539                                if (sibling.getNodeType() == Node.COMMENT_NODE && sibling.getPreviousSibling() != null && sibling.getPreviousSibling().getNodeType() == Node.TEXT_NODE && StringUtils.containsLineDelimiter(sibling.getPreviousSibling().getNodeValue()))
540                                        break;
541                                sibling = (IDOMNode) sibling.getPreviousSibling();
542                        }
543                        if (sibling == null) {
544                                sibling = (IDOMNode) node.getNextSibling();
545                                while ((sibling != null) && (sibling.getNodeType() == Node.TEXT_NODE))
546                                        sibling = (IDOMNode) sibling.getNextSibling();
547                        }
548                        String singleIndent = getFormatPreferences().getIndent();
549                        String parentLineIndent = getNodeIndent(node.getParentNode());
550 
551                        if (sibling != null) {
552                                String siblingIndent = getNodeIndent(sibling);
553                                if (siblingIndent.length() > 0)
554                                        result = siblingIndent;
555                                else {
556                                        String nodeIndent = getNodeIndent(node);
557                                        if (nodeIndent.length() > parentLineIndent.length())
558                                                // this node is indented from its parent, its
559                                                // indentation will be used
560                                                result = nodeIndent;
561                                        else
562                                                result = parentLineIndent + singleIndent;
563                                }
564                        }
565                        else {
566                                String nodeIndent = getNodeIndent(node);
567                                if (nodeIndent.length() > parentLineIndent.length())
568                                        // this node is indented from its parent, its indentation
569                                        // will be used
570                                        result = nodeIndent;
571                                else
572                                        result = parentLineIndent + singleIndent;
573                        }
574                }
575 
576                return result;
577        }
