480                // determine which breakpoint was hit, and set the thread's breakpoint
481                int lastSpace = event.lastIndexOf(' ');
482                if (lastSpace > 0) {
483                        String line = event.substring(lastSpace + 1);
484                        int lineNumber = Integer.parseInt(line);
485                        IBreakpoint[] breakpoints = DebugPlugin.getDefault()
486                                        .getBreakpointManager().getBreakpoints(
487                                                        IXSLConstants.ID_XSL_DEBUG_MODEL);
488                        for (IBreakpoint breakpoint : breakpoints) {
489                                if (supportsBreakpoint(breakpoint)) {
490                                        if (breakpoint instanceof ILineBreakpoint) {
491                                                ILineBreakpoint lineBreakpoint = (ILineBreakpoint) breakpoint;
492                                                try {
493                                                        if (lineBreakpoint.getLineNumber() == lineNumber) {
494                                                                thread
495                                                                                .setBreakpoints(new IBreakpoint[] { breakpoint });
496                                                                break;
497                                                        }
498                                                } catch (CoreException e) {
499                                                }
500                                        }
501                                }
502                        }
503                }
504                suspended(DebugEvent.BREAKPOINT);
505        }
