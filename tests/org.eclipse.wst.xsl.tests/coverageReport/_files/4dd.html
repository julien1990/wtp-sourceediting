526                for (int i = 0; i < listLength; i++) {

542        public StructuredDocumentEvent updateRegion(Object requester, IStructuredDocumentRegion structuredDocumentRegion, String changes, int requestStart, int lengthToReplace) {
543                StructuredDocumentEvent result = null;
544                int lengthDifference = Utilities.calculateLengthDifference(changes, lengthToReplace);
545                // Get the region pointed to by the requestStart postion, and give
546                // that region a chance to effect
547                // the update.
548                ITextRegion region = getRegionAtCharacterOffset(requestStart);
549                // if there is no region, then the requested changes must come right
550                // after the
551                // node (and right after the last region). This happens, for example,
552                // when someone
553                // types something at the end of the document, or more commonly, when
554                // they are right
555                // at the beginning of one node, and the dirty start is therefore
556                // calculated to be the
557                // previous node.
558                // So, in this case, we'll give the last region a chance to see if it
559                // wants to
560                // swallow the requested changes -- but only for inserts -- deletes
561                // and "replaces"
562                // should be reparsed if they are in these border regions, and only if
563                // the
564                if ((region == null) && (lengthToReplace == 0)) {
565                        region = _getRegions().get(_getRegions().size() - 1);
566                        // make sure the region is contiguous
567                        if (getEndOffset(region) == requestStart) {
568                                result = region.updateRegion(requester, this, changes, requestStart, lengthToReplace);
569                        }
570                }
571                else {
572                        if (region != null) {
573                                //
574                                // If the requested change spans more than one region, then
575                                // we don't give the region a chance to update.
576                                if ((containsOffset(region, requestStart)) && (containsOffset(region, requestStart + lengthToReplace))) {
577                                        result = region.updateRegion(requester, this, changes, requestStart, lengthToReplace);
578                                }
579                        }
580                }
581                // if result is not null, then we need to update the start and end
582                // postions of the regions that follow this one
583                // if result is null, then apply the flatnode specific checks on what
584                // it can change
585                // (i.e. more than one region, but no change to the node itself)
586                if (result != null) {
587                        // That is, a region decided it could handle the change and
588                        // created
589                        // a region changed event.
590                        Assert.isTrue(result instanceof RegionChangedEvent, "Program Error"); //$NON-NLS-1$
591                        updateDownStreamRegions(((RegionChangedEvent) result).getRegion(), lengthDifference);
592                        // PLUS, we need to update our own node end point (length)
593                        setLength(getLength() + lengthDifference);
594                }
595 
596                return result;
597        }
