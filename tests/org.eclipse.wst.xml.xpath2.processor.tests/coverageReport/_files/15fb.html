413         * @param addedPositions the added positions
414         * @param removedPositions the removed positions

416        public void updatePresentation(TextPresentation textPresentation, HighlightedPosition[] addedPositions, HighlightedPosition[] removedPositions) {
417                if (fSourceViewer == null)
418                        return;
419 
420//                checkOrdering("added positions: ", Arrays.asList(addedPositions)); //$NON-NLS-1$
421//                checkOrdering("removed positions: ", Arrays.asList(removedPositions)); //$NON-NLS-1$
422//                checkOrdering("old positions: ", fPositions); //$NON-NLS-1$
423 
424                // TODO: double-check consistency with document.getPositions(...)
425                // TODO: reuse removed positions
426                if (isCanceled())
427                        return;
428 
429                IDocument document= fSourceViewer.getDocument();
430                if (document == null)
431                        return;
432 
433                String positionCategory= getPositionCategory();
434 
435                List removedPositionsList= Arrays.asList(removedPositions);
436 
437                try {
438                        synchronized (fPositionLock) {
439                                List oldPositions= fPositions;
440                                int newSize= Math.max(fPositions.size() + addedPositions.length - removedPositions.length, 10);
441 
442                                /*
443                                 * The following loop is a kind of merge sort: it merges two List<Position>, each
444                                 * sorted by position.offset, into one new list. The first of the two is the
445                                 * previous list of positions (oldPositions), from which any deleted positions get
446                                 * removed on the fly. The second of two is the list of added positions. The result
447                                 * is stored in newPositions.
448                                 */
449                                List newPositions= new ArrayList(newSize);
450                                Position position= null;
451                                Position addedPosition= null;
452                                for (int i= 0, j= 0, n= oldPositions.size(), m= addedPositions.length; i < n || position != null || j < m || addedPosition != null;) {
453                                        // loop variant: i + j < old(i + j)
454 
455                                        // a) find the next non-deleted Position from the old list
456                                        while (position == null && i < n) {
457                                                position= (Position) oldPositions.get(i++);
458                                                if (position.isDeleted() || contain(removedPositionsList, position)) {
459                                                        document.removePosition(positionCategory, position);
460                                                        position= null;
461                                                }
462                                        }
463 
464                                        // b) find the next Position from the added list
465                                        if (addedPosition == null && j < m) {
466                                                addedPosition= addedPositions[j++];
467                                                document.addPosition(positionCategory, addedPosition);
468                                        }
469 
470                                        // c) merge: add the next of position/addedPosition with the lower offset
471                                        if (position != null) {
472                                                if (addedPosition != null)
473                                                        if (position.getOffset() <= addedPosition.getOffset()) {
474                                                                newPositions.add(position);
475                                                                position= null;
476                                                        } else {
477                                                                newPositions.add(addedPosition);
478                                                                addedPosition= null;
479                                                        }
480                                                else {
481                                                        newPositions.add(position);
482                                                        position= null;
483                                                }
484                                        } else if (addedPosition != null) {
485                                                newPositions.add(addedPosition);
486                                                addedPosition= null;
487                                        }
488                                }
489                                fPositions= newPositions;
490                        }
491                } catch (BadPositionCategoryException e) {
492                        // Should not happen
493                        Logger.logException(e);
494                } catch (BadLocationException e) {
495                        // Should not happen
496                        Logger.logException(e);
497                }
498//                checkOrdering("new positions: ", fPositions); //$NON-NLS-1$
499 
500                if (textPresentation != null)
501                        fSourceViewer.changeTextPresentation(textPresentation, false);
502                else
503                        fSourceViewer.invalidateTextPresentation();
504        }
