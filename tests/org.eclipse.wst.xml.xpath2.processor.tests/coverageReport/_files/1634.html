339                                                        result = catalog.resolvePublic(key, systemId);
340                                                        break;
341                                                case IDelegateCatalog.DELEGATE_TYPE_SYSTEM:
342                                                        result = catalog.resolveSystem(systemId);
343                                                        break;
344                                                case IDelegateCatalog.DELEGATE_TYPE_URI:
345                                                        result = catalog.resolveURI(systemId);
346                                                        break;
347                                                default:
348                                                        break;
349                                                }
350                                                if (result != null)
351                                                {
352                                                        return result;
353                                                }
354                                        }
355                                }
356                        }
357                        return null;
