429        void updateStatus(TextHoverManager.TextHoverDescriptor hoverConfig) {
430                IStatus status = new StatusInfo();
431 
432                if (hoverConfig != null && hoverConfig.isEnabled() && EditorUtility.computeStateMask(hoverConfig.getModifierString()) == -1)
433                        status = new StatusInfo(IStatus.ERROR, NLS.bind(SSEUIMessages.TextHoverPreferenceTab_modifierIsNotValid, new String[]{hoverConfig.getModifierString()}));
434 
435                int i = 0;
436                HashMap stateMasks = new HashMap(fTextHovers.length);
437                while (status.isOK() && i < fTextHovers.length) {
438                        if (fTextHovers[i].isEnabled()) {
439                                String label = fTextHovers[i].getLabel();
440                                Integer stateMask = new Integer(EditorUtility.computeStateMask(fTextHovers[i].getModifierString()));
441                                if (stateMask.intValue() == -1)
442                                        status = new StatusInfo(IStatus.ERROR, NLS.bind(SSEUIMessages.TextHoverPreferenceTab_modifierIsNotValidForHover, new String[]{fTextHovers[i].getModifierString(), label}));
443                                else if (stateMasks.containsKey(stateMask))
444                                        status = new StatusInfo(IStatus.ERROR, NLS.bind(SSEUIMessages.TextHoverPreferenceTab_duplicateModifier, new String[]{label, (String) stateMasks.get(stateMask)}));
445                                else
446                                        stateMasks.put(stateMask, label);
447                        }
448                        i++;
449                }
450 
451                updateStatus(status);
452        }
