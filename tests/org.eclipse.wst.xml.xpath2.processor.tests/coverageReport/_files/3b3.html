564        private void reportTaglibDuplicatePrefixes(IFile file, IReporter reporter, IStructuredDocument document) {
565                if (fSeverityTaglibDuplicatePrefixWithDifferentURIs == ValidationMessage.IGNORE && fSeverityTaglibDuplicatePrefixWithSameURIs == ValidationMessage.IGNORE)
566                        return;

569                for (int prefixNumber = 0; prefixNumber < prefixes.length; prefixNumber++) {
570                        int severity = fSeverityTaglibDuplicatePrefixWithSameURIs;
571 
572                        Object o = fTaglibPrefixesInUse.get(prefixes[prefixNumber]);
573                        /*

584                                        if (uriValueRegion == null) {
