626        private boolean isSingleControlStatementWithoutBlock() {
627                List statements= fDeclaration.getBody().statements();
628                int size= statements.size();
629                if (size != 1)
630                        return false;
631                Statement statement= (Statement) statements.get(size - 1);
632                int nodeType= statement.getNodeType();
633                if (nodeType == ASTNode.IF_STATEMENT) {
634                        IfStatement ifStatement= (IfStatement) statement;
635                        return !(ifStatement.getThenStatement() instanceof Block) 
636                                && !(ifStatement.getElseStatement() instanceof Block);
637                } else if (nodeType == ASTNode.FOR_STATEMENT) {
638                        return !(((ForStatement)statement).getBody() instanceof Block);
639                } else if (nodeType == ASTNode.WHILE_STATEMENT) {
640                        return !(((WhileStatement)statement).getBody() instanceof Block);
641                } else if (nodeType == ASTNode.WITH_STATEMENT) {
642                        return !(((WithStatement)statement).getBody() instanceof Block);
643                } else if (nodeType == ASTNode.FOR_IN_STATEMENT) {
644                        return !(((ForInStatement)statement).getBody() instanceof Block);
645                }
646                return false;
647        }
