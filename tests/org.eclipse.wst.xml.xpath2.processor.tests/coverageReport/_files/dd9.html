334                                // TODO instead of finding the right insert position the call inliner should

341                                change.setEdit(delete);

344                }

347                IJavaScriptProject javaProject= fInitialTypeRoot.getJavaScriptProject();
348                if (javaProject != null)

354                        flags|= RefactoringDescriptor.MULTI_CHANGE;

356                final String header= Messages.format(RefactoringCoreMessages.InlineMethodRefactoring_descriptor_description, new String[] { BindingLabelProvider.getBindingLabel(binding, JavaScriptElementLabels.ALL_FULLY_QUALIFIED), BindingLabelProvider.getBindingLabel(declaring, JavaScriptElementLabels.ALL_FULLY_QUALIFIED)});
357                final JDTRefactoringDescriptorComment comment= new JDTRefactoringDescriptorComment(project, this, header);
358                comment.addSetting(Messages.format(RefactoringCoreMessages.InlineMethodRefactoring_original_pattern, BindingLabelProvider.getBindingLabel(binding, JavaScriptElementLabels.ALL_FULLY_QUALIFIED)));
359                if (fDeleteSource)
360                        comment.addSetting(RefactoringCoreMessages.InlineMethodRefactoring_remove_method);
361                if (fCurrentMode == Mode.INLINE_ALL)
362                        comment.addSetting(RefactoringCoreMessages.InlineMethodRefactoring_replace_references);
363                final JDTRefactoringDescriptor descriptor= new JDTRefactoringDescriptor(IJavaScriptRefactorings.INLINE_METHOD, project, description, comment.asString(), arguments, flags);
364                arguments.put(JDTRefactoringDescriptor.ATTRIBUTE_INPUT, descriptor.elementToHandle(fInitialTypeRoot));
365                arguments.put(JDTRefactoringDescriptor.ATTRIBUTE_SELECTION, new Integer(fSelectionStart).toString() + " " + new Integer(fSelectionLength).toString()); //$NON-NLS-1$
366                arguments.put(ATTRIBUTE_DELETE, Boolean.valueOf(fDeleteSource).toString());
367                arguments.put(ATTRIBUTE_MODE, new Integer(fCurrentMode == Mode.INLINE_ALL ? 1 : 0).toString());
368                return new DynamicValidationRefactoringChange(descriptor, RefactoringCoreMessages.InlineMethodRefactoring_edit_inlineCall, fChangeManager.getAllChanges());

371        private static SourceProvider resolveSourceProvider(RefactoringStatus status, ITypeRoot typeRoot, ASTNode invocation) throws JavaScriptModelException {
372                JavaScriptUnit root= (JavaScriptUnit)invocation.getRoot();
373                IFunctionBinding methodBinding= Invocations.resolveBinding(invocation);
374                if (methodBinding == null) {
375                        status.addFatalError(RefactoringCoreMessages.InlineMethodRefactoring_error_noMethodDeclaration); 
376                        return null;
377                }
378                FunctionDeclaration declaration= (FunctionDeclaration)root.findDeclaringNode(methodBinding);
379                if (declaration != null) {
380                        return new SourceProvider(typeRoot, declaration);
381                }
382                IFunction method= (IFunction)methodBinding.getJavaElement();
383                if (method != null) {
384                        JavaScriptUnit methodDeclarationAstRoot;
385                        IJavaScriptUnit methodCu= method.getJavaScriptUnit();
386                        if (methodCu != null) {
387                                methodDeclarationAstRoot= new RefactoringASTParser(AST.JLS3).parse(methodCu, true);
388                        } else {
389                                IClassFile classFile= method.getClassFile();
390                                if (! JavaElementUtil.isSourceAvailable(classFile)) {
391                                        String methodLabel= JavaScriptElementLabels.getTextLabel(method, JavaScriptElementLabels.M_FULLY_QUALIFIED | JavaScriptElementLabels.M_PARAMETER_TYPES);
392                                        status.addFatalError(Messages.format(RefactoringCoreMessages.InlineMethodRefactoring_error_classFile, methodLabel)); 
393                                        return null;
394                                }                                
395                                methodDeclarationAstRoot= new RefactoringASTParser(AST.JLS3).parse(classFile, true);
396                        }
397                        ASTNode node= methodDeclarationAstRoot.findDeclaringNode(methodBinding.getMethodDeclaration().getKey());
398                        if (node instanceof FunctionDeclaration) {
399                                return new SourceProvider(methodDeclarationAstRoot.getTypeRoot(), (FunctionDeclaration) node);
400                        }
401                }
402                status.addFatalError(RefactoringCoreMessages.InlineMethodRefactoring_error_noMethodDeclaration); 
403                return null;
404        }

418        private static ASTNode checkNode(ASTNode node, IJavaScriptElement unit) {
419                if (node == null)
420                        return null;
421                if (node.getNodeType() == ASTNode.SIMPLE_NAME) {
422                        node= node.getParent();
423                } else if (node.getNodeType() == ASTNode.EXPRESSION_STATEMENT) {
424                        node= ((ExpressionStatement)node).getExpression();
425                }
426                switch(node.getNodeType()) {
427                        case ASTNode.FUNCTION_DECLARATION:
428                                        return node;
429                        case ASTNode.FUNCTION_INVOCATION:
430                        case ASTNode.SUPER_METHOD_INVOCATION:
431                        case ASTNode.CONSTRUCTOR_INVOCATION:
432                                return unit instanceof IJavaScriptUnit ? node : null; // don't start on invocations in binary
433                }
434                return null;
435        }

507        private ASTNode[] removeNestedCalls(RefactoringStatus status, IJavaScriptUnit unit, ASTNode[] invocations) {
508                if (invocations.length <= 1)
509                        return invocations;
510                ASTNode[] parents= new ASTNode[invocations.length];
511                for (int i= 0; i < invocations.length; i++) {
512                        parents[i]= invocations[i].getParent();
513                }
514                for (int i= 0; i < invocations.length; i++) {
515                        removeNestedCalls(status, unit, parents, invocations, i);
516                }
517                List result= new ArrayList();
518                for (int i= 0; i < invocations.length; i++) {
519                        if (invocations[i] != null)
520                                result.add(invocations[i]);
521                }
522                return (ASTNode[])result.toArray(new ASTNode[result.size()]);
523        }

525        private void removeNestedCalls(RefactoringStatus status, IJavaScriptUnit unit, ASTNode[] parents, ASTNode[] invocations, int index) {
526                ASTNode invocation= invocations[index];
527                for (int i= 0; i < parents.length; i++) {
528                        ASTNode parent= parents[i];
529                        while (parent != null) {
530                                if (parent == invocation) {
531                                        status.addError(RefactoringCoreMessages.InlineMethodRefactoring_nestedInvocation,  
532                                                JavaStatusContext.create(unit, parent));
533                                        invocations[index]= null;
534                                }
535                                parent= parent.getParent();
536                        }
537                }
