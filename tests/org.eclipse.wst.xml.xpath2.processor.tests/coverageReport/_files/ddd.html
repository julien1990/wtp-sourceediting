399         * Must only be called <i>after</i> checkActivation().
400         * The first proposal should be used as "best guess" (if it exists).
401         * @return proposed variable names (may be empty, but not null).
402         */
403        public String[] guessParameterNames() {
404                LinkedHashSet proposals= new LinkedHashSet(); //retain ordering, but prevent duplicates
405                if (fSelectedExpression instanceof FunctionInvocation){
406                        proposals.addAll(guessTempNamesFromMethodInvocation((FunctionInvocation) fSelectedExpression, fExcludedParameterNames));

409                return (String[]) proposals.toArray(new String[proposals.size()]);
410        }

412        private List/*<String>*/ guessTempNamesFromMethodInvocation(FunctionInvocation selectedMethodInvocation, String[] excludedVariableNames) {
413                SimpleName name = selectedMethodInvocation.getName();
414                
415                String methodName;
416                if (name!=null)
417                {
418                        methodName= name.getIdentifier();
419                for (int i= 0; i < KNOWN_METHOD_NAME_PREFIXES.length; i++) {
420                        String prefix= KNOWN_METHOD_NAME_PREFIXES[i];
421                        if (! methodName.startsWith(prefix))
422                                continue; //not this prefix
423                        if (methodName.length() == prefix.length())
424                                return Collections.EMPTY_LIST; // prefix alone -> don't take method name
425                        char firstAfterPrefix= methodName.charAt(prefix.length());
426                        if (! Character.isUpperCase(firstAfterPrefix))
427                                continue; //not uppercase after prefix
428                        //found matching prefix
429                        String proposal= Character.toLowerCase(firstAfterPrefix) + methodName.substring(prefix.length() + 1);
430                        methodName= proposal;
431                        break;
432                }
433                }
434                else
435                        methodName="indirectFunctionCall"; //$NON-NLS-1$
436                String[] proposals= StubUtility.getLocalNameSuggestions(fSourceCU.getJavaScriptProject(), methodName, 0, excludedVariableNames);
437                return Arrays.asList(proposals);
438        }

440        private List/*<String>*/ guessTempNamesFromExpression(Expression selectedExpression, String[] excluded) {
441                ITypeBinding expressionBinding= Bindings.normalizeForDeclarationUse(
442                        selectedExpression.resolveTypeBinding(),
443                        selectedExpression.getAST());
444                String typeName= getQualifiedName(expressionBinding);
445                if (typeName.length() == 0)
446                        typeName= expressionBinding.getName();
447                if (typeName.length() == 0)                        
448                        return Collections.EMPTY_LIST;
449                int typeParamStart= typeName.indexOf("<"); //$NON-NLS-1$
450                if (typeParamStart != -1)
451                        typeName= typeName.substring(0, typeParamStart);
452                String[] proposals= StubUtility.getLocalNameSuggestions(fSourceCU.getJavaScriptProject(), typeName, expressionBinding.getDimensions(), excluded);
453                return Arrays.asList(proposals);
454        }

494                fChangeSignatureRefactoring.setValidationContext(getValidationContext());
495                Change result;
496                try {
497                        result= fChangeSignatureRefactoring.createChange(pm);
498                } finally {
499                        fChangeSignatureRefactoring.setValidationContext(null);
500                }
501                if (result != null) {
502                        final ChangeDescriptor descriptor= result.getDescriptor();
503                        if (descriptor instanceof RefactoringChangeDescriptor) {
504                                final RefactoringDescriptor refactoringDescriptor= ((RefactoringChangeDescriptor) descriptor).getRefactoringDescriptor();
505                                if (refactoringDescriptor instanceof JDTRefactoringDescriptor) {
506                                        final JDTRefactoringDescriptor extended= (JDTRefactoringDescriptor) refactoringDescriptor;

509                                        arguments.put(JDTRefactoringDescriptor.ATTRIBUTE_SELECTION, new Integer(fSelectionStart).toString() + " " + new Integer(fSelectionLength).toString()); //$NON-NLS-1$

524                                }
525                        }
526                }
527                return result;
528        }

530        public RefactoringStatus initialize(final RefactoringArguments arguments) {
531                fArguments= arguments;
532                if (arguments instanceof JavaRefactoringArguments) {
533                        final JavaRefactoringArguments extended= (JavaRefactoringArguments) arguments;
534                        final String selection= extended.getAttribute(JDTRefactoringDescriptor.ATTRIBUTE_SELECTION);
535                        if (selection != null) {
536                                int offset= -1;
537                                int length= -1;
538                                final StringTokenizer tokenizer= new StringTokenizer(selection);
539                                if (tokenizer.hasMoreTokens())
540                                        offset= Integer.valueOf(tokenizer.nextToken()).intValue();
541                                if (tokenizer.hasMoreTokens())
542                                        length= Integer.valueOf(tokenizer.nextToken()).intValue();
543                                if (offset >= 0 && length >= 0) {
544                                        fSelectionStart= offset;
545                                        fSelectionLength= length;
546                                } else
547                                        return RefactoringStatus.createFatalErrorStatus(Messages.format(RefactoringCoreMessages.InitializableRefactoring_illegal_argument, new Object[] { selection, JDTRefactoringDescriptor.ATTRIBUTE_SELECTION}));
548                        } else
549                                return RefactoringStatus.createFatalErrorStatus(Messages.format(RefactoringCoreMessages.InitializableRefactoring_argument_not_exist, JDTRefactoringDescriptor.ATTRIBUTE_SELECTION));
550                        final String handle= extended.getAttribute(JDTRefactoringDescriptor.ATTRIBUTE_INPUT);
551                        if (handle != null) {
552                                final IJavaScriptElement element= JDTRefactoringDescriptor.handleToElement(extended.getProject(), handle, false);
553                                if (element == null || !element.exists() || element.getElementType() != IJavaScriptElement.JAVASCRIPT_UNIT)
554                                        return createInputFatalStatus(element, IJavaScriptRefactorings.INTRODUCE_PARAMETER);
555                                else
556                                        fSourceCU= ((IFunction) element).getJavaScriptUnit();
557                        } else
558                                return RefactoringStatus.createFatalErrorStatus(Messages.format(RefactoringCoreMessages.InitializableRefactoring_argument_not_exist, JDTRefactoringDescriptor.ATTRIBUTE_INPUT));
559                        final String name= extended.getAttribute(ATTRIBUTE_ARGUMENT);
560                        if (name != null && !"".equals(name)) //$NON-NLS-1$
561                                fParameterName= name;
562                        else
563                                return RefactoringStatus.createFatalErrorStatus(Messages.format(RefactoringCoreMessages.InitializableRefactoring_argument_not_exist, ATTRIBUTE_ARGUMENT));
564                } else
565                        return RefactoringStatus.createFatalErrorStatus(RefactoringCoreMessages.InitializableRefactoring_inacceptable_arguments);
566                return new RefactoringStatus();
567        }
