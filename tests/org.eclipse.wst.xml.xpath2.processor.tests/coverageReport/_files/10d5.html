438        private void handleVerifyKeyPressed(VerifyEvent event) {
439                if (!event.doit)

441 
442                if (event.stateMask != SWT.MOD1)
443                        return;
444                        
445                switch (event.character) {
446                        case ' ':
447                                fPatternEditor.doOperation(ISourceViewer.CONTENTASSIST_PROPOSALS);
448                                event.doit= false;

450 
451                        // CTRL-Z
452                        case 'z' - 'a' + 1:
453                                fPatternEditor.doOperation(ITextOperationTarget.UNDO);

455                                break;                                
456                }
