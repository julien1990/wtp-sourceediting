345        public static Test suiteForComplianceLevel(String complianceLevel, Class setupClass, ArrayList testClasses) {
346                TestSuite suite;
347                Class testClass;
348                if (testClasses.size() == 1) {
349                        suite = new TestSuite(testClass = (Class)testClasses.get(0), complianceLevel);
350                        TESTS_COUNTERS.put(testClass.getName(), new Integer(suite.countTestCases()));
351                } else {

353                        for (int i = 0, length = testClasses.size(); i < length; i++) {
354                                TestSuite innerSuite = new TestSuite(testClass = (Class)testClasses.get(i));
355                                TESTS_COUNTERS.put(testClass.getName(), new Integer(innerSuite.countTestCases()));
356                                suite.addTest(innerSuite);
357                        }
358                }
359 
360                // call the setup constructor with the suite and compliance level
361                try {
362                        Constructor constructor = setupClass.getConstructor(new Class[]{Test.class, String.class});
363                        Test setUp = (Test)constructor.newInstance(new Object[]{suite, complianceLevel});
364                        return setUp;
365                } catch (IllegalAccessException e) {
366                        e.printStackTrace();
367                } catch (InstantiationException e) {

370                        e.getTargetException().printStackTrace();
371                } catch (NoSuchMethodException e) {
372                        e.printStackTrace();
373                }

375                return null;
376        }
