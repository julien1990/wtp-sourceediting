599        private CMDocument loadDocument(String baseLocation, Node taglib) {
600                Node root = taglib;
601 
602                // create the CMDocument
603                CMDocumentImpl document = new CMDocumentImpl();
604                document.setBaseLocation(baseLocation);
605 
606                if (root == null) {
607                        if (_debug) {
608                                System.out.println("null \"taglib\" element for TLD " + baseLocation); //$NON-NLS-1$
609                        }
610                        return document;
611                }
612 
613                // populate the CMDocument
614                Node child = root.getFirstChild();
615                while (child != null) {
616                        if (child.getNodeType() != Node.ELEMENT_NODE) {
617                                child = child.getNextSibling();
618                                continue;
619                        }
620                        String nodeName = child.getNodeName();
621                        if(nodeName.indexOf(':') > 0) {
622                                nodeName = nodeName.substring(nodeName.indexOf(':'));
623                        }
624                        // tag
625                        if (nodeName.equals(JSP11TLDNames.TAG)) {
626                                CMElementDeclaration ed = createElementDeclaration(document, child);
627                                if (ed != null) {
628                                        document.fElements.setNamedItem(ed.getNodeName(), ed);
629                                }
630                        }
631                        // tag-file
632                        else if (nodeName.equals(JSP20TLDNames.TAG_FILE) && child.getNodeType() == Node.ELEMENT_NODE && child.hasChildNodes()) {
633                                Element tagFileElement = (Element) child;
634                                Node path = tagFileElement.getFirstChild();
635                                while (path != null) {
636                                        if (path.getNodeType() == Node.ELEMENT_NODE && (JSP20TLDNames.PATH.equals(path.getNodeName()) || JSP20TLDNames.PATH.equals(path.getLocalName()))) {
637                                                String pathValue = getContainedText(path);
638                                                if (pathValue != null && pathValue.length() > 0) {
639                                                        CMElementDeclarationImpl ed = (CMElementDeclarationImpl) createElementDeclaration(document, tagFileElement, pathValue);
640                                                        if (ed != null) {
641                                                                document.fElements.setNamedItem(ed.getNodeName(), ed);
642                                                        }
643                                                }
644                                        }
645                                        path = path.getNextSibling();
646                                }
647                        }
648                        // other one-of-a-kind children
649                        // JSP version
650                        else if ((nodeName.equals(JSP11TLDNames.JSPVERSION) || nodeName.equals(JSP12TLDNames.JSP_VERSION)) && child.hasChildNodes()) {
651                                document.setJspversion(getContainedText(child));
652                        }
653                        // tag library version
654                        else if ((nodeName.equals(JSP11TLDNames.TLIBVERSION) || nodeName.equals(JSP12TLDNames.TLIB_VERSION)) && child.hasChildNodes()) {
655                                document.setTlibversion(getContainedText(child));
656                        }
657                        // short name
658                        else if ((nodeName.equals(JSP11TLDNames.SHORTNAME) || nodeName.equals(JSP12TLDNames.SHORT_NAME)) && child.hasChildNodes()) {
659                                document.setShortname(getContainedText(child));
660                        }
661                        // URI/URN
662                        else if ((nodeName.equals(JSP11TLDNames.URI) || nodeName.equals(JSP11TLDNames.URN)) && child.hasChildNodes()) { //$NON-NLS-1$
663                                document.setUri(getContainedText(child));
664                        }
665                        // info
666                        else if (nodeName.equals(JSP11TLDNames.INFO) && child.hasChildNodes()) {
667                                document.setInfo(getContainedText(child));
668                        }
669                        // New JSP 1.2

672                                document.setDescription(getContainedText(child));
673                        }
674                        // display name
675                        else if (nodeName.equals(JSP12TLDNames.DISPLAY_NAME) && child.hasChildNodes()) {

681                        }
