584        private static boolean isVisibleFrom(IMember member, IType newMemberDeclaringType, IType accessingType) throws JavaScriptModelException{
585                int memberVisibility= JdtFlags.getVisibilityCode(newMemberDeclaringType);
586                        
587                IType declaringType= newMemberDeclaringType.getDeclaringType();
588                while (declaringType != null) { //get lowest visibility in all parent types of newMemberDeclaringType
589                        memberVisibility= JdtFlags.getLowerVisibility(
590                                        memberVisibility, JdtFlags.getVisibilityCode(declaringType));
591                        declaringType= declaringType.getDeclaringType();
592                }
593                
594                switch (memberVisibility) {
595                        case Modifier.PRIVATE :
596                                return isEqualOrEnclosedType(accessingType, newMemberDeclaringType);
597                                
598                        case Modifier.NONE :
599                                return JavaModelUtil.isSamePackage(accessingType.getPackageFragment(), newMemberDeclaringType.getPackageFragment());
600                                        
601                        case Modifier.PROTECTED :
602                                return JavaModelUtil.isSamePackage(accessingType.getPackageFragment(), newMemberDeclaringType.getPackageFragment())
603                                                || accessingType.newSupertypeHierarchy(null).contains(newMemberDeclaringType);
604                        
605                        case Modifier.PUBLIC :
606                                return true;
607 
608                        default:
609                                Assert.isTrue(false);
610                                return false;
611                }
612        }

647                if (javaProject != null)
648                        project= javaProject.getElementName();

651                        header= Messages.format(RefactoringCoreMessages.MoveStaticMembersProcessor_descriptor_description_single, new String[] { JavaScriptElementLabels.getElementLabel(members[0], JavaScriptElementLabels.ALL_FULLY_QUALIFIED), JavaScriptElementLabels.getElementLabel(fDestinationType, JavaScriptElementLabels.ALL_FULLY_QUALIFIED) });

656                try {

658                                flags|= JavaScriptRefactoringDescriptor.JAR_SOURCE_ATTACHMENT;
659                } catch (JavaScriptModelException exception) {
660                        JavaScriptPlugin.log(exception);
661                }

663                final JDTRefactoringDescriptorComment comment= new JDTRefactoringDescriptorComment(project, this, header);
664                comment.addSetting(Messages.format(RefactoringCoreMessages.MoveStaticMembersProcessor_target_element_pattern, JavaScriptElementLabels.getElementLabel(fDestinationType, JavaScriptElementLabels.ALL_FULLY_QUALIFIED)));
665                final MoveStaticMembersDescriptor descriptor= new MoveStaticMembersDescriptor();
666                descriptor.setProject(project);
667                descriptor.setDescription(description);
668                descriptor.setComment(comment.asString());
669                descriptor.setFlags(flags);
670                descriptor.setDestinationType(fDestinationType);
671                descriptor.setKeepOriginal(fDelegateUpdating);
672                descriptor.setDeprecateDelegate(fDelegateDeprecation);
673                descriptor.setMembers(members);
674                fChange= new DynamicValidationRefactoringChange(descriptor, RefactoringCoreMessages.MoveMembersRefactoring_move_members);
675                fTarget= getCuRewrite(fDestinationType.getJavaScriptUnit());
676                ITypeBinding targetBinding= getDestinationBinding();
677                if (targetBinding == null) {
678                        status.addFatalError(Messages.format(RefactoringCoreMessages.MoveMembersRefactoring_compile_errors, fTarget.getCu().getElementName()));
679                        monitor.done();

686                        SubProgressMonitor sub= new SubProgressMonitor(monitor, 1, SubProgressMonitor.SUPPRESS_SUBTASK_LABEL);
687                        sub.beginTask(RefactoringCoreMessages.MoveMembersRefactoring_creating, fMembersToMove.length);
688                        Set rewritten= new HashSet();
689                        for (int index= 0; index < fMembersToMove.length; index++) {
690                                member= fMembersToMove[index];
691                                final MemberVisibilityAdjustor adjustor= new MemberVisibilityAdjustor(fDestinationType, member);
692                                adjustor.setAdjustments(adjustments);
693                                adjustor.setStatus(status);
694                                adjustor.setVisibilitySeverity(RefactoringStatus.WARNING);
695                                adjustor.setFailureSeverity(RefactoringStatus.WARNING);
696                                adjustor.setRewrite(fSource.getASTRewrite(), fSource.getRoot());
697                                adjustor.adjustVisibility(new NullProgressMonitor());
698 
699                                if (fDelegateUpdating && isDelegateCreationAvailable(member)) {
700                                        // Add a visibility adjustment so the moved member
701                                        // will be visible from within the delegate
702                                        ModifierKeyword threshold= adjustor.getVisibilityThreshold(member, fDestinationType, new NullProgressMonitor());
703                                        IncomingMemberVisibilityAdjustment adjustment= (IncomingMemberVisibilityAdjustment) adjustments.get(member);
704                                        ModifierKeyword kw= (adjustment != null) ? adjustment.getKeyword() : ModifierKeyword.fromFlagValue(JdtFlags.getVisibilityCode(member));
705                                        if (MemberVisibilityAdjustor.hasLowerVisibility(kw, threshold)) {
706                                                adjustments.put(member, new MemberVisibilityAdjustor.IncomingMemberVisibilityAdjustment(member, threshold, RefactoringStatus.createWarningStatus(Messages.format(MemberVisibilityAdjustor.getMessage(member), new String[] { MemberVisibilityAdjustor.getLabel(member), MemberVisibilityAdjustor.getLabel(threshold)}), JavaStatusContext.create(member))));

712                                status.merge(checkMovedMemberAvailability(member, new SubProgressMonitor(sub, 1)));
713                                // Put rewrite info into code and into status
714                                for (final Iterator iterator= rewritten.iterator(); iterator.hasNext();) {
715                                        adjustments.remove(iterator.next());
716                                }

721                        // First update references in moved members, in order to extract the
722                        // source.
723                        String[] memberSources= getUpdatedMemberSource(status, fMemberDeclarations, targetBinding);
724                        monitor.worked(1);
725                        if (status.hasFatalError())
726                                return;
727 
728                        final RefactoringSearchEngine2 engine= new RefactoringSearchEngine2();
729                        engine.setPattern(fMembersToMove, IJavaScriptSearchConstants.ALL_OCCURRENCES);
730                        engine.setGranularity(RefactoringSearchEngine2.GRANULARITY_COMPILATION_UNIT);
731                        engine.setFiltering(true, true);
732                        engine.setScope(RefactoringScopeFactory.create(fMembersToMove));
733                        engine.setStatus(status);
734                        engine.searchPattern(new NullProgressMonitor());
735                        IJavaScriptUnit[] units= engine.getAffectedCompilationUnits();
736                        modifiedCus.addAll(Arrays.asList(units));
737                        final MemberVisibilityAdjustor adjustor= new MemberVisibilityAdjustor(fDestinationType, fDestinationType);
738                        sub= new SubProgressMonitor(monitor, 1);
739                        sub.beginTask(RefactoringCoreMessages.MoveMembersRefactoring_creating, units.length);
740                        for (int index= 0; index < units.length; index++) {
741                                IJavaScriptUnit unit= units[index];
742                                CompilationUnitRewrite rewrite= getCuRewrite(unit);
743                                adjustor.setRewrites(Collections.singletonMap(unit, rewrite));
744                                adjustor.setAdjustments(adjustments);
745                                adjustor.rewriteVisibility(unit, new SubProgressMonitor(sub, 1));
746                                ReferenceAnalyzer analyzer= new ReferenceAnalyzer(rewrite, fMemberBindings, targetBinding, fSourceBinding);
747                                rewrite.getRoot().accept(analyzer);
748                                status.merge(analyzer.getStatus());
749                                if (status.hasFatalError()) {
750                                        fChange= null;
751                                        return;
752                                }
753                                if (!(fSource.getCu().equals(unit) || fTarget.getCu().equals(unit)))
754                                        fChange.add(rewrite.createChange());
755                        }
756                        status.merge(moveMembers(fMemberDeclarations, memberSources));
757                        fChange.add(fSource.createChange());
758                        modifiedCus.add(fSource.getCu());
759                        if (!fSource.getCu().equals(fTarget.getCu())) {
760                                fChange.add(fTarget.createChange());
761                                modifiedCus.add(fTarget.getCu());
762                        }
763                        monitor.worked(1);
764                } catch (BadLocationException exception) {
765                        JavaScriptPlugin.log(exception);
766                }
767        }

797        private String[] getUpdatedMemberSource(RefactoringStatus status, BodyDeclaration[] members, ITypeBinding target) throws CoreException, BadLocationException {
798                List typeRefs= new ArrayList();
799                boolean targetNeedsSourceImport= false;
800                boolean isSourceNotTarget= fSource != fTarget;
801                Set exclude= new HashSet();
802                for (int i= 0; i < members.length; i++) {
803                        BodyDeclaration declaration= members[i];
804                        if (declaration instanceof AbstractTypeDeclaration) {
805                                AbstractTypeDeclaration type= (AbstractTypeDeclaration) declaration;
806                                ITypeBinding binding= type.resolveBinding();
807                                if (binding != null)
808                                        exclude.add(binding);
809                        } else if (declaration instanceof FunctionDeclaration) {
810                                FunctionDeclaration method= (FunctionDeclaration) declaration;
811                                IFunctionBinding binding= method.resolveBinding();
812                                if (binding != null)
813                                        exclude.add(binding);
814                        } else if (declaration instanceof FieldDeclaration) {
815                                FieldDeclaration field= (FieldDeclaration) declaration;
816                                for (final Iterator iterator= field.fragments().iterator(); iterator.hasNext();) {
817                                        VariableDeclarationFragment fragment= (VariableDeclarationFragment) iterator.next();
818                                        IVariableBinding binding= fragment.resolveBinding();
819                                        if (binding != null)
820                                                exclude.add(binding);
821                                }
822                        }
823                }
824                for (int i= 0; i < members.length; i++) {
825                        BodyDeclaration declaration= members[i];
826                        if (isSourceNotTarget)
827                                typeRefs.addAll(TypeReferenceFinder.perform(declaration));
828                        MovedMemberAnalyzer analyzer= new MovedMemberAnalyzer(fSource, fMemberBindings, fSourceBinding, target);
829                        declaration.accept(analyzer);
830                        ImportRewriteUtil.addImports(fTarget, declaration, new HashMap(), new HashMap(), exclude, false);
831                        
832                        ITrackedNodePosition trackedPosition= fSource.getASTRewrite().track(declaration);
833                        declaration.setProperty(TRACKED_POSITION_PROPERTY, trackedPosition);
834                        targetNeedsSourceImport|= analyzer.targetNeedsSourceImport();
835                        status.merge(analyzer.getStatus());
836                }
837                // Adjust imports
838                if (targetNeedsSourceImport && isSourceNotTarget) {
839                        fTarget.getImportRewrite().addImport(fSourceBinding);
840                }
841                if (isSourceNotTarget) {
842                        for (Iterator iter= typeRefs.iterator(); iter.hasNext();) {
843                                ITypeBinding binding= (ITypeBinding) iter.next();
844                                fTarget.getImportRewrite().addImport(binding);
845                        }
846                }
847                // extract updated members
848                String[] updatedMemberSources= new String[members.length];
849                IDocument document= new Document(fSource.getCu().getBuffer().getContents());
850                TextEdit edit= fSource.getASTRewrite().rewriteAST(document, fSource.getCu().getJavaScriptProject().getOptions(true));
851                edit.apply(document, TextEdit.UPDATE_REGIONS);
852                for (int i= 0; i < members.length; i++) {
853                        updatedMemberSources[i]= getUpdatedMember(document, members[i]);
854                }
855                fSource.clearASTRewrite();
856                return updatedMemberSources;
857        }

861                return Strings.trimIndentation(document.get(trackedPosition.getStartPosition(), trackedPosition.getLength()), fPreferences.tabWidth, fPreferences.indentWidth, false);

864        private RefactoringStatus moveMembers(BodyDeclaration[] members, String[] sources) throws CoreException {
865                RefactoringStatus result= new RefactoringStatus();
866                AbstractTypeDeclaration destination= (AbstractTypeDeclaration)
867                ASTNodes.getParent(
868                        NodeFinder.perform(fTarget.getRoot(), fDestinationType.getNameRange()),
869                        AbstractTypeDeclaration.class);
870                ListRewrite containerRewrite= fTarget.getASTRewrite().getListRewrite(destination, destination.getBodyDeclarationsProperty());
871                
872                TextEditGroup delete= fSource.createGroupDescription(RefactoringCoreMessages.MoveMembersRefactoring_deleteMembers); 
873                TextEditGroup add= fTarget.createGroupDescription(RefactoringCoreMessages.MoveMembersRefactoring_addMembers); 
874                for (int i= 0; i < members.length; i++) {
875                        BodyDeclaration declaration= members[i];
876                        ASTNode removeImportsOf= null;
877                        boolean addedDelegate= false;
878                        
879                        if (fDelegateUpdating) {
880                                if (declaration instanceof FunctionDeclaration) {
881 
882                                        DelegateMethodCreator creator= new DelegateMethodCreator();
883                                        creator.setDeclaration(declaration);
884                                        creator.setDeclareDeprecated(fDelegateDeprecation);
885                                        creator.setSourceRewrite(fSource);
886                                        creator.setCopy(false);
887                                        creator.setNewLocation(getDestinationBinding());
888                                        creator.prepareDelegate();
889                                        creator.createEdit();
890 
891                                        removeImportsOf= ((FunctionDeclaration) declaration).getBody();
892                                        addedDelegate= true;
893                                }
894                                if (declaration instanceof FieldDeclaration) {
895 
896                                        // Note: this FieldDeclaration only has one fragment (@see #getASTMembers(RefactoringStatus))
897                                        final VariableDeclarationFragment frag= (VariableDeclarationFragment) ((FieldDeclaration) declaration).fragments().get(0);
898                                        
899                                        if (!Modifier.isFinal(declaration.getModifiers())) {
900                                                // Don't create a delegate for non-final fields
901                                                result.addInfo(Messages.format(RefactoringCoreMessages.DelegateCreator_cannot_create_field_delegate_not_final, frag.getName()), null);
902                                        } else if (frag.getInitializer() == null) {
903                                                // Don't create a delegate without an initializer.
904                                                result.addInfo(Messages.format(RefactoringCoreMessages.DelegateCreator_cannot_create_field_delegate_no_initializer, frag.getName()), null);
905                                        } else {
906                                                DelegateFieldCreator creator= new DelegateFieldCreator();
907                                                creator.setDeclaration(declaration);
908                                                creator.setDeclareDeprecated(fDelegateDeprecation);
909                                                creator.setSourceRewrite(fSource);
910                                                creator.setCopy(false);
911                                                creator.setNewLocation(getDestinationBinding());
912                                                creator.prepareDelegate();
913                                                creator.createEdit();
914 
915                                                removeImportsOf= frag.getInitializer();
916                                                addedDelegate= true;
917                                        }
918                                }
919                                if (declaration instanceof AbstractTypeDeclaration) {
920                                        result.addInfo(Messages.format(RefactoringCoreMessages.DelegateCreator_cannot_create_delegate_for_type, ((AbstractTypeDeclaration) declaration).getName().getIdentifier()),
921                                                        null);
922                                }
923                        } 
924                        
925                        if (!addedDelegate) {
926                                fSource.getASTRewrite().remove(declaration, delete);
927                                removeImportsOf= declaration;
928                        }
929 
930                        if (removeImportsOf != null && fSource != fTarget)
931                                fSource.getImportRemover().registerRemovedNode(removeImportsOf);
932                        
933                        ASTNode node= fTarget.getASTRewrite().createStringPlaceholder(sources[i], declaration.getNodeType());
934                        List container= containerRewrite.getRewrittenList();
935                        int insertionIndex= ASTNodes.getInsertionIndex((BodyDeclaration) node, container);
936                        containerRewrite.insertAt(node, insertionIndex, add);
937                }
938                return result;
939        }
