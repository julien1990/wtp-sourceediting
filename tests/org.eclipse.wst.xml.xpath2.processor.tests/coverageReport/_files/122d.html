271         * Computes and returns the invocation offset given a new
272         * position, the initial offset and the best invocation offset
273         * found so far.
274         * <p>
275         * The closest offset to the left of the initial offset is the
276         * best. If there is no offset on the left, the closest on the
277         * right is the best.</p>
278         * @return -1 is returned if the given offset is not closer or the new best offset
279         */
280        private static int computeBestOffset(int newOffset, int invocationLocation, int bestOffset) {
281                if (newOffset <= invocationLocation) {
282                        if (bestOffset > invocationLocation) {
283                                return newOffset; // closest was on the right, prefer on the left
284                        } else if (bestOffset <= newOffset) {
285                                return newOffset; // we are closer or equal
286                        }
287                        return -1; // further away
288                }
289 
290                if (newOffset <= bestOffset)
291                        return newOffset; // we are closer or equal
292 
293                return -1; // further away
294        }
