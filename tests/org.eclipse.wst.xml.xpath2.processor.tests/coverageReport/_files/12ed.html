467        /*
468         * Changes the input to a new type
469         * @param inputElement
470         */
471        private void updateInput(IJavaScriptElement inputElement) {
472                IJavaScriptElement prevInput= fInputElement;
473                
474                synchronized (this) {
475                        if (fRestoreStateJob != null) {
476                                fRestoreStateJob.cancel();
477                                try {
478                                        fRestoreStateJob.join();
479                                } catch (InterruptedException e) {
480                                        // ignore
481                                } finally {
482                                        fRestoreStateJob= null;
483                                }
484                        }
485                }
486                
487                // Make sure the UI got repainted before we execute a long running
488                // operation. This can be removed if we refresh the hierarchy in a 
489                // separate thread.
490                // Work-around for http://dev.eclipse.org/bugs/show_bug.cgi?id=30881
491                processOutstandingEvents();
492                if (inputElement == null) {        
493                        clearInput();
494                } else {
495                        fInputElement= inputElement;
496                        fNoHierarchyShownLabel.setText(Messages.format(TypeHierarchyMessages.TypeHierarchyViewPart_createinput, inputElement.getElementName())); 
497                        try {
498                                fHierarchyLifeCycle.ensureRefreshedTypeHierarchy(inputElement, JavaScriptPlugin.getActiveWorkbenchWindow());
499                                // fHierarchyLifeCycle.ensureRefreshedTypeHierarchy(inputElement, getSite().getWorkbenchWindow());
500                        } catch (InvocationTargetException e) {
501                                ExceptionHandler.handle(e, getSite().getShell(), TypeHierarchyMessages.TypeHierarchyViewPart_exception_title, TypeHierarchyMessages.TypeHierarchyViewPart_exception_message); 
502                                clearInput();
503                                return;
504                        } catch (InterruptedException e) {
505                                fNoHierarchyShownLabel.setText(TypeHierarchyMessages.TypeHierarchyViewPart_empty); 
506                                return;                                
507                        }
508                                
509                        if (inputElement.getElementType() != IJavaScriptElement.TYPE) {
510                                setHierarchyMode(HIERARCHY_MODE_CLASSIC);
511                        }
512                        // turn off member filtering
513                        fSelectInEditor= false;
514                        setMemberFilter(null);
515                        internalSelectType(null, false); // clear selection
516                        fIsEnableMemberFilter= false;
517                        if (!inputElement.equals(prevInput)) {
518                                updateHierarchyViewer(true);
519                        }
520                        IType root= getSelectableType(inputElement);
521                        internalSelectType(root, true);
522                        updateMethodViewer(root);
523                        updateToolbarButtons();
524                        updateTitle();
525                        showMembersInHierarchy(false);
526                        fPagebook.showPage(fTypeMethodsSplitter);
527                        fSelectInEditor= true;
528                }
529        }
