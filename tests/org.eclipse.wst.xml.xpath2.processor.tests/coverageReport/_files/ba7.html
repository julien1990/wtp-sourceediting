381        public final int recordInitializationStates(FlowInfo flowInfo) {
382 
383                if ((flowInfo.tagBits & FlowInfo.UNREACHABLE) != 0) return -1;
384 
385                UnconditionalFlowInfo unconditionalFlowInfo = flowInfo.unconditionalInitsWithoutSideEffect();
386                long[] extraInits = unconditionalFlowInfo.extra == null ?
387                                null : unconditionalFlowInfo.extra[0];
388                long inits = unconditionalFlowInfo.definiteInits;
389                checkNextEntry : for (int i = lastIndex; --i >= 0;) {
390                        if (definiteInits[i] == inits) {
391                                long[] otherInits = extraDefiniteInits[i];
392                                if ((extraInits != null) && (otherInits != null)) {
393                                        if (extraInits.length == otherInits.length) {
394                                                int j, max;
395                                                for (j = 0, max = extraInits.length; j < max; j++) {
396                                                        if (extraInits[j] != otherInits[j]) {
397                                                                continue checkNextEntry;
398                                                        }
399                                                }
400                                                return i;
401                                        }
402                                } else {
403                                        if ((extraInits == null) && (otherInits == null)) {
404                                                return i;
405                                        }
406                                }
407                        }
408                }
409 
410                // add a new entry
411                if (definiteInits.length == lastIndex) {
412                        // need a resize
413                        System.arraycopy(
414                                definiteInits,
415                                0,
416                                (definiteInits = new long[lastIndex + 20]),
417                                0,
418                                lastIndex);
419                        System.arraycopy(
420                                extraDefiniteInits,
421                                0,
422                                (extraDefiniteInits = new long[lastIndex + 20][]),
423                                0,
424                                lastIndex);
425                }
426                definiteInits[lastIndex] = inits;
427                if (extraInits != null) {
428                        extraDefiniteInits[lastIndex] = new long[extraInits.length];
429                        System.arraycopy(
430                                extraInits,
431                                0,
432                                extraDefiniteInits[lastIndex],
433                                0,
434                                extraInits.length);
435                }
436                return lastIndex++;
437        }
