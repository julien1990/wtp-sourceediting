402         * Request the indexing of entries that have been added, and remove the index for removed entries.
403         */
404        public void requestIndexing() {
405                IIncludePathEntry[] newResolvedClasspath = null;
406                try {
407                        newResolvedClasspath = this.project.getResolvedClasspath();
408                } catch (JavaScriptModelException e) {
409                        // project doesn't exist
410                        return;
411                }
412 
413                JavaModelManager manager = JavaModelManager.getJavaModelManager();
414                IndexManager indexManager = manager.indexManager;
415                if (indexManager == null)
416                        return;
417                DeltaProcessingState state = manager.deltaState;
418 
419                int newLength = newResolvedClasspath.length;
420                int oldLength = this.oldResolvedClasspath.length;
421                for (int i = 0; i < oldLength; i++) {
422                        int index = classpathContains(newResolvedClasspath, this.oldResolvedClasspath[i]);
423                        if (index == -1) {
424                                // remote projects are not indexed in this project
425                                if (this.oldResolvedClasspath[i].getEntryKind() == IIncludePathEntry.CPE_PROJECT){
426                                        continue;
427                                }
428 
429                                // Remove the .js files from the index for a source folder
430                                // For a lib folder or a .jar file, remove the corresponding index if not shared.
431                                IIncludePathEntry oldEntry = this.oldResolvedClasspath[i];
432                                final IPath path = oldEntry.getPath();
433                                int changeKind = this.oldResolvedClasspath[i].getEntryKind();
434                                switch (changeKind) {
435                                        case IIncludePathEntry.CPE_SOURCE:
436                                                char[][] inclusionPatterns = ((ClasspathEntry)oldEntry).fullInclusionPatternChars();
437                                                char[][] exclusionPatterns = ((ClasspathEntry)oldEntry).fullExclusionPatternChars();
438                                                indexManager.removeSourceFolderFromIndex(this.project, path, inclusionPatterns, exclusionPatterns);
439                                                break;
440                                        case IIncludePathEntry.CPE_LIBRARY:
441                                                if (state.otherRoots.get(path) == null) { // if root was not shared
442                                                        indexManager.discardJobs(path.toString());
443                                                        indexManager.removeIndex(path);
444                                                        // TODO (kent) we could just remove the in-memory index and have the indexing check for timestamps
445                                                }
446                                                break;
447                                }
448                        }
449                }
450 
451                for (int i = 0; i < newLength; i++) {
452                        int index = classpathContains(this.oldResolvedClasspath, newResolvedClasspath[i]);
453                        if (index == -1) {
454                                // remote projects are not indexed in this project
455                                if (newResolvedClasspath[i].getEntryKind() == IIncludePathEntry.CPE_PROJECT){
456                                        continue;
457                                }
458 
459                                // Request indexing
460                                int entryKind = newResolvedClasspath[i].getEntryKind();
461                                switch (entryKind) {
462                                        case IIncludePathEntry.CPE_LIBRARY:
463                                                boolean pathHasChanged = true;
464                                                IPath newPath = newResolvedClasspath[i].getPath();
465                                                for (int j = 0; j < oldLength; j++) {
466                                                        IIncludePathEntry oldEntry = this.oldResolvedClasspath[j];
467                                                        if (oldEntry.getPath().equals(newPath)) {
468                                                                pathHasChanged = false;
469                                                                break;
470                                                        }
471                                                }
472                                                if (pathHasChanged) {
473                                                        indexManager.indexLibrary(newResolvedClasspath[i], this.project.getProject());
474                                                }
475                                                break;
476                                        case IIncludePathEntry.CPE_SOURCE:
477                                                IIncludePathEntry entry = newResolvedClasspath[i];
478                                                IPath path = entry.getPath();
479                                                char[][] inclusionPatterns = ((ClasspathEntry)entry).fullInclusionPatternChars();
480                                                char[][] exclusionPatterns = ((ClasspathEntry)entry).fullExclusionPatternChars();
481                                                indexManager.indexSourceFolder(this.project, path, inclusionPatterns, exclusionPatterns);
482                                                break;
483                                }
484                        }
485                }
486        }
