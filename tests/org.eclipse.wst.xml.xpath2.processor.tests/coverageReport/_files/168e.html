373                                                return child;

376                                parent = child;
377                                child = (IDOMNode) parent.getFirstChild();
378                        }
379                }
380                else {
381                        // search from the last
382                        IDOMNode child = (IDOMNode) this.document.getLastChild();
383                        while (child != null) {
384                                if (child.getStartOffset() > offset) {
385                                        child = (IDOMNode) child.getPreviousSibling();
386                                        continue;
387                                }
