405public IJavaScriptElement getElementAtConsideringSibling(int position) throws JavaScriptModelException {
406        IPackageFragment fragment = (IPackageFragment)getParent();
407        PackageFragmentRoot root = (PackageFragmentRoot) fragment.getAncestor(IJavaScriptElement.PACKAGE_FRAGMENT_ROOT);
408        SourceMapper mapper = root.getSourceMapper();
409        if (mapper == null) {
410                return null;
411        } else {
412                int index = this.name.indexOf('$');
413                int prefixLength = index < 0 ? this.name.length() : index;
414 
415                IType type = null;
416                int start = -1;
417                int end = Integer.MAX_VALUE;
418                IJavaScriptElement[] children = fragment.getChildren();
419                for (int i = 0; i < children.length; i++) {
420                        String childName = children[i].getElementName();
421 
422                        int childIndex = childName.indexOf('$');
423                        int childPrefixLength = childIndex < 0 ? childName.indexOf('.') : childIndex;
424                        if (prefixLength == childPrefixLength && this.name.regionMatches(0, childName, 0, prefixLength)) {
425                                IClassFile classFile = (IClassFile) children[i];
426 
427                                // ensure this class file's buffer is open so that source ranges are computed
428                                classFile.getBuffer();
429 
430                                SourceRange range = mapper.getSourceRange(classFile.getType());
431                                if (range == SourceMapper.UNKNOWN_RANGE) continue;
432                                int newStart = range.offset;
433                                int newEnd = newStart + range.length - 1;
434                                if(newStart > start && newEnd < end
435                                                && newStart <= position && newEnd >= position) {
436                                        type = classFile.getType();
437                                        start = newStart;
438                                        end = newEnd;
439                                }
440                        }
441                }
442                if(type != null) {
443                        return findElement(type, position, mapper);
444                }
445                return null;

755 * or <code>null</code> if the given className is <code>null</code>.
756 * (This removes the package name, but not enclosing type names).
757 *

770                        return name;
771                }
772                count++;
773        }
774        return className;
