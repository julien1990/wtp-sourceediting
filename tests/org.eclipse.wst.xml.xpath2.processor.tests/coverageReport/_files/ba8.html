310For each inherited method identifier (message pattern - vm signature minus the return type)
311        if current method exists
312                if current's vm signature does not match an inherited signature then complain
313                else compare current's exceptions & visibility against each inherited method
314        else
315                if inherited methods = 1
316                        if inherited is abstract && type is NOT an interface or abstract, complain
317                else
318                        if vm signatures do not match complain
319                        else
320                                find the concrete implementation amongst the abstract methods (can only be 1)
321                                if one exists then
322                                        it must be a public instance method
323                                        compare concrete's exceptions against each abstract method
324                                else
325                                        complain about missing implementation only if type is NOT an interface or abstract
326*/
327void checkMethods() {
328        boolean mustImplementAbstractMethods = mustImplementAbstractMethods();
329        boolean skipInheritedMethods = mustImplementAbstractMethods && canSkipInheritedMethods(); // have a single concrete superclass so only check overridden methods
330        char[][] methodSelectors = this.inheritedMethods.keyTable;
331        nextSelector : for (int s = methodSelectors.length; --s >= 0;) {
332                if (methodSelectors[s] == null) continue nextSelector;
333 
334                MethodBinding[] current = (MethodBinding[]) this.currentMethods.get(methodSelectors[s]);
335                if (current == null && skipInheritedMethods)
336                        continue nextSelector;
337 
338                MethodBinding[] inherited = (MethodBinding[]) this.inheritedMethods.valueTable[s];
339                if (inherited.length == 1 && current == null) { // handle the common case
340                        if (mustImplementAbstractMethods && inherited[0].isAbstract())
341                                checkAbstractMethod(inherited[0]);
342                        continue nextSelector;
343                }
344 
345                int index = -1;
346                MethodBinding[] matchingInherited = new MethodBinding[inherited.length];
347                if (current != null) {
348                        for (int i = 0, length1 = current.length; i < length1; i++) {
349                                MethodBinding currentMethod = current[i];
350                                for (int j = 0, length2 = inherited.length; j < length2; j++) {
351                                        MethodBinding inheritedMethod = computeSubstituteMethod(inherited[j], currentMethod);
352                                        if (inheritedMethod != null) {
353                                                if (doesMethodOverride(currentMethod, inheritedMethod)) {
354                                                        matchingInherited[++index] = inheritedMethod;
355                                                        inherited[j] = null; // do not want to find it again
356                                                }
357                                        }
358                                }
359                                if (index >= 0) {
360                                        checkAgainstInheritedMethods(currentMethod, matchingInherited, index + 1, inherited); // pass in the length of matching
361                                        while (index >= 0) matchingInherited[index--] = null; // clear the contents of the matching methods
362                                }
363                        }
364                }
365 
366                for (int i = 0, length = inherited.length; i < length; i++) {
367                        MethodBinding inheritedMethod = inherited[i];
368                        if (inheritedMethod == null) continue;
369 
370                        matchingInherited[++index] = inheritedMethod;
371                        for (int j = i + 1; j < length; j++) {
372                                MethodBinding otherInheritedMethod = inherited[j];
373                                if (canSkipInheritedMethods(inheritedMethod, otherInheritedMethod))
374                                        continue;
375                                otherInheritedMethod = computeSubstituteMethod(otherInheritedMethod, inheritedMethod);
376                                if (otherInheritedMethod != null) {
377                                        if (doesMethodOverride(inheritedMethod, otherInheritedMethod)) {
378                                                matchingInherited[++index] = otherInheritedMethod;
379                                                inherited[j] = null; // do not want to find it again
380                                        }
381                                }
382                        }
383                        if (index == -1) continue;
384                        if (index > 0)
385                                checkInheritedMethods(matchingInherited, index + 1); // pass in the length of matching
386                        else if (mustImplementAbstractMethods && matchingInherited[0].isAbstract())
387                                checkAbstractMethod(matchingInherited[0]);
388                        while (index >= 0) matchingInherited[index--] = null; // clear the contents of the matching methods
389                }
390        }
391}
392void checkPackagePrivateAbstractMethod(MethodBinding abstractMethod) {
393        // check that the inherited abstract method (package private visibility) is implemented within the same package
394        PackageBinding necessaryPackage = abstractMethod.declaringClass.fPackage;
395        if (necessaryPackage == this.type.fPackage) return; // not a problem
396 
397        ReferenceBinding superType = this.type.superclass();
398        char[] selector = abstractMethod.selector;
399        do {
400                if (!superType.isValidBinding()) return;
401                if (!superType.isAbstract()) return; // closer non abstract super type will be flagged instead
402 
403                if (necessaryPackage == superType.fPackage) {
404                        MethodBinding[] methods = superType.getMethods(selector);
405                        nextMethod : for (int m = methods.length; --m >= 0;) {
406                                MethodBinding method = methods[m];
407                                if (method.isPrivate() || method.isConstructor() || method.isDefaultAbstract())
408                                        continue nextMethod;
409                                if (areMethodsCompatible(method, abstractMethod))
410                                        return; // found concrete implementation of abstract method in same package
411                        }
412                }
413        } while ((superType = superType.superclass()) != abstractMethod.declaringClass);
414 
415        // non visible abstract methods cannot be overridden so the type must be defined abstract
416        problemReporter().abstractMethodCannotBeOverridden(this.type, abstractMethod);
417}
418void computeInheritedMethods() {
419        ReferenceBinding superclass = this.type.isInterface()
420                ? this.type.scope.getJavaLangObject() // check interface methods against Object
421                : this.type.superclass(); // class or enum
422        computeInheritedMethods(superclass, type.superInterfaces());
423}
424/*
425Binding creation is responsible for reporting:
426        - all modifier problems (duplicates & multiple visibility modifiers + incompatible combinations)
427                - plus invalid modifiers given the context... examples:
428                        - interface methods can only be public
429                        - abstract methods can only be defined by abstract classes
430        - collisions... 2 methods with identical vmSelectors
431        - multiple methods with the same message pattern but different return types
432        - ambiguous, invisible or missing return/argument/exception types
433        - check the type of any array is not void
434        - check that each exception type is Throwable or a subclass of it
435*/
436void computeInheritedMethods(ReferenceBinding superclass, ReferenceBinding[] superInterfaces) {
437        // only want to remember inheritedMethods that can have an impact on the current type
438        // if an inheritedMethod has been 'replaced' by a supertype's method then skip it
439 
440        this.inheritedMethods = new HashtableOfObject(51); // maps method selectors to an array of methods... must search to match paramaters & return type
441        ReferenceBinding[] interfacesToVisit = null;
442        int nextPosition = 0;
443        ReferenceBinding[] itsInterfaces = superInterfaces;
444        if (itsInterfaces != Binding.NO_SUPERINTERFACES) {
445                nextPosition = itsInterfaces.length;
446                interfacesToVisit = itsInterfaces;
447        }
448 
449        ReferenceBinding superType = superclass;
450        HashtableOfObject nonVisibleDefaultMethods = new HashtableOfObject(3); // maps method selectors to an array of methods
451        boolean allSuperclassesAreAbstract = true;
452 
453        while (superType != null && superType.isValidBinding()) {
454            if (allSuperclassesAreAbstract) {
455                    if (superType.isAbstract()) {
456                                // only need to include superinterfaces if immediate superclasses are abstract
457                                if ((itsInterfaces = superType.superInterfaces()) != Binding.NO_SUPERINTERFACES) {
458                                        if (interfacesToVisit == null) {
459                                                interfacesToVisit = itsInterfaces;
460                                                nextPosition = interfacesToVisit.length;
461                                        } else {
462                                                int itsLength = itsInterfaces.length;
463                                                if (nextPosition + itsLength >= interfacesToVisit.length)
464                                                        System.arraycopy(interfacesToVisit, 0, interfacesToVisit = new ReferenceBinding[nextPosition + itsLength + 5], 0, nextPosition);
465                                                nextInterface : for (int a = 0; a < itsLength; a++) {
466                                                        ReferenceBinding next = itsInterfaces[a];
467                                                        for (int b = 0; b < nextPosition; b++)
468                                                                if (next == interfacesToVisit[b]) continue nextInterface;
469                                                        interfacesToVisit[nextPosition++] = next;
470                                                }
471                                        }
472                                }
473                        } else {
474                            allSuperclassesAreAbstract = false;
475                        }
476                }
477 
478                MethodBinding[] methods = superType.unResolvedMethods();
479                nextMethod : for (int m = methods.length; --m >= 0;) {
480                        MethodBinding inheritedMethod = methods[m];
481                        if (inheritedMethod.isPrivate() || inheritedMethod.isConstructor() || inheritedMethod.isDefaultAbstract())
482                                continue nextMethod;
483                        MethodBinding[] existingMethods = (MethodBinding[]) this.inheritedMethods.get(inheritedMethod.selector);
484                        if (existingMethods != null) {
485                                for (int i = 0, length = existingMethods.length; i < length; i++) {
486                                        if (existingMethods[i].declaringClass != inheritedMethod.declaringClass && areMethodsCompatible(existingMethods[i], inheritedMethod)) {
487                                                if (inheritedMethod.isDefault() && inheritedMethod.isAbstract())
488                                                        checkPackagePrivateAbstractMethod(inheritedMethod);
489                                                continue nextMethod;
490                                        }
491                                }
492                        }
493                        MethodBinding[] nonVisible = (MethodBinding[]) nonVisibleDefaultMethods.get(inheritedMethod.selector);
494                        if (nonVisible != null)
495                                for (int i = 0, l = nonVisible.length; i < l; i++)
496                                        if (areMethodsCompatible(nonVisible[i], inheritedMethod))
497                                                continue nextMethod;
498 
499                        if (!inheritedMethod.isDefault() || inheritedMethod.declaringClass.fPackage == type.fPackage) {
500                                if (existingMethods == null) {
501                                        existingMethods = new MethodBinding[] {inheritedMethod};
502                                } else {
503                                        int length = existingMethods.length;
504                                        System.arraycopy(existingMethods, 0, existingMethods = new MethodBinding[length + 1], 0, length);
505                                        existingMethods[length] = inheritedMethod;
506                                }
507                                this.inheritedMethods.put(inheritedMethod.selector, existingMethods);
508                        } else {
509                                if (nonVisible == null) {
510                                        nonVisible = new MethodBinding[] {inheritedMethod};
511                                } else {
512                                        int length = nonVisible.length;
513                                        System.arraycopy(nonVisible, 0, nonVisible = new MethodBinding[length + 1], 0, length);
514                                        nonVisible[length] = inheritedMethod;
515                                }
516                                nonVisibleDefaultMethods.put(inheritedMethod.selector, nonVisible);
517 
518                                if (inheritedMethod.isAbstract() && !this.type.isAbstract()) // non visible abstract methods cannot be overridden so the type must be defined abstract
519                                        problemReporter().abstractMethodCannotBeOverridden(this.type, inheritedMethod);
520 
521                                MethodBinding[] current = (MethodBinding[]) this.currentMethods.get(inheritedMethod.selector);
522                                if (current != null) { // non visible methods cannot be overridden so a warning is issued
523                                        foundMatch : for (int i = 0, length = current.length; i < length; i++) {
524                                                if (areMethodsCompatible(current[i], inheritedMethod)) {
525                                                        problemReporter().overridesPackageDefaultMethod(current[i], inheritedMethod);
526                                                        break foundMatch;
527                                                }
528                                        }
529                                }
530                        }
531                }
532                superType = superType.superclass();
533        }
534        if (nextPosition == 0) return;
535 
536        SimpleSet skip = findSuperinterfaceCollisions(superclass, superInterfaces);
537        for (int i = 0; i < nextPosition; i++) {
538                superType = interfacesToVisit[i];
539                if (superType.isValidBinding()) {
540                        if (skip != null && skip.includes(superType)) continue;
541                        if ((itsInterfaces = superType.superInterfaces()) != Binding.NO_SUPERINTERFACES) {
542                                int itsLength = itsInterfaces.length;
543                                if (nextPosition + itsLength >= interfacesToVisit.length)
544                                        System.arraycopy(interfacesToVisit, 0, interfacesToVisit = new ReferenceBinding[nextPosition + itsLength + 5], 0, nextPosition);
545                                nextInterface : for (int a = 0; a < itsLength; a++) {
546                                        ReferenceBinding next = itsInterfaces[a];
547                                        for (int b = 0; b < nextPosition; b++)
548                                                if (next == interfacesToVisit[b]) continue nextInterface;
549                                        interfacesToVisit[nextPosition++] = next;
550                                }
551                        }
552 
553                        MethodBinding[] methods = superType.unResolvedMethods();
554                        nextMethod : for (int m = methods.length; --m >= 0;) { // Interface methods are all abstract public
555                                MethodBinding inheritedMethod = methods[m];
556                                MethodBinding[] existingMethods = (MethodBinding[]) this.inheritedMethods.get(inheritedMethod.selector);
557                                if (existingMethods == null) {
558                                        existingMethods = new MethodBinding[] {inheritedMethod};
559                                } else {
560                                        int length = existingMethods.length;
561                                        // look to see if any of the existingMethods implement this inheritedMethod
562                                        for (int e = 0; e < length; e++)
563                                                if (isInterfaceMethodImplemented(inheritedMethod, existingMethods[e], superType))
564                                                        continue nextMethod; // skip interface method with the same signature if visible to its declaringClass
565                                        System.arraycopy(existingMethods, 0, existingMethods = new MethodBinding[length + 1], 0, length);
566                                        existingMethods[length] = inheritedMethod;
567                                }
568                                this.inheritedMethods.put(inheritedMethod.selector, existingMethods);
569                        }
570                }
571        }
572}

575        int size = methods.length;
576        this.currentMethods = new HashtableOfObject(size == 0 ? 1 : size); // maps method selectors to an array of methods... must search to match paramaters & return type
577        for (int m = size; --m >= 0;) {
578                MethodBinding method = methods[m];
579                if (!(method.isConstructor() || method.isDefaultAbstract())) { // keep all methods which are NOT constructors or default abstract

581                        if (existingMethods == null)
582                                existingMethods = new MethodBinding[1];
583                        else
584                                System.arraycopy(existingMethods, 0,

588                }
589        }

602int[] findOverriddenInheritedMethods(MethodBinding[] methods, int length) {
603        // NOTE assumes length > 1
604        // inherited methods are added as we walk up the superclass hierarchy, then each superinterface
605        // so method[1] from a class can NOT override method[0], but methods from superinterfaces can
606        // since superinterfaces can be added from different superclasses or other superinterfaces
607        int[] toSkip = null;
608        int i = 0;
609        ReferenceBinding declaringClass = methods[i].declaringClass;
610        if (!declaringClass.isInterface()) {
611                // in the first pass, skip overridden methods from superclasses
612                // only keep methods from the closest superclass, all others from higher superclasses can be skipped
613                // NOTE: methods were added in order by walking up the superclass hierarchy
614                ReferenceBinding declaringClass2 = methods[++i].declaringClass;
615                while (declaringClass == declaringClass2) {
616                        if (++i == length) return null;
617                        declaringClass2 = methods[i].declaringClass;
618                }
619                if (!declaringClass2.isInterface()) {
620                        // skip all methods from different superclasses
621                        toSkip = new int[length];
622                        do {
623                                toSkip[i] = -1;
624                                if (++i == length) return toSkip;
625                                declaringClass2 = methods[i].declaringClass;
626                        } while (!declaringClass2.isInterface());
627                }
628        }
629        // in the second pass, skip overridden methods from superinterfaces
630        // NOTE: superinterfaces can appear in 'random' order
631        nextMethod : for (; i < length; i++) {
632                if (toSkip != null && toSkip[i] == -1) continue nextMethod;
633                declaringClass = methods[i].declaringClass;
634                for (int j = i + 1; j < length; j++) {
635                        if (toSkip != null && toSkip[j] == -1) continue;
636                        ReferenceBinding declaringClass2 = methods[j].declaringClass;
637                        if (declaringClass == declaringClass2) continue;
638                        if (declaringClass.implementsInterface(declaringClass2, true)) {
639                                if (toSkip == null)
640                                        toSkip = new int[length];
641                                toSkip[j] = -1;
642                        } else if (declaringClass2.implementsInterface(declaringClass, true)) {
643                                if (toSkip == null)
644                                        toSkip = new int[length];
645                                toSkip[i] = -1;
646                                continue nextMethod;
647                        }
648                }
649        }
650        return toSkip;
651}
652boolean isAsVisible(MethodBinding newMethod, MethodBinding inheritedMethod) {
653        if (inheritedMethod.modifiers == newMethod.modifiers) return true;

658        if (newMethod.isProtected()) return true;

662}

673boolean mustImplementAbstractMethod(ReferenceBinding declaringClass) {
674        // if the type's superclass is an abstract class, then all abstract methods must be implemented
675        // otherwise, skip it if the type's superclass must implement any of the inherited methods
676        ReferenceBinding superclass = this.type.superclass();
677        if (declaringClass.isClass()) {
678                while (superclass.isAbstract() && superclass != declaringClass)
679                        superclass = superclass.superclass(); // find the first concrete superclass or the abstract declaringClass
680        } else {
681                if (this.type.implementsInterface(declaringClass, false)) {
682                        if (this.type.isAbstract()) return false; // leave it for the subclasses
683                        if (!superclass.implementsInterface(declaringClass, true)) // only if a superclass does not also implement the interface
684                                return true;
685                }
686                while (superclass.isAbstract() && !superclass.implementsInterface(declaringClass, false))
687                        superclass = superclass.superclass(); // find the first concrete superclass or the superclass which implements the interface
688        }
689        return superclass.isAbstract();                // if it is a concrete class then we have already reported problem against it
690}
691boolean mustImplementAbstractMethods() {
692        return !this.type.isInterface() && !this.type.isAbstract();
693}
694ProblemReporter problemReporter() {
695        return this.type.scope.problemReporter();
696}
697ProblemReporter problemReporter(MethodBinding currentMethod) {
698        ProblemReporter reporter = problemReporter();
699        if (currentMethod.declaringClass == type && currentMethod.sourceMethod() != null)        // only report against the currentMethod if its implemented by the type
700                reporter.referenceContext = currentMethod.sourceMethod();
701        return reporter;
702}
703/**
704 * Return true and report an incompatibleReturnType error if currentMethod's
705 * return type is strictly incompatible with inheritedMethod's, else return
706 * false and report an unchecked conversion warning. Do not call when
707 * areReturnTypesCompatible(currentMethod, inheritedMethod) returns true.
708 * @param currentMethod the (potentially) inheriting method
709 * @param inheritedMethod the inherited method
710 * @return true if currentMethod's return type is strictly incompatible with
711 *         inheritedMethod's
712 */
713boolean reportIncompatibleReturnTypeError(MethodBinding currentMethod, MethodBinding inheritedMethod) {
714        problemReporter(currentMethod).incompatibleReturnType(currentMethod, inheritedMethod);
715        return true;
716}
717ReferenceBinding[] resolvedExceptionTypesFor(MethodBinding method) {
718        ReferenceBinding[] exceptions = method.thrownExceptions;
719        if ((method.modifiers & ExtraCompilerModifiers.AccUnresolved) == 0)
720                return exceptions;
721 
722        if (!(method.declaringClass instanceof BinaryTypeBinding))
723                return Binding.NO_EXCEPTIONS; // safety check
724 
725        for (int i = exceptions.length; --i >= 0;)
726                exceptions[i] = BinaryTypeBinding.resolveType(exceptions[i], this.environment, true);
727        return exceptions;
728}
