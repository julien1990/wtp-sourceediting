364        if ((this.bits & Binding.FIELD) != 0) {
365                field = (FieldBinding) this.binding;
366                if (!field.isStatic()) {
367                        //must check for the static status....
368                        if (this.indexOfFirstFieldBinding > 1  //accessing to a field using a type as "receiver" is allowed only with static field
369                                         || scope.methodScope().isStatic) {         // the field is the first token of the qualified reference....
370                                scope.problemReporter().staticFieldAccessToNonStaticVariable(this, field);
371                                return null;
372                         }
373                } else {
374                        // indirect static reference ?
375                        if (this.indexOfFirstFieldBinding > 1
376                                        && field.declaringClass != this.actualReceiverType
377                                        && field.declaringClass.canBeSeenBy(scope)) {
378                                scope.problemReporter().indirectAccessToStaticField(this, field);
379                        }
380                }
381                // only last field is actually a write access if any
382                if (isFieldUseDeprecated(field, scope, (this.bits & ASTNode.IsStrictlyAssigned) != 0 && this.indexOfFirstFieldBinding == length))
383                        scope.problemReporter().deprecatedField(field, this);
384        } else {
385                field = null;
386        }
387        TypeBinding type = ((VariableBinding) this.binding).type;
388        int index = this.indexOfFirstFieldBinding;
389        if (index == length) { //        restrictiveFlag == FIELD
390                this.constant = ((FieldBinding) this.binding).constant();
391                // perform capture conversion if read access
392                return (type != null && (this.bits & ASTNode.IsStrictlyAssigned) == 0)
393                                ? type.capture(scope, this.sourceEnd)
394                                : type;
395        }
396        // allocation of the fieldBindings array        and its respective constants
397        int otherBindingsLength = length - index;
398        this.otherCodegenBindings = this.otherBindings = new FieldBinding[otherBindingsLength];
399        this.otherDepths = new int[otherBindingsLength];
400 
401        // fill the first constant (the one of the binding)
402        this.constant = ((VariableBinding) this.binding).constant();
403        // save first depth, since will be updated by visibility checks of other bindings
404        int firstDepth = (this.bits & ASTNode.DepthMASK) >> ASTNode.DepthSHIFT;
405        // iteration on each field
406        while (index < length) {
407                char[] token = this.tokens[index];
408                if (type == null)
409                        return null; // could not resolve type prior to this point
410 
411                this.bits &= ~ASTNode.DepthMASK; // flush previous depth if any
412                FieldBinding previousField = field;
413                field = scope.getField(type.capture(scope, (int)this.sourcePositions[index]), token, this);
414                int place = index - this.indexOfFirstFieldBinding;
415                this.otherBindings[place] = field;
416                this.otherDepths[place] = (this.bits & ASTNode.DepthMASK) >> ASTNode.DepthSHIFT;
417                if (field.isValidBinding()) {
418                        // set generic cast of for previous field (if any)
419                        if (previousField != null) {
420                                TypeBinding fieldReceiverType = type;
421                                TypeBinding receiverErasure = type.erasure();
422                                if (receiverErasure instanceof ReferenceBinding) {
423                                        if (receiverErasure.findSuperTypeWithSameErasure(field.declaringClass) == null) {
424                                                fieldReceiverType = field.declaringClass; // handle indirect inheritance thru variable secondary bound
425                                        }
426                                }
427                                FieldBinding originalBinding = previousField.original();
428                            if ((originalBinding.type.tagBits &  TagBits.HasTypeVariable) != 0 && fieldReceiverType.id != TypeIds.T_JavaLangObject) {
429                                    setGenericCast(index-1,originalBinding.type.genericCast(fieldReceiverType)); // type cannot be base-type even in boxing case
430                            }
431                    }
432                        // only last field is actually a write access if any
433                        if (isFieldUseDeprecated(field, scope, (this.bits & ASTNode.IsStrictlyAssigned) !=0 && index+1 == length)) {
434                                scope.problemReporter().deprecatedField(field, this);
435                        }
436                        // constant propagation can only be performed as long as the previous one is a constant too.
437                        if (this.constant != Constant.NotAConstant) {
438                                this.constant = field.constant();
439                        }
440 
441                        if (field.isStatic()) {
442                                // static field accessed through receiver? legal but unoptimal (optional warning)
443                                scope.problemReporter().nonStaticAccessToStaticField(this, field);
444                                // indirect static reference ?
445                                if (field.declaringClass != type) {
446                                        scope.problemReporter().indirectAccessToStaticField(this, field);
447                                }
448                        }
449                        type = field.type;
450                        index++;
451                } else {
452                        this.constant = Constant.NotAConstant; //don't fill other constants slots...
453                        scope.problemReporter().invalidField(this, field, index, type);
454                        setDepth(firstDepth);
455                        return null;
456                }
457        }
458        setDepth(firstDepth);
459        type = (this.otherBindings[otherBindingsLength - 1]).type;
460        // perform capture conversion if read access
461        return (type != null && (this.bits & ASTNode.IsStrictlyAssigned) == 0)
462                        ? type.capture(scope, this.sourceEnd)
463                        : type;

481public void manageSyntheticAccessIfNecessary(BlockScope currentScope, FieldBinding fieldBinding, TypeBinding lastReceiverType,         int index, FlowInfo flowInfo) {
482        if ((flowInfo.tagBits & FlowInfo.UNREACHABLE) != 0)        return;
483        // index == 0 denotes the first fieldBinding, index > 0 denotes one of the 'otherBindings', index < 0 denotes a write access (to last binding)
484        if (fieldBinding.constant() != Constant.NotAConstant)
485                return;
486 
487        // if field from parameterized type got found, use the original field at codegen time
488        FieldBinding originalField = fieldBinding.original();
489        if (originalField != fieldBinding) {
490                setCodegenBinding(index < 0 ? (this.otherBindings == null ? 0 : this.otherBindings.length) : index, originalField);
491        }
492 
493        if (fieldBinding.isPrivate()) { // private access
494            FieldBinding someCodegenBinding = getCodegenBinding(index < 0 ? (this.otherBindings == null ? 0 : this.otherBindings.length) : index);
495                if (someCodegenBinding.declaringClass != currentScope.enclosingSourceType()) {
496                    setSyntheticAccessor(fieldBinding, index,
497                            ((SourceTypeBinding) someCodegenBinding.declaringClass).addSyntheticMethod(someCodegenBinding, index >= 0 /*read-access?*/));
498                        currentScope.problemReporter().needToEmulateFieldAccess(someCodegenBinding, this, index >= 0 /*read-access?*/);
499                        return;
500                }
501        } else if (fieldBinding.isProtected()){
502            int depth = (index == 0 || (index < 0 && this.otherDepths == null))
503                            ? (this.bits & ASTNode.DepthMASK) >> ASTNode.DepthSHIFT
504                             : this.otherDepths[index < 0 ? this.otherDepths.length-1 : index-1];
505 
506                // implicit protected access
507                if (depth > 0 && (fieldBinding.declaringClass.getPackage() != currentScope.enclosingSourceType().getPackage())) {
508                    FieldBinding someCodegenBinding = getCodegenBinding(index < 0 ? (this.otherBindings == null ? 0 : this.otherBindings.length) : index);
509                    setSyntheticAccessor(fieldBinding, index,
510                            ((SourceTypeBinding) currentScope.enclosingSourceType().enclosingTypeAt(depth)).addSyntheticMethod(someCodegenBinding, index >= 0 /*read-access?*/));
511                        currentScope.problemReporter().needToEmulateFieldAccess(someCodegenBinding, this, index >= 0 /*read-access?*/);
512                        return;
513                }
514        }
515        // if the binding declaring class is not visible, need special action
516        // for runtime compatibility on 1.2 VMs : change the declaring class of the binding
517        // NOTE: from target 1.2 on, field's declaring class is touched if any different from receiver type
518        // and not from Object or implicit static field access.
519        if (fieldBinding.declaringClass != lastReceiverType
520                        && !lastReceiverType.isArrayType()
521                        && fieldBinding.declaringClass != null // array.length
522                        && fieldBinding.constant() == Constant.NotAConstant) {
523                CompilerOptions options = currentScope.compilerOptions();
524                if ((options.targetJDK >= ClassFileConstants.JDK1_2
525                                && (options.complianceLevel >= ClassFileConstants.JDK1_4 || !(index <= 1 &&  this.indexOfFirstFieldBinding == 1 && fieldBinding.isStatic()))
526                                && fieldBinding.declaringClass.id != TypeIds.T_JavaLangObject) // no change for Object fields
527                                || !fieldBinding.declaringClass.canBeSeenBy(currentScope)) {
528 
529                    setCodegenBinding(
530                            index < 0 ? (this.otherBindings == null ? 0 : this.otherBindings.length) : index,
531                            currentScope.enclosingSourceType().getUpdatedFieldBinding(
532                                    getCodegenBinding(index < 0 ? (this.otherBindings == null ? 0 : this.otherBindings.length) : index),
533                                    (ReferenceBinding)lastReceiverType.erasure()));
534                }
535        }

623public TypeBinding resolveType(BlockScope scope) {
624        // field and/or local are done before type lookups
625        // the only available value for the restrictiveFlag BEFORE
626        // the TC is Flag_Type Flag_LocalField and Flag_TypeLocalField
627        this.actualReceiverType = scope.enclosingReceiverType();
628        this.constant = Constant.NotAConstant;
629        if ((/*this.codegenBinding =*/ this.binding = scope.getBinding(this.tokens, this.bits & ASTNode.RestrictiveFlagMASK, this, true /*resolve*/)).isValidBinding()) {
630                switch (this.bits & ASTNode.RestrictiveFlagMASK) {
631                        case Binding.VARIABLE : //============only variable===========
632                        case Binding.TYPE | Binding.VARIABLE :
633                                if (this.binding instanceof LocalVariableBinding) {
634                                        if (!((LocalVariableBinding) this.binding).isFinal() && ((this.bits & ASTNode.DepthMASK) != 0))
635                                                scope.problemReporter().cannotReferToNonFinalOuterLocal(
636                                                        (LocalVariableBinding) this.binding,
637                                                        this);
638                                        this.bits &= ~ASTNode.RestrictiveFlagMASK; // clear bits
639                                        this.bits |= Binding.LOCAL;
640                                        return this.resolvedType = getOtherFieldBindings(scope);
641                                }
642                                if (this.binding instanceof FieldBinding) {
643                                        FieldBinding fieldBinding = (FieldBinding) this.binding;
644                                        MethodScope methodScope = scope.methodScope();
645                                        // check for forward references
646                                        if (this.indexOfFirstFieldBinding == 1
647                                                        && methodScope.enclosingSourceType() == fieldBinding.original().declaringClass
648                                                        && methodScope.lastVisibleFieldID >= 0
649                                                        && fieldBinding.id >= methodScope.lastVisibleFieldID
650                                                        && (!fieldBinding.isStatic() || methodScope.isStatic)) {
651                                                scope.problemReporter().forwardReference(this, 0, methodScope.enclosingSourceType());
652                                        }
653                                        if (!fieldBinding.isStatic()
654                                                        && this.indexOfFirstFieldBinding == 1
655                                                        && scope.compilerOptions().getSeverity(CompilerOptions.UnqualifiedFieldAccess) != ProblemSeverities.Ignore) {
656                                                scope.problemReporter().unqualifiedFieldAccess(this, fieldBinding);
657                                        }
658                                        this.bits &= ~ASTNode.RestrictiveFlagMASK; // clear bits
659                                        this.bits |= Binding.FIELD;
660 
661//                                                // check for deprecated receiver type
662//                                                // deprecation check for receiver type if not first token
663//                                                if (indexOfFirstFieldBinding > 1) {
664//                                                        if (isTypeUseDeprecated(this.actualReceiverType, scope))
665//                                                                scope.problemReporter().deprecatedType(this.actualReceiverType, this);
666//                                                }
667 
668                                        return this.resolvedType = getOtherFieldBindings(scope);
669                                }
670                                // thus it was a type
671                                this.bits &= ~ASTNode.RestrictiveFlagMASK; // clear bits
672                                this.bits |= Binding.TYPE;
673                        case Binding.TYPE : //=============only type ==============
674                            TypeBinding type = (TypeBinding) this.binding;
675//                                        if (isTypeUseDeprecated(type, scope))
676//                                                scope.problemReporter().deprecatedType(type, this);
677                                type = scope.environment().convertToRawType(type);
678                                return this.resolvedType = type;
679                }
680        }
681        //========error cases===============
682        return this.resolvedType = this.reportError(scope);
683}
