201public boolean isRecursive(ArrayList visited) {
202        if (this.binding == null
203                        || this.constructorCall == null
204                        || this.constructorCall.binding == null
205                        || this.constructorCall.isSuperAccess()
206                        || !this.constructorCall.binding.isValidBinding()) {
207                return false;
208        }

210        ConstructorDeclaration targetConstructor =
211                ((ConstructorDeclaration)this.scope.referenceType().declarationOf(this.constructorCall.binding.original()));
212        if (this == targetConstructor) return true; // direct case
213 
214        if (visited == null) { // lazy allocation
215                visited = new ArrayList(1);
216        } else {

218                if (index >= 0) return index == 0; // only blame if directly part of the cycle
219        }
