677         * Updates the outline page's context menu for the current input
678         */
679        private void updateContextMenuId() {
680                String computedContextMenuId = null;
681                // update outline view's context menu control and ID
682 
683                if (fEditor == null) {
684                        IWorkbenchPage page = getSite().getWorkbenchWindow().getActivePage();
685                        if (page != null) {
686                                fEditor = page.getActiveEditor();
687                        }
688                }
689 
690                computedContextMenuId = computeContextMenuID();
691 
692                if (computedContextMenuId == null) {
693                        computedContextMenuId = OUTLINE_CONTEXT_MENU_ID;
694                }
695 
696                /*
697                 * Update outline context menu id if updating to a new id or if
698                 * context menu is not already set up
699                 */
700                if (!computedContextMenuId.equals(fContextMenuId) || (fContextMenu == null)) {
701                        fContextMenuId = computedContextMenuId;
702 
703                        if (getControl() != null && !getControl().isDisposed()) {
704                                // dispose of previous context menu
705                                if (fContextMenu != null) {
706                                        fContextMenu.dispose();
707                                }
708                                if (fContextMenuManager != null) {
709                                        fContextMenuManager.removeMenuListener(fGroupAdder);
710                                        fContextMenuManager.removeAll();
711                                        fContextMenuManager.dispose();
712                                }
713 
714                                fContextMenuManager = new MenuManager(fContextMenuId, fContextMenuId);
715                                fContextMenuManager.setRemoveAllWhenShown(true);
716 
717                                fContextMenuManager.addMenuListener(fGroupAdder);
718 
719                                fContextMenu = fContextMenuManager.createContextMenu(getControl());
720                                getControl().setMenu(fContextMenu);
721 
722                                getSite().registerContextMenu(fContextMenuId, fContextMenuManager, this);
723 
724                                /*
725                                 * also register this menu for source page part and structured
726                                 * text outline view ids
727                                 */
728                                if (fEditor != null) {
729                                        String partId = fEditor.getSite().getId();
730                                        if (partId != null) {
731                                                getSite().registerContextMenu(partId + OUTLINE_CONTEXT_MENU_SUFFIX, fContextMenuManager, this);
732                                        }
733                                }
734                                getSite().registerContextMenu(OUTLINE_CONTEXT_MENU_ID, fContextMenuManager, this);
735                        }
