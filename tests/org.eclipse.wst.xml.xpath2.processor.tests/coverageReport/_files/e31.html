443                RefactoringStatus result= new RefactoringStatus();
444                
445                int referenceSearchTicks= fUpdateReferences || fUpdateSimilarElements ? 15 : 0;
446                int affectedCusTicks= fUpdateReferences || fUpdateSimilarElements ? 10 : 1;

462                                return result;
463                        pm.worked(1);
464

466                        pm.worked(1);
467

469                        pm.worked(1);
470                
471                        result.merge(checkImportedTypes());        
472                        pm.worked(1);
473

475                                result.merge(Checks.checkCompilationUnitNewName(fType.getJavaScriptUnit(), getNewElementName()));
476                        pm.worked(1);        
477                        
478                        if (isPrimaryType())
479                                result.merge(checkNewPathValidity());
480                        pm.worked(1);

492                        pm.worked(1);

495//                        pm.worked(1);        
496

498                        if (result.hasFatalError())

501                        result.merge(analyseEnclosedTypes());
502                        pm.worked(1);

546        /**
547         * Initializes the references to the type and the similarly named elements. This
548         * method creates both the fReferences and the fPreloadedElementToName
549         * fields.
550         * 
551         * May be called from the UI.
552         * @param monitor 
553         * @return initialization status 
554         * @throws JavaScriptModelException some fundamental error with the underlying model
555         * @throws OperationCanceledException if user canceled the task
556         * 
557         */
558        public RefactoringStatus initializeReferences(IProgressMonitor monitor) throws JavaScriptModelException, OperationCanceledException {
559 
560                Assert.isNotNull(fType);
561                Assert.isNotNull(getNewElementName());
562 
563                // Do not search again if the preconditions have not changed.
564                // Search depends on the type, the new name, the similarly named elements, and
565                // the strategy
566 
567                if (fReferences != null && (getNewElementName().equals(fCachedNewName)) && (fCachedRenameSimilarElements == getUpdateSimilarDeclarations()) && (fCachedRenamingStrategy == fRenamingStrategy))
568                        return fCachedRefactoringStatus;
569 
570                fCachedNewName= getNewElementName();
571                fCachedRenameSimilarElements= fUpdateSimilarElements;
572                fCachedRenamingStrategy= fRenamingStrategy;
573                fCachedRefactoringStatus= new RefactoringStatus();
574 
575                
576                try {
577                        SearchPattern pattern= SearchPattern.createPattern(fType, IJavaScriptSearchConstants.REFERENCES, SearchUtils.GENERICS_AGNOSTIC_MATCH_RULE);
578                        fReferences= RefactoringSearchEngine.search(
579                                        pattern,
580                                        RefactoringScopeFactory.create(fType),
581                                        new TypeOccurrenceCollector(fType),
582                                        monitor,
583                                        fCachedRefactoringStatus);
584                        fReferences= Checks.excludeCompilationUnits(fReferences, fCachedRefactoringStatus);
585 
586                        fPreloadedElementToName= new LinkedHashMap();
587                        fPreloadedElementToSelection= new HashMap();
588 
589                        final String unQualifiedTypeName= fType.getElementName();
590 
591                        monitor.beginTask("", fReferences.length); //$NON-NLS-1$
592 
593                        if (getUpdateSimilarDeclarations()) {
594 
595                                RenamingNameSuggestor sugg= new RenamingNameSuggestor(fRenamingStrategy);
596 
597                                for (int i= 0; i < fReferences.length; i++) {
598                                        final IJavaScriptUnit cu= fReferences[i].getCompilationUnit();
599                                        if (cu == null)
600                                                continue;
601 
602                                        final SearchMatch[] results= fReferences[i].getSearchResults();
603 
604                                        for (int j= 0; j < results.length; j++) {
605 
606                                                if (! (results[j] instanceof TypeReferenceMatch))
607                                                        continue;
608 
609                                                final TypeReferenceMatch match= (TypeReferenceMatch) results[j];
610                                                final List matches= new ArrayList();
611 
612                                                if (match.getLocalElement() != null)
613                                                        matches.add(match.getLocalElement());
614                                                else
615                                                        matches.add(match.getElement());
616 
617                                                final IJavaScriptElement[] others= match.getOtherElements();
618                                                if (others != null)
619                                                        matches.addAll(Arrays.asList(others));
620 
621                                                for (Iterator iter= matches.iterator(); iter.hasNext();) {
622                                                        final IJavaScriptElement element= (IJavaScriptElement) iter.next();
623 
624                                                        if (! (element instanceof IFunction) && ! (element instanceof IField) && ! (element instanceof ILocalVariable))
625                                                                continue;
626                                                        
627                                                        if (!isInDeclaredType(match.getOffset(), element))
628                                                                continue;
629 
630                                                        if (element instanceof IField) {
631                                                                final IField currentField= (IField) element;
632                                                                final String newFieldName= sugg.suggestNewFieldName(currentField.getJavaScriptProject(), currentField.getElementName(), Flags.isStatic(currentField.getFlags()),
633                                                                                unQualifiedTypeName, getNewElementName());
634 
635                                                                if (newFieldName != null)
636                                                                        fPreloadedElementToName.put(currentField, newFieldName);
637                                                        }
638 
639                                                        if (element instanceof IFunction) {
640                                                                final IFunction currentMethod= (IFunction) element;
641                                                                addMethodRename(unQualifiedTypeName, sugg, currentMethod);
642                                                        }
643 
644                                                        if (element instanceof ILocalVariable) {
645                                                                final ILocalVariable currentLocal= (ILocalVariable) element;
