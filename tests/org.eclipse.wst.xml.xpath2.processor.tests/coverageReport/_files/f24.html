338                FieldDeclaration field = fieldCount == 0 ? null : typeDeclaration.fields[fieldIndex];

340 
341                int methodIndex = 0, methodCount = (typeDeclaration.methods == null) ? 0 : typeDeclaration.methods.length;

343                int methodStart = method == null ? Integer.MAX_VALUE : method.declarationSourceStart;
344 
345                int typeIndex = 0, typeCount = (typeDeclaration.memberTypes == null) ? 0 : typeDeclaration.memberTypes.length;
346                TypeDeclaration type = typeCount == 0 ? null : typeDeclaration.memberTypes[typeIndex];
347                int typeStart = type == null ? Integer.MAX_VALUE : type.declarationSourceStart;
348

350                ASTNode[] members = new ASTNode[memberLength];
351                if (memberLength != 0) {
352                        int index = 0;
353                        int previousFieldStart = -1;
354                        do {
355                                if (fieldStart < methodStart && fieldStart < typeStart) {

357                                                // filter out enum constants

359                                                if (++fieldIndex < fieldCount) { // find next field if any

362                                                        fieldStart = Integer.MAX_VALUE;
363                                                }

365                                        }
366                                        // next member is a field
367                                        if (fieldStart == previousFieldStart){
368                                                ASTNode previousMember = members[index - 1];
369                                                if (previousMember instanceof MultiFieldDeclaration) {
370                                                        MultiFieldDeclaration multiField = (MultiFieldDeclaration) previousMember;
371                                                        int length = multiField.declarations.length;
372                                                        System.arraycopy(multiField.declarations, 0, multiField.declarations=new FieldDeclaration[length+1], 0, length);
373                                                        multiField.declarations[length] = field;
374                                                } else {
375                                                        members[index - 1] = new MultiFieldDeclaration(new FieldDeclaration[]{ (FieldDeclaration)previousMember, field});
376                                                }
377                                        } else {
378                                                members[index++] = field;
379                                        }
380                                        previousFieldStart = fieldStart;
381                                        if (++fieldIndex < fieldCount) { // find next field if any

390                                        }

409                        }
410                }

412        }

414        private boolean dumpBinaryExpression(
415                BinaryExpression binaryExpression,
416                int operator,
417                BlockScope scope) {
418 
419//                boolean addSpace= operator==TerminalTokens.TokenNamein || operator==TerminalTokens.TokenNameinstanceof;
420                final int numberOfParens = (binaryExpression.bits & ASTNode.ParenthesizedMASK) >> ASTNode.ParenthesizedSHIFT;
421 
422                if (numberOfParens > 0) {
423                        manageOpeningParenthesizedExpression(binaryExpression, numberOfParens);
424                }
425                BinaryExpressionFragmentBuilder builder = buildFragments(binaryExpression, scope);
426                final int fragmentsSize = builder.size();
427 
428                if ((builder.realFragmentsSize() > 1 || builder.size() > 4) && numberOfParens == 0) {
429                        this.scribe.printComment();
430                        Alignment binaryExpressionAlignment = this.scribe.createAlignment("binaryExpressionAlignment", this.preferences.alignment_for_binary_expression, Alignment.R_OUTERMOST, fragmentsSize, this.scribe.scanner.currentPosition); //$NON-NLS-1$
431                        this.scribe.enterAlignment(binaryExpressionAlignment);
432                        boolean ok = false;
433                        ASTNode[] fragments = builder.fragments();
434                        int[] operators = builder.operators();
435                        do {
436                                try {
437                                        for (int i = 0; i < fragmentsSize - 1; i++) {
438                                                ASTNode fragment = fragments[i];
439                                                fragment.traverse(this, scope);
440                                                this.scribe.printTrailingComment();
441                                                if (this.scribe.lastNumberOfNewLines == 1) {
442                                                        // a new line has been inserted by printTrailingComment()
443                                                        this.scribe.indentationLevel = binaryExpressionAlignment.breakIndentationLevel;
444                                                }
445                                                if (this.preferences.wrap_before_binary_operator) {
446                                                        this.scribe.alignFragment(binaryExpressionAlignment, i);
447                                                        this.scribe.printNextToken(operators[i], this.preferences.insert_space_before_binary_operator);
448                                                } else {
449                                                        this.scribe.printNextToken(operators[i], this.preferences.insert_space_before_binary_operator);
450                                                        this.scribe.alignFragment(binaryExpressionAlignment, i);
451                                                }
452                                                if (operators[i] == TerminalTokens.TokenNameMINUS && isNextToken(TerminalTokens.TokenNameMINUS)) {
453                                                        // the next character is a minus (unary operator)
454                                                        this.scribe.space();
455                                                }
456                                                if (this.preferences.insert_space_after_binary_operator) {
457                                                        this.scribe.space();
458                                                }
459                                        }
460                                        fragments[fragmentsSize - 1].traverse(this, scope);
461                                        this.scribe.printTrailingComment();
462                                        ok = true;
463                                } catch(AlignmentException e){
464                                        this.scribe.redoAlignment(e);
465                                }
466                        } while (!ok);
467                        this.scribe.exitAlignment(binaryExpressionAlignment, true);
468                } else {
469                        binaryExpression.left.traverse(this, scope);
470                        this.scribe.printNextToken(operator, this.preferences.insert_space_before_binary_operator);
471                        if (operator == TerminalTokens.TokenNameMINUS && isNextToken(TerminalTokens.TokenNameMINUS)) {
472                                // the next character is a minus (unary operator)
473                                this.scribe.space();
474                        }
475                        if (this.preferences.insert_space_after_binary_operator) {
476                                this.scribe.space();
477                        }
478                        binaryExpression.right.traverse(this, scope);
479                }
480                if (numberOfParens > 0) {
481                        manageClosingParenthesizedExpression(binaryExpression, numberOfParens);
482                }
483                return false;
484        }
