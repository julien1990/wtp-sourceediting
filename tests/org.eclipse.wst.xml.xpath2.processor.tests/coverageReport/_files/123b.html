403        private Type evaluateVariableType(AST ast, ImportRewrite imports, IBinding targetContext) throws CoreException {
404                if (fOriginalNode.getParent() instanceof FunctionInvocation) {
405                        FunctionInvocation parent= (FunctionInvocation) fOriginalNode.getParent();
406                        if (parent.getExpression() == fOriginalNode) {
407                                // _x_.foo() -> guess qualifier type by looking for a type with method 'foo'
408                                ITypeBinding[] bindings= ASTResolving.getQualifierGuess(fOriginalNode.getRoot(), parent.getName().getIdentifier(), parent.arguments(), targetContext);
409                                if (bindings.length > 0) {
410                                        for (int i= 0; i < bindings.length; i++) {
411                                                addLinkedPositionProposal(KEY_TYPE, bindings[i]);
412                                        }
413                                        return imports.addImport(bindings[0], ast);
414                                }
415                        }
416                }
417 
418                ITypeBinding binding= ASTResolving.guessBindingForReference(fOriginalNode);
419                if (binding != null) {
420                        if (binding.isWildcardType()) {
421                                binding= ASTResolving.normalizeWildcardType(binding, isVariableAssigned(), ast);
422                                if (binding == null) {
423                                        // only null binding applies
424                                        binding= ast.resolveWellKnownType("java.lang.Object"); //$NON-NLS-1$ 
425                                }
426                        }
427                        
428                        if (isVariableAssigned()) {
429                                ITypeBinding[] typeProposals= ASTResolving.getRelaxingTypes(ast, binding);
430                                for (int i= 0; i < typeProposals.length; i++) {
431                                        addLinkedPositionProposal(KEY_TYPE, typeProposals[i]);
432                                }
433                        }
434                        return imports.addImport(binding, ast);
435                }
436                // no binding, find type AST node instead -> ABC a= x-> use 'ABC' as is
437                Type type= ASTResolving.guessTypeForReference(ast, fOriginalNode);
438                if (type != null) {
439                        return type;
440                }
441                if (fVariableKind == CONST_FIELD) {
442                        return ast.newSimpleType(ast.newSimpleName("String")); //$NON-NLS-1$
443                }
444                return ast.newSimpleType(ast.newSimpleName("Object")); //$NON-NLS-1$
445        }

448                ASTNode parent= fOriginalNode.getParent();
449                return (parent instanceof Assignment) && (fOriginalNode == ((Assignment) parent).getLeftHandSide());
450        }

453        private int evaluateFieldModifiers(ASTNode newTypeDecl) {
454                if (fSenderBinding.isAnnotation()) {
455                        return 0;
456                }
457                if (fSenderBinding.isInterface()) {
458                        // for interface members copy the modifiers from an existing field
459                        FieldDeclaration[] fieldDecls= ((TypeDeclaration) newTypeDecl).getFields();
460                        if (fieldDecls.length > 0) {
461                                return fieldDecls[0].getModifiers();
462                        }
463                        return 0;
464                }
465                int modifiers= 0;
466 
467                if (fVariableKind == CONST_FIELD) {
468                        modifiers |= Modifier.FINAL | Modifier.STATIC;
469                } else {
470                        ASTNode parent= fOriginalNode.getParent();
471                        if (parent instanceof QualifiedName) {
472                                IBinding qualifierBinding= ((QualifiedName)parent).getQualifier().resolveBinding();
473                                if (qualifierBinding instanceof ITypeBinding) {
474                                        modifiers |= Modifier.STATIC;
475                                }
476                        } else if (ASTResolving.isInStaticContext(fOriginalNode)) {
477                                modifiers |= Modifier.STATIC;
478                        }
479                }
480                ASTNode node= ASTResolving.findParentType(fOriginalNode, true);
481                if (newTypeDecl.equals(node)) {
482                        modifiers |= Modifier.PRIVATE;
483                } else if (node instanceof AnonymousClassDeclaration) {
484                        modifiers |= Modifier.PROTECTED;
485                } else {
486                        modifiers |= Modifier.PUBLIC;
487                }
488 
489                return modifiers;
490        }
