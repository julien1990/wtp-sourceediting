517         * Returns a segmentation of the line of the given document appropriate for
518         * BIDI rendering. The default implementation returns only the string literals of a java code
519         * line as segments.
520         *
521         * @param document the document
522         * @param lineOffset the offset of the line
523         * @return the line's BIDI segmentation
524         * @throws BadLocationException in case lineOffset is not valid in document
525         */
526        protected static int[] getBidiLineSegments(IDocument document, int lineOffset) throws BadLocationException {
527 
528                if (document == null)
529                        return null;
530 
531                IRegion line= document.getLineInformationOfOffset(lineOffset);
532                ITypedRegion[] linePartitioning= TextUtilities.computePartitioning(document, IJavaScriptPartitions.JAVA_PARTITIONING, lineOffset, line.getLength(), false);
533 
534                List segmentation= new ArrayList();
535                for (int i= 0; i < linePartitioning.length; i++) {
536                        if (IJavaScriptPartitions.JAVA_STRING.equals(linePartitioning[i].getType()))
537                                segmentation.add(linePartitioning[i]);
538                }
539 
540 
541                if (segmentation.size() == 0)
542                        return null;
543 
544                int size= segmentation.size();
545                int[] segments= new int[size * 2 + 1];
546 
547                int j= 0;
548                for (int i= 0; i < size; i++) {
549                        ITypedRegion segment= (ITypedRegion) segmentation.get(i);
550 
551                        if (i == 0)
552                                segments[j++]= 0;
553 
554                        int offset= segment.getOffset() - lineOffset;
555                        if (offset > segments[j - 1])
556                                segments[j++]= offset;
557 
558                        if (offset + segment.getLength() >= line.getLength())
559                                break;
560 
561                        segments[j++]= offset + segment.getLength();
562                }
563 
564                if (j < segments.length) {
565                        int[] result= new int[j];
566                        System.arraycopy(segments, 0, result, 0, j);
567                        segments= result;
568                }
569 
570                return segments;
571        }
