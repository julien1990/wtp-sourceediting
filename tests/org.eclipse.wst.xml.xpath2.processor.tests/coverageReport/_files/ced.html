318/*
319 * Returns the unqualified name without parameters from the given type name.
320 */
321private char[] getSimpleName(char[] typeName) {
322        int lastDot = -1, lastGenericStart = -1;
323        int depthCount = 0;
324        int length = typeName.length;
325        lastDotLookup: for (int i = length -1; i >= 0; i--) {
326                switch (typeName[i]) {
327                        case '.':
328                                if (depthCount == 0) {
329                                        lastDot = i;
330                                        break lastDotLookup;
331                                }
332                                break;
333                        case '<':
334                                depthCount--;
335                                if (depthCount == 0) lastGenericStart = i;
336                                break;
337                        case '>':
338                                depthCount++;
339                                break;
340                }
341        }
342        if (lastGenericStart < 0) {
343                if (lastDot < 0) {
344                        return typeName;
345                }
346                return  CharOperation.subarray(typeName, lastDot + 1, length);
347        }
348        return  CharOperation.subarray(typeName, lastDot + 1, lastGenericStart);
349}
