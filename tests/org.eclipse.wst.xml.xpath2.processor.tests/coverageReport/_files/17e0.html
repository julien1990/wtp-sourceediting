582        private TextAttribute getAttributeFor(String namedStyle) {
583                TextAttribute ta = new TextAttribute(fDefaultForeground, fDefaultBackground, SWT.NORMAL);
584 
585                if (namedStyle != null && fOverlayStore != null) {
586                        // note: "namedStyle" *is* the preference key
587                        String prefString = getOverlayStore().getString(namedStyle);
588                        String[] stylePrefs = ColorHelper.unpackStylePreferences(prefString);
589                        if (stylePrefs != null) {
590                                RGB foreground = ColorHelper.toRGB(stylePrefs[0]);
591                                RGB background = ColorHelper.toRGB(stylePrefs[1]);
592 
593                                int fontModifier = SWT.NORMAL;
594 
595                                if (stylePrefs.length > 2) {
596                                        boolean on = Boolean.valueOf(stylePrefs[2]).booleanValue();
597                                        if (on)
598                                                fontModifier = fontModifier | SWT.BOLD;
599                                }
600                                if (stylePrefs.length > 3) {
601                                        boolean on = Boolean.valueOf(stylePrefs[3]).booleanValue();
602                                        if (on)
603                                                fontModifier = fontModifier | SWT.ITALIC;
604                                }
605                                if (stylePrefs.length > 4) {
606                                        boolean on = Boolean.valueOf(stylePrefs[4]).booleanValue();
607                                        if (on)
608                                                fontModifier = fontModifier | TextAttribute.STRIKETHROUGH;
609                                }
610                                if (stylePrefs.length > 5) {
611                                        boolean on = Boolean.valueOf(stylePrefs[5]).booleanValue();
612                                        if (on)
613                                                fontModifier = fontModifier | TextAttribute.UNDERLINE;
614                                }
615 
616                                ta = new TextAttribute((foreground != null) ? EditorUtility.getColor(foreground) : null, (background != null) ? EditorUtility.getColor(background) : null, fontModifier);
617                        }
618                }

626        private String getNamedStyleAtOffset(int offset) {
627                // ensure the offset is clean
628                if (offset >= fDocument.getLength())
629                        return getNamedStyleAtOffset(fDocument.getLength() - 1);
630                else if (offset < 0)
631                        return getNamedStyleAtOffset(0);
632                IStructuredDocumentRegion documentRegion = fDocument.getFirstStructuredDocumentRegion();
633                while (documentRegion != null && !documentRegion.containsOffset(offset)) {
634                        documentRegion = documentRegion.getNext();
635                }
636                if (documentRegion != null) {
637                        // find the ITextRegion's Context at this offset
638                        ITextRegion interest = documentRegion.getRegionAtCharacterOffset(offset);
639                        if (interest == null)
640                                return null;
641                        if (offset > documentRegion.getTextEndOffset(interest))
642                                return null;
643                        String regionContext = interest.getType();
644                        if (regionContext == null)
645                                return null;
646                        // find the named style (internal/selectable name) for that
647                        // context
648                        String namedStyle = (String) fContextToStyleMap.get(regionContext);
649                        if (namedStyle != null) {
650                                return namedStyle;
651                        }
652                }
653                return null;
654        }
