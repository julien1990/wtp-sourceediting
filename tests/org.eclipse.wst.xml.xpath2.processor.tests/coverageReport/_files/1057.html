597        /**
598         * 
599         */
600        private static boolean mustSaveDirtyEditor(IEditorPart ep, IEditorInput input, boolean saveUnknownEditors) {
601                /*
602                 * Goal: save all editors that could interfere with refactoring operations.
603                 * 
604                 * If <code>saveUnknownEditors</code> is <code>false</code>, save all editors
605                 * for compilation units that are not working copies.
606                 * 
607                 * If <code>saveUnknownEditors</code> is <code>true</code>, save all editors
608                 * whose implementation is probably not based on file buffers.
609                 */
610                IResource resource= (IResource) input.getAdapter(IResource.class);
611                if (resource == null)
612                        return saveUnknownEditors;
613 
614                IJavaScriptElement javaElement= JavaScriptCore.create(resource);
615                if (javaElement instanceof IJavaScriptUnit) {
616                        IJavaScriptUnit cu= (IJavaScriptUnit) javaElement;
617                        if (!cu.isWorkingCopy()) {
618                                return true;
619                        }
620                }
621                
622                if (! (ep instanceof ITextEditor))
623                        return saveUnknownEditors;
624                
625                ITextEditor textEditor= (ITextEditor) ep;
626                IDocumentProvider documentProvider= textEditor.getDocumentProvider();
627                if (! (documentProvider instanceof TextFileDocumentProvider))
628                        return saveUnknownEditors;
629                
630                return false;
631        }
