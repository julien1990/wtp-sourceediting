557public int resolveLevel(Binding binding) {
558        if (binding == null) return INACCURATE_MATCH;
559        if (!(binding instanceof MethodBinding)) return IMPOSSIBLE_MATCH;
560 
561        MethodBinding method = (MethodBinding) binding;
562        boolean skipVerif = this.pattern.findDeclarations && this.mayBeGeneric;
563        int methodLevel = matchMethod(method, skipVerif);
564        if (methodLevel == IMPOSSIBLE_MATCH) {
565                if (method != method.original()) methodLevel = matchMethod(method.original(), skipVerif);
566                if (methodLevel == IMPOSSIBLE_MATCH) {
567                        return IMPOSSIBLE_MATCH;
568                } else {
569                        method = method.original();
570                }
571        }
572 
573        // declaring type
574        char[] qualifiedPattern = qualifiedPattern(this.pattern.declaringSimpleName, this.pattern.declaringQualification);
575        if (qualifiedPattern == null) return methodLevel; // since any declaring class will do
576 
577        boolean subType = !method.isStatic() && !method.isPrivate();
578        if (subType && this.pattern.declaringQualification != null && method.declaringClass != null && method.declaringClass.fPackage != null) {
579                subType = CharOperation.compareWith(this.pattern.declaringQualification, method.declaringClass.fPackage.shortReadableName()) == 0;
580        }
581        int declaringLevel = subType
582                ? resolveLevelAsSubtype(qualifiedPattern, method.declaringClass, null)
583                : resolveLevelForType(qualifiedPattern, method.declaringClass);
584        return methodLevel > declaringLevel ? declaringLevel : methodLevel; // return the weaker match
585}
586protected int resolveLevel(MessageSend messageSend) {
587        MethodBinding method = messageSend.binding;
588        if (method == null) {
589                return INACCURATE_MATCH;
590        }
591        if (messageSend.resolvedType == null) {
592                // Closest match may have different argument numbers when ProblemReason is NotFound
593                // see MessageSend#resolveType(BlockScope)
594                // see bug https://bugs.eclipse.org/bugs/show_bug.cgi?id=97322
595                int argLength = messageSend.arguments == null ? 0 : messageSend.arguments.length;
596                if (pattern.parameterSimpleNames == null || argLength == pattern.parameterSimpleNames.length) {
597                        return INACCURATE_MATCH;
598                }
599                return IMPOSSIBLE_MATCH;
600        }
601 
602        int methodLevel = matchMethod(method, false);
603        if (methodLevel == IMPOSSIBLE_MATCH) {
604                if (method != method.original()) methodLevel = matchMethod(method.original(), false);
605                if (methodLevel == IMPOSSIBLE_MATCH) return IMPOSSIBLE_MATCH;
606                method = method.original();
607        }
608 
609        // receiver type
610        char[] qualifiedPattern = qualifiedPattern(this.pattern.declaringSimpleName, this.pattern.declaringQualification);
611        if (qualifiedPattern == null) return methodLevel; // since any declaring class will do
612 
613        int declaringLevel;
614        if (isVirtualInvoke(method, messageSend) && (messageSend.actualReceiverType instanceof ReferenceBinding)) {
615                ReferenceBinding methodReceiverType = (ReferenceBinding) messageSend.actualReceiverType;
616                declaringLevel = resolveLevelAsSubtype(qualifiedPattern, methodReceiverType, method.parameters);
617                if (declaringLevel == IMPOSSIBLE_MATCH) {
618                        if (method.declaringClass == null || this.allSuperDeclaringTypeNames == null) {
619                                declaringLevel = INACCURATE_MATCH;
620                        } else {
621                                char[][] compoundName = methodReceiverType.compoundName;
622                                for (int i = 0, max = this.allSuperDeclaringTypeNames.length; i < max; i++) {
623                                        if (CharOperation.equals(this.allSuperDeclaringTypeNames[i], compoundName)) {
624                                                return methodLevel // since this is an ACCURATE_MATCH so return the possibly weaker match
625                                                        | SUPER_INVOCATION_FLAVOR; // this is an overridden method => add flavor to returned level
626                                        }
627                                }
628                                /* Do not return interfaces potential matches
629                                 * @see "https://bugs.eclipse.org/bugs/show_bug.cgi?id=157814#c8"
630                                if (methodReceiverType.isInterface()) {
631                                        // all methods interface with same name and parameters are potential matches
632                                        // see bug https://bugs.eclipse.org/bugs/show_bug.cgi?id=156491
633                                        return INACCURATE_MATCH | POLYMORPHIC_FLAVOR;
634                                }
635                                */
636                        }
637                }
638                if ((declaringLevel & FLAVORS_MASK) != 0) {
639                        // level got some flavors => return it
640                        return declaringLevel;
641                }
642        } else {
643                declaringLevel = resolveLevelForType(qualifiedPattern, method.declaringClass);
644        }
645        return methodLevel > declaringLevel ? declaringLevel : methodLevel; // return the weaker match
646}
647/**
648 * Returns whether the given reference type binding matches or is a subtype of a type
649 * that matches the given qualified pattern.
650 * Returns ACCURATE_MATCH if it does.
651 * Returns INACCURATE_MATCH if resolve fails
652 * Returns IMPOSSIBLE_MATCH if it doesn't.
653 */
654protected int resolveLevelAsSubtype(char[] qualifiedPattern, ReferenceBinding type, TypeBinding[] argumentTypes) {
655        if (type == null) return INACCURATE_MATCH;
656 
657        int level = resolveLevelForType(qualifiedPattern, type);
658        if (level != IMPOSSIBLE_MATCH) {
659                if (!type.isAbstract() && !type.isInterface()) { // if concrete class, then method is overridden
660                        level |= OVERRIDDEN_METHOD_FLAVOR;
661                }
662                return level;
663        }
664 
665        // matches superclass
666        if (!type.isInterface() && !CharOperation.equals(type.compoundName, TypeConstants.JAVA_LANG_OBJECT)) {
667                level = resolveLevelAsSubtype(qualifiedPattern, type.superclass(), argumentTypes);
668                if (level != IMPOSSIBLE_MATCH) {
669                        if (argumentTypes != null) {
670                                // need to verify if method may be overridden
671                                MethodBinding[] methods = type.getMethods(this.pattern.selector);
672                                for (int i=0, length=methods.length; i<length; i++) {
673                                        MethodBinding method = methods[i];
674                                        TypeBinding[] parameters = method.parameters;
675                                        if (argumentTypes.length == parameters.length) {
676                                                boolean found = true;
677                                                for (int j=0,l=parameters.length; j<l; j++) {
678                                                        if (parameters[j].erasure() != argumentTypes[j].erasure()) {
679                                                                found = false;
680                                                                break;
681                                                        }
682                                                }
683                                                if (found) { // one method match in hierarchy
684                                                        if ((level & OVERRIDDEN_METHOD_FLAVOR) != 0) {
685                                                                // this method is already overridden on a super class, current match is impossible
686                                                                return IMPOSSIBLE_MATCH;
687                                                        }
688                                                        if (!method.isAbstract() && !type.isInterface()) {
689                                                                // store the fact that the method is overridden
690                                                                level |= OVERRIDDEN_METHOD_FLAVOR;
691                                                        }
692                                                }
693                                        }
694                                }
695                        }
696                        return level | SUB_INVOCATION_FLAVOR; // add flavor to returned level
697                }
698        }
699 
700        // matches interfaces
701        ReferenceBinding[] interfaces = type.superInterfaces();
702        if (interfaces == null) return INACCURATE_MATCH;
703        for (int i = 0; i < interfaces.length; i++) {
704                level = resolveLevelAsSubtype(qualifiedPattern, interfaces[i], null);
705                if (level != IMPOSSIBLE_MATCH) {
706                        if (!type.isAbstract() && !type.isInterface()) { // if concrete class, then method is overridden
707                                level |= OVERRIDDEN_METHOD_FLAVOR;
708                        }
709                        return level | SUB_INVOCATION_FLAVOR; // add flavor to returned level
710                }
711        }
712        return IMPOSSIBLE_MATCH;
713}

718public int match(InferredMethod inferredMethod, MatchingNodeSet nodeSet) {
719        if (!this.pattern.findDeclarations) return IMPOSSIBLE_MATCH;
720 
721        // Verify method name
722        if (!matchesName(this.pattern.selector, inferredMethod.name)) return IMPOSSIBLE_MATCH;
723 
724        // Verify parameters types
725        boolean resolve = ((InternalSearchPattern)this.pattern).mustResolve;
726        if (this.pattern.parameterSimpleNames != null) {
727                int length = this.pattern.parameterSimpleNames.length;
728                ASTNode[] args = ((AbstractMethodDeclaration)inferredMethod.getFunctionDeclaration()).arguments;
729                int argsLength = args == null ? 0 : args.length;
730                if (length != argsLength) return IMPOSSIBLE_MATCH;
731                for (int i = 0; i < argsLength; i++) {
732                        if (args != null && !matchesTypeReference(this.pattern.parameterSimpleNames[i], ((Argument) args[i]).type)) {
733                                // Do not return as impossible when source level is at least 1.5
734                                if (this.mayBeGeneric) {
735                                        if (!((InternalSearchPattern)this.pattern).mustResolve) {
