678        private RefactoringStatus findCommonParent(ITypeBinding typeBinding) throws JavaScriptModelException {
679 
680                RefactoringStatus status= new RefactoringStatus();
681 
682                ITypeBinding highest= fIntermediaryFirstParameterType;
683                ITypeBinding current= typeBinding;
684 
685                if (current.equals(highest) || Bindings.isSuperType(highest, current))
686                        // current is the same as highest or highest is already a supertype of current in the same hierarchy => no change
687                        return status;
688 
689                // find lowest common supertype with the method
690                // search in bottom-up order 
691                ITypeBinding[] currentAndSupers= getTypeAndAllSuperTypes(current);
692                ITypeBinding[] highestAndSupers= getTypeAndAllSuperTypes(highest);
693 
694                ITypeBinding foundBinding= null;
695                for (int i1= 0; i1 < currentAndSupers.length; i1++) {
696                        for (int i2= 0; i2 < highestAndSupers.length; i2++) {
697                                if (highestAndSupers[i2].isEqualTo(currentAndSupers[i1])
698                                                && (Bindings.findMethodInHierarchy(highestAndSupers[i2], fTargetMethodBinding.getName(), fTargetMethodBinding.getParameterTypes()) != null)) {
699                                        foundBinding= highestAndSupers[i2];
700                                        break;
701                                }
702                        }
703                        if (foundBinding != null)
704                                break;
705                }
706 
707                if (foundBinding != null) {
708                        fIntermediaryFirstParameterType= foundBinding;
709                } else {

991         *

993         *

1019                        if (ASTNodes.isParent(originalInvocation, methodDeclaration.getParent()))
1020                                currentTypeBinding= methodBinding.getDeclaringClass();

1023                } else {

1048                }
