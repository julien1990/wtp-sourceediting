252        private boolean keepWalkingUp(ASTNode node) {
253                if (node == null)
254                        return false;
255                if (isAnyInstanceOf(STOPPERS, node))
256                        return false;
257                if (fLabel != null && LabeledStatement.class.isInstance(node)){
258                        LabeledStatement ls= (LabeledStatement)node;
259                        return ! areEqualLabels(ls.getLabel(), fLabel);
260                }
261                if (fLabel == null &&  fIsBreak && isAnyInstanceOf(BREAKTARGETS, node))
262                        return false;
263                if (fLabel == null && !fIsBreak && isAnyInstanceOf(CONTINUETARGETS, node))
264                        return false;
265                return true;
266        }
