225                if (packName.length() > 0) {
226                        IStatus val= validatePackageName(packName);
227                        if (val.getSeverity() == IStatus.ERROR) {
228                                status.setError(Messages.format(NewWizardMessages.NewPackageWizardPage_error_InvalidPackageName, val.getMessage()));

230                        } else if (val.getSeverity() == IStatus.WARNING) {
231                                status.setWarning(Messages.format(NewWizardMessages.NewPackageWizardPage_warning_DiscouragedPackageName, val.getMessage())); 
232                        }
233                } else {

235                        return status;

239                if (root != null && root.getJavaScriptProject().exists()) {

241                        try {
242                                if (pack.exists()) {
243                                        if (pack.containsJavaResources() || !pack.hasSubpackages()) {
244                                                status.setError(NewWizardMessages.NewPackageWizardPage_error_PackageExists); 
245                                        } else {
246                                                status.setError(NewWizardMessages.NewPackageWizardPage_error_PackageNotShown);  
247                                        }
248                                } else {
249                                        URI location= pack.getResource().getLocationURI();
250                                        if (location != null) {

252                                                if (store.fetchInfo().exists()) {
253                                                        status.setError(NewWizardMessages.NewPackageWizardPage_error_PackageExistsDifferentCase); 
254                                                }
255                                        }
256                                }
257                        } catch (CoreException e) {
258                                JavaScriptPlugin.log(e);
259                        }
