319

324                this.units[i] = null; // release unnecessary reference to the parsed unit
325        }

327        this.lastCompletedUnitIndex = this.lastUnitIndex;

643public AnnotationBinding createAnnotation(ReferenceBinding annotationType, ElementValuePair[] pairs) {
644        // cached info is array of already created annotation binding for this type
645        AnnotationBinding[] cachedInfo = (AnnotationBinding[])this.uniqueAnnotationBindings.get(annotationType);
646        boolean needToGrow = false;

652                nextCachedType :

654                        for (int max = cachedInfo.length; index < max; index++){

658                                int length2 = pairs.length;
659                                if (length2 != elementValuePairs.length) continue nextCachedType;
660                                loop: for (int i = 0; i < length2; i++) {

665                                                if (pair.binding == pair2.binding) {

668                                                                        continue loop;
669                                                                }
670                                                                continue nextCachedType;
671                                                        } else {
672                                                                if (pair2.value == null
673                                                                                || !pair2.value.equals(pair.value)) {
674                                                                        continue nextCachedType;
675                                                                }
676                                                        }
677                                                        continue loop;
678                                                }
679                                        }
680                                        // no match found for pair so we create a new annotation binding
681                                        continue nextCachedType;
682                                }
683                                // cached type match, reuse current
684                                return cachedType;
685                }

690        }

696        }
697        // add new binding

699        cachedInfo[index] = annotationBinding;
700        return annotationBinding;
