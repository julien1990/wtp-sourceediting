338                 * If invoking the StringMatcher with string literals in Java, don't
339                 * forget escape characters are represented by "\\".
340                 * 
341                 * @param pattern
342                 *            the pattern to match text against
343                 */
344                StringMatcher(String pattern) {
345                        if (pattern == null)
346                                throw new IllegalArgumentException();
347                        this.pattern = pattern;
348                        patternLength = pattern.length();
349                        parseWildCards();
350                }

352                /**
353                 * @param text
354                 *            a simple regular expression that may only contain '?'(s)
355                 * @param start
356                 *            the starting index in the text for search, inclusive
357                 * @param end
358                 *            the stopping point of search, exclusive
359                 * @param p
360                 *            a simple regular expression that may contain '?'
361                 * @return the starting index in the text of the pattern , or -1 if
362                 *         not found
363                 */
364                private int findPosition(String text, int start, int end, String p) {
365                        boolean hasWildCard = p.indexOf(SINGLE_WILD_CARD) >= 0;
366                        int plen = p.length();
367                        for (int i = start, max = end - plen; i <= max; ++i) {
368                                if (hasWildCard) {
369                                        if (regExpRegionMatches(text, i, p, 0, plen))
370                                                return i;
371                                }
372                                else {
373                                        if (text.regionMatches(true, i, p, 0, plen))
374                                                return i;
375                                }
376                        }
377                        return -1;
378                }

380                /**
381                 * Given the starting (inclusive) and the ending (exclusive) positions
382                 * in the <code>text</code>, determine if the given substring matches
383                 * with aPattern
384                 * 
385                 * @return true if the specified portion of the text matches the
386                 *         pattern
387                 * @param text
388                 *            a String object that contains the substring to match
389                 */
390                public boolean match(String text) {
391                        if (text == null)
392                                return false;
393                        final int end = text.length();
394                        final int segmentCount = segments.length;
395                        if (segmentCount == 0 && (hasLeadingStar || hasTrailingStar)) // pattern
396                                // contains
397                                // only
398                                // '*'(s)
399                                return true;
400                        if (end == 0)
401                                return patternLength == 0;
402                        if (patternLength == 0)
403                                return false;
404                        int currentTextPosition = 0;
405                        if ((end - bound) < 0)
406                                return false;
407                        int segmentIndex = 0;
408                        String current = segments[segmentIndex];
409 
410                        /* process first segment */
411                        if (!hasLeadingStar) {
412                                int currentLength = current.length();
413                                if (!regExpRegionMatches(text, 0, current, 0, currentLength))
414                                        return false;
415                                segmentIndex++;
416                                currentTextPosition = currentTextPosition + currentLength;
417                        }
418                        if ((segmentCount == 1) && (!hasLeadingStar) && (!hasTrailingStar)) {
419                                // only one segment to match, no wild cards specified
420                                return currentTextPosition == end;
421                        }
422                        /* process middle segments */
423                        while (segmentIndex < segmentCount) {
424                                current = segments[segmentIndex];
425                                int currentMatch = findPosition(text, currentTextPosition, end, current);
426                                if (currentMatch < 0)
427                                        return false;
428                                currentTextPosition = currentMatch + current.length();
429                                segmentIndex++;
430                        }
431 
432                        /* process final segment */
433                        if (!hasTrailingStar && currentTextPosition != end) {
434                                int currentLength = current.length();
435                                return regExpRegionMatches(text, end - currentLength, current, 0, currentLength);
436                        }
437                        return segmentIndex == segmentCount;
438                }

440                /**
441                 * Parses the pattern into segments separated by wildcard '*'
442                 * characters.
443                 */
444                private void parseWildCards() {
445                        if (pattern.startsWith("*"))//$NON-NLS-1$
446                                hasLeadingStar = true;
447                        if (pattern.endsWith("*")) {//$NON-NLS-1$
448                                /* make sure it's not an escaped wildcard */
449                                if (patternLength > 1 && pattern.charAt(patternLength - 2) != '\\') {
450                                        hasTrailingStar = true;
451                                }
452                        }
453 
454                        ArrayList temp = new ArrayList();
455 
456                        int pos = 0;
457                        StringBuffer buf = new StringBuffer();
458                        while (pos < patternLength) {
459                                char c = pattern.charAt(pos++);
460                                switch (c) {
461                                        case '\\' :
462                                                if (pos >= patternLength) {
463                                                        buf.append(c);
464                                                }
465                                                else {
466                                                        char next = pattern.charAt(pos++);
467                                                        /* if it's an escape sequence */
468                                                        if (next == '*' || next == '?' || next == '\\') {
469                                                                buf.append(next);
470                                                        }
471                                                        else {
472                                                                /*
473                                                                 * not an escape sequence, just insert
474                                                                 * literally
475                                                                 */
476                                                                buf.append(c);
477                                                                buf.append(next);
478                                                        }
479                                                }
480                                                break;
481                                        case '*' :
482                                                if (buf.length() > 0) {
483                                                        /* new segment */
484                                                        temp.add(buf.toString());
485                                                        bound += buf.length();
486                                                        buf.setLength(0);
487                                                }
488                                                break;
489                                        case '?' :
490                                                /*
491                                                 * append special character representing single match
492                                                 * wildcard
493                                                 */
494                                                buf.append(SINGLE_WILD_CARD);
495                                                break;
496                                        default :
497                                                buf.append(c);
498                                }
499                        }
500 
501                        /* add last buffer to segment list */
502                        if (buf.length() > 0) {
503                                temp.add(buf.toString());
504                                bound += buf.length();
505                        }
506                        segments = (String[]) temp.toArray(new String[temp.size()]);
507                }

509                /**
510                 * 
511                 * @return boolean
512                 * @param text
513                 *            a String to match
514                 * @param tStart
515                 *            the starting index of match, inclusive
516                 * @param p
517                 *            a simple regular expression that may contain '?'
518                 * @param pStart
519                 *            The start position in the pattern
520                 * @param plen
521                 *            The length of the pattern
522                 */
523                private boolean regExpRegionMatches(String text, int tStart, String p, int pStart, int plen) {
524                        while (plen-- > 0) {
525                                char tchar = text.charAt(tStart++);
526                                char pchar = p.charAt(pStart++);
527 
528                                // process wild cards, skipping single wild cards
529                                if (pchar == SINGLE_WILD_CARD)
530                                        continue;
531                                if (pchar == tchar)
532                                        continue;
533                                if (Character.toUpperCase(tchar) == Character.toUpperCase(pchar))
534                                        continue;
535                                // comparing after converting to upper case doesn't handle all
536                                // cases;
537                                // also compare after converting to lower case
538                                if (Character.toLowerCase(tchar) == Character.toLowerCase(pchar))
539                                        continue;
540                                return false;
541                        }
542                        return true;
543                }
