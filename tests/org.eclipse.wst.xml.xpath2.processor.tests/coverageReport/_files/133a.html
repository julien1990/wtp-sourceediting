552        public static CPListElement createFromExisting(Object parent, IIncludePathEntry curr, IJavaScriptProject project) {
553                IPath path= curr.getPath();
554                IWorkspaceRoot root= ResourcesPlugin.getWorkspace().getRoot();
555 
556                // get the resource
557                IResource res= null;
558                boolean isMissing= false;
559                IPath linkTarget= null;
560                
561                switch (curr.getEntryKind()) {
562                        case IIncludePathEntry.CPE_CONTAINER:
563                                try {
564                                        isMissing= project != null && (JavaScriptCore.getJsGlobalScopeContainer(path, project) == null);
565                                } catch (JavaScriptModelException e) {
566                                        isMissing= true;
567                                }
568                                break;
569                        case IIncludePathEntry.CPE_VARIABLE:
570                                IPath resolvedPath= JavaScriptCore.getResolvedVariablePath(path);
571                                isMissing=  root.findMember(resolvedPath) == null && !resolvedPath.toFile().isFile();
572                                break;
573                        case IIncludePathEntry.CPE_LIBRARY:
574                                res= root.findMember(path);
575                                if (res == null) {
576                                        if (!ArchiveFileFilter.isArchivePath(path)) {
577                                                if (root.getWorkspace().validatePath(path.toString(), IResource.FOLDER).isOK()
578                                                                && root.getProject(path.segment(0)).exists()) {
579                                                        res= root.getFolder(path);
580                                                }
581                                        }
582                                        isMissing= !path.toFile().isFile(); // look for external JARs
583                                } else if (res.isLinked()) {
584                                        linkTarget= res.getLocation();
585                                }
586                                break;
587                        case IIncludePathEntry.CPE_SOURCE:
588                                path= path.removeTrailingSeparator();
589                                res= root.findMember(path);
590                                if (res == null) {
591                                        if (root.getWorkspace().validatePath(path.toString(), IResource.FOLDER).isOK()) {
592                                                res= root.getFolder(path);
593                                        }
594                                        isMissing= true;
595                                } else if (res.isLinked()) {
596                                        linkTarget= res.getLocation();
597                                }
598                                break;
599                        case IIncludePathEntry.CPE_PROJECT:
600                                res= root.findMember(path);
601                                isMissing= (res == null);
602                                break;
603                }
604                CPListElement elem= new CPListElement(parent, project, curr.getEntryKind(), path, res, linkTarget);
605                elem.setExported(curr.isExported());
606                elem.setAttribute(EXCLUSION, curr.getExclusionPatterns());
607                elem.setAttribute(INCLUSION, curr.getInclusionPatterns());
608                elem.setAttribute(ACCESSRULES, curr.getAccessRules());
609                elem.setAttribute(COMBINE_ACCESSRULES, new Boolean(curr.combineAccessRules())); 
610                
611                IIncludePathAttribute[] extraAttributes= curr.getExtraAttributes();
612                for (int i= 0; i < extraAttributes.length; i++) {
613                        IIncludePathAttribute attrib= extraAttributes[i];
614                        CPListElementAttribute attribElem= elem.findAttributeElement(attrib.getName());
615                        if (attribElem == null) {
616                                elem.createAttributeElement(attrib.getName(), attrib.getValue(), false);
617                        } else {
618                                attribElem.setValue(attrib.getValue());
619                        }
620                }
621                
622                if (project != null && project.exists()) {
623                        elem.setIsMissing(isMissing);
624                }
625                return elem;
626        }
