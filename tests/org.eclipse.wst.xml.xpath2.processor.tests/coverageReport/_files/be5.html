856/*
857 * Given the current configuration, answers which category the problem
858 * falls into:
859 *                ProblemSeverities.Error | ProblemSeverities.Warning | ProblemSeverities.Ignore
860 * when different from Ignore, severity can be coupled with ProblemSeverities.Optional
861 * to indicate that this problem is configurable through options
862 */
863public int computeSeverity(int problemID){
864         
865 
866        /*
867         * If semantic validation is not enabled and this is anything but a
868         * syntax or task problem, ignore.
869         */
870        if (!this.options.enableSemanticValidation && (problemID & IProblem.Syntax) == 0 && problemID != IProblem.Task) {
871                return ProblemSeverities.Ignore;
872        }
873        
874        switch (problemID) {
875                case IProblem.Task :
876                        return ProblemSeverities.Warning;
877                 case IProblem.TypeCollidesWithPackage :
878                        return ProblemSeverities.Warning;
879 
880                /*
881                 * JS Type mismatch is set to default as Warning
882                 */
883                 case IProblem.NotAFunction:
884                 case IProblem.TypeMismatch:
885                         return ProblemSeverities.Warning;
886 
887//                 case IProblem.UndefinedName:
888//                 case IProblem.UndefinedFunction:
889//                 case IProblem.UndefinedMethod:
890//                 case IProblem.UndefinedField:
891//                 case IProblem.UndefinedConstructor:
892//                         break;
893                /*
894                 * Javadoc tags resolved references errors
895                 */
896                case IProblem.JavadocInvalidParamName:
897                case IProblem.JavadocDuplicateParamName:
898                case IProblem.JavadocMissingParamName:
899                case IProblem.JavadocMissingIdentifier:
900                case IProblem.JavadocInvalidMemberTypeQualification:
901                case IProblem.JavadocInvalidThrowsClassName:
902                case IProblem.JavadocDuplicateThrowsClassName:
903                case IProblem.JavadocMissingThrowsClassName:
904                case IProblem.JavadocMissingSeeReference:
905                case IProblem.JavadocInvalidValueReference:
906                case IProblem.JavadocUndefinedField:
907                case IProblem.JavadocAmbiguousField:
908                case IProblem.JavadocUndefinedConstructor:
909                case IProblem.JavadocAmbiguousConstructor:
910                case IProblem.JavadocUndefinedMethod:
911                case IProblem.JavadocAmbiguousMethod:
912                case IProblem.JavadocAmbiguousMethodReference:
913                case IProblem.JavadocParameterMismatch:
914                case IProblem.JavadocUndefinedType:
915                case IProblem.JavadocAmbiguousType:
916                case IProblem.JavadocInternalTypeNameProvided:
917                case IProblem.JavadocNoMessageSendOnArrayType:
918                case IProblem.JavadocNoMessageSendOnBaseType:
919                case IProblem.JavadocInheritedMethodHidesEnclosingName:
920                case IProblem.JavadocInheritedFieldHidesEnclosingName:
921                case IProblem.JavadocInheritedNameHidesEnclosingTypeName:
922                case IProblem.JavadocNonStaticTypeFromStaticInvocation:
923                case IProblem.JavadocEmptyReturnTag:
924                        if (!this.options.reportInvalidJavadocTags) {
925                                return ProblemSeverities.Ignore;
926                        }
927                        break;
928                /*
929                 * Javadoc invalid tags due to deprecated references
930                 */
931                case IProblem.JavadocUsingDeprecatedField:
932                case IProblem.JavadocUsingDeprecatedConstructor:
933                case IProblem.JavadocUsingDeprecatedMethod:
934                case IProblem.JavadocUsingDeprecatedType:
935                        if (!(this.options.reportInvalidJavadocTags && this.options.reportInvalidJavadocTagsDeprecatedRef)) {
936                                return ProblemSeverities.Ignore;
937                        }
938                        break;
939                /*
940                 * Javadoc invalid tags due to non-visible references
941                 */
942                case IProblem.JavadocNotVisibleField:
943                case IProblem.JavadocNotVisibleConstructor:
944                case IProblem.JavadocNotVisibleMethod:
945                case IProblem.JavadocNotVisibleType:
946                case IProblem.JavadocHiddenReference:
947                        if (!(this.options.reportInvalidJavadocTags && this.options.reportInvalidJavadocTagsNotVisibleRef)) {
948                                return ProblemSeverities.Ignore;
949                        }
950                        break;
951        }
952        long irritant = getIrritant(problemID);
953        if (irritant != 0) {
954                if ((problemID & IProblem.Javadoc) != 0 && !this.options.docCommentSupport)
955                        return ProblemSeverities.Ignore;
956                return this.options.getSeverity(irritant);
957        }
