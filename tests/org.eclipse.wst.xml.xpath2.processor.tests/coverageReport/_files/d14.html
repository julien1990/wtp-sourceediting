339                int idx = 0;
340                for (; idx<length; idx++) {
341                        if (localDeclarations[idx] == null) break;
342                        if (reference.sourceStart == localDeclarations[idx].declarationSourceStart) {
343                                localElement = locator.createHandle(localDeclarations[idx], element);
344                                break;
345                        }
346                        if (idx>0 && localDeclarations[idx].sourceStart > reference.sourceStart) {
347                                localElement = locator.createHandle(localDeclarations[idx-1], element);
348                                break;
349                        }
350                }
351                if (localElement == null && idx > 0) {
352                        if (reference.sourceEnd < localDeclarations[idx-1].declarationEnd) {
353                                localElement = locator.createHandle(localDeclarations[idx-1], element);
354                        }
355                }
356 
357                // Store other local variable declarations in other elements
358                int size = 0;
359                for (int j=1; j<length; j++) {
360                        if (localDeclarations[j] == null) break;
361                        if (reference.sourceStart == localDeclarations[j].declarationSourceStart) {
362                                if (otherElements == null) {
363                                        otherElements = new IJavaScriptElement[length-j];
364                                }
365                                otherElements[size++] = locator.createHandle(localDeclarations[j], element);
366                        }
367                }
368                if (size > 0 && size != (length-1)) {
369                        System.arraycopy(otherElements, 0, otherElements = new IJavaScriptElement[size], 0, size);
370                }
371        }
372 
373        // Report match with local and other elements if any
374        matchReportReference(reference, element, localElement, otherElements, elementBinding, accuracy, locator);

376protected void matchReportReference(QualifiedNameReference qNameRef, IJavaScriptElement element, Binding elementBinding, int accuracy, MatchLocator locator) throws CoreException {
377        Binding binding = qNameRef.binding;
378        TypeBinding typeBinding = null;

385                case Binding.TYPE : //=============only type ==============
386                        if (binding instanceof TypeBinding)
387                                typeBinding = (TypeBinding) binding;
388                        break;
389                case Binding.VARIABLE : //============unbound cases===========
390                case Binding.TYPE | Binding.VARIABLE :
391                        if (binding instanceof ProblemReferenceBinding) {
392                                typeBinding = (TypeBinding) binding;

396                        } else if (binding instanceof ProblemBinding) {
397                                typeBinding = ((ProblemBinding) binding).searchType;

400        }

423                                        }

428                                        match.setLength(end-start+1);

431                                        matchReportReference(qNameRef, lastIndex, refBinding, locator);
432                                }
433                                return;
434                        }
435                        lastIndex--;
436                        refBinding = refBinding.enclosingType();
437                }
438        }
439        locator.reportAccurateTypeReference(match, qNameRef, this.pattern.simpleName);

486void matchReportReference(Expression expr, int lastIndex, TypeBinding refBinding, MatchLocator locator) throws CoreException {

488        // Look if there's a need to special report for parameterized type

496                if (match.getRule() == 0) return; // impossible match

659protected int resolveLevel(NameReference nameRef) {

663                if (binding instanceof ProblemReferenceBinding)
664                        binding = ((ProblemReferenceBinding) binding).closestMatch();

668        }

670        TypeBinding typeBinding = null;

672        switch (qNameRef.bits & ASTNode.RestrictiveFlagMASK) {
673                case Binding.FIELD : // reading a field
674                        if (qNameRef.tokens.length < (qNameRef.otherBindings == null ? 2 : qNameRef.otherBindings.length + 2))
675                                return IMPOSSIBLE_MATCH; // must be at least A.x

678                case Binding.LOCAL : // reading a local variable

680                case Binding.TYPE : //=============only type ==============

688                case Binding.VARIABLE : //============unbound cases===========
