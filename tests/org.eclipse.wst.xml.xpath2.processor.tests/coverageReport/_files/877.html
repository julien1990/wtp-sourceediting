966                        return false;
967

969                        case TokenDO :
970                        case TokenELSE :
971                                return true;
972                        case TokenRPAREN :
973                                position = findOpeningPeer(fPos, LPAREN, RPAREN);
974                                if (position > 0) {
975                                        switch (previousToken(position - 1, bound)) {
976                                                case TokenIF :
977                                                case TokenFOR :
978                                                case TokenWHILE :
979                                                        return true;
980                                        }
981                                }
982                }
