370                        String compliance= getStoredValue(PREF_COMPLIANCE); // get actual value
371                        IVMInstall install= null;
372                        if (fProject != null) { // project specific settings: only test if a 50 JRE is installed
373                                try {

375                                } catch (CoreException e) {
376                                        JavaScriptPlugin.log(e);
377                                }
378                        } else {
379                                install= JavaRuntime.getDefaultVMInstall();
380                        }
381                        if (install instanceof IVMInstall2) {
382                                String compilerCompliance= JavaModelUtil.getCompilerCompliance((IVMInstall2) install, compliance);
383                                if (JavaModelUtil.isVersionLessThan(compilerCompliance, compliance)) { // Discourage using compiler with version less than compliance
384                                        String[] args= { getVersionLabel(compliance), getVersionLabel(compilerCompliance) };
385                                        if (fProject == null) {
386                                                fJRE50InfoText.setText(Messages.format(PreferencesMessages.ComplianceConfigurationBlock_jrecompliance_info, args));
387                                        } else {
388                                                fJRE50InfoText.setText(Messages.format(PreferencesMessages.ComplianceConfigurationBlock_jrecompliance_info_project, args));
389                                        }
390                                        isVisible= true;
391                                }
392                        }
393                        fJRE50InfoText.setVisible(isVisible);
394                }

407        private IStatus validateCompliance() {
408                StatusInfo status= new StatusInfo();
409                String compliance= getValue(PREF_COMPLIANCE);
410                String source= getValue(PREF_SOURCE_COMPATIBILITY);
411                String target= getValue(PREF_CODEGEN_TARGET_PLATFORM);
412                
413                // compliance must not be smaller than source or target
414                if (JavaModelUtil.isVersionLessThan(compliance, source)) {
415                        status.setError(PreferencesMessages.ComplianceConfigurationBlock_src_greater_compliance); 
416                        return status;
417                }
418                
419                if (JavaModelUtil.isVersionLessThan(compliance, target)) {
420                        status.setError(PreferencesMessages.ComplianceConfigurationBlock_classfile_greater_compliance); 
421                        return status;
422                }
423                
424                // target must not be smaller than source
425                if (!VERSION_1_3.equals(source) && JavaModelUtil.isVersionLessThan(target, source)) {
426                        status.setError(PreferencesMessages.ComplianceConfigurationBlock_classfile_greater_source); 
427                        return status;
428                }
429                
430                return status;
431        }

509        /*
510         * Set the default compliance values derived from the chosen level
511         */        
512        private void updateComplianceDefaultSettings(boolean rememberOld, String oldComplianceLevel) {
513                String assertAsId, enumAsId, source, target;
514                boolean isDefault= checkValue(INTR_DEFAULT_COMPLIANCE, DEFAULT_CONF);
515                String complianceLevel= getValue(PREF_COMPLIANCE);
516                
517                if (isDefault) {
518                        if (rememberOld) {
519                                if (oldComplianceLevel == null) {
520                                        oldComplianceLevel= complianceLevel;
521                                }
522                                
523                                fRememberedUserCompliance[IDX_ASSERT_AS_IDENTIFIER]= getValue(PREF_PB_ASSERT_AS_IDENTIFIER);
524                                fRememberedUserCompliance[IDX_ENUM_AS_IDENTIFIER]= getValue(PREF_PB_ENUM_AS_IDENTIFIER);
525                                fRememberedUserCompliance[IDX_SOURCE_COMPATIBILITY]= getValue(PREF_SOURCE_COMPATIBILITY);
526                                fRememberedUserCompliance[IDX_CODEGEN_TARGET_PLATFORM]= getValue(PREF_CODEGEN_TARGET_PLATFORM);
527                                fRememberedUserCompliance[IDX_COMPLIANCE]= oldComplianceLevel;
528                        }
529 
530                        if (VERSION_1_4.equals(complianceLevel)) {
531                                assertAsId= WARNING;
532                                enumAsId= WARNING;
533                                source= VERSION_1_3;
534                                target= VERSION_1_2;
535                        } else if (VERSION_1_5.equals(complianceLevel)) {
536                                assertAsId= ERROR;
537                                enumAsId= ERROR;
538                                source= VERSION_1_5;
539                                target= VERSION_1_5;
540                        } else if (VERSION_1_6.equals(complianceLevel)) {
541                                assertAsId= ERROR;
542                                enumAsId= ERROR;
543                                source= VERSION_1_6;
544                                target= VERSION_1_6;                                
545                        } else {
546                                assertAsId= IGNORE;
547                                enumAsId= IGNORE;
548                                source= VERSION_1_3;
549                                target= VERSION_1_1;
550                        }
551                } else {
552                        if (rememberOld && complianceLevel.equals(fRememberedUserCompliance[IDX_COMPLIANCE])) {
553                                assertAsId= fRememberedUserCompliance[IDX_ASSERT_AS_IDENTIFIER];
554                                enumAsId= fRememberedUserCompliance[IDX_ENUM_AS_IDENTIFIER];
555                                source= fRememberedUserCompliance[IDX_SOURCE_COMPATIBILITY];
556                                target= fRememberedUserCompliance[IDX_CODEGEN_TARGET_PLATFORM];
557                        } else {
558                                updateInlineJSREnableState();
559                                updateAssertEnumAsIdentifierEnableState();
560                                return;
561                        }
562                }
563                setValue(PREF_PB_ASSERT_AS_IDENTIFIER, assertAsId);
564                setValue(PREF_PB_ENUM_AS_IDENTIFIER, enumAsId);
565                setValue(PREF_SOURCE_COMPATIBILITY, source);
566                setValue(PREF_CODEGEN_TARGET_PLATFORM, target);
567                updateControls();
568                updateInlineJSREnableState();
569                updateAssertEnumAsIdentifierEnableState();
570        }

574         */
575        private String getCurrentCompliance() {
576                Object complianceLevel= getValue(PREF_COMPLIANCE);

578                                && IGNORE.equals(getValue(PREF_PB_ASSERT_AS_IDENTIFIER))
579                                && IGNORE.equals(getValue(PREF_PB_ENUM_AS_IDENTIFIER))

581                                && VERSION_1_1.equals(getValue(PREF_CODEGEN_TARGET_PLATFORM)))
582                        || (VERSION_1_4.equals(complianceLevel)
583                                && WARNING.equals(getValue(PREF_PB_ASSERT_AS_IDENTIFIER))
584                                && WARNING.equals(getValue(PREF_PB_ENUM_AS_IDENTIFIER))
585                                && VERSION_1_3.equals(getValue(PREF_SOURCE_COMPATIBILITY))
586                                && VERSION_1_2.equals(getValue(PREF_CODEGEN_TARGET_PLATFORM)))
587                        || (VERSION_1_5.equals(complianceLevel)
588                                && ERROR.equals(getValue(PREF_PB_ASSERT_AS_IDENTIFIER))
589                                && ERROR.equals(getValue(PREF_PB_ENUM_AS_IDENTIFIER))
590                                && VERSION_1_5.equals(getValue(PREF_SOURCE_COMPATIBILITY))
591                                && VERSION_1_5.equals(getValue(PREF_CODEGEN_TARGET_PLATFORM)))
592                        || (VERSION_1_6.equals(complianceLevel)
593                                && ERROR.equals(getValue(PREF_PB_ASSERT_AS_IDENTIFIER))
594                                && ERROR.equals(getValue(PREF_PB_ENUM_AS_IDENTIFIER))
595                                && VERSION_1_6.equals(getValue(PREF_SOURCE_COMPATIBILITY))
596                                && VERSION_1_6.equals(getValue(PREF_CODEGEN_TARGET_PLATFORM)))) {
597                        return DEFAULT_CONF;
598                }
599                return USER_CONF;
600        }
