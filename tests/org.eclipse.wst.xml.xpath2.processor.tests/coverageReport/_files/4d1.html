576        void valueChanged(CSSNodeImpl node, String oldValue) {
577                if (!fStructuredDocumentUpdate) {
578                        CSSModelUpdater updater = getUpdater();
579                        updater.valueChanged(node, oldValue);
580                }
581 
582                ICSSSelector removed[] = null, added[] = null;
583                if (node != null) {
584                        if (node.getNodeType() == ICSSNode.ATTR_NODE && ((CSSAttrImpl) node).getName().equals(ICSSStyleRule.SELECTOR)) {
585                                CSSAttrImpl attr = (CSSAttrImpl) node;
586                                // collect changed selector
587                                ICSSSelectorList list = new CSSSelectorListImpl(attr.getValue());
588                                added = new ICSSSelector[list.getLength()];
589                                for (int i = 0; i < list.getLength(); i++)
590                                        added[i] = list.getSelector(i);
591 
592                                // get old value
593                                list = new CSSSelectorListImpl(oldValue);
594                                removed = new ICSSSelector[list.getLength()];
595                                for (int i = 0; i < list.getLength(); i++)
596                                        removed[i] = list.getSelector(i);
597                        }
598                        else if (node instanceof ICSSValue) {
599                                ICSSNode rule = node;
600                                while (rule != null) {
601                                        if (rule instanceof ICSSStyleRule)
602                                                break;
603                                        rule = rule.getParentNode();
604                                }
605                                if (rule != null) {
606                                        ICSSSelectorList list = ((ICSSStyleRule) rule).getSelectors();
607                                        added = new ICSSSelector[list.getLength()];
608                                        for (int i = 0; i < list.getLength(); i++)
609                                                added[i] = list.getSelector(i);
610                                }
611                        }
612                }
613                if (removed != null || added != null || getDocument().getNodeType() == ICSSNode.STYLEDECLARATION_NODE) {
614                        // send selector changed event
615                        getStyleNotifier().fire(removed, added, null);
616                }
617                // for href attribute
618                if (getStyleListeners() != null && getStyleListeners().size() > 0) {
619                        if (node != null && node.getNodeType() == ICSSNode.ATTR_NODE && ((CSSAttrImpl) node).getName().equals(ICSSImportRule.HREF)) {
620                                ((ICSSImportRule) ((ICSSAttr) node).getOwnerCSSNode()).getStyleSheet();
621                        }
622                }
623        }
