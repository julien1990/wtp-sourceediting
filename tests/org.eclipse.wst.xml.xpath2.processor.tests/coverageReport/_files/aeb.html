356private void quickPrioritize(CategorizedProblem[] problemList, int left, int right) {
357        if (left >= right) return;
358 
359        // sort the problems by their priority... starting with the highest priority
360        int original_left = left;
361        int original_right = right;
362        int mid = computePriority(problemList[left + (right - left) / 2]);
363        do {
364                while (computePriority(problemList[right]) < mid)
365                        right--;
366                while (mid < computePriority(problemList[left]))
367                        left++;
368                if (left <= right) {
369                        CategorizedProblem tmp = problemList[left];
370                        problemList[left] = problemList[right];
371                        problemList[right] = tmp;
372                        left++;
373                        right--;
374                }
375        } while (left <= right);
376        if (original_left < right)
377                quickPrioritize(problemList, original_left, right);
378        if (left < original_right)
379                quickPrioritize(problemList, left, original_right);
