380                if (! isVisibilitySameAsInitial())

400         *         the signatures are too much alike
401         * @throws JavaScriptModelException
402         */
403        public boolean isSignatureClashWithInitial() throws JavaScriptModelException {
404 
405                if (!isMethodNameSameAsInitial())
406                        return false; // name has changed.
407 
408                if (fMethod.getNumberOfParameters() == 0 && fParameterInfos.isEmpty())
409                        return true; // name is equal and both parameter lists are empty
410 
411                // name is equal and there are some parameters.
412                // check if there are more or less parameters than before
413 
414                int no= getNotDeletedInfos().size();
415 
416                if (fMethod.getNumberOfParameters() != no)
417                        return false;
418 
419                // name is equal and parameter count is equal.
420                // check whether types remained the same
421                
422                if (isOrderSameAsInitial())
423                        return areParameterTypesSameAsInitial();
424                else
425                        return false; // could be more specific here
426        }
