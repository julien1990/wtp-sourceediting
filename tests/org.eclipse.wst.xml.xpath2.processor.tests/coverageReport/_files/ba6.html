303MethodBinding computeSubstitutedMethod(MethodBinding method, LookupEnvironment env) {

319/*
320 * declaringUniqueKey dot selector genericSignature
321 * p.X { <T> void bar(X<T> t) } --> Lp/X;.bar<T:Ljava/lang/Object;>(LX<TT;>;)V
322 */
323public char[] computeUniqueKey(boolean isLeaf) {
324        // declaring class
325        char[] declaringKey = this.declaringClass.computeUniqueKey(false/*not a leaf*/);
326        int declaringLength = declaringKey.length;
327 
328        // selector
329        int selectorLength =  
330                (this.selector == TypeConstants.INIT || this.selector==null) ? 0 : this.selector.length;
331 
332        // generic signature
333        char[] sig = genericSignature();
334        boolean isGeneric = sig != null;
335        if (!isGeneric) sig = signature();
336        int signatureLength = sig.length;
337 
338        // thrown exceptions
339        int thrownExceptionsLength = this.thrownExceptions.length;
340        int thrownExceptionsSignatureLength = 0;
341        char[][] thrownExceptionsSignatures = null;
342        boolean addThrownExceptions = thrownExceptionsLength > 0 && (!isGeneric || CharOperation.lastIndexOf('^', sig) < 0);
343        if (addThrownExceptions) {
344                thrownExceptionsSignatures = new char[thrownExceptionsLength][];
345                for (int i = 0; i < thrownExceptionsLength; i++) {
346                        if (this.thrownExceptions[i] != null) {
347                                thrownExceptionsSignatures[i] = this.thrownExceptions[i].signature();
348                                thrownExceptionsSignatureLength += thrownExceptionsSignatures[i].length + 1;        // add one char for separator
349                        }
350                }
351        }
352 
353        char[] uniqueKey = new char[declaringLength + 1 + selectorLength + signatureLength + thrownExceptionsSignatureLength];
354        int index = 0;
355        System.arraycopy(declaringKey, 0, uniqueKey, index, declaringLength);
356        index = declaringLength;
357        uniqueKey[index++] = '.';
358        if (this.selector!=null)
359          System.arraycopy(this.selector, 0, uniqueKey, index, selectorLength);
360        index += selectorLength;
361        System.arraycopy(sig, 0, uniqueKey, index, signatureLength);
362        if (thrownExceptionsSignatureLength > 0) {
363                index += signatureLength;
364                for (int i = 0; i < thrownExceptionsLength; i++) {
365                        char[] thrownExceptionSignature = thrownExceptionsSignatures[i];
366                        if (thrownExceptionSignature != null) {
367                                uniqueKey[index++] = '|';
368                                int length = thrownExceptionSignature.length;
369                                System.arraycopy(thrownExceptionSignature, 0, uniqueKey, index, length);
370                                index += length;
371                        }
372                }
373        }
374 
375        return uniqueKey;
376}

430                }
431        }

817public String toString() {
818        String s = (returnType != null) ? returnType.debugName() : "NULL TYPE"; //$NON-NLS-1$
819        s += " "; //$NON-NLS-1$
820        s += (selector != null) ? new String(selector) : "UNNAMED METHOD"; //$NON-NLS-1$

822        s += "("; //$NON-NLS-1$
823        if (parameters != null) {
824                if (parameters != Binding.NO_PARAMETERS) {
825                        for (int i = 0, length = parameters.length; i < length; i++) {
826                                if (i  > 0)
827                                        s += ", "; //$NON-NLS-1$
828                                s += (parameters[i] != null) ? parameters[i].debugName() : "NULL TYPE"; //$NON-NLS-1$
829                        }
830                }
831        } else {
832                s += "NULL PARAMETERS"; //$NON-NLS-1$
833        }
834        s += ") "; //$NON-NLS-1$
835 
836        if (thrownExceptions != null) {
837                if (thrownExceptions != Binding.NO_EXCEPTIONS) {
838                        s += "throws "; //$NON-NLS-1$
839                        for (int i = 0, length = thrownExceptions.length; i < length; i++) {
840                                if (i  > 0)
841                                        s += ", "; //$NON-NLS-1$
842                                s += (thrownExceptions[i] != null) ? thrownExceptions[i].debugName() : "NULL TYPE"; //$NON-NLS-1$
843                        }
844                }
845        } else {
846                s += "NULL THROWN EXCEPTIONS"; //$NON-NLS-1$
847        }
848        return s;
