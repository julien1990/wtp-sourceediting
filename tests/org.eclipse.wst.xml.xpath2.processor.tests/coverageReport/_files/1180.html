303                switch (c) {
304                        case '\b' :
305                                return "\\b";//$NON-NLS-1$
306                        case '\t' :
307                                return "\\t";//$NON-NLS-1$
308                        case '\n' :
309                                return "\\n";//$NON-NLS-1$
310                        case '\f' :
311                                return "\\f";//$NON-NLS-1$        
312                        case '\r' :
313                                return "\\r";//$NON-NLS-1$
314                        case '\\' :

320        private static String windEscapeChars(String s) {
321                if (s == null)
322                        return null;
323 
324                char aChar;
325                int len= s.length();
326                StringBuffer outBuffer= new StringBuffer(len);
327 
328                for (int x= 0; x < len;) {
329                        aChar= s.charAt(x++);
330                        if (aChar == '\\') {
331                                aChar= s.charAt(x++);
332                                if (aChar == 'u') {
333                                        // Read the xxxx
334                                        int value= 0;
335                                        for (int i= 0; i < 4; i++) {
336                                                aChar= s.charAt(x++);
337                                                switch (aChar) {
338                                                        case '0': case '1': case '2': case '3': case '4': case '5': case '6': case '7': case '8': case '9':
339                                                                value= (value << 4) + aChar - '0';
340                                                                break;
341                                                        case 'a': case 'b': case 'c': case 'd': case 'e': case 'f':
342                                                                value= (value << 4) + 10 + aChar - 'a';
343                                                                break;
344                                                        case 'A': case 'B': case 'C': case 'D': case 'E': case 'F':
345                                                                value= (value << 4) + 10 + aChar - 'A';
346                                                                break;
347                                                        default:
348                                                                throw new IllegalArgumentException("Malformed \\uxxxx encoding."); //$NON-NLS-1$
349                                                }
350                                        }
351                                        outBuffer.append((char) value);
352                                } else {
353                                        if (aChar == 't') {
354                                                outBuffer.append('\t');
355                                        } else {
356                                                if (aChar == 'r') {
357                                                        outBuffer.append('\r');
358                                                } else {
359                                                        if (aChar == 'n') {
360                                                                outBuffer.append('\n');
361                                                        } else {
362                                                                if (aChar == 'f') {
363                                                                        outBuffer.append('\f');
364                                                                } else {
365                                                                        outBuffer.append(aChar);
366                                                                }
367                                                        }
368                                                }
369                                        }
370                                }
371                        } else
372                                outBuffer.append(aChar);
373                }
374                return outBuffer.toString();
375        }
