411                        System.arraycopy(typeDeclaration.methods, 0, methodDeclarations, 0, defaultConstructorIndex);
412                }
413                if (defaultConstructorIndex != typeDeclaration.methods.length-1){
414                        System.arraycopy(
415                                typeDeclaration.methods,
416                                defaultConstructorIndex+1,
417                                methodDeclarations,
418                                defaultConstructorIndex,
419                                typeDeclaration.methods.length - defaultConstructorIndex - 1);
420                }
421                typeDeclaration.methods = methodDeclarations;
422        } else {
423                int kind = TypeDeclaration.kind(typeDeclaration.modifiers);
424                if (!hasConstructor && kind != TypeDeclaration.INTERFACE_DECL && kind != TypeDeclaration.ANNOTATION_TYPE_DECL) {// if was already reduced, then constructor
425                        boolean insideFieldInitializer = false;
426                        RecoveredElement parentElement = this.parent;
427                        while (parentElement != null){
428                                if (parentElement instanceof RecoveredField){
429                                                insideFieldInitializer = true;
430                                                break;
431                                }
432                                parentElement = parentElement.parent;
433                        }
434                        typeDeclaration.createDefaultConstructor(!parser().diet || insideFieldInitializer, true);
435                }
436        }
437        if (parent instanceof RecoveredType){
438                typeDeclaration.bits |= ASTNode.IsMemberType;
439        } else if (parent instanceof RecoveredMethod){
440                typeDeclaration.bits |= ASTNode.IsLocalType;
441        }
442        if(typeDeclaration.declarationSourceEnd == 0) {
443                typeDeclaration.declarationSourceEnd = lastEnd;
444                typeDeclaration.bodyEnd = lastEnd;
445        }
446        return typeDeclaration;
447}
448/*
449 * Update the corresponding parse node from parser state which
450 * is about to disappear because of restarting recovery
451 */
452public void updateFromParserState(){
453 
454        // anymous type and enum constant doesn't need to be updated
455        if(this.bodyStartsAtHeaderEnd() && typeDeclaration.allocation == null){
456                Parser parser = this.parser();
457                /* might want to recover implemented interfaces */
458                // protection for bugs 15142
459                if (parser.listLength > 0 && parser.astLengthPtr > 0){ // awaiting interface type references
460                        int length = parser.astLengthStack[parser.astLengthPtr];
461                        int astPtr = parser.astPtr - length;
462                        boolean canConsume = astPtr >= 0;
463                        if(canConsume) {
464                                if((!(parser.astStack[astPtr] instanceof TypeDeclaration))) {
465                                        canConsume = false;
466                                }
467                                for (int i = 1, max = length + 1; i < max; i++) {
468                                        if(!(parser.astStack[astPtr + i ] instanceof TypeReference)) {
469                                                canConsume = false;
470                                        }
471                                }
472                        }
473                        if(canConsume) {
474                                parser.consumeClassHeaderImplements();
475                                // will reset typeListLength to zero
476                                // thus this check will only be performed on first errorCheck after class X implements Y,Z,
477                        }
478                } else if (parser.listTypeParameterLength > 0) {
479                        int length = parser.listTypeParameterLength;
480                        int genericsPtr = parser.genericsPtr;
481                        boolean canConsume = genericsPtr + 1 >= length && parser.astPtr > -1;
482                        if(canConsume) {
483                                if (!(parser.astStack[parser.astPtr] instanceof TypeDeclaration)) {
484                                        canConsume = false;
485                                }
486                                while(genericsPtr + 1 > length && !(parser.genericsStack[genericsPtr] instanceof TypeParameter)) {
487                                        genericsPtr--;
488                                }
489                                for (int i = 0; i < length; i++) {
490                                        if(!(parser.genericsStack[genericsPtr - i] instanceof TypeParameter)) {
491                                                canConsume = false;
492                                        }
493                                }
494                        }
495                        if(canConsume) {
496                                TypeDeclaration typeDecl = (TypeDeclaration)parser.astStack[parser.astPtr];
497                                System.arraycopy(parser.genericsStack, genericsPtr - length + 1, typeDecl.typeParameters = new TypeParameter[length], 0, length);
498                                typeDecl.bodyStart = typeDecl.typeParameters[length-1].declarationSourceEnd + 1;
499                                parser.listTypeParameterLength = 0;
500                                parser.lastCheckPoint = typeDecl.bodyStart;
501                        }
502                }
503        }
504}
505/*
506 * A closing brace got consumed, might have closed the current element,
507 * in which case both the currentElement is exited
508 */
509public RecoveredElement updateOnClosingBrace(int braceStart, int braceEnd){
510        if ((--bracketBalance <= 0) && (parent != null)){
511                this.updateSourceEndIfNecessary(braceStart, braceEnd);
512                this.bodyEnd = braceStart - 1;
513                return parent;
514        }
515        return this;
516}
517/*
518 * An opening brace got consumed, might be the expected opening one of the current element,
519 * in which case the bodyStart is updated.
520 */
521public RecoveredElement updateOnOpeningBrace(int braceStart, int braceEnd){
522        /* in case the opening brace is not close enough to the signature, ignore it */
523        if (bracketBalance == 0){
524                /*
525                        if (parser.scanner.searchLineNumber(typeDeclaration.sourceEnd)
526                                != parser.scanner.searchLineNumber(braceEnd)){
527                 */
528                Parser parser = this.parser();
529                switch(parser.lastIgnoredToken){
530                        case -1 :
531                        case TokenNameextends :
532                        case TokenNameimplements :
533                        case TokenNameGREATER :
534                        case TokenNameRIGHT_SHIFT :
535                        case TokenNameUNSIGNED_RIGHT_SHIFT :
536                                if (parser.recoveredStaticInitializerStart == 0) break;
537                        default:
538                                this.foundOpeningBrace = true;
539                                bracketBalance = 1; // pretend the brace was already there
540                }
541        }
542        // might be an initializer
543        if (this.bracketBalance == 1){
544                Block block = new Block(0);
545                Parser parser = this.parser();
546                block.sourceStart = parser.scanner.startPosition;
547                Initializer init;
548                if (parser.recoveredStaticInitializerStart == 0){
549                        init = new Initializer(block, ClassFileConstants.AccDefault);
550                } else {
551                        init = new Initializer(block, ClassFileConstants.AccStatic);
552                        init.declarationSourceStart = parser.recoveredStaticInitializerStart;
553                }
554                init.bodyStart = parser.scanner.currentPosition;
555                return this.add(init, 1);
556        }
557        return super.updateOnOpeningBrace(braceStart, braceEnd);
558}
