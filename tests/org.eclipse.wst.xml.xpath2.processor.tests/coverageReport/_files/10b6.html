357                protected void setSelectionToWidget(ISelection selection, boolean reveal) {
358                        if (true) {
359                                super.setSelectionToWidget(selection, reveal);
360                                return;
361                        }
362                        if (!fInPreserveSelection || !(selection instanceof ITreeSelection)) {
363                                super.setSelectionToWidget(selection, reveal);
364                                return;
365                        }
366                        IContentProvider cp= getContentProvider();
367                        if (!(cp instanceof IMultiElementTreeContentProvider)) {
368                                super.setSelectionToWidget(selection, reveal);
369                                return;
370                        }
371                        IMultiElementTreeContentProvider contentProvider= (IMultiElementTreeContentProvider)cp;
372                        ITreeSelection toRestore= (ITreeSelection)selection;
373                        List pathsToSelect= new ArrayList();
374                        for (Iterator iter= toRestore.iterator(); iter.hasNext();) {
375                                Object element= iter.next();
376                                TreePath[] pathsToRestore= toRestore.getPathsFor(element);
377                                CustomHashtable currentParents= createRootAccessedMap(contentProvider.getTreePaths(element));
378                                for (int i= 0; i < pathsToRestore.length; i++) {
379                                        TreePath path= pathsToRestore[i];
380                                        Object root= path.getFirstSegment();
381                                        if (root != null && path.equals((TreePath)currentParents.get(root), getComparer())) {
382                                                pathsToSelect.add(path);
383                                        }
384                                }
385                        }
386                        List toSelect= new ArrayList();
387                        for (Iterator iter= pathsToSelect.iterator(); iter.hasNext();) {
388                                TreePath path= (TreePath)iter.next();
389                                int size= path.getSegmentCount();
390                                if (size == 0)
391                                        continue;
392                                Widget current= getTree();
393                                int last= size - 1;
394                                Object segment;
395                                for (int i= 0; i < size && current != null && (segment= path.getSegment(i)) != null; i++) {
396                                        internalExpandToLevel(current, 1);
397                                        current= internalFindChild(current, segment);
398                                        if (i == last && current != null)
399                                                toSelect.add(current);
400                                }
401                        }
402                        getTree().setSelection((TreeItem[])toSelect.toArray(new TreeItem[toSelect.size()]));
403                }
404            private Widget internalFindChild(Widget parent, Object element) {
405                Item[] items = getChildren(parent);
406                for (int i = 0; i < items.length; i++) {
407                    Item item = items[i];
408                    Object data = item.getData();
409                    if (data != null && equals(data, element))
410                        return item;
411                }
412                return null;
413            }

418                                Object root= path.getFirstSegment();
419                                if (root != null) {
420                                        result.put(root, path);
421                                }
422                        }
423                        return result;
424                }
