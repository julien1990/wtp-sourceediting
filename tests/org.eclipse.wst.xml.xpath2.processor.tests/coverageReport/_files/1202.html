699        private IJavaScriptElement[] getInputForCategories() {
700                if (fInput == null)
701                        return new IJavaScriptElement[0];
702                
703                if (fOutlineContentProvider.isShowingInheritedMembers()) {
704                        IJavaScriptElement p= fInput;
705                        if (p instanceof ITypeRoot) {
706                                p= ((ITypeRoot)p).findPrimaryType();
707                        }
708                        while (p != null && !(p instanceof IType)) {
709                                p= p.getParent();
710                        }
711                        if (!(p instanceof IType))
712                                return new IJavaScriptElement[] {fInput};
713                        
714                        ITypeHierarchy hierarchy= getSuperTypeHierarchy((IType)p);
715                        if (hierarchy == null)
716                                return new IJavaScriptElement[] {fInput};
717                        
718                        IType[] supertypes= hierarchy.getAllSupertypes((IType)p);
719                        IJavaScriptElement[] result= new IJavaScriptElement[supertypes.length + 1];
720                        result[0]= fInput;
721                        System.arraycopy(supertypes, 0, result, 1, supertypes.length);
722                        return result;
723                } else {
724                        return new IJavaScriptElement[] {fInput};
725                }
