216        private ICompletionProposal[][] guessParameters() throws JavaScriptModelException {
217                // find matches in reverse order.  Do this because people tend to declare the variable meant for the last
218                // parameter last.  That is, local variables for the last parameter in the method completion are more
219                // likely to be closer to the point of code completion. As an example consider a "delegation" completion:
220                //
221                //                 public void myMethod(int param1, int param2, int param3) {
222                //                         someOtherObject.yourMethod(param1, param2, param3);
223                //                }
224                //
225                // The other consideration is giving preference to variables that have not previously been used in this
226                // code completion (which avoids "someOtherObject.yourMethod(param1, param1, param1)";
227                
228                char[][] parameterNames= fProposal.findParameterNames(null);
229                int count= parameterNames.length;
230                fPositions= new Position[count];
231                fChoices= new ICompletionProposal[count][];
232 
233                IDocument document= fInvocationContext.getDocument();
234                IJavaScriptUnit cu= fInvocationContext.getCompilationUnit();
235                JavaModelUtil.reconcile(cu);
236                String[][] parameterTypes= getParameterSignatures();
237                ParameterGuesser guesser= new ParameterGuesser(fProposal.getCompletionLocation() + 1, cu);
238                
239                for (int i= count - 1; i >= 0; i--) {
240                        String paramName= new String(parameterNames[i]);
241                        Position position= new Position(0,0);
242                        
243                        ICompletionProposal[] argumentProposals= guesser.parameterProposals(parameterTypes[i][0], parameterTypes[i][1], paramName, position, document);
244                        if (argumentProposals.length == 0)
245                                argumentProposals= new ICompletionProposal[] {new JavaCompletionProposal(paramName, 0, paramName.length(), null, paramName, 0)};
246                        
247                        fPositions[i]= position;
248                        fChoices[i]= argumentProposals;
249                }
250
