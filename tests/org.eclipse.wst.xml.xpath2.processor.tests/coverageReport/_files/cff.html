363/**
364 * Parses the method bodies in the given compilation unit
365 * @param unit CompilationUnitDeclaration
366 */
367public void parseBodies(CompilationUnitDeclaration unit) {
368        TypeDeclaration[] types = unit.types;
369        if (types != null)
370          for (int i = 0; i < types.length; i++) {
371                TypeDeclaration type = types[i];
372                this.patternLocator.match(type, this.nodeSet);
373                this.parseBodies(type, unit);
374        }
375 
376 
377        ProgramElement[] statements = unit.statements;
378        if (statements != null)
379           for (int i = 0; i < statements.length; i++) {
380                if (statements[i] instanceof LocalDeclaration)
381                {
382                        ((LocalDeclaration)statements[i]).traverse(localDeclarationVisitor, null);
383                        if (patternLocator instanceof FieldLocator)
384                                ((FieldLocator)patternLocator).matchLocalDeclaration((LocalDeclaration)statements[i], this.nodeSet);
385                }
386                else if (statements[i] instanceof AbstractMethodDeclaration)
387                {
388                        AbstractMethodDeclaration methodDeclaration=(AbstractMethodDeclaration)statements[i];
389//                        this.parse(methodDeclaration, unit);
390                        methodDeclaration.traverse(localDeclarationVisitor, (Scope) null);
391                        if (this.patternLocator instanceof MethodLocator)
392                                ((MethodLocator)this.patternLocator).match((MethodDeclaration)statements[i], this.nodeSet);
393 
394                }
395 
396        }
397 
398        unit.traverseInferredTypes(localDeclarationVisitor,  null);
399}
