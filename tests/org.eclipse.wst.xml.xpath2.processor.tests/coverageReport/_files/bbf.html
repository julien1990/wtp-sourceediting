603// JLS3: 4.5.1.1
604public boolean isTypeArgumentContainedBy(TypeBinding otherType) {
605        if (this == otherType)
606                return true;
607        switch (otherType.kind()) {
608        // allow wildcard containment
609        case Binding.WILDCARD_TYPE:
610                TypeBinding lowerBound = this;
611                TypeBinding upperBound = this;
612                switch (this.kind()) {
613                case Binding.WILDCARD_TYPE:
614                        WildcardBinding wildcard = (WildcardBinding) this;
615                        switch (wildcard.boundKind) {
616                        case Wildcard.EXTENDS:
617                                if (wildcard.otherBounds != null) // intersection type
618                                        break;
619                                upperBound = wildcard.bound;
620                                lowerBound = null;
621                                break;
622                        case Wildcard.SUPER:
623                                upperBound = wildcard;
624                                lowerBound = wildcard.bound;
625                                break;
626                        case Wildcard.UNBOUND:
627                                upperBound = wildcard;
628                                lowerBound = null;
629                        }
630                        break;
631                case Binding.TYPE_PARAMETER:
632                        if (this.isCapture()) {
633                                CaptureBinding capture = (CaptureBinding) this;
634                                if (capture.lowerBound != null)
635                                        lowerBound = capture.lowerBound;
636                        }
637                }
638                WildcardBinding otherWildcard = (WildcardBinding) otherType;
639                if (otherWildcard.otherBounds != null)
640                        return false; // not a true wildcard (intersection type)
641                TypeBinding otherBound = otherWildcard.bound;
642                switch (otherWildcard.boundKind) {
643                case Wildcard.EXTENDS:
644                        if (otherBound == this)
645                                return true; // ? extends T  <=  ? extends ? extends T
646                        if (upperBound == null)
647                                return false;
648                        TypeBinding match = upperBound.findSuperTypeWithSameErasure(otherBound);
649                        if (match != null && (match = match.leafComponentType()).isRawType()) {
650                                return match == otherBound.leafComponentType(); // forbide: Collection <=  ? extends Collection<?>
651                                                                                                                                                                                                                // forbide: Collection[] <=  ? extends Collection<?>[]
652                        }
653                        return upperBound.isCompatibleWith(otherBound);
654 
655                case Wildcard.SUPER:
656                        if (otherBound == this)
657                                return true; // ? super T  <=  ? super ? super T
658                        if (lowerBound == null)
659                                return false;
660                        match = otherBound.findSuperTypeWithSameErasure(lowerBound);
661                        if (match != null && (match = match.leafComponentType()).isRawType()) {
662                                return match == lowerBound.leafComponentType(); // forbide: Collection <=  ? super Collection<?>
663                                                                                                                                                                                                                // forbide: Collection[] <=  ? super Collection<?>[]
664                        }
665                        return otherBound.isCompatibleWith(lowerBound);
666 
667                case Wildcard.UNBOUND:
668                default:
669                        return true;
670                }
671                // allow List<?> to match List<? extends Object> (and reciprocally)
672        case Binding.PARAMETERIZED_TYPE:
673                if (!this.isParameterizedType())
674                        return false;
675                ParameterizedTypeBinding paramType = (ParameterizedTypeBinding) this;
676                ParameterizedTypeBinding otherParamType = (ParameterizedTypeBinding) otherType;
677                if (paramType.actualType() != otherParamType.actualType())
678                        return false;
679                if (!paramType.isStatic()) { // static member types do not compare their enclosing
680                        ReferenceBinding enclosing = enclosingType();
681                        if (enclosing != null) {
682                                ReferenceBinding otherEnclosing = otherParamType
683                                                .enclosingType();
684                                if (otherEnclosing == null)
685                                        return false;
686                                if ((otherEnclosing.tagBits & TagBits.HasDirectWildcard) == 0) {
687                                        if (enclosing != otherEnclosing)
688                                                return false;
689                                } else {
690                                        if (!enclosing.isEquivalentTo(otherParamType
691                                                        .enclosingType()))
692                                                return false;
693                                }
694                        }
695                }
696                int length = paramType.arguments == null ? 0
697                                : paramType.arguments.length;
698                TypeBinding[] otherArguments = otherParamType.arguments;
699                int otherLength = otherArguments == null ? 0
700                                : otherArguments.length;
701                if (otherLength != length)
702                        return false;
703                nextArgument: for (int i = 0; i < length; i++) {
704                        TypeBinding argument = paramType.arguments[i];
705                        TypeBinding otherArgument = otherArguments[i];
706                        if (argument == otherArgument)
707                                continue nextArgument;
708                        int kind = argument.kind();
709                        if (otherArgument.kind() != kind)
710                                return false;
711                        switch (kind) {
712                        case Binding.PARAMETERIZED_TYPE:
713                                if (argument.isTypeArgumentContainedBy(otherArgument)) // recurse
714                                        continue nextArgument;
715                                break;
716                        case Binding.WILDCARD_TYPE:
717                                WildcardBinding wildcard = (WildcardBinding) argument;
718                                otherWildcard = (WildcardBinding) otherArgument;
719                                switch (wildcard.boundKind) {
720                                case Wildcard.EXTENDS:
721                                        // match "? extends <upperBound>" with "?"
722                                        if (otherWildcard.boundKind == Wildcard.UNBOUND
723                                                        && wildcard.bound == wildcard.typeVariable()
724                                                                        .upperBound())
725                                                continue nextArgument;
726                                        break;
727                                case Wildcard.SUPER:
728                                        break;
729                                case Wildcard.UNBOUND:
730                                        // match "?" with "? extends <upperBound>"
731                                        if (otherWildcard.boundKind == Wildcard.EXTENDS
732                                                        && otherWildcard.bound == otherWildcard
733                                                                        .typeVariable().upperBound())
734                                                continue nextArgument;
735                                        break;
736                                }
737                                break;
738                        }
739                        return false;
740                }
741                return true;
742        }
743        return false;
744}
