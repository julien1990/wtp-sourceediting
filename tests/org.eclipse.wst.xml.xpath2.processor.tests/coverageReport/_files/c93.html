445/* Return the list of projects for which it requires a resource delta. This builder's project
446* is implicitly included and need not be specified. Builders must re-specify the list
447* of interesting projects every time they are run as this is not carried forward
448* beyond the next build. Missing projects should be specified but will be ignored until
449* they are added to the workspace.
450*/
451private IProject[] getRequiredProjects(boolean includeBinaryPrerequisites) {
452        if (javaProject == null || workspaceRoot == null) return new IProject[0];
453 
454        ArrayList projects = new ArrayList();
455        try {
456                IIncludePathEntry[] entries = javaProject.getExpandedClasspath();
457                for (int i = 0, l = entries.length; i < l; i++) {
458                        IIncludePathEntry entry = entries[i];
459                        IPath path = entry.getPath();
460                        IProject p = null;
461                        switch (entry.getEntryKind()) {
462                                case IIncludePathEntry.CPE_PROJECT :
463                                        p = workspaceRoot.getProject(path.lastSegment()); // missing projects are considered too
464                                        if (((ClasspathEntry) entry).isOptional() && !JavaProject.hasJavaNature(p)) // except if entry is optional
465                                                p = null;
466                                        break;
467                                case IIncludePathEntry.CPE_LIBRARY :
468                                        if (includeBinaryPrerequisites && path.segmentCount() > 1) {
469                                                // some binary resources on the class path can come from projects that are not included in the project references
470                                                IResource resource = workspaceRoot.findMember(path.segment(0));
471                                                if (resource instanceof IProject)
472                                                        p = (IProject) resource;
473                                        }
474                        }
475                        if (p != null && !projects.contains(p))
476                                projects.add(p);
477                }
478        } catch(JavaScriptModelException e) {
479                return new IProject[0];
480        }
481        IProject[] result = new IProject[projects.size()];
482        projects.toArray(result);
483        return result;
484}

494private boolean hasClasspathChanged() {
495        ClasspathMultiDirectory[] newSourceLocations = nameEnvironment.sourceLocations;
496        ClasspathMultiDirectory[] oldSourceLocations = lastState.sourceLocations;
497        int newLength = newSourceLocations.length;
498        int oldLength = oldSourceLocations.length;
499        int n, o;
500        for (n = o = 0; n < newLength && o < oldLength; n++, o++) {
501                if (newSourceLocations[n].equals(oldSourceLocations[o])) continue; // checks source & output folders
502                try {
503                        if (newSourceLocations[n].sourceFolder.members().length == 0) { // added new empty source folder
504                                o--;
505                                continue;
506                        }
507                } catch (CoreException ignore) { // skip it
508                }
509                if (DEBUG) {
510                        System.out.println("New location: " + newSourceLocations[n] + "\n!= old location: " + oldSourceLocations[o]); //$NON-NLS-1$ //$NON-NLS-2$
511                        printLocations(newSourceLocations, oldSourceLocations);
512                }
513                return true;
514        }
515        while (n < newLength) {
516                try {
517                        if (newSourceLocations[n].sourceFolder.members().length == 0) { // added new empty source folder
518                                n++;
519                                continue;
520                        }
521                } catch (CoreException ignore) { // skip it
522                }
523                if (DEBUG) {
524                        System.out.println("Added non-empty source folder"); //$NON-NLS-1$
525                        printLocations(newSourceLocations, oldSourceLocations);
526                }
527                return true;
528        }
529        if (o < oldLength) {
530                if (DEBUG) {
531                        System.out.println("Removed source folder"); //$NON-NLS-1$
532                        printLocations(newSourceLocations, oldSourceLocations);
533                }
534                return true;
535        }
536 
537        ClasspathLocation[] newBinaryLocations = nameEnvironment.binaryLocations;
538        ClasspathLocation[] oldBinaryLocations = lastState.binaryLocations;
539        newLength = newBinaryLocations.length;
540        oldLength = oldBinaryLocations.length;
541        for (n = o = 0; n < newLength && o < oldLength; n++, o++) {
542                if (newBinaryLocations[n].equals(oldBinaryLocations[o])) continue;
543                if (DEBUG) {
544                        System.out.println("New location: " + newBinaryLocations[n] + "\n!= old location: " + oldBinaryLocations[o]); //$NON-NLS-1$ //$NON-NLS-2$
545                        printLocations(newBinaryLocations, oldBinaryLocations);
546                }
547                return true;
548        }
549        if (n < newLength || o < oldLength) {
550                if (DEBUG) {
551                        System.out.println("Number of binary folders/jar files has changed:"); //$NON-NLS-1$
552                        printLocations(newBinaryLocations, oldBinaryLocations);
553                }
554                return true;
555        }
556        return false;
557}

589private int initializeBuilder(int kind, boolean forBuild) throws CoreException {

591        this.javaProject = (JavaProject) JavaScriptCore.create(currentProject);
592        this.workspaceRoot = currentProject.getWorkspace().getRoot();

594        if (forBuild) {

599                                if (this.participants[i].aboutToBuild(this.javaProject) == ValidationParticipant.NEEDS_FULL_BUILD)

605                        JavaModel.flushExternalFileCache();

618                        : null;

620                        this.extraResourceFileFilters = null;
621                        this.extraResourceFolderFilters = null;

635                                        extraResourceFolderFilters[--folderCount] = new String(f, 0, f.length - 1);
636                                else
637                                        extraResourceFileFilters[--fileCount] = f;
638                        }
639                }
640        }
641        return kind;
