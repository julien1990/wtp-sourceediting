384        public static IFix[] createRemoveBlockFix(JavaScriptUnit compilationUnit, ASTNode node) {
385                Statement statement= ASTResolving.findParentStatement(node);
386                if (statement == null) {
387                        return null;
388                }
389                
390                if (statement instanceof Block) {
391                        Block block= (Block)statement;
392                        if (block.statements().size() != 1)
393                                return null;
394                        
395                        ASTNode parent= block.getParent();
396                        if (!(parent instanceof Statement))
397                                return null;
398                        
399                        statement= (Statement)parent;
400                }
401                
402                if (statement instanceof IfStatement) {
403                        List result= new ArrayList();
404                        
405                        List removeAllList= new ArrayList();
406                        
407                        IfElseIterator iter= new IfElseIterator((IfStatement)statement);
408                        IfStatement item= null;
409                        while (iter.hasNext()) {
410                                item= iter.next();
411                                if (RemoveBlockOperation.satisfiesQuickAssistPrecondition(item, IfStatement.THEN_STATEMENT_PROPERTY)) {
412                            RemoveBlockOperation op= new RemoveBlockOperation(item, IfStatement.THEN_STATEMENT_PROPERTY);
413                                        removeAllList.add(op);
414                                        if (item == statement)
415                                                result.add(new ControlStatementsFix(FixMessages.ControlStatementsFix_removeIfBlock_proposalDescription, compilationUnit, new IFixRewriteOperation[] {op}));
416                    }
417                        }
418                        
419                        if (RemoveBlockOperation.satisfiesQuickAssistPrecondition(item, IfStatement.ELSE_STATEMENT_PROPERTY)) {
420                    RemoveBlockOperation op= new RemoveBlockOperation(item, IfStatement.ELSE_STATEMENT_PROPERTY);
421                                removeAllList.add(op);
422                                if (item == statement)
423                                        result.add(new ControlStatementsFix(FixMessages.ControlStatementsFix_removeElseBlock_proposalDescription, compilationUnit, new IFixRewriteOperation[] {op}));
424            }
425            
426                        if (removeAllList.size() > 1) {
427                                IFixRewriteOperation[] allConvert= (IFixRewriteOperation[])removeAllList.toArray(new IFixRewriteOperation[removeAllList.size()]);
428                                result.add(new ControlStatementsFix(FixMessages.ControlStatementsFix_removeIfElseBlock_proposalDescription, compilationUnit, allConvert));
429            }
430            
431            return (IFix[])result.toArray(new IFix[result.size()]);
432                } else if (statement instanceof WhileStatement) {
433                        if (RemoveBlockOperation.satisfiesQuickAssistPrecondition(statement, WhileStatement.BODY_PROPERTY)) {
434                                RemoveBlockOperation op= new RemoveBlockOperation(statement, WhileStatement.BODY_PROPERTY);
435                                return new IFix[] {new ControlStatementsFix(FixMessages.ControlStatementsFix_removeBrackets_proposalDescription, compilationUnit, new IFixRewriteOperation[] {op})};
436                        }
437                } else if (statement instanceof ForStatement) {
438                        if (RemoveBlockOperation.satisfiesQuickAssistPrecondition(statement, ForStatement.BODY_PROPERTY)) {
439                                RemoveBlockOperation op= new RemoveBlockOperation(statement, ForStatement.BODY_PROPERTY);
440                                return new IFix[] {new ControlStatementsFix(FixMessages.ControlStatementsFix_removeBrackets_proposalDescription, compilationUnit, new IFixRewriteOperation[] {op})};
441                        }
442                } else if (statement instanceof ForInStatement) {
443                        if (RemoveBlockOperation.satisfiesQuickAssistPrecondition(statement, ForInStatement.BODY_PROPERTY)) {
444                                RemoveBlockOperation op= new RemoveBlockOperation(statement, ForInStatement.BODY_PROPERTY);
445                                return new IFix[] {new ControlStatementsFix(FixMessages.ControlStatementsFix_removeBrackets_proposalDescription, compilationUnit, new IFixRewriteOperation[] {op})};
446                        }
447                } else if (statement instanceof EnhancedForStatement) {
448                        if (RemoveBlockOperation.satisfiesQuickAssistPrecondition(statement, EnhancedForStatement.BODY_PROPERTY)) {
449                                RemoveBlockOperation op= new RemoveBlockOperation(statement, EnhancedForStatement.BODY_PROPERTY);
450                                return new IFix[] {new ControlStatementsFix(FixMessages.ControlStatementsFix_removeBrackets_proposalDescription, compilationUnit, new IFixRewriteOperation[] {op})};
451                        }
452                } else if (statement instanceof DoStatement) {
453                        if (RemoveBlockOperation.satisfiesQuickAssistPrecondition(statement, DoStatement.BODY_PROPERTY)) {
454                                RemoveBlockOperation op= new RemoveBlockOperation(statement, DoStatement.BODY_PROPERTY);
455                                return new IFix[] {new ControlStatementsFix(FixMessages.ControlStatementsFix_removeBrackets_proposalDescription, compilationUnit, new IFixRewriteOperation[] {op})};
456                        }
457                }
458                
459                return null;
