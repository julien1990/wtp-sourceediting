384                        return true;

585         * Subclasses should implement for specific handling of dirty regions. The
586         * method is invoked for each dirty region in the Job's queue.
587         * 
588         * @param dirtyRegion
589         */
590        protected void process(DirtyRegion dirtyRegion) {
591                if (isInRewriteSession()) {
592                        return;
593                }
594                /*
595                 * Break the dirty region into a sequence of partitions and find the
596                 * corresponding strategy to reconcile those partitions. If a strategy
597                 * implements INonIncrementalReconcilingStrategy, only call it once
598                 * regardless of the number and types of partitions.
599                 */
600                ITypedRegion[] partitions = computePartitioning(dirtyRegion);
601                for (int i = 0; i < partitions.length; i++) {
602                        IReconcilingStrategy strategy = getReconcilingStrategy(partitions[i].getType());
603                        if (strategy != null) {
604                                strategy.reconcile(partitions[i]);
605                        }
606                }
607        }

611         *

613         */
614        public final void processDirtyRegion(DirtyRegion dr) {
615                if (dr == null || !isInstalled())

617 
618                cancel();
619                addRequest(dr);
620                schedule(getDelay());
621 
622                if (DEBUG) {
623                        System.out.println("added request for: [" + dr.getText() + "]"); //$NON-NLS-1$ //$NON-NLS-2$
624                        System.out.println("queue size is now: " + getDirtyRegionQueue().size()); //$NON-NLS-1$

628        protected IStatus run(IProgressMonitor monitor) {
629                IStatus status = Status.OK_STATUS;
630                if (!PlatformUI.isWorkbenchRunning())
631                        return status;
632 
633                Thread.currentThread().setPriority(Thread.MIN_PRIORITY);
634                try {
635                        beginProcessing();
636 
637                        DirtyRegion[] toRefresh = getRequests();
638                        for (int i = 0; i < toRefresh.length && fDocument != null; i++) {
639                                if (monitor.isCanceled())
640                                        throw new OperationCanceledException();
641                                process(toRefresh[i]);
642                        }
643                }
644                catch (Exception e) {
645                        // https://bugs.eclipse.org/bugs/show_bug.cgi?id=103676
646                        // stop errors from popping up a dialog
647                        // from the job manager
648 
649                        // https://bugs.eclipse.org/bugs/show_bug.cgi?id=106052
650                        // don't log OperationCanceledException
651                        if (!(e instanceof OperationCanceledException))
652                                Logger.logException("problem with reconciling", e); //$NON-NLS-1$
653                }
654                finally {
655                        endProcessing();
656 
657                        monitor.done();
658                }
659                return status;
660        }
