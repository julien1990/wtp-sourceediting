416protected int resolveLevel(NameReference nameRef) {
417        if (nameRef instanceof SingleNameReference)
418                return resolveLevel(nameRef.binding);
419 
420        Binding binding = nameRef.binding;
421        QualifiedNameReference qNameRef = (QualifiedNameReference) nameRef;
422        FieldBinding fieldBinding = null;
423        if (binding instanceof FieldBinding) {
424                fieldBinding = (FieldBinding) binding;
425                char[] bindingName = fieldBinding.name;
426                int lastDot = CharOperation.lastIndexOf('.', bindingName);
427                if (lastDot > -1)
428                        bindingName = CharOperation.subarray(bindingName, lastDot+1, bindingName.length);
429                if (matchesName(this.pattern.name, bindingName)) {
430                        int level = matchField(fieldBinding, false);
431                        if (level != IMPOSSIBLE_MATCH) return level;
432                }
433        }
434        int otherMax = qNameRef.otherBindings == null ? 0 : qNameRef.otherBindings.length;
435        for (int i = 0; i < otherMax; i++) {
436                char[] token = qNameRef.tokens[i + qNameRef.indexOfFirstFieldBinding];
437                if (matchesName(this.pattern.name, token)) {
438                        FieldBinding otherBinding = qNameRef.otherBindings[i];
439                        int level = matchField(otherBinding, false);
440                        if (level != IMPOSSIBLE_MATCH) return level;
441                }
442        }
443        return IMPOSSIBLE_MATCH;

462public int matchLocalDeclaration(LocalDeclaration node, MatchingNodeSet nodeSet) {
463        int referencesLevel = IMPOSSIBLE_MATCH;
464        if (this.pattern.findReferences)
465                // must be a write only access with an initializer
466                if (this.pattern.writeAccess && !this.pattern.readAccess && node.initialization != null)
467                        if (matchesName(this.pattern.name, node.name))
468                                referencesLevel = ((InternalSearchPattern)this.pattern).mustResolve ? POSSIBLE_MATCH : ACCURATE_MATCH;
469 
470        int declarationsLevel = IMPOSSIBLE_MATCH;
471        if (this.pattern.findDeclarations)
472                if (matchesName(this.pattern.name, node.name))
473                                declarationsLevel = ((InternalSearchPattern)this.pattern).mustResolve ? POSSIBLE_MATCH : ACCURATE_MATCH;
474 
475        return nodeSet.addMatch(node, referencesLevel >= declarationsLevel ? referencesLevel : declarationsLevel); // use the stronger match
476}
