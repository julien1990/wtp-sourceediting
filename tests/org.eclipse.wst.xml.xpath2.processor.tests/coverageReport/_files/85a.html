428        public IHyperlink[] detectHyperlinks(ITextViewer textViewer, IRegion region, boolean canShowMultipleHyperlinks) {
429                if (textViewer != null && region != null) {
430                        IDocument document = textViewer.getDocument();
431                        if (document != null) {
432                                Node currentNode = getCurrentNode(document, region.getOffset());
433                                if (currentNode != null && currentNode.getNodeType() == Node.ELEMENT_NODE) {
434                                        Element element = (Element) currentNode;
435                                                IStructuredDocumentRegion documentRegion = ((IStructuredDocument) document).getRegionAtCharacterOffset(region.getOffset());
436                                                ITextRegion textRegion = documentRegion.getRegionAtCharacterOffset(region.getOffset());
437                                                ITextRegion nameRegion = null;
438                                                ITextRegion valueRegion = null;
439                                                String name = null;
440                                                String value = null;
441                                                if (DOMRegionContext.XML_TAG_ATTRIBUTE_VALUE.equals(textRegion.getType())) {
442                                                        ITextRegionList regions = documentRegion.getRegions();
443                                                        /*
444                                                         * Could use 2, but there needs to be the tag open
445                                                         * and name regions
446                                                         */
447                                                        int index = regions.indexOf(textRegion);
448                                                        if (index >= 4) {
449                                                                nameRegion = regions.get(index - 2);
450                                                                valueRegion = textRegion;
451                                                                name = documentRegion.getText(nameRegion);
452                                                                value = StringUtils.strip(documentRegion.getText(valueRegion));
453                                                        }
454                                                }
455                                                else if (DOMRegionContext.XML_TAG_ATTRIBUTE_NAME.equals(textRegion.getType())) {
456                                                        ITextRegionList regions = documentRegion.getRegions();
457                                                        int index = regions.indexOf(textRegion);
458                                                        // minus 3 to leave equal and value regions
459                                                        if (index <= regions.size() - 3) {
460                                                                nameRegion = textRegion;
461                                                                valueRegion = regions.get(index + 2);
462                                                                name = documentRegion.getText(nameRegion);
463                                                                value = StringUtils.strip(documentRegion.getText(valueRegion));
464                                                        }
465                                                }
466                                                if (name != null && value != null) {
467                                                        if (HTML40Namespace.ATTR_NAME_HREF.equalsIgnoreCase(name) && value.indexOf("#") >= 0) { //$NON-NLS-1$
468                                                                return createHyperlinksToAnchorNamed(textViewer, createHyperlinkRegion(documentRegion, valueRegion), element, value, canShowMultipleHyperlinks);
469                                                        }
470                                                        if (HTML40Namespace.ATTR_NAME_NAME.equalsIgnoreCase(name)||HTML40Namespace.ATTR_NAME_ID.equalsIgnoreCase(name)) {
471                                                                return createReferrerHyperlinks(textViewer, createHyperlinkRegion(documentRegion, valueRegion), element, value, canShowMultipleHyperlinks);
472                                                        }
473                                                }
474                                }
475                        }
476                }
477                return null;
478        }
