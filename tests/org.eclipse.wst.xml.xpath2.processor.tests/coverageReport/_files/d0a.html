316protected int matchNameValue(char[] pattern, char[] name) {
317        if (pattern == null) return ACCURATE_MATCH; // null is as if it was "*"
318        if (name == null) return IMPOSSIBLE_MATCH; // cannot match null name
319        if (name.length == 0) { // empty name
320                if (pattern.length == 0) { // can only matches empty pattern
321                        return ACCURATE_MATCH;
322                }
323                return IMPOSSIBLE_MATCH;
324        } else if (pattern.length == 0) {
325                return IMPOSSIBLE_MATCH; // need to have both name and pattern length==0 to be accurate
326        }
327        boolean matchFirstChar = !this.isCaseSensitive || pattern[0] == name[0];
328        boolean sameLength = pattern.length == name.length;
329        boolean canBePrefix = name.length >= pattern.length;
330        if (this.isCamelCase && matchFirstChar && CharOperation.camelCaseMatch(pattern, name)) {
331                return POSSIBLE_MATCH;
332        }
333        switch (this.matchMode) {
334                case SearchPattern.R_EXACT_MATCH:
335                        if (!this.isCamelCase) {
336                                if (sameLength && matchFirstChar && CharOperation.equals(pattern, name, this.isCaseSensitive)) {
337                                        return POSSIBLE_MATCH | EXACT_FLAVOR;
338                                }
339                                break;
340                        }
341                        // fall through next case to match as prefix if camel case failed
342                case SearchPattern.R_PREFIX_MATCH:
343                        if (canBePrefix && matchFirstChar && CharOperation.prefixEquals(pattern, name, this.isCaseSensitive)) {
344                                return POSSIBLE_MATCH;
345                        }
346                        break;
347                case SearchPattern.R_PATTERN_MATCH:
348                        if (!this.isCaseSensitive) {
349                                pattern = CharOperation.toLowerCase(pattern);
350                        }
351                        if (CharOperation.match(pattern, name, this.isCaseSensitive)) {
352                                return POSSIBLE_MATCH;
353                        }
354                        break;
355                case SearchPattern.R_REGEXP_MATCH :
356                        // TODO (frederic) implement regular expression match
357                        break;
358        }
359        return IMPOSSIBLE_MATCH;
360}

487protected void updateMatch(ParameterizedTypeBinding parameterizedBinding, char[][][] patternTypeArguments, MatchLocator locator) {

493protected void updateMatch(ParameterizedTypeBinding parameterizedBinding, char[][][] patternTypeArguments, boolean patternHasTypeParameters, int depth, MatchLocator locator) {
494        // Only possible if locator has an unit scope.
495        if (locator.unitScope == null) return;
496 
497        // Set match raw flag
498        boolean endPattern = patternTypeArguments==null  ? true  : depth>=patternTypeArguments.length;
499        TypeBinding[] argumentsBindings = parameterizedBinding.arguments;
500        boolean isRaw = parameterizedBinding.isRawType()|| (argumentsBindings==null && parameterizedBinding.genericType().isGenericType());
501        if (isRaw && !match.isRaw()) {
502                match.setRaw(isRaw);
503        }
504 
505        // Update match
506        if (!endPattern && patternTypeArguments != null) {
507                // verify if this is a reference to the generic type itself
508                if (!isRaw && patternHasTypeParameters && argumentsBindings != null) {
509                        boolean needUpdate = false;
510                        TypeVariableBinding[] typeVariables = parameterizedBinding.genericType().typeVariables();
511                        for (int i=0, l=argumentsBindings.length; i<l; i++) {
512                                if (argumentsBindings[i] != typeVariables[i]) {
513                                        needUpdate = true;
514                                        break;
515                                }
516                        }
517                        if (needUpdate) {
518                                char[][] patternArguments =  patternTypeArguments[depth];
519                                updateMatch(argumentsBindings, locator, patternArguments, patternHasTypeParameters);
520                        }
521                } else {
522                        char[][] patternArguments =  patternTypeArguments[depth];
523                        updateMatch(argumentsBindings, locator, patternArguments, patternHasTypeParameters);
524                }
525        }
526 
527        // Recurse
528        TypeBinding enclosingType = parameterizedBinding.enclosingType();
529        if (enclosingType != null && (enclosingType.isParameterizedType() || enclosingType.isRawType())) {
530                updateMatch((ParameterizedTypeBinding)enclosingType, patternTypeArguments, patternHasTypeParameters, depth+1, locator);
531        }
532}
533/*
534 * Update pattern locator match comparing type arguments with pattern ones.
535 * Try to resolve pattern and look for compatibility with type arguments
536 * to set match rule.
537 */
538protected void updateMatch(TypeBinding[] argumentsBinding, MatchLocator locator, char[][] patternArguments, boolean hasTypeParameters) {
539        // Only possible if locator has an unit scope.
540        if (locator.unitScope == null) return;
541 
542        // First compare lengthes
543        int patternTypeArgsLength = patternArguments==null ? 0 : patternArguments.length;
544        int typeArgumentsLength = argumentsBinding == null ? 0 : argumentsBinding.length;
545 
546        // Initialize match rule
547        int matchRule = match.getRule();
548        if (match.isRaw()) {
549                if (patternTypeArgsLength != 0) {
550                        matchRule &= ~SearchPattern.R_FULL_MATCH;
551                }
552        }
553        if (hasTypeParameters) {
554                matchRule = SearchPattern.R_ERASURE_MATCH;
555        }
556 
557        // Compare arguments lengthes
558        if (patternTypeArgsLength == typeArgumentsLength) {
559                if (!match.isRaw() && hasTypeParameters) {
560                        // generic patterns are always not compatible match
561                        match.setRule(SearchPattern.R_ERASURE_MATCH);
562                        return;
563                }
564        } else {
565                if (patternTypeArgsLength==0) {
566                        if (!match.isRaw() || hasTypeParameters) {
567                                match.setRule(matchRule & ~SearchPattern.R_FULL_MATCH);
568                        }
569                } else  if (typeArgumentsLength==0) {
570                        // raw binding is always compatible
571                        match.setRule(matchRule & ~SearchPattern.R_FULL_MATCH);
572                } else {
573                        match.setRule(0); // impossible match
574                }
575                return;
576        }
577        if (argumentsBinding == null || patternArguments == null) {
578                match.setRule(matchRule);
579                return;
580        }
581 
582        // Compare binding for each type argument only if pattern is not erasure only and at first level
583        if (!hasTypeParameters && !match.isRaw() && (match.isEquivalent() || match.isExact())) {
584                for (int i=0; i<typeArgumentsLength; i++) {
585                        // Get parameterized type argument binding
586                        TypeBinding argumentBinding = argumentsBinding[i];
587                        if (argumentBinding instanceof CaptureBinding) {
588                                WildcardBinding capturedWildcard = ((CaptureBinding)argumentBinding).wildcard;
589                                if (capturedWildcard != null) argumentBinding = capturedWildcard;
590                        }
591                        // Get binding for pattern argument
592                        char[] patternTypeArgument = patternArguments[i];
593                        char patternWildcard = patternTypeArgument[0];
594                        char[] patternTypeName = patternTypeArgument;
595                        
596                        patternTypeName = Signature.toCharArray(patternTypeName);
597                        TypeBinding patternBinding = locator.getType(patternTypeArgument, patternTypeName);
598 
599                        // If have no binding for pattern arg, then we won't be able to refine accuracy
600                        if (patternBinding == null) {
601                                if (argumentBinding.isWildcard()) {
602                                        WildcardBinding wildcardBinding = (WildcardBinding) argumentBinding;
603                                        if (wildcardBinding.boundKind == Wildcard.UNBOUND) {
604                                                matchRule &= ~SearchPattern.R_FULL_MATCH;
605                                        } else {
606                                                match.setRule(SearchPattern.R_ERASURE_MATCH);
607                                                return;
608                                        }
609                                }
610                                continue;
611                        }
612 
613                        // Verify tha pattern binding is compatible with match type argument binding
614                        switch (patternWildcard) {
615                                default:
616                                        if (argumentBinding.isWildcard()) {
617                                                WildcardBinding wildcardBinding = (WildcardBinding) argumentBinding;
618                                                switch (wildcardBinding.boundKind) {
619                                                        case Wildcard.EXTENDS:
620                                                                if (wildcardBinding.bound== null || patternBinding.isCompatibleWith(wildcardBinding.bound)) {
621                                                                        // valid only when arg extends a superclass of pattern
622                                                                        matchRule &= ~SearchPattern.R_FULL_MATCH;
623                                                                        continue;
624                                                                }
625                                                                break;
626                                                        case Wildcard.SUPER:
627                                                                if (wildcardBinding.bound== null || wildcardBinding.bound.isCompatibleWith(patternBinding)) {
628                                                                        // valid only when arg super a subclass of pattern
629                                                                        matchRule &= ~SearchPattern.R_FULL_MATCH;
630                                                                        continue;
631                                                                }
632                                                                break;
633                                                        case Wildcard.UNBOUND:
634                                                                matchRule &= ~SearchPattern.R_FULL_MATCH;
635                                                                continue;
636                                                }
637                                        } else if (argumentBinding == patternBinding)
638                                                // valid only when arg is equals to pattern
639                                                continue;
640                                        break;
641                        }
642 
643                        // Argument does not match => erasure match will be the only possible one
644                        match.setRule(SearchPattern.R_ERASURE_MATCH);
645                        return;
646                }
647        }
648 
649        // Set match rule
650        match.setRule(matchRule);
651}

675                return ACCURATE_MATCH;

679        int level = resolveLevelForType(qualifiedPattern, binding);

714/**

718 * Returns IMPOSSIBLE_MATCH if it doesn't.
719 */
720protected int resolveLevelForType(char[] qualifiedPattern, TypeBinding type) {
721        if (qualifiedPattern == null) return ACCURATE_MATCH;
722        if (type == null) return INACCURATE_MATCH;
723 
724        // Type variable cannot be specified through pattern => this kind of binding cannot match it (see bug 79803)
725        if (type.isTypeVariable()) return IMPOSSIBLE_MATCH;

732        {
733                filePath=CharOperation.subarray(qualifiedPattern, 0, index);
734                qualifiedPattern=CharOperation.subarray(qualifiedPattern, index+1, qualifiedPattern.length);
735                bindingPath=type.getFileName();
736                index=CharOperation.lastIndexOf('/', bindingPath);
737                if (index>-1)
738                        bindingPath=CharOperation.subarray(bindingPath, 0, index);
739        }
740        
741        char[] qualifiedPackageName = type.qualifiedPackageName();
742        char[] qualifiedSourceName = qualifiedSourceName(type);
743        char[] fullyQualifiedTypeName = qualifiedPackageName.length == 0
744                ? qualifiedSourceName
745                : CharOperation.concat(qualifiedPackageName, qualifiedSourceName, '.');
746        if (CharOperation.match(qualifiedPattern, fullyQualifiedTypeName, this.isCaseSensitive))
747        {
748                if (filePath.length>0)
749                {
750                   return (CharOperation.endsWith(bindingPath, filePath)) ? ACCURATE_MATCH:IMPOSSIBLE_MATCH;        
751                }
752                return ACCURATE_MATCH;
753        }
754        return IMPOSSIBLE_MATCH;
755}
756/* (non-Javadoc)
757 * Resolve level for type with a given binding with all pattern information.
758 */
759protected int resolveLevelForType (char[] simpleNamePattern,
760                                                                        char[] qualificationPattern,
761                                                                        char[][][] patternTypeArguments,
762                                                                        int depth,
763                                                                        TypeBinding type) {
764        // standard search with no generic additional information must succeed
765        int level = resolveLevelForType(simpleNamePattern, qualificationPattern, type);
766        if (level == IMPOSSIBLE_MATCH) return IMPOSSIBLE_MATCH;
767        if (type == null || patternTypeArguments == null || patternTypeArguments.length == 0 || depth >= patternTypeArguments.length) {
768                return level;
769        }
770 
771        // if pattern is erasure match (see bug 79790), commute impossible to erasure
772        int impossible = this.isErasureMatch ? ERASURE_MATCH : IMPOSSIBLE_MATCH;
773 
774        // pattern has type parameter(s) or type argument(s)
775        if (type.isGenericType()) {
776                // Binding is generic, get its type variable(s)
777                TypeVariableBinding[] typeVariables = null;
778                if (type instanceof SourceTypeBinding) {
779                        SourceTypeBinding sourceTypeBinding = (SourceTypeBinding) type;
780                        typeVariables = sourceTypeBinding.typeVariables;
781                } else if (type instanceof BinaryTypeBinding) {
782                        BinaryTypeBinding binaryTypeBinding = (BinaryTypeBinding) type;
783                        if (this.mustResolve)
784                                typeVariables = binaryTypeBinding.typeVariables(); // TODO (frederic) verify performance
785                }
786                if (patternTypeArguments[depth] != null && patternTypeArguments[depth].length > 0 &&
787                        typeVariables != null && typeVariables.length > 0) {
788                        if (typeVariables.length != patternTypeArguments[depth].length) return IMPOSSIBLE_MATCH;
789                }
790                // TODO (frederic) do we need to verify each parameter?
791                return level; // we can't do better
792        } else if (type.isRawType()) {
793                return level; // raw type always match
794        } else {
795                TypeBinding leafType = type.leafComponentType();
796                if (!leafType.isParameterizedType()) {
797                        // Standard types (ie. neither generic nor parameterized nor raw types)
798                        // cannot match pattern with type parameters or arguments
799                        return (patternTypeArguments[depth]==null || patternTypeArguments[depth].length==0) ? level : IMPOSSIBLE_MATCH;
800                }
801                ParameterizedTypeBinding paramTypeBinding = (ParameterizedTypeBinding) leafType;
802 
803                // Compare arguments only if there ones on both sides
804                if (patternTypeArguments[depth] != null && patternTypeArguments[depth].length > 0 &&
805                        paramTypeBinding.arguments != null && paramTypeBinding.arguments.length > 0) {
806 
807                        // type parameters length must match at least specified type names length
808                        int length = patternTypeArguments[depth].length;
809                        if (paramTypeBinding.arguments.length != length) return IMPOSSIBLE_MATCH;
810 
811                        // verify each pattern type parameter
812                        nextTypeArgument: for (int i= 0; i<length; i++) {
813                                char[] patternTypeArgument = patternTypeArguments[depth][i];
814                                TypeBinding argTypeBinding = paramTypeBinding.arguments[i];
815 
816                                // get pattern type argument from its signature
817                                patternTypeArgument = Signature.toCharArray(patternTypeArgument);
818                                if (!this.isCaseSensitive) patternTypeArgument = CharOperation.toLowerCase(patternTypeArgument);
819                                boolean patternTypeArgHasAnyChars = CharOperation.contains(new char[] {'*', '?'}, patternTypeArgument);
820 
821                                // Verify that names match...
822                                // ...special case for wildcard
823                                if (argTypeBinding instanceof CaptureBinding) {
824                                        WildcardBinding capturedWildcard = ((CaptureBinding)argTypeBinding).wildcard;
825                                        if (capturedWildcard != null) argTypeBinding = capturedWildcard;
826                                }
827                                if (argTypeBinding.isWildcard()) {
828                                        WildcardBinding wildcardBinding = (WildcardBinding) argTypeBinding;
829                                        switch (wildcardBinding.boundKind) {
830                                                case Wildcard.EXTENDS:
831                                                        // Invalid if type argument is not exact
832                                                        if (patternTypeArgHasAnyChars) return impossible;
833                                                case Wildcard.UNBOUND:
834                                                        // there's no bound name to match => valid
835                                                        continue nextTypeArgument;
836                                        }
837                                        // Look if bound name match pattern type argument
838                                        ReferenceBinding boundBinding = (ReferenceBinding) wildcardBinding.bound;
839                                        if (CharOperation.match(patternTypeArgument, boundBinding.shortReadableName(), this.isCaseSensitive) ||
840                                                CharOperation.match(patternTypeArgument, boundBinding.readableName(), this.isCaseSensitive)) {
841                                                // found name in hierarchy => match
842                                                continue nextTypeArgument;
843                                        }
844 
845                                        // If pattern is not exact then match fails
846                                        if (patternTypeArgHasAnyChars) return impossible;
847 
848                                        // Look for bound name in type argument superclasses
849                                        boundBinding = boundBinding.superclass();
850                                        while (boundBinding != null) {
851                                                if (CharOperation.equals(patternTypeArgument, boundBinding.shortReadableName(), this.isCaseSensitive) ||
852                                                        CharOperation.equals(patternTypeArgument, boundBinding.readableName(), this.isCaseSensitive)) {
853                                                        // found name in hierarchy => match
854                                                        continue nextTypeArgument;
855                                                } else if (boundBinding.isLocalType() || boundBinding.isMemberType()) {
856                                                        // for local or member type, verify also source name (bug 81084)
857                                                        if (CharOperation.match(patternTypeArgument, boundBinding.sourceName(), this.isCaseSensitive))
858                                                                continue nextTypeArgument;
859                                                }
860                                                boundBinding = boundBinding.superclass();
861                                        }
862                                        return impossible;
863                                }
864 
865                                // See if names match
866                                if (CharOperation.match(patternTypeArgument, argTypeBinding.shortReadableName(), this.isCaseSensitive) ||
867                                        CharOperation.match(patternTypeArgument, argTypeBinding.readableName(), this.isCaseSensitive)) {
868                                        continue nextTypeArgument;
869                                } else if (argTypeBinding.isLocalType() || argTypeBinding.isMemberType()) {
870                                        // for local or member type, verify also source name (bug 81084)
871                                        if (CharOperation.match(patternTypeArgument, argTypeBinding.sourceName(), this.isCaseSensitive))
872                                                continue nextTypeArgument;
873                                }
874 
875                                // If pattern is not exact then match fails
876                                if (patternTypeArgHasAnyChars) return impossible;
877 
878                                // Scan hierarchy
879                                TypeBinding leafTypeBinding = argTypeBinding.leafComponentType();
880                                if (leafTypeBinding.isBaseType()) return impossible;
881                                ReferenceBinding refBinding = ((ReferenceBinding) leafTypeBinding).superclass();
882                                while (refBinding != null) {
883                                        if (CharOperation.equals(patternTypeArgument, refBinding.shortReadableName(), this.isCaseSensitive) ||
884                                                CharOperation.equals(patternTypeArgument, refBinding.readableName(), this.isCaseSensitive)) {
885                                                // found name in hierarchy => match
886                                                continue nextTypeArgument;
887                                        } else if (refBinding.isLocalType() || refBinding.isMemberType()) {
888                                                // for local or member type, verify also source name (bug 81084)
889                                                if (CharOperation.match(patternTypeArgument, refBinding.sourceName(), this.isCaseSensitive))
890                                                        continue nextTypeArgument;
891                                        }
892                                        refBinding = refBinding.superclass();
893                                }
894                                return impossible;
895                        }
896                }
897 
898                // Recurse on enclosing type
899                TypeBinding enclosingType = paramTypeBinding.enclosingType();
900                if (enclosingType != null && enclosingType.isParameterizedType() && depth < patternTypeArguments.length && qualificationPattern != null) {
901                        int lastDot = CharOperation.lastIndexOf('.', qualificationPattern);
902                        char[] enclosingQualificationPattern = lastDot==-1 ? null : CharOperation.subarray(qualificationPattern, 0, lastDot);
903                        char[] enclosingSimpleNamePattern = lastDot==-1 ? qualificationPattern : CharOperation.subarray(qualificationPattern, lastDot+1, qualificationPattern.length);
904                        int enclosingLevel = resolveLevelForType(enclosingSimpleNamePattern, enclosingQualificationPattern, patternTypeArguments, depth+1, enclosingType);
905                        if (enclosingLevel == impossible) return impossible;
906                        if (enclosingLevel == IMPOSSIBLE_MATCH) return IMPOSSIBLE_MATCH;
907                }
908                return level;
909        }
