647                 */
648                public void resourceChanged(IResourceChangeEvent event) {
649 
650                        if (isInitializing())
651                                return;
652 
653                        // ignore resource changes if already rebuilding
654                        if (getIndexState() == S_REBUILDING)
655                                return;
656                        // previously canceled, needs entire index rebuild
657                        if (getIndexState() == S_CANCELED) {
658                                // rebuildIndex();
659                                // just resume indexing
660                                getProcessFilesJob().schedule(500);
661                                //System.out.println("schedule: resource changed, previously canceled");
662                                return;
663                        }
664 
665                        IResourceDelta delta = event.getDelta();
666                        if (delta != null) {
667                                // only care about adds or changes right now...
668                                int kind = delta.getKind();
669                                boolean added = (kind & IResourceDelta.ADDED) == IResourceDelta.ADDED;
670                                boolean changed = (kind & IResourceDelta.CHANGED) == IResourceDelta.CHANGED;
671                                if (added || changed) {
672 
673                                        // only analyze the full (starting at root) delta
674                                        // hierarchy
675                                        if (delta.getFullPath().toString().equals("/")) { //$NON-NLS-1$
676                                                try {
677                                                        JSPResourceVisitor v = getVisitor();
678                                                        // clear from last run
679                                                        v.reset();
680                                                        // count files, possibly do this in a job too...
681                                                        // don't include PHANTOM resources
682                                                        delta.accept(v, false);
683 
684                                                        // process files from this delta
685                                                        IFile[] files = v.getFiles();
686                                                        if (files.length > 0) {
687                                                                /*
688                                                                 * Job change listener should set back to
689                                                                 * stable when finished
690                                                                 */
691                                                                setUpdatingState();
692                                                                // processFiles(files);
693                                                                indexFiles(files);
694                                                        }
695                                                }
696                                                catch (CoreException e) {
697                                                        // need to set state here somehow, and reindex
698                                                        // otherwise index will be unreliable
699                                                        if (DEBUG)
700                                                                Logger.logException(e);
701                                                }

703                                                        // need to set state here somehow, and reindex
