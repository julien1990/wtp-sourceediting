691                public void resourceChanged(IResourceChangeEvent event) {
692 
693                        if (isInitializing()) {
694                                return;
695                        }
696 
697                        // ignore resource changes if already rebuilding
698                        if (getIndexState() == S_REBUILDING) {
699                                return;
700                        }
701                        // previously canceled, needs entire index rebuild
702                        if (getIndexState() == S_CANCELED) {
703                                // rebuildIndex();
704                                // just resume indexing
705                                getProcessFilesJob().schedule(500);
706                                //System.out.println("schedule: resource changed, previously canceled");

708                        }
709 
710                        IResourceDelta delta = event.getDelta();
711                        if (delta != null) {
712                                // only care about adds or changes right now...
713                                int kind = delta.getKind();
714                                boolean added = (kind & IResourceDelta.ADDED) == IResourceDelta.ADDED;
715                                boolean changed = (kind & IResourceDelta.CHANGED) == IResourceDelta.CHANGED;
716                                if (added || changed) {
717 
718                                        // only analyze the full (starting at root) delta
719                                        // hierarchy
720                                        if (delta.getFullPath().toString().equals("/")) { //$NON-NLS-1$
721                                                try {
