416        /**
417         * Originally from ReconcileStepForJava.  Creates an ICompilationUnit from the contents of the JSP document.
418         * 
419         * @return an ICompilationUnit from the contents of the JSP document
420         */
421        private ICompilationUnit createCompilationUnit() throws JavaModelException {
422                
423                IPackageFragment packageFragment = null;
424                IJavaElement je = getJavaProject();
425 
426                if (je == null || !je.exists())
427                        return null;
428 
429                switch (je.getElementType()) {
430                        case IJavaElement.PACKAGE_FRAGMENT :
431                                je = je.getParent();
432                        // fall through
433 
434                        case IJavaElement.PACKAGE_FRAGMENT_ROOT :
435                                IPackageFragmentRoot packageFragmentRoot = (IPackageFragmentRoot) je;
436                                packageFragment = packageFragmentRoot.getPackageFragment(IPackageFragmentRoot.DEFAULT_PACKAGEROOT_PATH);
437                                break;
438 
439                        case IJavaElement.JAVA_PROJECT :
440                                IJavaProject jProject = (IJavaProject) je;
441 
442                                if (!jProject.exists()) {
443                                        if(DEBUG) {
444                                                System.out.println("** Abort create working copy: cannot create working copy: JSP is not in a Java project"); //$NON-NLS-1$
445                                        }
446                                        return null;
447                                }
448 
449                                packageFragmentRoot = null;
450                                IPackageFragmentRoot[] packageFragmentRoots = jProject.getPackageFragmentRoots();
451                                int i = 0;
452                                while (i < packageFragmentRoots.length) {
453                                        if (!packageFragmentRoots[i].isArchive() && !packageFragmentRoots[i].isExternal()) {
454                                                packageFragmentRoot = packageFragmentRoots[i];
455                                                break;
456                                        }
457                                        i++;
458                                }
459                                if (packageFragmentRoot == null) {
460                                        if(DEBUG) {
461                                                System.out.println("** Abort create working copy: cannot create working copy: JSP is not in a Java project with source package fragment root"); //$NON-NLS-1$
462                                        }
463                                        return null;
464                                }
465                                packageFragment = packageFragmentRoot.getPackageFragment(IPackageFragmentRoot.DEFAULT_PACKAGEROOT_PATH);
466                                break;
467 
468                        default :
469                                return null;
470                }
471                
472                ICompilationUnit cu = packageFragment.getCompilationUnit(getClassname() + ".java").getWorkingCopy(getWorkingCopyOwner(), getProgressMonitor()); //$NON-NLS-1$
473                setContents(cu);
474 
475                if(DEBUG) {
476                        System.out.println("+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++"); //$NON-NLS-1$
477                        System.out.println("(+) JSPTranslation ["+ this + "] finished creating CompilationUnit: " + cu); //$NON-NLS-1$ //$NON-NLS-2$
478                        System.out.println("+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++"); //$NON-NLS-1$
479                }
480                
481                return cu;
482        }
