384public IResource getUnderlyingResource() throws JavaScriptModelException {
385        IResource rootResource = this.parent.getUnderlyingResource();
386        if (rootResource == null) {
387                //jar package fragment root that has no associated resource
388                return null;
389        }
390        // the underlying resource may be a folder or a project (in the case that the project folder
391        // is atually the package fragment root)
392        if (rootResource.getType() == IResource.FOLDER || rootResource.getType() == IResource.PROJECT) {
393                IContainer folder = (IContainer) rootResource;
394                String[] segs = this.names;
395                for (int i = 0; i < segs.length; ++i) {
396                        IResource child = folder.findMember(segs[i]);
397                        if (child == null || child.getType() != IResource.FOLDER) {
398                                throw newNotPresentException();
399                        }
400                        folder = (IFolder) child;
401                }
402                return folder;
403        } else {
404                return rootResource;
405        }

423        IJavaScriptElement[] packages= ((IPackageFragmentRoot)getParent()).getChildren();
424        int namesLength = this.names.length;
425        nextPackage: for (int i= 0, length = packages.length; i < length; i++) {
426                String[] otherNames = ((PackageFragment) packages[i]).names;

501public String getAttachedJavadoc(IProgressMonitor monitor) throws JavaScriptModelException {
502        PerProjectInfo projectInfo = JavaModelManager.getJavaModelManager().getPerProjectInfoCheckExistence(this.getJavaScriptProject().getProject());
503        String cachedJavadoc = null;
504        synchronized (projectInfo.javadocCache) {
505                cachedJavadoc = (String) projectInfo.javadocCache.get(this);
506        }
507        if (cachedJavadoc != null) {
508                return cachedJavadoc;
509        }
510        URL baseLocation= getJavadocBaseLocation();
511        if (baseLocation == null) {
512                return null;
513        }
514        StringBuffer pathBuffer = new StringBuffer(baseLocation.toExternalForm());
515 
516        if (!(pathBuffer.charAt(pathBuffer.length() - 1) == '/')) {
517                pathBuffer.append('/');
518        }
519        String packPath= this.getElementName().replace('.', '/');
520        pathBuffer.append(packPath).append('/').append(JavadocConstants.PACKAGE_FILE_NAME);
521 
522        if (monitor != null && monitor.isCanceled()) throw new OperationCanceledException();
523        final String contents = getURLContents(String.valueOf(pathBuffer));
524        if (monitor != null && monitor.isCanceled()) throw new OperationCanceledException();
525        if (contents == null) throw new JavaScriptModelException(new JavaModelStatus(IJavaScriptModelStatusConstants.CANNOT_RETRIEVE_ATTACHED_JSDOC, this));
526        synchronized (projectInfo.javadocCache) {
527                projectInfo.javadocCache.put(this, contents);
528        }
529        return contents;
