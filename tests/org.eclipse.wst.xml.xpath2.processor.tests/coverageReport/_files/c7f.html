292                                if (firstChar == JEM_FIELD || firstChar == JEM_INITIALIZER || firstChar == JEM_METHOD || firstChar == JEM_TYPE || firstChar == JEM_COUNT) {
293                                        token = typeName;
294                                        typeName = ""; //$NON-NLS-1$
295                                } else {
296                                        token = null;
297                                }
298                        } else {
299                                typeName = ""; //$NON-NLS-1$
300                                token = null;
301                        }
302                        JavaElement type = (JavaElement)getType(typeName);
303                        if (token == null) {
304                                return type.getHandleFromMemento(memento, workingCopyOwner);
305                        } else {
306                                return type.getHandleFromMemento(token, memento, workingCopyOwner);
307                        }
308 
309        }
310        return null;
311}
