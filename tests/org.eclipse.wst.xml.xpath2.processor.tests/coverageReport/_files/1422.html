688        /**
689         * Returns a version of <code>descriptor</code> decorated according to
690         * the passed <code>modifier</code> flags.
691         *
692         * @param descriptor the image descriptor to decorate
693         * @param proposal the proposal
694         * @return an image descriptor for a method proposal
695         * @see Flags
696         */
697        private ImageDescriptor decorateImageDescriptor(ImageDescriptor descriptor, CompletionProposal proposal) {
698                int adornments= 0;
699                int flags= proposal.getFlags();
700                int kind= proposal.getKind();
701 
702                if (Flags.isDeprecated(flags))
703                        adornments |= JavaScriptElementImageDescriptor.DEPRECATED;
704 
705                if (kind == CompletionProposal.FIELD_REF || kind == CompletionProposal.METHOD_DECLARATION || kind == CompletionProposal.METHOD_DECLARATION || kind == CompletionProposal.METHOD_NAME_REFERENCE || kind == CompletionProposal.METHOD_REF)
706                        if (Flags.isStatic(flags))
707                                adornments |= JavaScriptElementImageDescriptor.STATIC;
708 
709                if (kind == CompletionProposal.TYPE_REF && Flags.isAbstract(flags))
710                        adornments |= JavaScriptElementImageDescriptor.ABSTRACT;
711 
712                return new JavaScriptElementImageDescriptor(descriptor, adornments, JavaElementImageProvider.SMALL_SIZE);
713        }
