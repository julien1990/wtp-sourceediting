428                                String unprefixedName = DOMNamespaceHelper.getUnprefixedName(attrName);
429                                if ((XMLNS.equals(prefix)) || (XMLNS.equals(unprefixedName))) {

442                                        // else check if xmlns publicId = value

472                        return true;

476                if ((XSI_NAMESPACE_URI.equals(DOMNamespaceHelper.getNamespaceURI(attr))) && ((SCHEMA_LOCATION.equals(unprefixedName)) || (NO_NAMESPACE_SCHEMA_LOCATION.equals(unprefixedName)))) {
477                        return true;
478                }
479 
480                // determine if attribute value is of type URI
481                if (cmElement != null) {
482                        CMNamedNodeMap attrDecls = cmElement.getAttributes();
483                        CMNamedNodeMapImpl allAttributes = new CMNamedNodeMapImpl(attrDecls);
484                        List nodes = ModelQueryUtil.getModelQuery(attr.getOwnerDocument()).getAvailableContent(attr.getOwnerElement(), cmElement, ModelQuery.INCLUDE_ATTRIBUTES);
485                        for (int k = 0; k < nodes.size(); k++) {
486                                CMNode cmnode = (CMNode) nodes.get(k);
487                                if (cmnode.getNodeType() == CMNode.ATTRIBUTE_DECLARATION) {

494                        if ((attrDecl != null) && (attrDecl.getAttrType() != null) && (CMDataType.URI.equals(attrDecl.getAttrType().getDataTypeName()))) {
495                                return true;
496                        }
497                }
498                return false;
499        }
