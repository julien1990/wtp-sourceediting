660        private static void addAllCus(HashSet/*<IJavaScriptUnit>*/ collector, IJavaScriptElement javaElement) throws JavaScriptModelException {
661                switch (javaElement.getElementType()) {
662                        case IJavaScriptElement.JAVASCRIPT_PROJECT:
663                                IJavaScriptProject javaProject= (IJavaScriptProject) javaElement;
664                                IPackageFragmentRoot[] packageFragmentRoots= javaProject.getPackageFragmentRoots();
665                                for (int i= 0; i < packageFragmentRoots.length; i++)
666                                        addAllCus(collector, packageFragmentRoots[i]);
667                                return;
668                
669                        case IJavaScriptElement.PACKAGE_FRAGMENT_ROOT:
670                                IPackageFragmentRoot packageFragmentRoot= (IPackageFragmentRoot) javaElement;
671                                if (packageFragmentRoot.getKind() != IPackageFragmentRoot.K_SOURCE)
672                                        return;
673                                IJavaScriptElement[] packageFragments= packageFragmentRoot.getChildren();
674                                for (int j= 0; j < packageFragments.length; j++)
675                                        addAllCus(collector, packageFragments[j]);
676                                return;
677                
678                        case IJavaScriptElement.PACKAGE_FRAGMENT:
679                                IPackageFragment packageFragment= (IPackageFragment) javaElement;
680                                collector.addAll(Arrays.asList(packageFragment.getJavaScriptUnits()));
681                                return;
682                        
683                        case IJavaScriptElement.JAVASCRIPT_UNIT:
684                                collector.add(javaElement);
685                                return;
686                                
687                        default:
688                                IJavaScriptElement cu= javaElement.getAncestor(IJavaScriptElement.JAVASCRIPT_UNIT);
689                                if (cu != null)
690                                        collector.add(cu);
691                }
692        }
