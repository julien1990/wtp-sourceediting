382        private boolean canRunOn(IField[] fields) throws JavaScriptModelException {
383                if (fields == null || fields.length == 0)
384                        return false;
385                int count= 0;
386                for (int index= 0; index < fields.length; index++) {
387                        if (!hasPrimitiveType(fields[index]) || isArray(fields[index]))
388                                count++;

390                if (count == 0)
391                        MessageDialog.openInformation(getShell(), DIALOG_TITLE, ActionMessages.AddDelegateMethodsAction_not_applicable); 
392                return (count > 0);
393        }

403        private IField[] getSelectedFields(IStructuredSelection selection) {
404                List elements= selection.toList();
405                if (elements.size() > 0) {
406                        IField[] result= new IField[elements.size()];
407                        IJavaScriptUnit unit= null;
408                        for (int index= 0; index < elements.size(); index++) {
409                                if (elements.get(index) instanceof IField) {
410                                        IField field= (IField) elements.get(index);
411 
412                                        if (index == 0) {
413                                                // remember the CU of the first element
414                                                unit= field.getJavaScriptUnit();
415                                                if (unit == null) {
416                                                        return null;
417                                                }
418                                        } else if (!unit.equals(field.getJavaScriptUnit())) {
419                                                // all fields must be in the same CU
420                                                return null;
421                                        }
422 
423                                        result[index]= field;
424                                } else {
425                                        return null;
426                                }
427                        }
428                        return result;
429                }
430                return null;
431        }

447                                run(JavaElementUtil.getMainType((IJavaScriptUnit) firstElement), new IField[0], false);
448                        else if (!(firstElement instanceof IField))
449                                MessageDialog.openInformation(getShell(), DIALOG_TITLE, ActionMessages.AddDelegateMethodsAction_not_applicable); 
450                } catch (CoreException e) {

459        public void run(ITextSelection selection) {
460                try {
461                        if (!ActionUtil.isProcessable(fEditor))
462                                return;
463 
464                        IJavaScriptElement[] elements= SelectionConverter.codeResolveForked(fEditor, true);
465                        if (elements.length == 1 && (elements[0] instanceof IField)) {
466                                IField field= (IField) elements[0];
467                                run(field.getDeclaringType(), new IField[] { field}, true);
468                                return;
469                        }
470                        IJavaScriptElement element= SelectionConverter.getElementAtOffset(fEditor);
471                        if (element != null) {
472                                IType type= (IType) element.getAncestor(IJavaScriptElement.TYPE);
473                                if (type != null) {
474                                        if (type.getFields().length > 0) {
475                                                run(type, new IField[0], true);
476                                                return;
477                                        }
478                                }
479                        }
480                        MessageDialog.openInformation(getShell(), DIALOG_TITLE, ActionMessages.AddDelegateMethodsAction_not_applicable); 
481                } catch (CoreException e) {
482                        ExceptionHandler.handle(e, getShell(), DIALOG_TITLE, ActionMessages.AddDelegateMethodsAction_error_actionfailed); 
483                } catch (InvocationTargetException e) {
484                        ExceptionHandler.handle(e, getShell(), DIALOG_TITLE, ActionMessages.AddDelegateMethodsAction_error_actionfailed); 
485                } catch (InterruptedException e) {
486                        // cancelled
487                }

524        private void showUI(IType type, IField[] fields) {
525                try {
526                        AddDelegateMethodsContentProvider provider= new AddDelegateMethodsContentProvider(type, fields);
527                        SourceActionDialog dialog= new AddDelegateMethodsDialog(getShell(), new AddDelegateMethodsLabelProvider(), provider, fEditor, type, false);
528                        dialog.setValidator(new AddDelegateMethodsActionStatusValidator(provider.getCount()));
529                        AddDelegateMethodsViewerComparator comparator= new AddDelegateMethodsViewerComparator();
530                        dialog.setComparator(comparator);
531                        dialog.setInput(new Object());
532                        dialog.setContainerMode(true);
533                        dialog.setMessage(ActionMessages.AddDelegateMethodsAction_message); 
534                        dialog.setTitle(ActionMessages.AddDelegateMethodsAction_title); 
535                        IVariableBinding[] expanded= provider.getExpandedElements();
536                        if (expanded.length > 0) {
537                                dialog.setExpandedElements(expanded);
538                        } else {
539                                Object[] elements= provider.getElements(null);
540                                if (elements.length > 0) {
541                                        comparator.sort(null, elements);
542                                        Object[] expand= { elements[0]};
543                                        dialog.setExpandedElements(expand);
544                                }
545                        }
546                        dialog.setInitialSelections(provider.getInitiallySelectedElements());
547                        dialog.setSize(60, 18);
548                        int result= dialog.open();
549                        if (result == Window.OK) {
550                                Object[] object= dialog.getResult();
551                                if (object == null) {
552                                        notifyResult(false);
553                                        return;
554                                }
555                                List tuples= new ArrayList(object.length);
556                                for (int index= 0; index < object.length; index++) {
557                                        if (object[index] instanceof IBinding[])
558                                                tuples.add(object[index]);
559                                }
560                                IEditorPart part= JavaScriptUI.openInEditor(type);
561                                IRewriteTarget target= (IRewriteTarget) part.getAdapter(IRewriteTarget.class);
562                                try {
563                                        if (target != null)
564                                                target.beginCompoundChange();
565                                        CodeGenerationSettings settings= JavaPreferencesSettings.getCodeGenerationSettings(type.getJavaScriptProject());
566                                        settings.createComments= dialog.getGenerateComment();
567                                        final int size= tuples.size();
568                                        String[] methodKeys= new String[size];
569                                        String[] variableKeys= new String[size];
570                                        for (int index= 0; index < size; index++) {
571                                                final IBinding[] tuple= (IBinding[]) tuples.get(index);
572                                                variableKeys[index]= tuple[0].getKey();
573                                                methodKeys[index]= tuple[1].getKey();
574                                        }
575                                        AddDelegateMethodsOperation operation= new AddDelegateMethodsOperation(type, dialog.getElementPosition(), provider.getCompilationUnit(), variableKeys, methodKeys, settings, true, false);
576                                        IRunnableContext context= JavaScriptPlugin.getActiveWorkbenchWindow();
577                                        if (context == null)
578                                                context= new BusyIndicatorRunnableContext();
579                                        try {
580                                                PlatformUI.getWorkbench().getProgressService().runInUI(context, new WorkbenchRunnableAdapter(operation, operation.getSchedulingRule()), operation.getSchedulingRule());
581                                        } catch (InterruptedException exception) {
582                                                // User interruption
583                                        }
584                                } finally {
585                                        if (target != null)
586                                                target.endCompoundChange();
587                                }
588                        }
589                        notifyResult(result == Window.OK);
590                } catch (CoreException exception) {
591                        ExceptionHandler.handle(exception, DIALOG_TITLE, ActionMessages.AddDelegateMethodsAction_error_actionfailed); 
592                } catch (InvocationTargetException e) {
593                        ExceptionHandler.handle(e, DIALOG_TITLE, ActionMessages.AddDelegateMethodsAction_error_actionfailed); 
594                }
595        }
