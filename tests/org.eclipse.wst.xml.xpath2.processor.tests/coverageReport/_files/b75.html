465public static final int
466  CAN_ONLY_NULL_NON_NULL = 0x0000,
467          // check against null and non null, with definite values -- comparisons
468  CAN_ONLY_NULL = 0x0001,
469          // check against null, with definite values -- comparisons

471        // check against non null, with definite values -- comparisons
472  MAY_NULL = 0x0003,

474  CHECK_MASK = 0x00FF,

484/**
485 * Record a null reference for use by deferred checks. Only looping or
486 * finally contexts really record that information. The context may
487 * emit an error immediately depending on the status of local against
488 * flowInfo and its nature (only looping of finally contexts defer part
489 * of the checks; nonetheless, contexts that are nested into a looping or a
490 * finally context get affected and delegate some checks to their enclosing
491 * context).
492 * @param scope the scope into which the check is performed
493 * @param local the local variable involved in the check
494 * @param reference the expression within which local lies
495 * @param checkType the status against which the check must be performed; one
496 *                 of {@link #CAN_ONLY_NULL CAN_ONLY_NULL}, {@link #CAN_ONLY_NULL_NON_NULL
497 *                 CAN_ONLY_NULL_NON_NULL}, {@link #MAY_NULL MAY_NULL}, potentially
498 *      combined with a context indicator (one of {@link #IN_COMPARISON_NULL},
499 *      {@link #IN_COMPARISON_NON_NULL}, {@link #IN_ASSIGNMENT} or {@link #IN_INSTANCEOF})
500 * @param flowInfo the flow info at the check point; deferring contexts will
501 *          perform supplementary checks against flow info instances that cannot
502 *          be known at the time of calling this method (they are influenced by
503 *                 code that follows the current point)
504 */
505public void recordUsingNullReference(Scope scope, LocalVariableBinding local,
506                Expression reference, int checkType, FlowInfo flowInfo) {
507        if ((flowInfo.tagBits & FlowInfo.UNREACHABLE) != 0 ||
508                        flowInfo.isDefinitelyUnknown(local)) {
509                return;
510        }
511        switch (checkType) {
512                case CAN_ONLY_NULL_NON_NULL | IN_COMPARISON_NULL:
513                case CAN_ONLY_NULL_NON_NULL | IN_COMPARISON_NON_NULL:
514                        if (flowInfo.isDefinitelyNonNull(local)) {
515                                if (checkType == (CAN_ONLY_NULL_NON_NULL | IN_COMPARISON_NON_NULL)) {
516                                        scope.problemReporter().localVariableRedundantCheckOnNonNull(local, reference);
517                                } else {
518                                        scope.problemReporter().localVariableNonNullComparedToNull(local, reference);
519                                }
520                                return;
521                        }
522                        else if (flowInfo.cannotBeDefinitelyNullOrNonNull(local)) {
523                                return;
524                        }
525                case CAN_ONLY_NULL | IN_COMPARISON_NULL:
526                case CAN_ONLY_NULL | IN_COMPARISON_NON_NULL:
527                case CAN_ONLY_NULL | IN_ASSIGNMENT:
528                case CAN_ONLY_NULL | IN_INSTANCEOF:
529                        if (flowInfo.isDefinitelyNull(local)) {
530                                switch(checkType & CONTEXT_MASK) {
531                                        case FlowContext.IN_COMPARISON_NULL:
532                                                scope.problemReporter().localVariableRedundantCheckOnNull(local, reference);
533                                                return;
534                                        case FlowContext.IN_COMPARISON_NON_NULL:
535                                                scope.problemReporter().localVariableNullComparedToNonNull(local, reference);
536                                                return;
537                                        case FlowContext.IN_ASSIGNMENT:
538                                                scope.problemReporter().localVariableRedundantNullAssignment(local, reference);
539                                                return;
540                                        case FlowContext.IN_INSTANCEOF:
541                                                scope.problemReporter().localVariableNullInstanceof(local, reference);
542                                                return;
543                                }
544                        } else if (flowInfo.cannotBeDefinitelyNullOrNonNull(local)) {
545                                return;
546                        }
547                        break;
548                case MAY_NULL :
549                        if (flowInfo.isDefinitelyNull(local)) {
550                                scope.problemReporter().localVariableNullReference(local, reference);
551                                return;
552                        }
553                        if (flowInfo.isPotentiallyNull(local)) {
554                                scope.problemReporter().localVariablePotentialNullReference(local, reference);
555                                return;
556                        }
557                        break;
558                default:
559                        // never happens
560        }
561        if (this.parent != null) {
562                this.parent.recordUsingNullReference(scope, local, reference, checkType,
563                                flowInfo);
564        }
565}

575public String toString() {
576        StringBuffer buffer = new StringBuffer();
577        FlowContext current = this;
578        int parentsCount = 0;
579        while ((current = current.parent) != null) {
580                parentsCount++;
581        }
582        FlowContext[] parents = new FlowContext[parentsCount + 1];
583        current = this;
584        int index = parentsCount;
585        while (index >= 0) {
586                parents[index--] = current;
587                current = current.parent;
588        }
589        for (int i = 0; i < parentsCount; i++) {
590                for (int j = 0; j < i; j++)
591                        buffer.append('\t');
592                buffer.append(parents[i].individualToString()).append('\n');
593        }
594        buffer.append('*');
595        for (int j = 0; j < parentsCount + 1; j++)
596                buffer.append('\t');
597        buffer.append(individualToString()).append('\n');
598        return buffer.toString();
599}
