152                                        case IIncludePathEntry.CPE_CONTAINER:
153                                                res.add(JavaScriptCore.newContainerEntry(path, isExported));
154                                                break;
155                                }                                                                
156                        } catch (NumberFormatException e) {

168        public static String encodeJRELibrary(String desc, IIncludePathEntry[] cpentries) {
169                StringBuffer buf= new StringBuffer();
170                for (int i= 0; i < cpentries.length; i++) {
171                        IIncludePathEntry entry= cpentries[i];
172                        buf.append(encode(desc));
173                        buf.append(' ');
174                        buf.append(entry.getEntryKind());
175                        buf.append(' ');
176                        buf.append(encodePath(entry.getPath()));
177                        buf.append(' ');
178                        buf.append(encodePath(entry.getSourceAttachmentPath()));
179                        buf.append(' ');
180                        buf.append(encodePath(entry.getSourceAttachmentRootPath()));
181                        buf.append(' ');
182                        buf.append(entry.isExported());
183                        buf.append(' ');
184                }
185                return buf.toString();
186        }

188        private static String encodePath(IPath path) {
189                if (path == null) {
190                        return "#"; //$NON-NLS-1$
191                } else if (path.isEmpty()) {
192                        return "&"; //$NON-NLS-1$
193                } else {
194                        return encode(path.toPortableString());
195                }
196        }

198        private static IPath decodePath(String str) {
199                if ("#".equals(str)) { //$NON-NLS-1$
200                        return null;
201                } else if ("&".equals(str)) { //$NON-NLS-1$
202                        return Path.EMPTY;
203                } else {
204                        return Path.fromPortableString(decode(str));
205                }

395                boolean useFolders= fFoldersAsSourceFolder.getSelection();

397                fSrcFolderNameText.setEnabled(useFolders);

411                        IStatus status;

413                        if (srcName.length() != 0) {
414                                status= workspace.validatePath(srcPath.toString(), IResource.FOLDER);
415                                if (!status.isOK()) {
416                                        String message= Messages.format(PreferencesMessages.NewJavaProjectPreferencePage_folders_error_invalidsrcname, status.getMessage());

420                        }

462                IPreferenceStore store= getPreferenceStore();
463                for (int i= 0; i < fCheckBoxes.size(); i++) {
464                        Button button= (Button) fCheckBoxes.get(i);
465                        String key= (String) button.getData();
466                        button.setSelection(store.getDefaultBoolean(key));
467                }
468                for (int i= 0; i < fRadioButtons.size(); i++) {
469                        Button button= (Button) fRadioButtons.get(i);
470                        String[] info= (String[]) button.getData();
471                        button.setSelection(info[1].equals(store.getDefaultString(info[0])));
472                }
473                for (int i= 0; i < fTextControls.size(); i++) {
474                        Text text= (Text) fTextControls.get(i);
475                        String key= (String) text.getData();
476                        text.setText(store.getDefaultString(key));
477                }
478                if (fJRECombo != null) {
479                        fJRECombo.select(store.getDefaultInt(CLASSPATH_JRELIBRARY_INDEX));
480                }
481                
482                validateFolders();
483                super.performDefaults();
484        }
