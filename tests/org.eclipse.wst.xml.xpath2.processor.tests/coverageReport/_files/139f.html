395                                        String name = names[j];
396                                        File jar = new File(extDir, name);

398                                                int length = name.length();
399                                                if (length > 4) {
400                                                        String suffix = name.substring(length - 4);

402                                                                try {

407                                                                        Util.log(e, ""); //$NON-NLS-1$
408                                                                }
409                                                        }
410                                                }
