582         * Creates an image descriptor for the given path in a bundle. The path can contain variables
583         * like $NL$.
584         * If no image could be found, <code>useMissingImageDescriptor</code> decides if either
585         * the 'missing image descriptor' is returned or <code>null</code>.
586         * Added for 3.1.1.
587         */
588        public static ImageDescriptor createImageDescriptor(Bundle bundle, IPath path, boolean useMissingImageDescriptor) {
589                URL url= FileLocator.find(bundle, path, null);
590                if (url != null) {
591                        return ImageDescriptor.createFromURL(url);
592                }
593                if (useMissingImageDescriptor) {
594                        return ImageDescriptor.getMissingImageDescriptor();
595                }
596                return null;
597        }
