609         *        from

611         *        spaces to tabs

622

628                                        spaces++;

630                                                ret.append('\t');
631                                                spaces= 0;
632                                        }

634                                        ret.append(' ');

638                        }

640                        while (spaces-- > 0)
641                                ret.append(' ');

644                }
645

647        }

733         *
734         * <p>If <code>peekNextChar</code> is <code>true</code>, the next token after
735         * <code>offset</code> is read and taken into account when computing the
736         * indentation. Currently, if the next token is the first token on the line
737         * (i.e. only preceded by whitespace), the following tokens are specially
738         * handled:
739         * <ul>
740         *         <li><code>switch</code> labels are indented relative to the switch block</li>
741         *         <li>opening curly braces are aligned correctly with the introducing code</li>

744         *         <li>closing parenthesis' are aligned with their opening peer</li>
745         *         <li>the <code>else</code> keyword is aligned with its <code>if</code>, anything
746         *                 else is aligned normally (i.e. with the base of any introducing statements).</li>
747         *  <li>if there is no token on the same line after <code>offset</code>, the indentation
748         *                 is the same as for an <code>else</code> keyword</li>
749         * </ul>
750         *
751         * @param offset the offset for which the reference is computed
752         * @param nextToken the next token to assume in the document
753         * @return the reference statement relative to which <code>offset</code>
754         *         should be indented, or {@link JavaHeuristicScanner#NOT_FOUND}

765                // if they are on a line by themselves, the indentation gets adjusted
766                // accordingly
767                //
768                // also account for a dangling else
769                if (offset < fDocument.getLength()) {
770                        try {
771                                IRegion line= fDocument.getLineInformationOfOffset(offset);
772                                int lineOffset= line.getOffset();
773                                int prevPos= Math.max(offset - 1, 0);
774                                boolean isFirstTokenOnLine= fDocument.get(lineOffset, prevPos + 1 - lineOffset).trim().length() == 0;
775                                int prevToken= fScanner.previousToken(prevPos, JavaHeuristicScanner.UNBOUND);
776                                boolean bracelessBlockStart= fScanner.isBracelessBlockStart(prevPos, JavaHeuristicScanner.UNBOUND);
777 
778                                switch (nextToken) {
779                                        case Symbols.TokenELSE:
