271public void checkComment() {
272        // discard obsolete comments while inside methods or fields initializer (see bug 74369)
273        if (!(this.diet && this.dietInt==0) && this.scanner.commentPtr >= 0) {
274                flushCommentsDefinedPriorTo(this.endStatementPosition);
275        }
276 
277        int lastComment = this.scanner.commentPtr;
278 
279        if (this.modifiersSourceStart >= 0) {
280                // eliminate comments located after modifierSourceStart if positionned
281                while (lastComment >= 0 && Math.abs(this.scanner.commentStarts[lastComment]) > this.modifiersSourceStart) lastComment--;
282        }
283        if (lastComment >= 0) {
284                // consider all remaining leading comments to be part of current declaration
285                this.modifiersSourceStart = Math.abs(this.scanner.commentStarts[0]);
286 
287                // check deprecation in last comment if javadoc (can be followed by non-javadoc comments which are simply ignored)
288                while (lastComment >= 0 && this.scanner.commentStops[lastComment] < 0) lastComment--; // non javadoc comment have negative end positions
289                if (lastComment >= 0 && this.javadocParser != null) {
290                        int commentEnd = this.scanner.commentStops[lastComment] - 1; //stop is one over,
291                        // do not report problem before last parsed comment while recovering code...
292                        this.javadocParser.reportProblems = this.currentElement == null || commentEnd > this.lastJavadocEnd;
293                        if (this.javadocParser.checkDeprecation(lastComment)) {
294                                checkAndSetModifiers(ClassFileConstants.AccDeprecated);
295                        }
296                        this.javadoc = this.javadocParser.docComment;        // null if check javadoc is not activated
297                        if (currentElement == null) this.lastJavadocEnd = commentEnd;
298                }
299        }
300 
301        if (this.reportReferenceInfo && this.javadocParser.checkDocComment && this.javadoc != null) {
302                // Report reference info in javadoc comment @throws/@exception tags
303                TypeReference[] thrownExceptions = this.javadoc.exceptionReferences;
304                if (thrownExceptions != null) {
305                        for (int i = 0, max=thrownExceptions.length; i < max; i++) {
306                                TypeReference typeRef = thrownExceptions[i];
307                                if (typeRef instanceof JavadocSingleTypeReference) {
308                                        JavadocSingleTypeReference singleRef = (JavadocSingleTypeReference) typeRef;
309                                        this.requestor.acceptTypeReference(singleRef.token, singleRef.sourceStart);
310                                } else if (typeRef instanceof JavadocQualifiedTypeReference) {
311                                        JavadocQualifiedTypeReference qualifiedRef = (JavadocQualifiedTypeReference) typeRef;
312                                        this.requestor.acceptTypeReference(qualifiedRef.tokens, qualifiedRef.sourceStart, qualifiedRef.sourceEnd);
313                                }
314                        }
315                }
316 
317                // Report reference info in javadoc comment @see tags
318                Expression[] references = this.javadoc.seeReferences;
319                if (references != null) {
320                        for (int i = 0, max=references.length; i < max; i++) {
321                                Expression reference = references[i];
322                                acceptJavadocTypeReference(reference);
323                                if (reference instanceof JavadocFieldReference) {
324                                        JavadocFieldReference fieldRef = (JavadocFieldReference) reference;
325                                        this.requestor.acceptFieldReference(fieldRef.token, fieldRef.sourceStart);
326                                        if (fieldRef.receiver != null && !fieldRef.receiver.isThis()) {
327                                                acceptJavadocTypeReference(fieldRef.receiver);
328                                        }
329                                } else if (reference instanceof JavadocMessageSend) {
330                                        JavadocMessageSend messageSend = (JavadocMessageSend) reference;
331                                        int argCount = messageSend.arguments == null ? 0 : messageSend.arguments.length;
332                                        this.requestor.acceptMethodReference(messageSend.selector, argCount, messageSend.sourceStart);
333                                        this.requestor.acceptConstructorReference(messageSend.selector, argCount, messageSend.sourceStart);
334                                        if (messageSend.receiver != null && !messageSend.receiver.isThis()) {
335                                                acceptJavadocTypeReference(messageSend.receiver);
336                                        }
337                                } else if (reference instanceof JavadocAllocationExpression) {
338                                        JavadocAllocationExpression constructor = (JavadocAllocationExpression) reference;
339                                        int argCount = constructor.arguments == null ? 0 : constructor.arguments.length;
340                                        if (constructor.type != null) {
341                                                char[][] compoundName = constructor.type.getParameterizedTypeName();
342                                                this.requestor.acceptConstructorReference(compoundName[compoundName.length-1], argCount, constructor.sourceStart);
343                                                if (!constructor.type.isThis()) {
344                                                        acceptJavadocTypeReference(constructor.type);
345                                                }
346                                        }
347                                }
348                        }
349                }
350        }
