357                                        continue;
358                                inliner= new CallInliner(unit, (JavaScriptUnit) bodies[0].getRoot(), fSourceProvider);
359                                for (int b= 0; b < bodies.length; b++) {
360                                        BodyDeclaration body= bodies[b];
361                                        inliner.initialize(body);
362                                        RefactoringStatus nestedInvocations= new RefactoringStatus();
363                                        ASTNode[] invocations= removeNestedCalls(nestedInvocations, unit, 
364                                                fTargetProvider.getInvocations(body, new SubProgressMonitor(sub, 2)));
365                                        for (int i= 0; i < invocations.length; i++) {
366                                                ASTNode invocation= invocations[i];
367                                                result.merge(inliner.initialize(invocation, fTargetProvider.getStatusSeverity()));
368                                                if (result.hasFatalError())
369                                                        break;
370                                                if (result.getSeverity() < fTargetProvider.getStatusSeverity()) {
371                                                        added= true;
372                                                        TextEditGroup group= new TextEditGroup(RefactoringCoreMessages.InlineMethodRefactoring_edit_inline); 
373                                                        change.addTextEditGroup(group);
374                                                        result.merge(inliner.perform(group)); 
375                                                }
376                                        }
377                                        // do this after we have inlined the method calls. We still want
378                                        // to generate the modifications.
379                                        result.merge(nestedInvocations);
380                                }
381                                if (!added) {
382                                        fChangeManager.remove(unit);
383                                } else {
384                                        root.addChild(inliner.getModifications());
385                                        ImportRewrite rewrite= inliner.getImportEdit();
386                                        if (rewrite.hasRecordedChanges()) {
387                                                TextEdit edit= rewrite.rewriteImports(null);
388                                                if (edit instanceof MultiTextEdit ? ((MultiTextEdit)edit).getChildrenSize() > 0 : true) {
389                                                        root.addChild(edit);
390                                                        change.addTextEditGroup(
391                                                                new TextEditGroup(RefactoringCoreMessages.InlineMethodRefactoring_edit_import, new TextEdit[] {edit})); 
392                                                }
393                                        }
394                                }
395                        } finally {
396                                if (inliner != null)
397                                        inliner.dispose();
398                        }
399                        sub.worked(1);
400                        if (sub.isCanceled())
401                                throw new OperationCanceledException();
402                }
403                result.merge(searchStatus);
404                sub.done();
405                pm.done();
406                return result;
407        }

495        private ASTNode[] removeNestedCalls(RefactoringStatus status, IJavaScriptUnit unit, ASTNode[] invocations) {
496                if (invocations.length <= 1)
497                        return invocations;
498                ASTNode[] parents= new ASTNode[invocations.length];
499                for (int i= 0; i < invocations.length; i++) {
500                        parents[i]= invocations[i].getParent();
501                }
502                for (int i= 0; i < invocations.length; i++) {
503                        removeNestedCalls(status, unit, parents, invocations, i);
504                }
505                List result= new ArrayList();
506                for (int i= 0; i < invocations.length; i++) {
507                        if (invocations[i] != null)
508                                result.add(invocations[i]);
509                }
510                return (ASTNode[])result.toArray(new ASTNode[result.size()]);
511        }

513        private void removeNestedCalls(RefactoringStatus status, IJavaScriptUnit unit, ASTNode[] parents, ASTNode[] invocations, int index) {
514                ASTNode invocation= invocations[index];
515                for (int i= 0; i < parents.length; i++) {
516                        ASTNode parent= parents[i];
517                        while (parent != null) {
518                                if (parent == invocation) {
519                                        status.addError(RefactoringCoreMessages.InlineMethodRefactoring_nestedInvocation,  
520                                                JavaStatusContext.create(unit, parent));
521                                        invocations[index]= null;
522                                }
523                                parent= parent.getParent();
524                        }
525                }

528        public RefactoringStatus initialize(final RefactoringArguments arguments) {
529                if (arguments instanceof JavaRefactoringArguments) {
530                        final JavaRefactoringArguments generic= (JavaRefactoringArguments) arguments;
531                        final String value= generic.getAttribute(ATTRIBUTE_MODE);
532                        if (value != null && !"".equals(value)) {//$NON-NLS-1$

542                                        return RefactoringStatus.createFatalErrorStatus(exception.getLocalizedMessage());
543                                }
544                        }
545                } else
546                        return RefactoringStatus.createFatalErrorStatus(RefactoringCoreMessages.InitializableRefactoring_inacceptable_arguments);
547                return new RefactoringStatus();
