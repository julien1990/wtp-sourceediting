350                if (declarationBinding != null) {
351                        final int modifiers= declarationBinding.getModifiers();
352                        if (Modifier.isAbstract(modifiers)) {
353                                result.addFatalError(RefactoringCoreMessages.InlineMethodRefactoring_SourceAnalyzer_abstract_methods, JavaStatusContext.create(fTypeRoot, fDeclaration));
354                                return result;
355                        } else if (Modifier.isNative(modifiers)) {
356                                result.addFatalError(RefactoringCoreMessages.InlineMethodRefactoring_SourceAnalyzer_native_methods, JavaStatusContext.create(fTypeRoot, fDeclaration));
357                                return result;
358                        }
359                } else {
360                        result.addFatalError(RefactoringCoreMessages.InlineMethodRefactoring_SourceAnalyzer_methoddeclaration_has_errors, JavaStatusContext.create(fTypeRoot));
361                        return result;
362                }
363                ActivationAnalyzer analyzer= new ActivationAnalyzer();
364                fDeclaration.accept(analyzer);
365                result.merge(analyzer.status);
366                if (!result.hasFatalError()) {
367                        List parameters= fDeclaration.parameters();
368                        fParameters= new HashMap(parameters.size() * 2);
369                        for (Iterator iter= parameters.iterator(); iter.hasNext();) {
370                                SingleVariableDeclaration element= (SingleVariableDeclaration) iter.next();
371                                IVariableBinding binding= element.resolveBinding();
372                                if (binding == null) {
373                                        result.addFatalError(
374                                                RefactoringCoreMessages.InlineMethodRefactoring_SourceAnalyzer_declaration_has_errors, 
375                                                JavaStatusContext.create(fTypeRoot, fDeclaration));
376                                        return result;
377                                }
378                                fParameters.put(binding, element.getProperty(ParameterData.PROPERTY));
379                        }
380                        fNames= new HashMap();
381                        fImplicitReceivers= new ArrayList(2);
382                        
383                        fTypeParameterReferences= new ArrayList(0);
384                        fTypeParameterMapping= new HashMap();
385                        ITypeBinding declaringType= declarationBinding.getDeclaringClass();
386                        if (declaringType == null) {
387                                result.addFatalError(
388                                        RefactoringCoreMessages.InlineMethodRefactoring_SourceAnalyzer_typedeclaration_has_errors, 
389                                        JavaStatusContext.create(fTypeRoot));
390                                return result;
391                        }
392                        ITypeBinding[] typeParameters= declaringType.getTypeParameters();
393                        for (int i= 0; i < typeParameters.length; i++) {
394                                NameData data= new NameData(typeParameters[i].getName());
395                                fTypeParameterReferences.add(data);
396                                fTypeParameterMapping.put(typeParameters[i], data);
397                        }
398                        
399                        fMethodTypeParameterReferences= new ArrayList(0);
400                        fMethodTypeParameterMapping= new HashMap();
401                        IFunctionBinding method= declarationBinding;
402                        typeParameters= method.getTypeParameters();
403                        for (int i= 0; i < typeParameters.length; i++) {
404                                NameData data= new NameData(typeParameters[i].getName());
405                                fMethodTypeParameterReferences.add(data);
406                                fMethodTypeParameterMapping.put(typeParameters[i], data);
407                        }
408                        
409                }
410                if (fDeclaration.isVarargs()) {
411                        List parameters= fDeclaration.parameters();
412                        VarargAnalyzer vAnalyzer= new VarargAnalyzer(
413                                ((SingleVariableDeclaration)parameters.get(parameters.size() - 1)).getName().resolveBinding());
414                        fDeclaration.getBody().accept(vAnalyzer);
415                }
416                return result;
417        }

419        public void initialize() {
420                Block body= fDeclaration.getBody();
421                // first collect the static imports. This is necessary to not mark
422                // static imported fields and methods as implicit visible.
423                fTypesToImport= new ArrayList();
424                fStaticsToImport= new ArrayList();
425                ImportReferencesCollector collector= new ImportReferencesCollector(
426                        fTypeRoot.getJavaScriptProject(), null, fTypesToImport, fStaticsToImport);
427                body.accept(collector);
428                
429                // Now collect implicit references and name references
430                body.accept(new UpdateCollector());
431                
432                int numberOfLocals= LocalVariableIndex.perform(fDeclaration);
433                FlowContext context= new FlowContext(0, numberOfLocals + 1);
434                context.setConsiderAccessMode(true);
435                context.setComputeMode(FlowContext.MERGE);
436                InOutFlowAnalyzer flowAnalyzer= new InOutFlowAnalyzer(context);
437                FlowInfo info= flowAnalyzer.perform(getStatements());
438                
439                for (Iterator iter= fDeclaration.parameters().iterator(); iter.hasNext();) {
440                        SingleVariableDeclaration element= (SingleVariableDeclaration) iter.next();
441                        IVariableBinding binding= element.resolveBinding();
442                        ParameterData data= (ParameterData)element.getProperty(ParameterData.PROPERTY);
443                        data.setAccessMode(info.getAccessMode(context, binding));
444                }
445        }
