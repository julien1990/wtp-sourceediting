431        protected void processDelta(IJavaScriptElementDelta delta) throws JavaScriptModelException {
432 
433                int kind = delta.getKind();
434                final IJavaScriptElement element = delta.getElement();
435 
436                if (isClassPathChange(delta)) {
437                        Object input= fViewer.getInput();
438                        if (input != null) {
439                                if (fInputIsProject && input.equals(element.getJavaScriptProject())) {
440                                        postRefresh(input);
441                                        return;
442                                } else if (!fInputIsProject && input.equals(element)) {
443                                        if (element.exists())
444                                                postRefresh(input);
445                                        else
446                                                postRemove(input);
447                                        return;
448                                }
449                        }
450                }
451 
452                if (kind == IJavaScriptElementDelta.REMOVED) {
453                        Object input= fViewer.getInput();
454                        if (input != null && input.equals(element)) {
455                                        postRemove(input);
456                                        return;
457                                }
458                }
459 
460                if (element instanceof IPackageFragment) {
461                        final IPackageFragment frag = (IPackageFragment) element;
462 
463                        //if fragment was in LogicalPackage refresh,
464                        //otherwise just remove
465                        if (kind == IJavaScriptElementDelta.REMOVED) {
466                                removeElement(frag);
467                                return;
468 
469                        } else if (kind == IJavaScriptElementDelta.ADDED) {
470 
471                                Object parent= getParent(frag);
472                                addElement(frag, parent);
473                                return;
474 
475                        } else if (kind == IJavaScriptElementDelta.CHANGED) {
476                                //just refresh
477                                LogicalPackage logicalPkg= findLogicalPackage(frag);
478                                //in case changed object is filtered out
479                                if (logicalPkg != null)
480                                        postRefresh(findElementToRefresh(logicalPkg));
481                                else
482                                        postRefresh(findElementToRefresh(frag));
483                                return;
484                        }
485                }
486 
487                processAffectedChildren(delta);
488        }

491                Object toBeRefreshed= object;
492                if (fViewer.testFindItem(object) == null) {
493                         Object parent= getParent(object);
494                         if(parent instanceof IPackageFragmentRoot && fInputIsProject)
495                                 parent= ((IPackageFragmentRoot)parent).getJavaScriptProject();
496 
497                        if(parent != null)
498                                toBeRefreshed= parent;
499                }
500                return toBeRefreshed;

561        private void addElement(IPackageFragment frag, Object parent) {
562 
563                String key= getKey(frag);
564                LogicalPackage lp= (LogicalPackage)fMapToLogicalPackage.get(key);
565 
566                //if fragment must be added to an existing LogicalPackage
567                if (lp != null && lp.belongs(frag)){
568                        lp.add(frag);
569                        return;
570                }
571 
572                //if a new LogicalPackage must be created
573                IPackageFragment iPackageFragment= (IPackageFragment)fMapToPackageFragments.get(key);
574                if (iPackageFragment!= null && !iPackageFragment.equals(frag)){
575                        lp= new LogicalPackage(iPackageFragment);
576                        lp.add(frag);
577                        //add new LogicalPackage to LogicalPackages map
578                        fMapToLogicalPackage.put(key, lp);
579 
580                        //determine who to refresh
581                        if (parent instanceof IPackageFragmentRoot){
582                                IPackageFragmentRoot root= (IPackageFragmentRoot) parent;
583                                if (fInputIsProject){
584                                        postRefresh(root.getJavaScriptProject());
585                                } else {
586                                        postRefresh(root);
587                                }
588                        } else {
589                                //@Improve: Should this be replaced by a refresh?
590                                postAdd(lp, parent);
591                                postRemove(iPackageFragment);
592                        }
593 
594                }
595                //if this is a new Package Fragment
596                else {
597                        fMapToPackageFragments.put(key, frag);
598 
599                        //determine who to refresh
600                        if (parent instanceof IPackageFragmentRoot) {
601                                IPackageFragmentRoot root= (IPackageFragmentRoot) parent;
602                                if (fInputIsProject) {
603                                        postAdd(frag, root.getJavaScriptProject());
604                                } else
605                                        postAdd(frag, root);
606                        } else {
607                                postAdd(frag, parent);
608                        }
609                }
610        }

612        private void removeElement(IPackageFragment frag) {
613 
614                String key= getKey(frag);
615                LogicalPackage lp= (LogicalPackage)fMapToLogicalPackage.get(key);
616 
617                if(lp != null){
618                        lp.remove(frag);
619                        //if the LogicalPackage needs to revert back to a PackageFragment
620                        //remove it from the LogicalPackages map and add the PackageFragment
621                        //to the PackageFragment map
622                        if (lp.getFragments().length == 1) {
623                                IPackageFragment fragment= lp.getFragments()[0];
624                                fMapToPackageFragments.put(key, fragment);
625                                fMapToLogicalPackage.remove(key);
626 
627                                //remove the LogicalPackage from viewer
628                                postRemove(lp);
629 
630                                Object parent= getParent(fragment);
631                                if (parent instanceof IPackageFragmentRoot) {
632                                        parent= ((IPackageFragmentRoot)parent).getJavaScriptProject();
633                                }
634                                postAdd(fragment, parent);
635                        }
636 
637                } else {
638                        //remove the fragment from the fragment map and viewer
639                        IPackageFragment fragment= (IPackageFragment) fMapToPackageFragments.get(key);
640                        if (fragment!= null && fragment.equals(frag)) {
641                                fMapToPackageFragments.remove(key);
642                                postRemove(frag);
643                        }
644                }
645        }
