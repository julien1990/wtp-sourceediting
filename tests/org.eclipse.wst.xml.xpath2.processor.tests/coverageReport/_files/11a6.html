376                ISourceViewer viewer= fEditor.getViewer();
377                final IDocument document= viewer.getDocument();
378                
379                try {
380                        if (! fOriginalName.equals(newName)) {
381                                fEditor.getSite().getWorkbenchWindow().run(false, true, new IRunnableWithProgress() {
382                                        public void run(IProgressMonitor monitor) throws InvocationTargetException, InterruptedException {
383                                                LinkedPosition[] positions= fLinkedPositionGroup.getPositions();
384                                                Arrays.sort(positions, new Comparator() {
385                                                        public int compare(Object o1, Object o2) {
386                                                                return ((LinkedPosition) o1).offset - ((LinkedPosition) o2).offset;
387                                                        }
388                                                });
389                                                int correction= 0;
390                                                int originalLength= fOriginalName.length();
391                                                for (int i= 0; i < positions.length; i++) {
392                                                        LinkedPosition position= positions[i];
393                                                        try {
394                                                                int length= position.getLength();
395                                                                document.replace(position.getOffset() + correction, length, fOriginalName);
396                                                                correction= correction - length + originalLength;
397                                                        } catch (BadLocationException e) {
398                                                                throw new InvocationTargetException(e);
399                                                        }
400                                                }
401                                                if (fOriginalSaved) {
402                                                        fEditor.doSave(monitor); // started saved -> end saved
403                                                }
404                                        }
405                                });
406                        }
407                } catch (InvocationTargetException e) {
408                        throw new CoreException(new Status(IStatus.ERROR, JavaScriptPlugin.getPluginId(), ReorgMessages.RenameLinkedMode_error_saving_editor, e));
409                } catch (InterruptedException e) {
410                        // cancelling is OK
411                        return null;
412                } finally {
413                        JavaModelUtil.reconcile(getCompilationUnit());
414                }
415                
416                viewer.setSelectedRange(fOriginalSelection.x, fOriginalSelection.y);
417                
418                RenameJavaScriptElementDescriptor descriptor= createRenameDescriptor(fJavaElement, newName);
419                RenameSupport renameSupport= RenameSupport.create(descriptor);
420                return renameSupport;
421        }

443         * @param javaElement
444         * @param newName
445         * @return a rename descriptor with current settings as used in the refactoring dialogs 
446         * @throws JavaScriptModelException
447         */
448        private RenameJavaScriptElementDescriptor createRenameDescriptor(IJavaScriptElement javaElement, String newName) throws JavaScriptModelException {
449                String contributionId;
450                // see RefactoringExecutionStarter#createRenameSupport(..):
451                int elementType= javaElement.getElementType();
452                switch (elementType) {
453                        case IJavaScriptElement.JAVASCRIPT_PROJECT:
454                                contributionId= IJavaScriptRefactorings.RENAME_JAVA_PROJECT;
455                                break;
456                        case IJavaScriptElement.PACKAGE_FRAGMENT_ROOT:
457                                contributionId= IJavaScriptRefactorings.RENAME_SOURCE_FOLDER;
458                                break;
459                        case IJavaScriptElement.PACKAGE_FRAGMENT:
460                                contributionId= IJavaScriptRefactorings.RENAME_PACKAGE;
461                                break;
462                        case IJavaScriptElement.JAVASCRIPT_UNIT:
463                                contributionId= IJavaScriptRefactorings.RENAME_JAVASCRIPT_UNIT;
464                                break;
465                        case IJavaScriptElement.TYPE:
466                                contributionId= IJavaScriptRefactorings.RENAME_TYPE;
467                                break;
468                        case IJavaScriptElement.METHOD:
469                                final IFunction method= (IFunction) javaElement;
470                                if (method.isConstructor())
471                                        return createRenameDescriptor(method.getDeclaringType(), newName);
472                                else
473                                        contributionId= IJavaScriptRefactorings.RENAME_METHOD;
474                                break;
475                        case IJavaScriptElement.FIELD:
476                                contributionId= IJavaScriptRefactorings.RENAME_FIELD;
477                                break;
478                        case IJavaScriptElement.LOCAL_VARIABLE:
479                                contributionId= IJavaScriptRefactorings.RENAME_LOCAL_VARIABLE;
480                                break;
481                        default:
482                                return null;
483                }
484                
485                RenameJavaScriptElementDescriptor descriptor= (RenameJavaScriptElementDescriptor) RefactoringCore.getRefactoringContribution(contributionId).createDescriptor();
486                descriptor.setJavaElement(javaElement);
487                descriptor.setNewName(newName);
488                if (elementType != IJavaScriptElement.PACKAGE_FRAGMENT_ROOT)
489                        descriptor.setUpdateReferences(true);
490                
491                IDialogSettings javaSettings= JavaScriptPlugin.getDefault().getDialogSettings();
492                IDialogSettings refactoringSettings= javaSettings.getSection(RefactoringWizardPage.REFACTORING_SETTINGS); //TODO: undocumented API
493                if (refactoringSettings == null) {
494                        refactoringSettings= javaSettings.addNewSection(RefactoringWizardPage.REFACTORING_SETTINGS); 
495                }
496                
497                switch (elementType) {
498                        case IJavaScriptElement.METHOD:
499                        case IJavaScriptElement.FIELD:
500                                descriptor.setDeprecateDelegate(refactoringSettings.getBoolean(DelegateUIHelper.DELEGATE_DEPRECATION));
501                                descriptor.setKeepOriginal(refactoringSettings.getBoolean(DelegateUIHelper.DELEGATE_UPDATING));
502                }
503                switch (elementType) {
504                        case IJavaScriptElement.TYPE:
505//                        case IJavaScriptElement.JAVASCRIPT_UNIT: // TODO
506                                descriptor.setUpdateSimilarDeclarations(refactoringSettings.getBoolean(RenameRefactoringWizard.TYPE_UPDATE_SIMILAR_ELEMENTS));
507                                int strategy;
508                                try {
509                                        strategy= refactoringSettings.getInt(RenameRefactoringWizard.TYPE_SIMILAR_MATCH_STRATEGY);
510                                } catch (NumberFormatException e) {
511                                        strategy= RenamingNameSuggestor.STRATEGY_EXACT;
512                                }
513                                descriptor.setMatchStrategy(strategy);
514                }
515                switch (elementType) {
516                        case IJavaScriptElement.PACKAGE_FRAGMENT:
517                                descriptor.setUpdateHierarchy(refactoringSettings.getBoolean(RenameRefactoringWizard.PACKAGE_RENAME_SUBPACKAGES));
518                }
519                switch (elementType) {
520                        case IJavaScriptElement.PACKAGE_FRAGMENT:
521                        case IJavaScriptElement.TYPE:
522                                String fileNamePatterns= refactoringSettings.get(RenameRefactoringWizard.QUALIFIED_NAMES_PATTERNS);
523                                if (fileNamePatterns != null && fileNamePatterns.length() != 0) {
524                                        descriptor.setFileNamePatterns(fileNamePatterns);
525                                        boolean updateQualifiedNames= refactoringSettings.getBoolean(RenameRefactoringWizard.UPDATE_QUALIFIED_NAMES);
526                                        descriptor.setUpdateQualifiedNames(updateQualifiedNames);
527                                        fShowPreview|= updateQualifiedNames;
528                                }
529                }
530                switch (elementType) {
531                        case IJavaScriptElement.PACKAGE_FRAGMENT:
532                        case IJavaScriptElement.TYPE:
533                        case IJavaScriptElement.FIELD:
534                                boolean updateTextualOccurrences= refactoringSettings.getBoolean(RenameRefactoringWizard.UPDATE_TEXTUAL_MATCHES);
535                                descriptor.setUpdateTextualOccurrences(updateTextualOccurrences);
536                                fShowPreview|= updateTextualOccurrences;
537                }
538                switch (elementType) {
539                        case IJavaScriptElement.FIELD:
540                                descriptor.setRenameGetters(refactoringSettings.getBoolean(RenameRefactoringWizard.FIELD_RENAME_GETTER));
541                                descriptor.setRenameSetters(refactoringSettings.getBoolean(RenameRefactoringWizard.FIELD_RENAME_SETTER));
542                }
543                return descriptor;
