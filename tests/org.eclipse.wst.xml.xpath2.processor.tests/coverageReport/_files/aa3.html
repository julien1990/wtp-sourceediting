328                if(this.result && this.parent == null && astNode != this.searchedNode) {
329                        if(!(astNode instanceof AllocationExpression && ((AllocationExpression) astNode).type == this.searchedNode)
330                                && !(astNode instanceof ConditionalExpression && ((ConditionalExpression) astNode).valueIfTrue == this.searchedNode)
331                                && !(astNode instanceof ConditionalExpression && ((ConditionalExpression) astNode).valueIfFalse == this.searchedNode)) {
332                                this.parent = astNode;

334                }
