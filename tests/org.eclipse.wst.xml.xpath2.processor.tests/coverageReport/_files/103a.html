773                }

792        }

794        private void addProxyOptions(List vmOptions) {

796                String hostPrefix= "-J-Dhttp.proxyHost="; //$NON-NLS-1$
797                String portPrefix= "-J-Dhttp.proxyPort="; //$NON-NLS-1$
798                for (int i= 0; i < vmOptions.size(); i++) {

801                                return;
802                        }
803                }

805                if (proxyHost != null) {
806                        vmOptions.add(hostPrefix + proxyHost); 
807                }
808                
809                String proxyPort= System.getProperty("http.proxyPort"); //$NON-NLS-1$
810                if (proxyPort != null) {
811                        vmOptions.add(portPrefix + proxyPort); 
812                }
813        }
