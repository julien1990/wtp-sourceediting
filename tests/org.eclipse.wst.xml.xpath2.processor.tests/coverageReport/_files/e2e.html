285                        private String[] resolveParameterTypes(IFunction method) {
286                                final String[] oldParameterTypes= method.getParameterTypes();
287                                final String[] newparams= new String[oldParameterTypes.length];
288 
289                                final String[] possibleOldSigs= new String[2];
290                                //using type signature, since there is no package signature
291                                possibleOldSigs[0]= Signature.createTypeSignature(fPackage.getElementName(), false);
292                                possibleOldSigs[1]= Signature.createTypeSignature(fPackage.getElementName(), true);
293 
294                                final String[] possibleNewSigs= new String[2];
295                                possibleNewSigs[0]= Signature.createTypeSignature(getNewElementName(), false);
296                                possibleNewSigs[1]= Signature.createTypeSignature(getNewElementName(), true);
297 
298                                // Textually replace all occurrences
299                                // This handles stuff like Map<SomeClass, some.package.SomeClass>
300                                for (int i= 0; i < oldParameterTypes.length; i++) {
301                                        newparams[i]= oldParameterTypes[i];
302                                        for (int j= 0; j < possibleOldSigs.length; j++) {
303                                                newparams[i]= replaceAll(newparams[i], possibleOldSigs[j], possibleNewSigs[j]);
304                                        }
305                                }
306                                return newparams;
307                        }

309                    private String replaceAll(final String source, final String replaceFrom, final String replaceTo) {
310                                final StringBuffer buf= new StringBuffer(source.length());
311                                int currentIndex= 0;
312                                int matchIndex;
313                                while ((matchIndex= source.indexOf(replaceFrom, currentIndex)) != -1) {
314                                        buf.append(source.substring(currentIndex, matchIndex));
315                                        buf.append(replaceTo);
316                                        currentIndex= matchIndex + replaceFrom.length();
317                                }
318                                buf.append(source.substring(currentIndex));
319                                return buf.toString();
320                        }
321                }.transplantHandle(element);
322        }

640                void doRename(IProgressMonitor pm, RefactoringStatus result) throws CoreException {

643                                pm.setTaskName(RefactoringCoreMessages.RenamePackageRefactoring_searching);         
644                                fOccurrences= getReferences(new SubProgressMonitor(pm, 4), result);        
645                                fReferencesToTypesInNamesakes= getReferencesToTypesInNamesakes(new SubProgressMonitor(pm, 4), result);
646                                fReferencesToTypesInPackage= getReferencesToTypesInPackage(new SubProgressMonitor(pm, 4), result);
647                                pm.setTaskName(RefactoringCoreMessages.RenamePackageRefactoring_checking); 
648                                result.merge(analyzeAffectedCompilationUnits());
649                                pm.worked(1);
650                        } else {
651                                fOccurrences= new SearchResultGroup[0];
652                                pm.worked(13);

656                                return;

658                        if (fProcessor.getUpdateReferences())
659                                addReferenceUpdates(new SubProgressMonitor(pm, 3));
