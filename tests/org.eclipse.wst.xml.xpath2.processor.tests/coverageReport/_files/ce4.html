567        Object[] keyTable = this.indexes.keyTable;
568        Object[] valueTable = this.indexes.valueTable;
569        IPath[] locations = null;

571        int count = 0;
572        for (int i = 0, l = keyTable.length; i < l; i++) {

588        }

590                for (int i = 0; i < count; i++)
591                        this.indexes.removeKey(locations[i]);

593        }

640public void saveIndex(Index index) throws IOException {
641        // must have permission to write from the write monitor
642        if (index.hasChanged()) {
643                if (VERBOSE)
644                        Util.verbose("-> saving index " + index.getIndexFile()); //$NON-NLS-1$
645                index.save();
646        }
647        synchronized (this) {
648                IPath containerPath = new Path(index.containerPath);
649                if (this.jobEnd > this.jobStart) {
650                        for (int i = this.jobEnd; i > this.jobStart; i--) { // skip the current job
651                                IJob job = this.awaitingJobs[i];
652                                if (job instanceof IndexRequest)
653                                        if (((IndexRequest) job).containerPath.equals(containerPath)) return;
654                        }
655                }
656                IPath indexLocation = computeIndexLocation(containerPath);
657                updateIndexState(indexLocation, SAVED_STATE);
658        }

660/**
661 * Commit all index memory changes to disk
662 */
663public void saveIndexes() {
664        // only save cached indexes... the rest were not modified
665        ArrayList toSave = new ArrayList();
666        synchronized(this) {
667                Object[] valueTable = this.indexes.valueTable;
668                for (int i = 0, l = valueTable.length; i < l; i++) {
669                        Index index = (Index) valueTable[i];
670                        if (index != null)
671                                toSave.add(index);
672                }
673        }
674 
675        boolean allSaved = true;
676        for (int i = 0, length = toSave.size(); i < length; i++) {
677                Index index = (Index) toSave.get(i);
678                ReadWriteMonitor monitor = index.monitor;
679                if (monitor == null) continue; // index got deleted since acquired
680                try {
681                        // take read lock before checking if index has changed
682                        // don't take write lock yet since it can cause a deadlock (see https://bugs.eclipse.org/bugs/show_bug.cgi?id=50571)
683                        monitor.enterRead();
684                        if (index.hasChanged()) {
685                                if (monitor.exitReadEnterWrite()) {
686                                        try {
687                                                saveIndex(index);
688                                        } catch(IOException e) {
689                                                if (VERBOSE) {
690                                                        Util.verbose("-> got the following exception while saving:", System.err); //$NON-NLS-1$
691                                                        e.printStackTrace();
692                                                }
693                                                allSaved = false;
694                                        } finally {
695                                                monitor.exitWriteEnterRead();
696                                        }
697                                } else {
698                                        allSaved = false;
699                                }
700                        }
701                } finally {
702                        monitor.exitRead();
703                }
704        }
705        this.needToSave = !allSaved;
706}
707public void scheduleDocumentIndexing(final SearchDocument searchDocument, IPath container, final IPath indexLocation, final SearchParticipant searchParticipant) {
708        request(new IndexRequest(container, this) {
709                public boolean execute(IProgressMonitor progressMonitor) {
710                        if (this.isCancelled || progressMonitor != null && progressMonitor.isCanceled()) return true;
711 
712                        /* ensure no concurrent write access to index */
713                        Index index = getIndex(this.containerPath, indexLocation, true, /*reuse index file*/ true /*create if none*/);
714                        if (index == null) return true;
715                        ReadWriteMonitor monitor = index.monitor;
716                        if (monitor == null) return true; // index got deleted since acquired
717 
718                        try {
719                                monitor.enterWrite(); // ask permission to write
720                                indexDocument(searchDocument, searchParticipant, index, indexLocation);
721                        } finally {
722                                monitor.exitWrite(); // free write lock
723                        }
724                        return true;
725                }
726                public String toString() {
727                        return "indexing " + searchDocument.getPath(); //$NON-NLS-1$
728                }
729        });
730}

765private synchronized void removeIndexesState(IPath[] locations) {
766        getIndexStates(); // ensure the states are initialized

776                }
777        }
778        if (!changed) return;

782private synchronized void updateIndexState(IPath indexLocation, Integer indexState) {
783        if (indexLocation.isEmpty())
784                throw new IllegalArgumentException();
785 
786        getIndexStates(); // ensure the states are initialized
787        if (indexState != null) {
788                if (indexState.equals(indexStates.get(indexLocation))) return; // not changed
789                indexStates.put(indexLocation, indexState);
790        } else {
791                if (!indexStates.containsKey(indexLocation)) return; // did not exist anyway
792                indexStates.removeKey(indexLocation);
793        }
794 
795        writeSavedIndexNamesFile();
796 
797        if (VERBOSE) {
798                String state = "?"; //$NON-NLS-1$
799                if (indexState == SAVED_STATE) state = "SAVED"; //$NON-NLS-1$
800                else if (indexState == UPDATING_STATE) state = "UPDATING"; //$NON-NLS-1$
801                else if (indexState == UNKNOWN_STATE) state = "UNKNOWN"; //$NON-NLS-1$
802                else if (indexState == REBUILDING_STATE) state = "REBUILDING"; //$NON-NLS-1$
803                Util.verbose("-> index state updated to: " + state + " for: "+indexLocation); //$NON-NLS-1$ //$NON-NLS-2$
804        }
805}
806private void writeSavedIndexNamesFile() {

815                Object[] states = indexStates.valueTable;

817                        IPath key = (IPath) keys[i];
818                        if (key != null && !key.isEmpty() && states[i] == SAVED_STATE) {
819                                writer.write(key.lastSegment());
820                                writer.write('\n');
821                        }
822                }
823        } catch (IOException ignored) {
824                if (VERBOSE)
825                        Util.verbose("Failed to write saved index file names", System.err); //$NON-NLS-1$
826        } finally {
827                if (writer != null) {
828                        try {
829                                writer.close();
830                        } catch (IOException e) {
831                                // ignore
832                        }
833                }
834        }
