460        private static ImportOrderEntry[] unpackOrderList(String str) {
461                ArrayList res= new ArrayList();
462                int start= 0;
463                do {
464                        int end= str.indexOf(';', start);
465                        if (end == -1) {
466                                end= str.length();
467                        }
468                        res.add(ImportOrderEntry.fromSerialized(str.substring(start, end)));
469                        start= end + 1;
470                } while (start < str.length());
471                
472                return (ImportOrderEntry[]) res.toArray(new ImportOrderEntry[res.size()]);
473        }

475        private static String packOrderList(List orderList) {
476                StringBuffer buf= new StringBuffer();
477                for (int i= 0; i < orderList.size(); i++) {
478                        ImportOrderEntry entry= (ImportOrderEntry) orderList.get(i);
479                        buf.append(entry.serialize());
480                        buf.append(';');
481                }
482                return buf.toString();
483        }
