489                if (selection.isEmpty())
490                        return false;
491                
492                int offset= selection.getOffset();
493                int length= selection.getLength();
494                
495                IDocument document= getDocument();
496                if (document == null)
497                        return false;
498                
499                try {
500                        IRegion firstLine= document.getLineInformationOfOffset(offset);
501                        int lineOffset= firstLine.getOffset();
502                        
503                        // either the selection has to be empty and the caret in the WS at the line start
504                        // or the selection has to extend over multiple lines
505                        if (length == 0)
506                                return document.get(lineOffset, offset - lineOffset).trim().length() == 0;
507                        else
508//                                return lineOffset + firstLine.getLength() < offset + length;
509                                return false; // only enable for empty selections for now
510                        
511                } catch (BadLocationException e) {
512                }

514                return false;
