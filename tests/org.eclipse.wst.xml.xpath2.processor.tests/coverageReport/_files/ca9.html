722        private RewriteEvent[] processListWithRanges(List rangeInfos, RewriteEvent[] childEvents, TargetSourceRangeComputer sourceRangeComputer) {
723                List newChildEvents= new ArrayList(childEvents.length);
724                NodeRangeInfo topInfo= null;
725                Stack newChildrenStack= new Stack();
726                Stack topInfoStack= new Stack();
727 
728                Iterator rangeInfoIterator= rangeInfos.iterator();
729                NodeRangeInfo nextInfo= (NodeRangeInfo) rangeInfoIterator.next();
730 
731                for (int k= 0; k < childEvents.length; k++) {
732                        RewriteEvent event= childEvents[k];
733                        ASTNode node= (ASTNode) event.getOriginalValue();
734                        // check for ranges and add a placeholder for them
735                        while (nextInfo != null && node == nextInfo.getStartNode()) { // is this child the beginning of a range?
736                                nextInfo.updatePlaceholderSourceRanges(sourceRangeComputer);
737 
738                                Block internalPlaceholder= nextInfo.getInternalPlaceholder();
739                                RewriteEvent newEvent;
740                                if (nextInfo.isMove()) {
741                                        newEvent= new NodeRewriteEvent(internalPlaceholder, nextInfo.replacingNode); // remove or replace
742                                } else {
743                                        newEvent= new NodeRewriteEvent(internalPlaceholder, internalPlaceholder); // unchanged
744                                }
745                                newChildEvents.add(newEvent);
746                                if (nextInfo.editGroup != null) {
747                                        setEventEditGroup(newEvent, nextInfo.editGroup);
748                                }
749 
750                                newChildrenStack.push(newChildEvents);
751                                topInfoStack.push(topInfo);
752 
753                                newChildEvents= new ArrayList(childEvents.length);
754                                topInfo= nextInfo;
755 
756                                nextInfo= rangeInfoIterator.hasNext() ? (NodeRangeInfo) rangeInfoIterator.next() : null;
757                        }
758 
759                        newChildEvents.add(event);
760 
761                        while (topInfo != null && node == topInfo.getEndNode()) {
762                                RewriteEvent[] placeholderChildEvents= (RewriteEvent[]) newChildEvents.toArray(new RewriteEvent[newChildEvents.size()]);
763                                Block internalPlaceholder= topInfo.getInternalPlaceholder();
764                                addEvent(internalPlaceholder, Block.STATEMENTS_PROPERTY, new ListRewriteEvent(placeholderChildEvents));
765 
766                                newChildEvents= (List) newChildrenStack.pop();
767                                topInfo= (NodeRangeInfo) topInfoStack.pop();
768                        }
769                }
770                return (RewriteEvent[]) newChildEvents.toArray(new RewriteEvent[newChildEvents.size()]);
771        }
