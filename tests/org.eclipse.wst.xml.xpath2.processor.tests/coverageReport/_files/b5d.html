371public MethodBinding createDefaultConstructorWithBinding(MethodBinding inheritedConstructorBinding) {
372        //Add to method'set, the default constuctor that just recall the
373        //super constructor with the same arguments
374        String baseName = "$anonymous"; //$NON-NLS-1$
375        TypeBinding[] argumentTypes = inheritedConstructorBinding.parameters;
376        int argumentsLength = argumentTypes.length;
377        //the constructor
378        ConstructorDeclaration constructor = new ConstructorDeclaration(this.compilationResult);
379        constructor.selector = new char[] { 'x' }; //no maining
380        constructor.sourceStart = this.sourceStart;
381        constructor.sourceEnd = this.sourceEnd;
382        int newModifiers = this.modifiers & ExtraCompilerModifiers.AccVisibilityMASK;
383        if (inheritedConstructorBinding.isVarargs()) {
384                newModifiers |= ClassFileConstants.AccVarargs;
385        }
386        constructor.modifiers = newModifiers;
387        constructor.bits |= ASTNode.IsDefaultConstructor;
388 
389        if (argumentsLength > 0) {
390                Argument[] arguments = (constructor.arguments = new Argument[argumentsLength]);
391                for (int i = argumentsLength; --i >= 0;) {
392                        arguments[i] = new Argument((baseName + i).toCharArray(), 0L, null /*type ref*/, ClassFileConstants.AccDefault);
393                }
394        }
395        //the super call inside the constructor
396        constructor.constructorCall = SuperReference.implicitSuperConstructorCall();
397        constructor.constructorCall.sourceStart = this.sourceStart;
398        constructor.constructorCall.sourceEnd = this.sourceEnd;
399 
400        if (argumentsLength > 0) {
401                Expression[] args;
402                args = constructor.constructorCall.arguments = new Expression[argumentsLength];
403                for (int i = argumentsLength; --i >= 0;) {
404                        args[i] = new SingleNameReference((baseName + i).toCharArray(), 0L);
405                }
406        }
407 
408        //adding the constructor in the methods list
409        if (this.methods == null) {
410                this.methods = new AbstractMethodDeclaration[] { constructor };
411        } else {
412                AbstractMethodDeclaration[] newMethods;
413                System.arraycopy(this.methods, 0, newMethods = new AbstractMethodDeclaration[this.methods.length + 1], 1, this.methods.length);
414                newMethods[0] = constructor;
415                this.methods = newMethods;
416        }
417 
418        //============BINDING UPDATE==========================
419        SourceTypeBinding sourceType = this.binding;
420        constructor.binding = new MethodBinding(
421                        constructor.modifiers, //methodDeclaration
422                        argumentsLength == 0 ? Binding.NO_PARAMETERS : argumentTypes, //arguments bindings
423                        inheritedConstructorBinding.thrownExceptions, //exceptions
424                        sourceType); //declaringClass
425 
426        constructor.binding.modifiers |= ExtraCompilerModifiers.AccIsDefaultConstructor;
427 
428        constructor.scope = new MethodScope(this.scope, constructor, true);
429        constructor.bindArguments();
430        constructor.constructorCall.resolve(constructor.scope);
431 
432        MethodBinding[] methodBindings = sourceType.methods(); // trigger sorting
433        int length;
434        System.arraycopy(methodBindings, 0, methodBindings = new MethodBinding[(length = methodBindings.length) + 1], 1, length);
435        methodBindings[0] = constructor.binding;
436        if (++length > 1)
437                ReferenceBinding.sortMethods(methodBindings, 0, length);        // need to resort, since could be valid methods ahead (140643) - DOM needs eager sorting
438        sourceType.setMethods(methodBindings);
439        //===================================================
440 
441        return constructor.binding;
442}

517/**
518 *        Common flow analysis for all types
519 */
520private void internalAnalyseCode(FlowContext flowContext, FlowInfo flowInfo) {
521        if ((this.binding.isPrivate() || (this.binding.tagBits & (TagBits.IsAnonymousType|TagBits.IsLocalType)) == TagBits.IsLocalType) && !this.binding.isUsed()) {
522                if (!this.scope.referenceCompilationUnit().compilationResult.hasSyntaxError) {
523                        this.scope.problemReporter().unusedPrivateType(this);
524                }
525        }
526        InitializationFlowContext initializerContext = new InitializationFlowContext(null, this, this.initializerScope);
527        InitializationFlowContext staticInitializerContext = new InitializationFlowContext(null, this, this.staticInitializerScope);
528        FlowInfo nonStaticFieldInfo = flowInfo.unconditionalFieldLessCopy();
529        FlowInfo staticFieldInfo = flowInfo.unconditionalFieldLessCopy();
530        if (this.fields != null) {
531                for (int i = 0, count = this.fields.length; i < count; i++) {
532                        FieldDeclaration field = this.fields[i];
533                        if (field.isStatic()) {
534                                if ((staticFieldInfo.tagBits & FlowInfo.UNREACHABLE) != 0)
535                                        field.bits &= ~ASTNode.IsReachable;
536 
537                                /*if (field.isField()){
538                                        staticInitializerContext.handledExceptions = NoExceptions; // no exception is allowed jls8.3.2
539                                } else {*/
540                                staticInitializerContext.handledExceptions = Binding.ANY_EXCEPTION; // tolerate them all, and record them
541                                /*}*/
542                                staticFieldInfo =
543                                        field.analyseCode(
544                                                this.staticInitializerScope,
545                                                staticInitializerContext,
546                                                staticFieldInfo);
547                                // in case the initializer is not reachable, use a reinitialized flowInfo and enter a fake reachable
548                                // branch, since the previous initializer already got the blame.
549                                if (staticFieldInfo == FlowInfo.DEAD_END) {
550                                        this.staticInitializerScope.problemReporter().initializerMustCompleteNormally(field);
551                                        staticFieldInfo = FlowInfo.initial(this.maxFieldCount).setReachMode(FlowInfo.UNREACHABLE);
552                                }
553                        } else {
554                                if ((nonStaticFieldInfo.tagBits & FlowInfo.UNREACHABLE) != 0)
555                                        field.bits &= ~ASTNode.IsReachable;
556 
557                                /*if (field.isField()){
558                                        initializerContext.handledExceptions = NoExceptions; // no exception is allowed jls8.3.2
559                                } else {*/
560                                        initializerContext.handledExceptions = Binding.ANY_EXCEPTION; // tolerate them all, and record them
561                                /*}*/
562                                nonStaticFieldInfo =
563                                        field.analyseCode(this.initializerScope, initializerContext, nonStaticFieldInfo);
564                                // in case the initializer is not reachable, use a reinitialized flowInfo and enter a fake reachable
565                                // branch, since the previous initializer already got the blame.
566                                if (nonStaticFieldInfo == FlowInfo.DEAD_END) {
567                                        this.initializerScope.problemReporter().initializerMustCompleteNormally(field);
568                                        nonStaticFieldInfo = FlowInfo.initial(this.maxFieldCount).setReachMode(FlowInfo.UNREACHABLE);
569                                }
570                        }
571                }
572        }
573        if (this.memberTypes != null) {
574                for (int i = 0, count = this.memberTypes.length; i < count; i++) {
575                        if (flowContext != null){ // local type
576                                this.memberTypes[i].analyseCode(this.scope, flowContext, nonStaticFieldInfo.copy().setReachMode(flowInfo.reachMode())); // reset reach mode in case initializers did abrupt completely
577                        } else {
578                                this.memberTypes[i].analyseCode(this.scope);
579                        }
580                }
581        }
582        if (this.methods != null) {
583                UnconditionalFlowInfo outerInfo = flowInfo.unconditionalFieldLessCopy();
584                FlowInfo constructorInfo = nonStaticFieldInfo.unconditionalInits().discardNonFieldInitializations().addInitializationsFrom(outerInfo);
585                for (int i = 0, count = this.methods.length; i < count; i++) {
586                        AbstractMethodDeclaration method = this.methods[i];
587                        if (method.ignoreFurtherInvestigation)
588                                continue;
589                        if (method.isInitializationMethod()) {
590                                if (method.isStatic()) { // <clinit>
591                                        method.analyseCode(
592                                                this.scope,
593                                                staticInitializerContext,
594                                                staticFieldInfo.unconditionalInits().discardNonFieldInitializations().addInitializationsFrom(outerInfo));
595                                } else { // constructor
596                                        ((ConstructorDeclaration)method).analyseCode(this.scope, initializerContext, constructorInfo.copy(), flowInfo.reachMode());
597                                }
598                        } else { // regular method
599                                method.analyseCode(this.scope, null, flowInfo.copy());
600                        }
601                }
602        }
603        // enable enum support ?
604        if (this.binding.isEnum() && !this.binding.isAnonymousType()) {
605                this.enumValuesSyntheticfield = this.binding.addSyntheticFieldForEnumValues();
606        }
607}
