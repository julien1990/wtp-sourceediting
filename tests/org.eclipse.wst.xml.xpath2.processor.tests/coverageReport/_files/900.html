576        private String getTypeNameFrom(Type type) {
577                if (type == null) return Util.EMPTY_STRING;
578                switch(type.getNodeType0()) {
579                        case ASTNode.ARRAY_TYPE :
580                                ArrayType arrayType = (ArrayType) type;
581                                type = arrayType.getElementType();
582                                return getTypeNameFrom(type);
583                        case ASTNode.PARAMETERIZED_TYPE :
584                                ParameterizedType parameterizedType = (ParameterizedType) type;
585                                StringBuffer buffer = new StringBuffer(getTypeNameFrom(parameterizedType.getType()));
586                                ITypeBinding[] tArguments = getTypeArguments();
587                                final int typeArgumentsLength = tArguments.length;
588                                if (typeArgumentsLength != 0) {
589                                        buffer.append('<');
590                                        for (int i = 0; i < typeArgumentsLength; i++) {
591                                                if (i > 0) {
592                                                        buffer.append(',');
593                                                }
594                                                buffer.append(tArguments[i].getName());
595                                        }
596                                        buffer.append('>');
597                                }
598                                return String.valueOf(buffer);
599                        case ASTNode.PRIMITIVE_TYPE :
600                                PrimitiveType primitiveType = (PrimitiveType) type;
601                                return primitiveType.getPrimitiveTypeCode().toString();
602                        case ASTNode.QUALIFIED_TYPE :
603                                QualifiedType qualifiedType = (QualifiedType) type;
604                                return qualifiedType.getName().getIdentifier();
605                        case ASTNode.SIMPLE_TYPE :
606                                SimpleType simpleType = (SimpleType) type;
607                                Name name = simpleType.getName();
608                                if (name.isQualifiedName()) {
609                                        QualifiedName qualifiedName = (QualifiedName) name;
610                                        return qualifiedName.getName().getIdentifier();
611                                }
612                                return ((SimpleName) name).getIdentifier();
613                }
614                return Util.EMPTY_STRING;
615        }

617        private Type getType() {
618                if (this.currentType != null) {
619                        return this.currentType;
620                }
621                if (this.variableDeclaration == null) return null;
622                switch(this.variableDeclaration.getNodeType()) {
623                        case ASTNode.SINGLE_VARIABLE_DECLARATION :
624                                SingleVariableDeclaration singleVariableDeclaration = (SingleVariableDeclaration) this.variableDeclaration;
625                                return singleVariableDeclaration.getType();
626                        default :
627                                // this is a variable declaration fragment
628                                ASTNode parent = this.variableDeclaration.getParent();
629                                switch(parent.getNodeType()) {
630                                        case ASTNode.VARIABLE_DECLARATION_EXPRESSION :
631                                                VariableDeclarationExpression variableDeclarationExpression = (VariableDeclarationExpression) parent;
632                                                return variableDeclarationExpression.getType();
633                                        case ASTNode.VARIABLE_DECLARATION_STATEMENT :
634                                                VariableDeclarationStatement statement = (VariableDeclarationStatement) parent;
635                                                return statement.getType();
636                                        case ASTNode.FIELD_DECLARATION :
637                                                FieldDeclaration fieldDeclaration  = (FieldDeclaration) parent;
638                                                return fieldDeclaration.getType();
639                                }
640                }
641                return null; // should not happen
642        }
