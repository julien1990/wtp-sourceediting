404        private void workingSetManagerChanged(PropertyChangeEvent event) {
405                String property= event.getProperty();
406                if (IWorkingSetManager.CHANGE_WORKING_SET_UPDATER_INSTALLED.equals(property) && event.getSource() == fLocalWorkingSetManager) {
407                        IWorkingSetUpdater updater= (IWorkingSetUpdater)event.getNewValue();
408                        if (updater instanceof OthersWorkingSetUpdater) {
409                                fOthersWorkingSetUpdater= (OthersWorkingSetUpdater)updater;
410                                fOthersWorkingSetUpdater.init(this);
411                        }
412                        return;
413                }
414                // don't handle working sets not managed by the model
415                if (!isAffected(event))
416                        return;
417 
418                if (IWorkingSetManager.CHANGE_WORKING_SET_CONTENT_CHANGE.equals(property)) {
419                        IWorkingSet workingSet= (IWorkingSet)event.getNewValue();
420                        IAdaptable[] elements= fElementMapper.refresh(workingSet);
421                        if (elements != null) {
422                                fireEvent(event);
423                        }
424                } else if (IWorkingSetManager.CHANGE_WORKING_SET_REMOVE.equals(property)) {
425                        IWorkingSet workingSet= (IWorkingSet)event.getOldValue();
426                        List elements= new ArrayList(fActiveWorkingSets);
427                        elements.remove(workingSet);
428                        setActiveWorkingSets((IWorkingSet[])elements.toArray(new IWorkingSet[elements.size()]));
429                } else if (IWorkingSetManager.CHANGE_WORKING_SET_NAME_CHANGE.equals(property)) {
430                        fireEvent(event);
431                }
432        }

442                if (fActiveWorkingSets == null)

445                Object newValue= event.getNewValue();
446                if ((oldValue != null && fActiveWorkingSets.contains(oldValue)) 
447                                || (newValue != null && fActiveWorkingSets.contains(newValue))) {
448                        return true;
