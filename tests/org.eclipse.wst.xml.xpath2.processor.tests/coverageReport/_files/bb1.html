524        public MethodBinding[] getMethods(char[] selector) {
525                if (this.methods != null) {
526                        long range;
527                        if ((range = ReferenceBinding.binarySearch(selector, this.methods)) >= 0) {

532                                //        return this.methods; // current set is already interesting subset
533                                MethodBinding[] result;

535                                return result;
536                        }
537                }
538                if ((tagBits & TagBits.AreMethodsComplete) != 0)
539                        return Binding.NO_METHODS; // have created all the methods and there are no matches
540

542                try {
543                    MethodBinding[] originalMethods = this.type.getMethods(selector);

545                    if (length == 0) return Binding.NO_METHODS;

548                    for (int i = 0; i < length; i++)
549                            // substitute methods, so as to get updated declaring class at least

555                    } else {
556                                int total = length + this.methods.length;
557                                MethodBinding[] temp = new MethodBinding[total];
558                                System.arraycopy(parameterizedMethods, 0, temp, 0, length);
559                                System.arraycopy(this.methods, 0, temp, length, this.methods.length);
560                                if (total > 1)
561                                        ReferenceBinding.sortMethods(temp, 0, total); // resort to ensure order is good
562                                this.methods = temp;
563                        }
564                    return parameterizedMethods;
565                } finally {
566                        // if the original methods cannot be retrieved (ex. AbortCompilation), then assume we do not have any methods
567                    if (parameterizedMethods == null)
568                        this.methods = parameterizedMethods = Binding.NO_METHODS;

635                    ParameterizedTypeBinding otherParamType = (ParameterizedTypeBinding) otherType;

645                                    } else {

647                                    }
648                            }
649                    }

651                            return otherParamType.arguments == null;
652                    }
653                    int length = this.arguments.length;
654                    TypeBinding[] otherArguments = otherParamType.arguments;
655                    if (otherArguments == null || otherArguments.length != length) return false;
656                    for (int i = 0; i < length; i++) {
657                            if (!this.arguments[i].isTypeArgumentContainedBy(otherArguments[i]))
658                                    return false;
659                    }
660                    return true;
