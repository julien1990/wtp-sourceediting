523        private Element findDocumentElement(String docName, Node node, Node[] firstFound, int max) {
524                int countSearch = 0;
525                for (Node child = node.getFirstChild(); child != null; child = child.getNextSibling()) {
526                        
527                        /* 
528                         * maxDocTypeSearch limits added via bug 151929
529                         * https://bugs.eclipse.org/bugs/show_bug.cgi?id=151929
530                         * but, in other contexts, 
531                         * if noMaxSearch is specified, then do not "break out" of long searches 
532                         * */
533                        if (max != noMaxSearch && countSearch++ > max) {
534                                break;
535                        }
536                        if (child.getNodeType() != ELEMENT_NODE)
537                                continue;
538                        ElementImpl element = (ElementImpl) child;
539                        if (element.isCommentTag()) {
540                                Element docElement = findDocumentElement(docName, element, firstFound, max);
541                                if (docElement != null) {
542                                        return docElement;
543                                }
544                                else {
545                                        // added 'else continue' to better handle cases where
546                                        // there is "more than one root" element
547                                        // especially complicated by CommentElements, which are
548                                        // sometimes treated as elements, but should
549                                        // be treated as comments in this context.
550                                        continue;
551                                }
552                        }
553                        // note: the "name" won't match in the event of a jsp tag ... but
554                        // incase
555                        // the name is null, we do not want the jsp element returned as
556                        // documentElement
557                        if (element.isJSPTag())
558                                continue;
559                        if (docName == null)
560                                return element;
561                        // use local name for namespace
562                        String localName = element.getLocalName();
563                        if (localName == null)
564                                continue;
565                        if (isXMLType()) {
566                                if (localName.equals(docName))
567                                        return element;
568                        }
569                        else {
570                                if (localName.equalsIgnoreCase(docName))
571                                        return element;
572                        }
573                        if (firstFound[0] == null)
574                                firstFound[0] = element;
575                }
576                return null;
577        }

579        /**
580         * getCharValue method
581         * 
582         * @return java.lang.String
583         * @param name
584         *            java.lang.String
585         */
586        protected String getCharValue(String name) {
587                if (name == null)
588                        return null;
589                int length = name.length();
590                if (length == 0)
591                        return null;
592 
593                if (name.charAt(0) == '#') { // character reference
594                        if (length == 1)
595                                return null;
596                        int radix = 10;
597                        String s = null;
598                        // now allow hexadecimal also for non XML document
599                        if (name.charAt(1) == 'x') { // hexadecimal
600                                radix = 16;
601                                s = name.substring(2);
602                        }
603                        else { // decimal
604                                s = name.substring(1);
605                        }
606                        if (s == null || s.length() == 0)
607                                return null;
608                        if (s.charAt(0) == '-')
609                                return null; // no minus accepted
610                        char c = 0;
611                        try {
612                                c = (char) Integer.parseInt(s, radix);
613                        }
614                        catch (NumberFormatException ex) {
615                        }
616                        if (c == 0)
617                                return null;
618                        return String.valueOf(c);
619                }
620 
621                // implicit character entities for XML
622                if (name.equals(IXMLCharEntity.LT_NAME))
623                        return IXMLCharEntity.LT_VALUE;
624                if (name.equals(IXMLCharEntity.GT_NAME))
625                        return IXMLCharEntity.GT_VALUE;
626                if (name.equals(IXMLCharEntity.AMP_NAME))
627                        return IXMLCharEntity.AMP_VALUE;
628                if (name.equals(IXMLCharEntity.QUOT_NAME))
629                        return IXMLCharEntity.QUOT_VALUE;
630                if (isXMLType()) {
631                        if (name.equals(IXMLCharEntity.APOS_NAME))
632                                return IXMLCharEntity.APOS_VALUE;
633                }
634 
635                CMDocument cm = getCMDocument();
636                if (cm != null) {
637                        CMNamedNodeMap map = cm.getEntities();
638                        if (map != null) {
639                                CMEntityDeclaration decl = (CMEntityDeclaration) map.getNamedItem(name);
640                                if (decl != null) {
641                                        String value = decl.getValue();
642                                        if (value == null)
643                                                return null;
644                                        int valueLength = value.length();
645                                        if (valueLength > 1 && value.charAt(0) == '&' && value.charAt(1) == '#' && value.charAt(valueLength - 1) == ';') {
646                                                // character reference
647                                                return getCharValue(value.substring(1, valueLength - 1));
648                                        }
649                                        return value;
650                                }
651                        }
652                }
653 
654                return null;
655        }
