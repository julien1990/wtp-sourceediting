429        private void removeAllProjects() throws CoreException, InterruptedException {
430                IWorkspace workspace = ResourcesPlugin.getWorkspace();
431                IProject[] projects = workspace.getRoot().getProjects();
432                IProject project = null;
433                for (int i = 0; i < projects.length; i++) {
434                        project = projects[i];
435                        deleteProject(project);
436                }

439        /**
440         * It's not easy to delete projects. If any of it's files are open by another thread, 
441         * the operation will fail. So, this method will make several attempts before giving up. 
442         * @param project
443         * @throws CoreException
444         * @throws InterruptedException
445         */
446        private void deleteProject(IProject project) throws CoreException, InterruptedException {
447                int nTrys = 0;
448                while (project != null && project.exists() && nTrys < MAX_RETRYS) {
449                        try {
450                                nTrys++;
451                                project.delete(true, true, null);
452                        }
453                        catch (ResourceException e) {
454                                if (DEBUG) {
455                                        System.out.println();
456                                        System.out.println("Could not delete project on attempt number: "+ nTrys);
457                                        IStatus eStatus = e.getStatus();
458                                        // should always be MultiStatus, but we'll check
459                                        if (eStatus instanceof MultiStatus) {
460                                                MultiStatus mStatus = (MultiStatus) eStatus;
461                                                IStatus[] iStatus = mStatus.getChildren();
462                                                for (int j = 0; j < iStatus.length; j++) {
463                                                        System.out.println("Status: " + j + " " + iStatus[j]);
464                                                }
465                                        }
466                                        else {
467                                                System.out.println("Status: " + eStatus);
468                                        }
469                                }
470                                /*
471                                 * If we could not delete the first time, wait a bit and
472                                 * re-try. If we could not delete, it is likely because
473                                 * another thread has a file open, or similar (such as the
474                                 * validation thread).
475                                 */
476                                Thread.sleep(PAUSE_TIME);
477                        }
478                }
479                
480                if (project != null && project.exists()) {
481                        fail("Error in test infrastructure. Could not delete project " + project + " after " + MAX_RETRYS + "attempts.");
482                }
483        }
