365        static void fireTaglibDelta(ITaglibIndexDelta delta) {
366                if (_debugEvents) {
367                        Logger.log(Logger.INFO, "TaglibIndex fired delta:" + delta + " [" + delta.getAffectedChildren().length + "]\n" + ((TaglibIndexDelta) delta).trigger); //$NON-NLS-1$
368                }
369                /*
370                 * Flush any shared cache entries, the TaglibControllers should handle
371                 * updating their documents as needed.
372                 */
373                ITaglibIndexDelta[] deltas = delta.getAffectedChildren();
374                for (int i = 0; i < deltas.length; i++) {
375                        ITaglibRecord taglibRecord = deltas[i].getTaglibRecord();
376                        if (taglibRecord != null) {
377                                Object uniqueIdentifier = TLDCMDocumentManager.getUniqueIdentifier(taglibRecord);
378                                if (uniqueIdentifier != null) {
379                                        TLDCMDocumentManager.getSharedDocumentCache().remove(uniqueIdentifier);
380                                }
381                                else {
382                                        Logger.log(Logger.ERROR, "identifier for " + taglibRecord + " was null");
383                                }
384                        }
385                }
386                synchronized (TLDCMDocumentManager.getSharedDocumentCache()) {
387                        Iterator values = TLDCMDocumentManager.getSharedDocumentCache().values().iterator();
388                        while (values.hasNext()) {
389                                Object o = values.next();
390                                if (o instanceof Reference) {
391                                        values.remove();
392                                }
393                        }
394                }
395 
396                if (_instance.isInitialized()) {
397                        ITaglibIndexListener[] listeners = _instance.fTaglibIndexListeners;
398                        if (listeners != null) {
399                                for (int j = 0; j < listeners.length; j++) {
400                                        try {
401                                                listeners[j].indexChanged(delta);
402                                        }
403                                        catch (Exception e) {
404                                                Logger.log(Logger.WARNING, e.getMessage());
405                                        }
406                                }
407                        }
408                }
409        }

490         *            - whether to search across projects (currently ignored)
491         *

493         *         known tag library descriptor

495         * @See ITaglibRecord

497        public static ITaglibRecord resolve(String basePath, String reference, boolean crossProjects) {
498                ITaglibRecord result = null;
499                if (getInstance().isInitialized()) {

501                }

506                        }
507                        else {
508                                switch (result.getRecordType()) {
509                                        case (ITaglibRecord.TLD) : {
510                                                ITLDRecord record = (ITLDRecord) result;
511                                                Logger.log(Logger.INFO, "TaglibIndex resolved " + basePath + ":" + reference + " = " + record.getPath()); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
512                                        }
513                                                break;
514                                        case (ITaglibRecord.JAR) : {
515                                                IJarRecord record = (IJarRecord) result;
516                                                Logger.log(Logger.INFO, "TaglibIndex resolved " + basePath + ":" + reference + " = " + record.getLocation()); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
517                                        }
518                                                break;
519                                        case (ITaglibRecord.TAGDIR) : {
520                                                ITagDirRecord record = (ITagDirRecord) result;
521                                                Logger.log(Logger.INFO, "TaglibIndex resolved " + basePath + ":" + reference + " = " + record.getPath()); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
522                                        }
523                                                break;
524                                        case (ITaglibRecord.URL) : {
525                                                IURLRecord record = (IURLRecord) result;
526                                                Logger.log(Logger.INFO, "TaglibIndex resolved " + basePath + ":" + reference + " = " + record.getURL()); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
527                                        }
528                                                break;
529                                }
530                        }
531                }
532                return result;
533        }

810        private ITaglibRecord internalResolve(String basePath, final String reference, boolean crossProjects) {
811                IProject project = null;
812                ITaglibRecord resolved = null;
813 
814                Path baseIPath = new Path(basePath);
815                IResource baseResource = FileBuffers.getWorkspaceFileAtLocation(baseIPath);
816 
817                if (baseResource == null) {
818                        IWorkspaceRoot workspaceRoot = ResourcesPlugin.getWorkspace().getRoot();
819                        // Try the base path as a folder first
820                        if (baseIPath.segmentCount() > 1) {
821                                baseResource = workspaceRoot.getFolder(baseIPath);
822                        }
823                        // If not a folder, then try base path as a file
824                        if (baseResource != null && !baseResource.exists() && baseIPath.segmentCount() > 1) {
825                                baseResource = workspaceRoot.getFile(baseIPath);
