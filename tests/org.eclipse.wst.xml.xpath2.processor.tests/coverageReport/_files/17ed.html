611                // Avoid cycling - can happen if a closing cell editor causes a
612                // refresh
613                // on the PropertySheet page and the setInput again asks the editor to
614                // close; besides, why apply the same value twice?
615                if (!fValuesBeingSet.isEmpty() && (fValuesBeingSet.peek() == nameObject)) {
616                        return;
617                }
618                fValuesBeingSet.push(nameObject);
619                String name = nameObject.toString();
620                String valueString = null;
621                if (value != null) {
622                        valueString = value.toString();
623                }
624                NamedNodeMap attrMap = fNode.getAttributes();
625                try {
626                        if (attrMap != null) {
627                                Attr attr = (Attr) attrMap.getNamedItem(name);
628                                if (attr != null) {
629                                        // EXISTING VALUE
630                                        // potential out of control loop if updating the value
631                                        // triggers a viewer update, forcing the
632                                        // active cell editor to save its value and causing the
633                                        // loop to continue
634                                        if ((attr.getValue() == null) || !attr.getValue().equals(valueString)) {
635                                                if (attr instanceof IDOMNode) {
636                                                        ((IDOMNode) attr).setValueSource(valueString);
637                                                }
638                                                else {
639                                                        attr.setValue(valueString);
640                                                }
641                                        }
642                                }
643                                else {
644                                        // NEW(?) value
645                                        if (value != null) { // never create an empty attribute
646                                                Attr newAttr = fNode.getOwnerDocument().createAttribute(name);
647                                                if (newAttr instanceof IDOMNode) {
648                                                        ((IDOMNode) newAttr).setValueSource(valueString);
649                                                }
650                                                else {
651                                                        newAttr.setValue(valueString);
652                                                }
653                                                attrMap.setNamedItem(newAttr);
654                                        }
655                                }
656                        }
657                        else {
658                                if (fNode instanceof Element) {
659                                        ((Element) fNode).setAttribute(name, valueString);
660                                }
661                        }
662                }
663                catch (DOMException e) {
664                        Display d = getDisplay();
665                        if (d != null) {
666                                d.beep();
667                        }
668                }
669                fValuesBeingSet.pop();
