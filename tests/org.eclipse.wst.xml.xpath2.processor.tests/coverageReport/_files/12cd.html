480                try {
481                        int replaceOffset= getReplaceOffset();
482                        if (offset >= replaceOffset) {
483                                String content= document.get(replaceOffset, offset - replaceOffset);
484                                String templateName= fTemplate.getName().toLowerCase();
485                                boolean valid= templateName.startsWith(content.toLowerCase());
486                                if (!valid && fContext instanceof JavaDocContext && templateName.startsWith("<")) { //$NON-NLS-1$
487                                        valid= templateName.startsWith(content.toLowerCase(), 1);
488                                }
489                                return valid;
490                        }
491                } catch (BadLocationException e) {
492                        // concurrent modification - ignore
493                }
494                return false;
495        }
