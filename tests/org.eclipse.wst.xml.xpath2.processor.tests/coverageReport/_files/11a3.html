407        /**
408         * @param snapPosition one of the SNAP_POSITION_* constants
409         * @return the location in display coordinates or <code>null</code> iff not visible
410         */
411        private Point computePopupLocation(int snapPosition) {
412                if (fPopup == null || fPopup.isDisposed())
413                        return null;
414                
415                switch (snapPosition) {
416                        case SNAP_POSITION_LOWER_RIGHT:
417                        {
418                                StyledText eWidget= fEditor.getViewer().getTextWidget();
419                                Rectangle eBounds= eWidget.getClientArea();
420                                Point eLowerRight= eWidget.toDisplay(eBounds.x + eBounds.width, eBounds.y + eBounds.height);
421                                Point pSize= getExtent();
422                                return new Point(eLowerRight.x - pSize.x - 5, eLowerRight.y - pSize.y - 5);
423                        }
424                        
425                        case SNAP_POSITION_UNDER_RIGHT_FIELD:
426                        case SNAP_POSITION_OVER_RIGHT_FIELD:
427                        {
428                                LinkedPosition position= fRenameLinkedMode.getCurrentLinkedPosition();
429                                if (position == null)
430                                        return null;
431                                ISourceViewer viewer= fEditor.getViewer();
432                                ITextViewerExtension5 viewer5= (ITextViewerExtension5) viewer;
433                                int widgetOffset= viewer5.modelOffset2WidgetOffset(position.offset + position.length);
434                                
435                                StyledText textWidget= viewer.getTextWidget();
436                                Point pos= textWidget.getLocationAtOffset(widgetOffset);
437                                Point pSize= getExtent();
438                                if (snapPosition == SNAP_POSITION_OVER_RIGHT_FIELD) {
439                                        pos.y-= pSize.y + GAP;
440                                } else {
441                                        pos.y+= textWidget.getLineHeight(widgetOffset) + GAP;
442                                }
443                                pos.x+= GAP;
444                                Point dPos= textWidget.toDisplay(pos);
445                                Rectangle displayBounds= textWidget.getDisplay().getClientArea();
446                                Rectangle dPopupRect= Geometry.createRectangle(dPos, pSize);
447                                Geometry.moveInside(dPopupRect, displayBounds);
448                                return new Point(dPopupRect.x, dPopupRect.y);
449                        }
450                        
451                        case SNAP_POSITION_UNDER_LEFT_FIELD:
452                        case SNAP_POSITION_OVER_LEFT_FIELD:
453                        default: // same as SNAP_POSITION_UNDER_LEFT_FIELD
454                        {
455                                LinkedPosition position= fRenameLinkedMode.getCurrentLinkedPosition();
456                                if (position == null)
457                                        return null;
458                                ISourceViewer viewer= fEditor.getViewer();
459                                ITextViewerExtension5 viewer5= (ITextViewerExtension5) viewer;
460                                int widgetOffset= viewer5.modelOffset2WidgetOffset(position.offset/* + position.length*/);
461                                
462                                StyledText textWidget= viewer.getTextWidget();
463                                Point pos= textWidget.getLocationAtOffset(widgetOffset);
464                                Point pSize= getExtent();
465                                pSize.y+= HAH + 1;
466                                pos.x-= HAO;
467                                if (snapPosition == SNAP_POSITION_OVER_LEFT_FIELD) {
468                                        pos.y-= pSize.y;
469                                } else {
470                                        pos.y+= textWidget.getLineHeight(widgetOffset);
471                                }
472                                Point dPos= textWidget.toDisplay(pos);
473                                Rectangle displayBounds= textWidget.getDisplay().getClientArea();
474                                Rectangle dPopupRect= Geometry.createRectangle(dPos, pSize);
475                                Geometry.moveInside(dPopupRect, displayBounds);
476                                return new Point(dPopupRect.x, dPopupRect.y);
477                        }
478                        
479                }

482        private void addMoveSupport(final Shell popupShell, final Control movedControl) {
483                movedControl.addMouseListener(new MouseAdapter() {
484 
485                        public void mouseDown(final MouseEvent downEvent) {
486                                if (downEvent.button != 1) {
487                                        return;
488                                }
489                                
490                                final Point POPUP_SOURCE= popupShell.getLocation();
491                                final StyledText textWidget= fEditor.getViewer().getTextWidget();
492                                Point pSize= getExtent();
493                                int originalSnapPosition= fSnapPosition;
494                                
495                                /*
496                                 * Feature in Tracker: it is not possible to directly control the feedback,
497                                 * see https://bugs.eclipse.org/bugs/show_bug.cgi?id=121300
498                                 * and https://bugs.eclipse.org/bugs/show_bug.cgi?id=121298#c1 .
499                                 * 
500                                 * Workaround is to have an offscreen rectangle for tracking mouse movement
501                                 * and a manually updated rectangle for the actual drop target.
502                                 */
503                                final Tracker tracker= new Tracker(textWidget, SWT.NONE);
504                                
505                                final Point[] LOCATIONS= {
506                                                textWidget.toControl(computePopupLocation(SNAP_POSITION_UNDER_RIGHT_FIELD)),
507                                                textWidget.toControl(computePopupLocation(SNAP_POSITION_OVER_RIGHT_FIELD)),
508                                                textWidget.toControl(computePopupLocation(SNAP_POSITION_UNDER_LEFT_FIELD)),
509                                                textWidget.toControl(computePopupLocation(SNAP_POSITION_OVER_LEFT_FIELD)),
510                                                textWidget.toControl(computePopupLocation(SNAP_POSITION_LOWER_RIGHT))
511                                };
512                                
513                                final Rectangle[] DROP_TARGETS= {
514                                        Geometry.createRectangle(LOCATIONS[0], pSize),
515                                        Geometry.createRectangle(LOCATIONS[1], pSize),
516                                        new Rectangle(LOCATIONS[2].x, LOCATIONS[2].y + HAH, pSize.x, pSize.y),
517                                        Geometry.createRectangle(LOCATIONS[3], pSize),
518                                        Geometry.createRectangle(LOCATIONS[4], pSize)
519                                };
520                                final Rectangle MOUSE_MOVE_SOURCE= new Rectangle(1000000, 0, 0, 0);
521                                tracker.setRectangles(new Rectangle[] { MOUSE_MOVE_SOURCE, DROP_TARGETS[fSnapPosition] });
522                                tracker.setStippled(true);
523                                
524                                ControlListener moveListener= new ControlAdapter() {
525                                        /*
526                                         * @see org.eclipse.swt.events.ControlAdapter#controlMoved(org.eclipse.swt.events.ControlEvent)
527                                         */
528                                        public void controlMoved(ControlEvent moveEvent) {
529                                                Rectangle[] currentRects= tracker.getRectangles();
530                                                final Rectangle mouseMoveCurrent= currentRects[0];
531                                                Point popupLoc= new Point(
532                                                                POPUP_SOURCE.x + mouseMoveCurrent.x - MOUSE_MOVE_SOURCE.x,
533                                                                POPUP_SOURCE.y + mouseMoveCurrent.y - MOUSE_MOVE_SOURCE.y);
534                                                
535                                                popupShell.setLocation(popupLoc);
536                                                
537                                                Point ePopupLoc= textWidget.toControl(popupLoc);
538                                                int minDist= Integer.MAX_VALUE;
539                                                for (int snapPos= 0; snapPos < DROP_TARGETS.length; snapPos++) {
540                                                        int dist= Geometry.distanceSquared(ePopupLoc, LOCATIONS[snapPos]);
541                                                        if (dist < minDist) {
542                                                                minDist= dist;
543                                                                fSnapPosition= snapPos;
544                                                                currentRects[1]= DROP_TARGETS[snapPos];
545                                                        }
546                                                }
547                                                tracker.setRectangles(currentRects);
548                                        }
549                                };
550                                tracker.addControlListener(moveListener);
551                                boolean committed= tracker.open();
552                                tracker.close();
553                                tracker.dispose();
554                                if (committed) {
555                                        getDialogSettings().put(SNAP_POSITION_KEY, fSnapPosition);
556                                } else {
557                                        fSnapPosition= originalSnapPosition;
558                                }
559                                updatePopupLocation(true);
560                                activateEditor();
561                        }
562                });
563        }
