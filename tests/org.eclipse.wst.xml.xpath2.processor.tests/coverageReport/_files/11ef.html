452        private IJavaScriptElement findElement(TreeItem[] items) {
453                ILabelProvider labelProvider= (ILabelProvider)fTreeViewer.getLabelProvider();
454                for (int i= 0; i < items.length; i++) {
455                        IJavaScriptElement element= (IJavaScriptElement)items[i].getData();
456                        if (fStringMatcher == null)
457                                return element;
458 
459                        if (element != null) {
460                                String label= labelProvider.getText(element);
461                                if (fStringMatcher.match(label))
462                                        return element;
463                        }
464 
465                        element= findElement(items[i].getItems());
466                        if (element != null)
467                                return element;
468                }
469                return null;
470        }
