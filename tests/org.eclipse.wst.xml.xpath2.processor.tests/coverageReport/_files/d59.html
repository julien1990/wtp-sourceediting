546        public static IProblem[] getProblems(ASTNode node, int scope, int severity) {
547                ASTNode root= node.getRoot();
548                if (!(root instanceof JavaScriptUnit))
549                        return EMPTY_PROBLEMS;
550                IProblem[] problems= ((JavaScriptUnit)root).getProblems();
551                if (root == node)
552                        return problems;
553                final int iterations= computeIterations(scope);
554                List result= new ArrayList(5);
555                for (int i= 0; i < problems.length; i++) {
556                        IProblem problem= problems[i];
557                        boolean consider= false;
558                        if ((severity & PROBLEMS) == PROBLEMS)
559                                consider= true;
560                        else if ((severity & WARNING) != 0)
561                                consider= problem.isWarning();
562                        else if ((severity & ERROR) != 0)
563                                consider= problem.isError();
564                        if (consider) {
565                                ASTNode temp= node;
566                                int count= iterations;
567                                do {
568                                        int nodeOffset= temp.getStartPosition();
569                                        int problemOffset= problem.getSourceStart();
570                                        if (nodeOffset <= problemOffset && problemOffset < nodeOffset + temp.getLength()) {
571                                                result.add(problem);
572                                                count= 0;
573                                        } else {
574                                                count--;
575                                        }
576                                } while ((temp= temp.getParent()) != null && count > 0);
577                        }
578                }
579                return (IProblem[]) result.toArray(new IProblem[result.size()]);
580        }

623        private static int getOrderPreference(BodyDeclaration member, MembersOrderPreferenceCache store) {
624                int memberType= member.getNodeType();
625                int modifiers= member.getModifiers();
626 
627                switch (memberType) {
628                        case ASTNode.TYPE_DECLARATION:
629                                return store.getCategoryIndex(MembersOrderPreferenceCache.TYPE_INDEX) * 2;
630                        case ASTNode.FIELD_DECLARATION:
631                                if (Modifier.isStatic(modifiers)) {
632                                        int index= store.getCategoryIndex(MembersOrderPreferenceCache.STATIC_FIELDS_INDEX) * 2;
633                                        if (Modifier.isFinal(modifiers)) {
634                                                return index; // first final static, then static
635                                        }
636                                        return index + 1;
637                                }
638                                return store.getCategoryIndex(MembersOrderPreferenceCache.FIELDS_INDEX) * 2;
639                        case ASTNode.INITIALIZER:
640                                if (Modifier.isStatic(modifiers)) {
641                                        return store.getCategoryIndex(MembersOrderPreferenceCache.STATIC_INIT_INDEX) * 2;
642                                }
643                                return store.getCategoryIndex(MembersOrderPreferenceCache.INIT_INDEX) * 2;
644                        case ASTNode.FUNCTION_DECLARATION:
645                                if (Modifier.isStatic(modifiers)) {
646                                        return store.getCategoryIndex(MembersOrderPreferenceCache.STATIC_METHODS_INDEX) * 2;
647                                }
648                                if (((FunctionDeclaration) member).isConstructor()) {
649                                        return store.getCategoryIndex(MembersOrderPreferenceCache.CONSTRUCTORS_INDEX) * 2;
650                                }
651                                return store.getCategoryIndex(MembersOrderPreferenceCache.METHOD_INDEX) * 2;
652                        default:
653                                return 100;
654                }
655        }

657        /**
658         * Computes the insertion index to be used to add the given member to the
659         * the list <code>container</code>.
660         * @param member the member to add
661         * @param container a list containing objects of type <code>BodyDeclaration</code>
662         * @return the insertion index to be used
663         */
664        public static int getInsertionIndex(BodyDeclaration member, List container) {
665                int containerSize= container.size();
666                
667                MembersOrderPreferenceCache orderStore= JavaScriptPlugin.getDefault().getMemberOrderPreferenceCache();
668                
669                int orderIndex= getOrderPreference(member, orderStore);
670                
671                int insertPos= containerSize;
672                int insertPosOrderIndex= -1;
673 
674                for (int i= containerSize - 1; i >= 0; i--) {
675                        int currOrderIndex= getOrderPreference((BodyDeclaration) container.get(i), orderStore);
676                        if (orderIndex == currOrderIndex) {
677                                if (insertPosOrderIndex != orderIndex) { // no perfect match yet
678                                        insertPos= i + 1; // after a same kind
679                                        insertPosOrderIndex= orderIndex; // perfect match
680                                }
681                        } else if (insertPosOrderIndex != orderIndex) { // not yet a perfect match
682                                if (currOrderIndex < orderIndex) { // we are bigger
683                                        if (insertPosOrderIndex == -1) {
684                                                insertPos= i + 1; // after
685                                                insertPosOrderIndex= currOrderIndex;
686                                        }
687                                } else {
688                                        insertPos= i; // before
689                                        insertPosOrderIndex= currOrderIndex;
690                                }
691                        }
692                }
693                return insertPos;
694        }
