363                    // infer from bounds of type parameters
364                        for (int i = 0; i < varLength; i++) {
365                                TypeVariableBinding originalVariable = originalVariables[i];
366                                TypeBinding argument = this.typeArguments[i];
367                                boolean argAlreadyInferred = argument != originalVariable;
368                                if (originalVariable.firstBound == originalVariable.superclass) {
369                                        TypeBinding substitutedBound = Scope.substitute(this, originalVariable.superclass);
370                                        argument.collectSubstitutes(scope, substitutedBound, inferenceContext, TypeConstants.CONSTRAINT_SUPER);
371                                        if (inferenceContext.status == InferenceContext.FAILED) return null; // impossible substitution
372                                        // JLS 15.12.2.8 claims reverse inference shouldn't occur, however it improves inference
373                                        // e.g. given: <E extends Object, S extends Collection<E>> S test1(S param)
374                                        //                   invocation: test1(new Vector<String>())    will infer: S=Vector<String>  and with code below: E=String
375                                        if (argAlreadyInferred) {
376                                                substitutedBound.collectSubstitutes(scope, argument, inferenceContext, TypeConstants.CONSTRAINT_EXTENDS);
377                                                if (inferenceContext.status == InferenceContext.FAILED) return null; // impossible substitution
378                                        }
379                                }
380                                for (int j = 0, max = originalVariable.superInterfaces.length; j < max; j++) {
381                                        TypeBinding substitutedBound = Scope.substitute(this, originalVariable.superInterfaces[j]);
382                                        argument.collectSubstitutes(scope, substitutedBound, inferenceContext, TypeConstants.CONSTRAINT_SUPER);
383                                        if (inferenceContext.status == InferenceContext.FAILED) return null; // impossible substitution
384                                        // JLS 15.12.2.8 claims reverse inference shouldn't occur, however it improves inference
385                                        if (argAlreadyInferred) {
386                                                substitutedBound.collectSubstitutes(scope, argument, inferenceContext, TypeConstants.CONSTRAINT_EXTENDS);
387                                                if (inferenceContext.status == InferenceContext.FAILED) return null; // impossible substitution
388                                        }
389                                }
390                        }
391                        if (inferenceContext.status == InferenceContext.RAW_SUBSTITUTION) {
392                            // raw generic method inferred
393                            this.isRaw = true;
394                                this.isUnchecked = false;
395                            for (int i = 0; i < varLength; i++) {
396                                    this.typeArguments[i] = originalVariables[i].upperBound();
397                            }
398                            break computeSubstitutes;
399                        }
400                        if (!resolveSubstituteConstraints(scope, originalVariables, inferenceContext, true/*consider Ti<:Uk*/))
401                                return null; // incompatible
402                        // this.typeArguments = substitutes; - no op since side effects got performed during #resolveSubstituteConstraints
403                    for (int i = 0; i < varLength; i++) {
404                            TypeBinding substitute = inferenceContext.substitutes[i];
405                            if (substitute != null) {
406                                    this.typeArguments[i] = inferenceContext.substitutes[i];
407                            } else {
408                                    // remaining unresolved variable are considered to be Object (or their bound actually)
409                                    this.typeArguments[i] = originalVariables[i].upperBound();
410                            }
411                    }
412                }
413                // may still need an extra substitution at the end (see https://bugs.eclipse.org/bugs/show_bug.cgi?id=121369)
414                // to properly substitute a remaining unresolved variable which also appear in a formal bound
415                this.typeArguments = Scope.substitute(this, this.typeArguments);
416                // adjust method types to reflect latest inference
417                TypeBinding oldReturnType = this.returnType;
418                this.returnType = Scope.substitute(this, this.returnType);
419                this.inferredReturnType = inferenceContext.hasExplicitExpectedType && this.returnType != oldReturnType;
420            this.parameters = Scope.substitute(this, this.parameters);
421            this.thrownExceptions = Scope.substitute(this, this.thrownExceptions);
422            return this;
423        }

433         * @see org.eclipse.wst.jsdt.internal.compiler.lookup.Substitution#substitute(org.eclipse.wst.jsdt.internal.compiler.lookup.TypeVariableBinding)
434         */
435        public TypeBinding substitute(TypeVariableBinding originalVariable) {
436        TypeVariableBinding[] variables = this.originalMethod.typeVariables;
437        int length = variables.length;
438        // check this variable can be substituted given parameterized type
439        if (originalVariable.rank < length && variables[originalVariable.rank] == originalVariable) {
440                        return this.typeArguments[originalVariable.rank];
441        }
442        if (!this.isStatic() && this.declaringClass instanceof Substitution) {
443                return ((Substitution)this.declaringClass).substitute(originalVariable);
444        }
445            return originalVariable;
446        }

448         * @see org.eclipse.wst.jsdt.internal.compiler.lookup.MethodBinding#tiebreakMethod()
449         */
450        public MethodBinding tiebreakMethod() {
451                if (this.tiebreakMethod == null) {
452                        TypeVariableBinding[] originalVariables = originalMethod.typeVariables;
453                        int length = originalVariables.length;
454                        TypeBinding[] newArguments = new TypeBinding[length];
455                        for (int i = 0; i < length; i++)
456                                newArguments[i] = environment.convertToRawType(originalVariables[i].upperBound());
457                        this.tiebreakMethod = this.environment.createParameterizedGenericMethod(this.originalMethod, newArguments);
458                }
459                return this.tiebreakMethod;
