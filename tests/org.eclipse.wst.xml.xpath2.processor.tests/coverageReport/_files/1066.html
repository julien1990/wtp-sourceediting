391                        fSharedState.fResult= IndentUtil.indentLines(document, selectionAfter, getProject(), fSharedState.fResult);
392                        
393                        // move the selection along
394                        IRegion region= getRegion(document, selectionAfter);
395                        selectAndReveal(viewer, region.getOffset(), region.getLength());
396                        
397                } catch (BadLocationException x) {
398                        // won't happen without concurrent modification - bail out
399                        return;
400                } finally {
401                        fSharedState.fIsChanging= false;
402                        if (fCopy)
403                                fSharedState.endCompoundEdit();
404                }
405        }

415        private ILineRange getLineRange(IDocument document, ITextSelection selection) throws BadLocationException {
416                final int offset= selection.getOffset();
417                int startLine= document.getLineOfOffset(offset);
418                int endOffset= offset + selection.getLength();
419                int endLine= document.getLineOfOffset(endOffset);
420                final int nLines= endLine - startLine + 1;
421                return new LineRange(startLine, nLines);
422        }

424        private IRegion getRegion(IDocument document, ILineRange lineRange) throws BadLocationException {
425                final int startLine= lineRange.getStartLine();
426                int offset= document.getLineOffset(startLine);
427                final int numberOfLines= lineRange.getNumberOfLines();
428                if (numberOfLines < 1)
429                        return new Region(offset, 0);
430                int endLine = startLine + numberOfLines - 1;
431                int endOffset;
432                if (fSharedState.fEditor.isBlockSelectionModeEnabled()) {
433                        // in column mode, don't select the last delimiter as we count an
434                        // empty selected line
435                        IRegion endLineInfo = document.getLineInformation(endLine);
436                        endOffset = endLineInfo.getOffset() + endLineInfo.getLength();
437                }
438                else {
439                        endOffset = document.getLineOffset(endLine) + document.getLineLength(endLine);
440                }
441                return new Region(offset, endOffset - offset);
442        }

444        /**
445         * Performs similar to AbstractTextEditor.selectAndReveal, but does not update
446         * the viewers highlight area.
447         *
448         * @param viewer the viewer that we want to select on
449         * @param offset the offset of the selection
450         * @param length the length of the selection
451         */
452        private void selectAndReveal(ITextViewer viewer, int offset, int length) {
453                // invert selection to avoid jumping to the end of the selection in st.showSelection()
454                viewer.setSelectedRange(offset + length, -length);
455                //viewer.revealRange(offset, length); // will trigger jumping
456                StyledText st= viewer.getTextWidget();
457                if (st != null)
458                        st.showSelection(); // only minimal scrolling
459        }
