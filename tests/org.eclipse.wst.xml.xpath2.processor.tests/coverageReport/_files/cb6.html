318        int count = -1;
319        for (int i = 0; i < onDiskLength;) {
320                switch(positions[i]) {
321                        case DELETED :
322                                i++; // skip over deleted... references are forgotten
323                                break;
324                        case RE_INDEXED :
325                                String newName = newDocNames[++count];
326                                if (newName.equals(onDiskNames[i])) {
327                                        indexedDocuments.put(newName, new Integer(count)); // the reindexed docName that was at position i is now at position count
328                                        i++;
329                                }
330                                break;
331                        default :
332                                if (newDocNames[++count].equals(onDiskNames[i]))
333                                        positions[i++] = count; // the unchanged docName that was at position i is now at position count
334                }
335        }
336        return newDocNames;
337}
338private void copyQueryResults(HashtableOfObject categoryToWords, int newPosition) {

435private void mergeCategory(char[] categoryName, DiskIndex onDisk, int[] positions, FileOutputStream stream) throws IOException {
436        HashtableOfObject wordsToDocs = (HashtableOfObject) this.categoryTables.get(categoryName);
437        if (wordsToDocs == null)
438                wordsToDocs = new HashtableOfObject(3);
439 
440        HashtableOfObject oldWordsToDocs = onDisk.readCategoryTable(categoryName, true);
441        if (oldWordsToDocs != null) {
442                char[][] oldWords = oldWordsToDocs.keyTable;
443                Object[] oldArrayOffsets = oldWordsToDocs.valueTable;
444                nextWord: for (int i = 0, l = oldWords.length; i < l; i++) {
445                        char[] oldWord = oldWords[i];
446                        if (oldWord != null) {
447                                int[] oldDocNumbers = (int[]) oldArrayOffsets[i];
448                                int length = oldDocNumbers.length;
449                                int[] mappedNumbers = new int[length];
450                                int count = 0;
451                                for (int j = 0; j < length; j++) {
452                                        int pos = positions[oldDocNumbers[j]];
453                                        if (pos > RE_INDEXED) // forget any reference to a document which was deleted or re_indexed
454                                                mappedNumbers[count++] = pos;
455                                }
456                                if (count < length) {
457                                        if (count == 0) continue nextWord; // skip words which no longer have any references
458                                        System.arraycopy(mappedNumbers, 0, mappedNumbers = new int[count], 0, count);
459                                }
460 
461                                Object o = wordsToDocs.get(oldWord);
462                                if (o == null) {
463                                        wordsToDocs.put(oldWord, mappedNumbers);
464                                } else {
465                                        IntList list = null;
466                                        if (o instanceof IntList) {
467                                                list = (IntList) o;
468                                        } else {
469                                                list = new IntList((int[]) o);
470                                                wordsToDocs.put(oldWord, list);
471                                        }
472                                        for (int j = 0; j < count; j++)
473                                                list.add(mappedNumbers[j]);
474                                }
475                        }
476                }
477                onDisk.categoryTables.put(categoryName, null); // flush cached table
478        }
479        writeCategoryTable(categoryName, wordsToDocs, stream);
480}

585                if (cachedTable != null) {

594        }
595 
596        FileInputStream stream = new FileInputStream(this.indexFile);
597        HashtableOfObject categoryTable = null;
598        char[][] matchingWords = null;
599        int count = 0;
600        int firstOffset = -1;
601        this.streamBuffer = new byte[BUFFER_READ_SIZE];

603                stream.skip(offset);

613                                System.err.println("--------------------   END   --------------------"); //$NON-NLS-1$
614                        }
615                        categoryTable = new HashtableOfObject(size);
616                } catch (OutOfMemoryError oom) {

618                        oom.printStackTrace();
619                        System.err.println("-------------------- DEBUG --------------------"); //$NON-NLS-1$
620                        System.err.println("file = "+this.indexFile); //$NON-NLS-1$
621                        System.err.println("offset = "+offset); //$NON-NLS-1$
622                        System.err.println("size = "+size); //$NON-NLS-1$
623                        System.err.println("--------------------   END   --------------------"); //$NON-NLS-1$
624                        throw oom;
625                }
626                int largeArraySize = 256;
627                for (int i = 0; i < size; i++) {
628                        char[] word = readStreamChars(stream);

657        } finally {

659        }

680private void readChunk(String[] docNames, FileInputStream stream, int index, int size) throws IOException {
681        String current = new String(readStreamChars(stream));
682        docNames[index++] = current;
683        for (int i = 1; i < size; i++) {
684                if (stream != null && this.bufferIndex + 2 >= this.bufferEnd)
685                        readStreamBuffer(stream);
686                int start = streamBuffer[this.bufferIndex++] & 0xFF;
687                int end = streamBuffer[this.bufferIndex++] & 0xFF;
688                String next  = new String(readStreamChars(stream));
689                if (start > 0) {
690                        if (end > 0) {
691                                int length = current.length();
692                                next = current.substring(0, start) + next + current.substring(length - end, length);
693                        } else {
694                                next = current.substring(0, start) + next;
695                        }
696                } else if (end > 0) {
697                        int length = current.length();
698                        next = next + current.substring(length - end, length);
699                }
700                docNames[index++] = next;
701                current = next;
702        }

704synchronized String readDocumentName(int docNumber) throws IOException {
705        if (this.cachedChunks == null)
706                this.cachedChunks = new String[this.numberOfChunks][];
707

812private void readStreamBuffer(FileInputStream stream) throws IOException {
813        // if we're about to read a known amount at the end of the existing buffer, but it does not completely fit
814        // so we need to shift the remaining bytes to be read, and fill the buffer from the stream
815        if (this.bufferEnd < this.streamBuffer.length)
816                return; // we're at the end of the stream - nothing left to read
817 
818        int bytesInBuffer = this.bufferEnd - this.bufferIndex;
819        if (bytesInBuffer > 0)
820                System.arraycopy(this.streamBuffer, this.bufferIndex, this.streamBuffer, 0, bytesInBuffer);
821        this.bufferEnd = bytesInBuffer + stream.read(this.streamBuffer, bytesInBuffer, this.bufferIndex);
822        this.bufferIndex = 0;
823}
824/**
825 * Reads in a string from the specified data input stream. The
826 * string has been encoded using a modified UTF-8 format.
827 * <p>
828 * The first two bytes are read as an unsigned short.
829 * This value gives the number of following bytes that are in the encoded string,
830 * not the length of the resulting string. The following bytes are then
831 * interpreted as bytes encoding characters in the UTF-8 format
832 * and are converted into characters.
833 * <p>
834 * This method blocks until all the bytes are read, the end of the
835 * stream is detected, or an exception is thrown.
836 *
837 * @param      stream   a data input stream.
838 * @return     UTF decoded string as a char array
839 * @exception  java.io.EOFException if this end of data input is reached while reading it.
840 * @exception  IOException if an I/O error occurs while reading data input.
841 * @exception  UTFDataFormatException  if the bytes do not represent a
842 *               valid UTF-8 encoding of a Unicode string.
843 */
844private char[] readStreamChars(FileInputStream stream) throws IOException {
845        // read chars array length
846        if (stream != null && this.bufferIndex + 2 >= this.bufferEnd)
847                readStreamBuffer(stream);
848        int length = (streamBuffer[this.bufferIndex++] & 0xFF) << 8;
849        length += this.streamBuffer[this.bufferIndex++] & 0xFF;
850 
851        // fill the chars from bytes buffer
852        char[] word = new char[length];
853        int i = 0;
854        while (i < length) {
855                // how many characters can be decoded without refilling the buffer?
856                int charsInBuffer = i + ((this.bufferEnd - this.bufferIndex) / 3);
857                // all the characters must already be in the buffer if we're at the end of the stream
858                if (charsInBuffer > length || this.bufferEnd != this.streamBuffer.length || stream == null)
859                        charsInBuffer = length;
860                while (i < charsInBuffer) {
861                        byte b = this.streamBuffer[this.bufferIndex++];
862                        switch (b & 0xF0) {
863                                case 0x00 :
864                                case 0x10 :
865                                case 0x20 :
866                                case 0x30 :
867                                case 0x40 :
868                                case 0x50 :
869                                case 0x60 :
870                                case 0x70 :
871                                        word[i++]= (char) b;
872                                        break;
873                                case 0xC0 :
874                                case 0xD0 :
875                                        char next = (char) this.streamBuffer[this.bufferIndex++];
876                                        if ((next & 0xC0) != 0x80) {
877                                                throw new UTFDataFormatException();
878                                        }
879                                        char ch = (char) ((b & 0x1F) << 6);
880                                        ch |= next & 0x3F;
881                                        word[i++] = ch;
882                                        break;
883                                case 0xE0 :
884                                        char first = (char) this.streamBuffer[this.bufferIndex++];
885                                        char second = (char) this.streamBuffer[this.bufferIndex++];
886                                        if ((first & second & 0xC0) != 0x80) {
887                                                throw new UTFDataFormatException();
888                                        }
889                                        ch = (char) ((b & 0x0F) << 12);
890                                        ch |= ((first& 0x3F) << 6);
891                                        ch |= second & 0x3F;
892                                        word[i++] = ch;
893                                        break;
894                                default:
895                                        throw new UTFDataFormatException();
896                        }
897                }
898                if (i < length && stream != null)
899                        readStreamBuffer(stream);
900        }
901        return word;
902}

953private void writeAllDocumentNames(String[] sortedDocNames, FileOutputStream stream) throws IOException {
954        if (sortedDocNames.length == 0)
955                throw new IllegalArgumentException();
956 
957        // assume the file was just created by initializeFrom()
958        this.streamBuffer = new byte[BUFFER_WRITE_SIZE];
959        this.bufferIndex = 0;
960        this.streamEnd = 0;
961 
962        // in order, write: SIGNATURE & headerInfoOffset place holder, then each compressed chunk of document names
963        writeStreamChars(stream, SIGNATURE_CHARS);
964        this.headerInfoOffset = this.streamEnd;
965        writeStreamInt(stream, -1); // will overwrite with correct value later
966 
967        int size = sortedDocNames.length;
968        this.numberOfChunks = (size / CHUNK_SIZE) + 1;
969        this.sizeOfLastChunk = size % CHUNK_SIZE;
970        if (this.sizeOfLastChunk == 0) {
971                this.numberOfChunks--;
972                this.sizeOfLastChunk = CHUNK_SIZE;
973        }
974        this.documentReferenceSize = size <= 0x7F ? 1 : (size <= 0x7FFF ? 2 : 4); // number of bytes used to encode a reference
975 
976        this.chunkOffsets = new int[this.numberOfChunks];
977        int lastIndex = this.numberOfChunks - 1;
978        for (int i = 0; i < this.numberOfChunks; i++) {
979                this.chunkOffsets[i] = this.streamEnd;
980 
981                int chunkSize = i == lastIndex ? this.sizeOfLastChunk : CHUNK_SIZE;
982                int chunkIndex = i * CHUNK_SIZE;
983                String current = sortedDocNames[chunkIndex];
984                writeStreamChars(stream, current.toCharArray());
985                for (int j = 1; j < chunkSize; j++) {
986                        String next = sortedDocNames[chunkIndex + j];
987                        int len1 = current.length();
988                        int len2 = next.length();
989                        int max = len1 < len2 ? len1 : len2;
990                        int start = 0; // number of identical characters at the beginning (also the index of first character that is different)
991                        while (current.charAt(start) == next.charAt(start)) {
992                                start++;
993                                if (max == start) break; // current is 'abba', next is 'abbab'
994                        }
995                        if (start > 255) start = 255;
996 
997                        int end = 0; // number of identical characters at the end
998                        while (current.charAt(--len1) == next.charAt(--len2)) {
999                                end++;
1000                                if (len2 == start) break; // current is 'abbba', next is 'abba'
1001                                if (len1 == 0) break; // current is 'xabc', next is 'xyabc'
1002                        }
1003                        if (end > 255) end = 255;
1004                        if ((this.bufferIndex + 2) >= BUFFER_WRITE_SIZE)  {
1005                                stream.write(this.streamBuffer, 0, this.bufferIndex);
1006                                this.bufferIndex = 0;
1007                        }
1008                        this.streamBuffer[this.bufferIndex++] = (byte) start;
1009                        this.streamBuffer[this.bufferIndex++] = (byte) end;
1010                        this.streamEnd += 2;
1011 
1012                        int last = next.length() - end;
1013                        writeStreamChars(stream, (start < last ? CharOperation.subarray(next.toCharArray(), start, last) : CharOperation.NO_CHAR));
1014                        current = next;
1015                }
1016        }
1017        this.startOfCategoryTables = this.streamEnd + 1;
1018}

1027private void writeCategoryTable(char[] categoryName, HashtableOfObject wordsToDocs, FileOutputStream stream) throws IOException {
1028        // the format of a category table is as follows:
1029        // any document number arrays with >= 256 elements are written before the table (the offset to each array is remembered)
1030        // then the number of word->int[] pairs in the table is written
1031        // for each word -> int[] pair, the word is written followed by:
1032        //                an int <= 0 if the array size == 1
1033        //                an int > 1 & < 256 for the size of the array if its > 1 & < 256, the document array follows immediately
1034        //                256 if the array size >= 256 followed by another int which is the offset to the array (written prior to the table)
1035 
1036        int largeArraySize = 256;
1037        Object[] values = wordsToDocs.valueTable;
1038        for (int i = 0, l = values.length; i < l; i++) {
1039                Object o = values[i];
1040                if (o != null) {
1041                        if (o instanceof IntList)
1042                                o = values[i] = ((IntList) values[i]).asArray();
1043                        int[] documentNumbers = (int[]) o;
1044                        if (documentNumbers.length >= largeArraySize) {
1045                                values[i] = new Integer(this.streamEnd);
1046                                writeDocumentNumbers(documentNumbers, stream);
1047                        }
1048                }
1049        }
1050 
1051        this.categoryOffsets.put(categoryName, this.streamEnd); // remember the offset to the start of the table
1052        this.categoryTables.put(categoryName, null); // flush cached table
1053        writeStreamInt(stream, wordsToDocs.elementSize);
1054        char[][] words = wordsToDocs.keyTable;
1055        for (int i = 0, l = words.length; i < l; i++) {
1056                Object o = values[i];
1057                if (o != null) {
1058                        writeStreamChars(stream, words[i]);
1059                        if (o instanceof int[]) {
1060                                int[] documentNumbers = (int[]) o;
1061                                if (documentNumbers.length == 1)
1062                                        writeStreamInt(stream, -documentNumbers[0]); // store an array of 1 element by negating the documentNumber (can be zero)
1063                                else
1064                                        writeDocumentNumbers(documentNumbers, stream);
1065                        } else {
1066                                writeStreamInt(stream, largeArraySize); // mark to identify that an offset follows
1067                                writeStreamInt(stream, ((Integer) o).intValue()); // offset in the file of the array of document numbers
1068                        }
1069                }
1070        }
1071}
1072private void writeDocumentNumbers(int[] documentNumbers, FileOutputStream stream) throws IOException {
1073        // must store length as a positive int to detect in-lined array of 1 element
1074        int length = documentNumbers.length;
1075        writeStreamInt(stream, length);
1076        Util.sort(documentNumbers);
1077        int start = 0;
1078        switch (this.documentReferenceSize) {
1079                case 1 :
1080                        while ((this.bufferIndex + length - start) >= BUFFER_WRITE_SIZE) {
1081                                // when documentNumbers is large, write BUFFER_WRITE_SIZE parts & fall thru to write the last part
1082                                int bytesLeft = BUFFER_WRITE_SIZE - this.bufferIndex;
1083                                for (int i=0; i < bytesLeft; i++) {
1084                                        this.streamBuffer[this.bufferIndex++] = (byte) documentNumbers[start++];
1085                                }
1086                                stream.write(this.streamBuffer, 0, this.bufferIndex);
1087                                this.bufferIndex = 0;
1088                        }
1089                        while (start < length) {
1090                                this.streamBuffer[this.bufferIndex++] = (byte) documentNumbers[start++];
1091                        }
1092                        this.streamEnd += length;
1093                        break;
1094                case 2 :
1095                        while ((this.bufferIndex + ((length - start) * 2)) >= BUFFER_WRITE_SIZE) {

1163/**

1165 * encoding in a machine-independent manner.
1166 * <p>
1167 * First, two bytes of the array are giving the number of bytes to
1168 * follow. This value is the number of bytes actually written out,
1169 * not the length of the string. Following the length, each character
1170 * of the string is put in the bytes array, in sequence, using the UTF-8
1171 * encoding for the character.
1172 * </p>
1173 * <p>
1174 * Then the entire byte array is written to the output stream
1175 * using {@link java.io.OutputStream#write(byte[], int, int)} method.
1176 * </p>

1180 *         the bytes array to the stream.

1189        this.streamBuffer[this.bufferIndex++] = (byte) (length & 0xFF); // this will allow to read it faster
1190        this.streamEnd += 2;

1205                        stream.write(this.streamBuffer, 0, this.bufferIndex);
