524 * Create method bindings for binary type, filtering out <clinit> and synthetics

526private void createMethods(ISourceMethod[] iMethods, long sourceLevel) {
527        int total = 0, initialTotal = 0, iClinit = -1;
528        int[] toSkip = null;
529        if (iMethods != null) {
530                total = initialTotal = iMethods.length;
531                boolean keepBridgeMethods = sourceLevel < ClassFileConstants.JDK1_5
532                        && this.environment.globalOptions.complianceLevel >= ClassFileConstants.JDK1_5;
533                for (int i = total; --i >= 0;) {
534                        ISourceMethod method = iMethods[i];
535                        if ((method.getModifiers() & ClassFileConstants.AccSynthetic) != 0) {
536                                if (keepBridgeMethods && (method.getModifiers() & ClassFileConstants.AccBridge) != 0)
537                                        continue; // want to see bridge methods as real methods
538                                // discard synthetics methods
539                                if (toSkip == null) toSkip = new int[iMethods.length];
540                                toSkip[i] = -1;
541                                total--;
542                        } else if (iClinit == -1) {
543                                char[] methodName =((SourceMethod)method).getElementName().toCharArray();
544                                if (methodName.length == 8 && methodName[0] == '<') {
545                                        // discard <clinit>
546                                        iClinit = i;
547                                        total--;
548                                }
549                        }
550                }
551        }
552        if (total == 0) {
553                this.methods = Binding.NO_METHODS;
554                return;
555        }
556 
557        boolean isViewedAsDeprecated = isViewedAsDeprecated();
558        boolean hasRestrictedAccess = hasRestrictedAccess();
559        this.methods = new MethodBinding[total];
560        if (total == initialTotal) {
561                for (int i = 0; i < initialTotal; i++) {
562                        MethodBinding method = createMethod(iMethods[i], sourceLevel);
563                        if (isViewedAsDeprecated && !method.isDeprecated())
564                                method.modifiers |= ExtraCompilerModifiers.AccDeprecatedImplicitly;
565                        if (hasRestrictedAccess)
566                                method.modifiers |= ExtraCompilerModifiers.AccRestrictedAccess;
567                        this.methods[i] = method;
568                }
569        } else {
570                for (int i = 0, index = 0; i < initialTotal; i++) {
571                        if (iClinit != i && (toSkip == null || toSkip[i] != -1)) {
572                                MethodBinding method = createMethod(iMethods[i], sourceLevel);
573                                if (isViewedAsDeprecated && !method.isDeprecated())

575                                if (hasRestrictedAccess)

578                        }
579                }

582private TypeVariableBinding[] createTypeVariables(SignatureWrapper wrapper, boolean assignVariables) {
583        // detect all type variables first
584        char[] typeSignature = wrapper.signature;
585        int depth = 0, length = typeSignature.length;
586        int rank = 0;
587        ArrayList variables = new ArrayList(1);
588        depth = 0;
589        boolean pendingVariable = true;
590        createVariables: {
591                for (int i = 1; i < length; i++) {
592                        switch(typeSignature[i]) {
593                                case '<' :
594                                        depth++;
595                                        break;
596                                case '>' :
597                                        if (--depth < 0)
598                                                break createVariables;
599                                        break;
600                                case ';' :
601                                        if ((depth == 0) && (i +1 < length) && (typeSignature[i+1] != ':'))
602                                                pendingVariable = true;
603                                        break;
604                                default:
605                                        if (pendingVariable) {
606                                                pendingVariable = false;
607                                                int colon = CharOperation.indexOf(':', typeSignature, i);
608                                                char[] variableName = CharOperation.subarray(typeSignature, i, colon);
609                                                variables.add(new TypeVariableBinding(variableName, this, rank++));
610                                        }
611                        }
612                }
613        }
614        // initialize type variable bounds - may refer to forward variables
615        TypeVariableBinding[] result;
616        variables.toArray(result = new TypeVariableBinding[rank]);
617        // when creating the type variables for a type, the type must remember them before initializing each variable
618        // see https://bugs.eclipse.org/bugs/show_bug.cgi?id=163680
619        if (assignVariables)
620                this.typeVariables = result;
621        for (int i = 0; i < rank; i++) {
622                initializeTypeVariable(result[i], result, wrapper);
623        }
624        return result;
625}

766// NOTE: the return type, arg & exception types of each method of a binary type are resolved when needed
767public MethodBinding[] getMethods(char[] selector) {
768        if ((this.tagBits & TagBits.AreMethodsComplete) != 0) {
769                long range;
770                if ((range = ReferenceBinding.binarySearch(selector, this.methods)) >= 0) {
771                        int start = (int) range, end = (int) (range >> 32);
772                        int length = end - start + 1;
773                        if ((this.tagBits & TagBits.AreMethodsComplete) != 0) {
774                                // simply clone method subset
775                                MethodBinding[] result;
776                                System.arraycopy(this.methods, start, result = new MethodBinding[length], 0, length);
777                                return result;
778                        }
779                }
780                return Binding.NO_METHODS;
781        }
782        // lazily sort methods
783        if ((this.tagBits & TagBits.AreMethodsSorted) == 0) {
784                int length = this.methods.length;
785                if (length > 1)
786                        ReferenceBinding.sortMethods(this.methods, 0, length);
787                this.tagBits |= TagBits.AreMethodsSorted;
788        }
789        long range;
790        if ((range = ReferenceBinding.binarySearch(selector, this.methods)) >= 0) {
791                int start = (int) range, end = (int) (range >> 32);
792                int length = end - start + 1;
793                MethodBinding[] result = new MethodBinding[length];
794                // iterate methods to resolve them
795                for (int i = start, index = 0; i <= end; i++, index++)
796                        result[index] = resolveTypesFor(methods[i]);
797                return result;
798        }
799        return Binding.NO_METHODS;
800}

810private void initializeTypeVariable(TypeVariableBinding variable, TypeVariableBinding[] existingVariables, SignatureWrapper wrapper) {
811        // ParameterSignature = Identifier ':' TypeSignature
812        //   or Identifier ':' TypeSignature(optional) InterfaceBound(s)
813        // InterfaceBound = ':' TypeSignature
814        int colon = CharOperation.indexOf(':', wrapper.signature, wrapper.start);
815        wrapper.start = colon + 1; // skip name + ':'
816        ReferenceBinding type, firstBound = null;
817        if (wrapper.signature[wrapper.start] == ':') {
818                type = environment.getResolvedType(TypeConstants.JAVA_LANG_OBJECT, null);
819        } else {
820                type = (ReferenceBinding) environment.getTypeFromTypeSignature(wrapper, existingVariables, this);
821                firstBound = type;
822        }
823 
824        // variable is visible to its bounds
825        variable.modifiers |= ExtraCompilerModifiers.AccUnresolved;
826        variable.superclass = type;
827 
828        ReferenceBinding[] bounds = null;
829        if (wrapper.signature[wrapper.start] == ':') {
830                java.util.ArrayList types = new java.util.ArrayList(2);
831                do {
832                        wrapper.start++; // skip ':'
833                        types.add(environment.getTypeFromTypeSignature(wrapper, existingVariables, this));
834                } while (wrapper.signature[wrapper.start] == ':');
835                bounds = new ReferenceBinding[types.size()];
836                types.toArray(bounds);
837        }
838 
839        variable.superInterfaces = bounds == null ? Binding.NO_SUPERINTERFACES : bounds;
840        if (firstBound == null) {
841                firstBound = variable.superInterfaces.length == 0 ? null : variable.superInterfaces[0];
842        }
843        variable.firstBound = firstBound;
844}
845/**

847 * or for generic types, true if compared to its raw type.
848 */
849public boolean isEquivalentTo(TypeBinding otherType) {
850        if (this == otherType) return true;
851        if (otherType == null) return false;
852        switch(otherType.kind()) {

855                case Binding.RAW_TYPE :
856                        return otherType.erasure() == this;
857        }
858        return false;
859}

903MethodBinding resolveTypesFor(MethodBinding method) {
