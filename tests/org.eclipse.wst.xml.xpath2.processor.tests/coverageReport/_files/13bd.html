472        private void run(Shell shell, IType type, boolean activatedFromEditor) throws CoreException {
473                if (!ElementValidator.check(type, getShell(), getDialogTitle(), activatedFromEditor)) {
474                        notifyResult(false);
475                        return;
476                }
477                if (!ActionUtil.isEditable(fEditor, getShell(), type)) {
478                        notifyResult(false);
479                        return;
480                }
481 
482                AddUnimplementedConstructorsContentProvider provider= new AddUnimplementedConstructorsContentProvider(type);
483                Object[] constructors= provider.getElements(null);
484                if (constructors.length == 0) {
485                        MessageDialog.openInformation(getShell(), getDialogTitle(), ActionMessages.AddUnimplementedConstructorsAction_error_nothing_found);
486                        notifyResult(false);
487                        return;
488                }
489 
490                AddUnimplementedConstructorsDialog dialog= new AddUnimplementedConstructorsDialog(shell, new BindingLabelProvider(), provider, fEditor, type);
491                dialog.setCommentString(ActionMessages.SourceActionDialog_createConstructorComment); 
492                dialog.setTitle(ActionMessages.AddUnimplementedConstructorsAction_dialog_title); 
493                dialog.setInitialSelections(constructors);
494                dialog.setContainerMode(true);
495                dialog.setComparator(new JavaScriptElementComparator());
496                dialog.setSize(60, 18);
497                dialog.setInput(new Object());
498                dialog.setMessage(ActionMessages.AddUnimplementedConstructorsAction_dialog_label); 
499                dialog.setValidator(new AddUnimplementedConstructorsValidator(constructors.length));
500 
501                final int dialogResult= dialog.open();
502                if (dialogResult == Window.OK) {
503                        Object[] elements= dialog.getResult();
504                        if (elements == null) {
505                                notifyResult(false);
506                                return;
507                        }
508                        
509                        ArrayList result= new ArrayList();
510                        for (int i= 0; i < elements.length; i++) {
511                                Object elem= elements[i];
512                                if (elem instanceof IFunctionBinding) {
513                                        result.add(elem);
514                                }
515                        }
516                        IFunctionBinding[] selected= (IFunctionBinding[]) result.toArray(new IFunctionBinding[result.size()]);
517 
518                        CodeGenerationSettings settings= JavaPreferencesSettings.getCodeGenerationSettings(type.getJavaScriptProject());
519                        settings.createComments= dialog.getGenerateComment();
520                        IEditorPart editor= JavaScriptUI.openInEditor(type, true, false);
521                        IRewriteTarget target= editor != null ? (IRewriteTarget) editor.getAdapter(IRewriteTarget.class) : null;
522                        if (target != null)
523                                target.beginCompoundChange();
524                        try {
525                                JavaScriptUnit astRoot= provider.getCompilationUnit();
526                                final ITypeBinding typeBinding= ASTNodes.getTypeBinding(astRoot, type);
527                                int insertPos= dialog.getInsertOffset();
528                                
529                                AddUnimplementedConstructorsOperation operation= (AddUnimplementedConstructorsOperation) createRunnable(astRoot, typeBinding, selected, insertPos, dialog.getGenerateComment(), dialog.getVisibilityModifier(), dialog.isOmitSuper());
530                                IRunnableContext context= JavaScriptPlugin.getActiveWorkbenchWindow();
531                                if (context == null)
532                                        context= new BusyIndicatorRunnableContext();
533                                PlatformUI.getWorkbench().getProgressService().runInUI(context, new WorkbenchRunnableAdapter(operation, operation.getSchedulingRule()), operation.getSchedulingRule());
534                                String[] created= operation.getCreatedConstructors();
535                                if (created == null || created.length == 0)
536                                        MessageDialog.openInformation(shell, getDialogTitle(), ActionMessages.AddUnimplementedConstructorsAction_error_nothing_found); 
537                        } catch (InvocationTargetException e) {
538                                ExceptionHandler.handle(e, shell, getDialogTitle(), null);
539                        } catch (InterruptedException e) {
540                                // Do nothing. Operation has been canceled by user.
541                        } finally {
542                                if (target != null) {
543                                        target.endCompoundChange();
544                                }
545                        }
546                }
547                notifyResult(dialogResult == Window.OK);
548        }
