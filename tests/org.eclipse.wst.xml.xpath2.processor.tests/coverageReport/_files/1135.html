680        private static void moveUp(List elements, List move) {
681                List res= new ArrayList(elements.size());
682                List deleted= new ArrayList();
683                Object floating= null;
684                for (Iterator iter= elements.iterator(); iter.hasNext();) {
685                        Object curr= iter.next();
686                        if (move.contains(curr)) {
687                                res.add(curr);
688                        } else if (((ParameterInfo) curr).isDeleted()) {
689                                deleted.add(curr);
690                        } else {
691                                if (floating != null)
692                                        res.add(floating);
693                                floating= curr;
694                        }
695                }
696                if (floating != null) {
697                        res.add(floating);
698                }
699                res.addAll(deleted);
700                elements.clear();
701                for (Iterator iter= res.iterator(); iter.hasNext();) {
702                        elements.add(iter.next());
703                }
704        }

706        private boolean canMove(boolean up) {
707                int notDeletedInfosCount= getNotDeletedInfosCount();
708                if (notDeletedInfosCount == 0)
709                        return false;
710                int[] indc= getTable().getSelectionIndices();
711                if (indc.length == 0)
712                        return false;
713                int invalid= up ? 0 : notDeletedInfosCount - 1;
714                for (int i= 0; i < indc.length; i++) {
715                        if (indc[i] == invalid)
716                                return false;
717                }
718                return true;
719        }
