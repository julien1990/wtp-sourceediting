478         */
479        protected Object computeInput(Object input) {
480                if (getControl() == null || ! (input instanceof IJavaScriptElement))
481                        return null;
482 
483                IJavaScriptElement je= (IJavaScriptElement)input;
484                String javadocHtml;
485 
486                switch (je.getElementType()) {
487                        case IJavaScriptElement.JAVASCRIPT_UNIT:
488                                try {
489                                        javadocHtml= getJavadocHtml(((IJavaScriptUnit)je).getTypes());
490                                } catch (JavaScriptModelException ex) {
491                                        javadocHtml= null;
492                                }
493                                break;
494                        case IJavaScriptElement.CLASS_FILE:
495                                javadocHtml= getJavadocHtml(new IJavaScriptElement[] {((IClassFile)je).getType()});
496                                break;
497                        default:
498                                javadocHtml= getJavadocHtml(new IJavaScriptElement[] { je });
499                }
500                
501                if (javadocHtml == null)
502                        return ""; //$NON-NLS-1$
503                
504                return javadocHtml;
505        }

508         * @see AbstractInfoView#setInput(Object)
509         */
510        protected void setInput(Object input) {
511                String javadocHtml= (String)input;
512 
513                if (fIsUsingBrowserWidget) {
514                        if (javadocHtml != null && javadocHtml.length() > 0) {
515                                boolean RTL= (getSite().getShell().getStyle() & SWT.RIGHT_TO_LEFT) != 0;
516                                if (RTL) {
517                                        StringBuffer buffer= new StringBuffer(javadocHtml);
518                                        HTMLPrinter.insertStyles(buffer, new String[] { "direction:rtl" } ); //$NON-NLS-1$
519                                        javadocHtml= buffer.toString();
520                                }
521                        }
522                        fBrowser.setText(javadocHtml);
523                } else {
524                        fPresentation.clear();
525                        Rectangle size=  fText.getClientArea();
526 
527                        try {
528                                javadocHtml= ((DefaultInformationControl.IInformationPresenterExtension)fPresenter).updatePresentation(getSite().getShell(), javadocHtml, fPresentation, size.width, size.height);
529                        } catch (IllegalArgumentException ex) {
530                                // the javadoc might no longer be valid
531                                return;
532                        }
533                        fText.setText(javadocHtml);
534                        TextPresentation.applyTextPresentation(fPresentation, fText);
535                }
536        }

538        /**
539         * Returns the Javadoc in HTML format.
540         *
541         * @param result the Java elements for which to get the Javadoc
542         * @return a string with the Javadoc in HTML format.
543         */
544        private String getJavadocHtml(IJavaScriptElement[] result) {
545                StringBuffer buffer= new StringBuffer();
546                int nResults= result.length;
547 
548                if (nResults == 0)
549                        return null;
550 
551                if (nResults > 1) {
552 
553                        for (int i= 0; i < result.length; i++) {
554                                HTMLPrinter.startBulletList(buffer);
555                                IJavaScriptElement curr= result[i];
556                                if (curr instanceof IMember)
557                                        HTMLPrinter.addBullet(buffer, getInfoText((IMember) curr));
558                                HTMLPrinter.endBulletList(buffer);
559                        }
560 
561                } else {
562 
563                        IJavaScriptElement curr= result[0];
564                        if (curr instanceof IMember) {
565                                IMember member= (IMember) curr;
566//                                HTMLPrinter.addSmallHeader(buffer, getInfoText(member));
567                                Reader reader;
568                                try {
569                                        reader= JSdocContentAccess.getHTMLContentReader(member, true, true);
570                                        
571                                        // Provide hint why there's no Javadoc
572                                        if (reader == null && member.isBinary()) {
573                                                boolean hasAttachedJavadoc= JavaDocLocations.getJavadocBaseLocation(member) != null;
574                                                IPackageFragmentRoot root= (IPackageFragmentRoot)member.getAncestor(IJavaScriptElement.PACKAGE_FRAGMENT_ROOT);
575                                                boolean hasAttachedSource= root != null && root.getSourceAttachmentPath() != null;
576                                                IOpenable openable= member.getOpenable();
577                                                boolean hasSource= openable.getBuffer() != null;
578                                                
579                                                if (!hasAttachedSource && !hasAttachedJavadoc)
580                                                        reader= new StringReader(InfoViewMessages.JavadocView_noAttachments);
581                                                else if (!hasAttachedJavadoc && !hasSource)
582                                                        reader= new StringReader(InfoViewMessages.JavadocView_noAttachedJavadoc);
583                                                else if (!hasAttachedSource)
584                                                        reader= new StringReader(InfoViewMessages.JavadocView_noAttachedSource);
585                                                else if (!hasSource)
586                                                        reader= new StringReader(InfoViewMessages.JavadocView_noInformation);
587                                        }
588                                        
589                                } catch (JavaScriptModelException ex) {
590                                        reader= new StringReader(InfoViewMessages.JavadocView_error_gettingJavadoc);
591                                        JavaScriptPlugin.log(ex.getStatus());
592                                }
593                                if (reader != null) {
594                                        HTMLPrinter.addParagraph(buffer, reader);
595                                }
596                        }
597                }
598 
599                boolean flushContent= true;
600                if (buffer.length() > 0 || flushContent) {
601                        HTMLPrinter.insertPageProlog(buffer, 0, fBackgroundColorRGB, fgStyleSheet);
602                        HTMLPrinter.addPageEpilog(buffer);
603                        return buffer.toString();
604                }
605 
606                return null;
607        }
