352                                        //Assert.isTrue(false, "Node type not covered: " + node.getClass().getName());
353                                        return null;
354                        }
355                }

357                String concatStr= prefix + str + suffix;
358                TextEdit edit= formatString(code, concatStr, prefix.length(), str.length(), indentationLevel);
359

361                        edit= shifEdit(edit, prefix.length());
362                }

364        }

366        private static TextEdit shifEdit(TextEdit oldEdit, int diff) {
367                TextEdit newEdit;
368                if (oldEdit instanceof ReplaceEdit) {
369                        ReplaceEdit edit= (ReplaceEdit) oldEdit;
370                        newEdit= new ReplaceEdit(edit.getOffset() - diff, edit.getLength(), edit.getText());
371                } else if (oldEdit instanceof InsertEdit) {
372                        InsertEdit edit= (InsertEdit) oldEdit;
373                        newEdit= new InsertEdit(edit.getOffset() - diff,  edit.getText());
374                } else if (oldEdit instanceof DeleteEdit) {
375                        DeleteEdit edit= (DeleteEdit) oldEdit;
376                        newEdit= new DeleteEdit(edit.getOffset() - diff,  edit.getLength());
377                } else if (oldEdit instanceof MultiTextEdit) {
378                        newEdit= new MultiTextEdit();
379                } else {
380                        return null; // not supported
381                }
382                TextEdit[] children= oldEdit.getChildren();
383                for (int i= 0; i < children.length; i++) {
384                        TextEdit shifted= shifEdit(children[i], diff);
385                        if (shifted != null) {
386                                newEdit.addChild(shifted);
387                        }
388                }
389                return newEdit;
390        }

392        private static Document createDocument(String string, Position[] positions) throws IllegalArgumentException {
393                Document doc= new Document(string);
394                try {
395                        if (positions != null) {
396                                final String POS_CATEGORY= "myCategory"; //$NON-NLS-1$
397 
398                                doc.addPositionCategory(POS_CATEGORY);
399                                doc.addPositionUpdater(new DefaultPositionUpdater(POS_CATEGORY) {
400                                        protected boolean notDeleted() {
401                                                int start= this.fOffset;
402                                                int end= start + this.fLength;
403                                                if (start < this.fPosition.offset && (this.fPosition.offset + this.fPosition.length < end)) {
404                                                        this.fPosition.offset= end; // deleted positions: set to end of remove
405                                                        return false;
406                                                }
407                                                return true;
408                                        }
409                                });
410                                for (int i= 0; i < positions.length; i++) {
411                                        try {
412                                                doc.addPosition(POS_CATEGORY, positions[i]);
413                                        } catch (BadLocationException e) {
414                                                throw new IllegalArgumentException("Position outside of string. offset: " + positions[i].offset + ", length: " + positions[i].length + ", string size: " + string.length());   //$NON-NLS-1$//$NON-NLS-2$//$NON-NLS-3$
415                                        }
416                                }
417                        }
418                } catch (BadPositionCategoryException cannotHappen) {
419                        // can not happen: category is correctly set up
420                }
421                return doc;
422        }
