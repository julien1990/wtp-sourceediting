480private static int consumeWhitespace(char[] typeName, int pos, int length) {
481    while (pos < length) {
482        char currentChar = typeName[pos];
483        if (currentChar != ' ' && !CharOperation.isWhitespace(currentChar)) {

490private static int encodeQualifiedName(char[] typeName, int pos, int length, StringBuffer buffer) {
491    int count = 0;
492    char lastAppendedChar = 0;
493    nameLoop: while (pos < length) {
494            char currentChar = typeName[pos];
495                switch (currentChar) {
496                    case '<' :
497                    case '>' :
498                    case '[' :
499                    case ',' :
500                        break nameLoop;
501                        case '.' :
502                            buffer.append(C_DOT);
503                                lastAppendedChar = C_DOT;
504                            count++;
505                            break;
506                        default:
507                            if (currentChar == ' ' || ScannerHelper.isWhitespace(currentChar)) {
508                                if (lastAppendedChar == C_DOT) { // allow spaces after a dot
509                                    pos = consumeWhitespace(typeName, pos, length) - 1; // will be incremented
510                                    break;
511                                }
512                                // allow spaces before a dot
513                                    int checkPos = checkNextChar(typeName, '.', pos, length, true);
514                                    if (checkPos > 0) {
515                                        buffer.append(C_DOT);                        // process dot immediately to avoid one iteration
516                                        lastAppendedChar = C_DOT;
517                                        count++;
518                                        pos = checkPos;
519                                        break;
520                                    }
521                                    break nameLoop;
522                            }
523                            buffer.append(currentChar);
524                            lastAppendedChar = currentChar;
525                                count++;
526                            break;
527                }
528            pos++;
529    }
530    if (count == 0) throw new IllegalArgumentException(new String(typeName));
531        return pos;
532}

542private static int checkArrayDimension(char[] typeName, int pos, int length) {

573private static int encodeTypeSignature(char[] typeName, int start, boolean isResolved, int length, StringBuffer buffer) {
574    int pos = start;
575    pos = consumeWhitespace(typeName, pos, length);
576    if (pos >= length) throw new IllegalArgumentException(new String(typeName));
577    int checkPos;
578    char currentChar = typeName[pos];
579    switch (currentChar) {
580                // primitive type?
581                case 'b' :
582                    checkPos = checkName(BOOLEAN, typeName, pos, length);
583                    if (checkPos > 0) {
584                        pos = encodeArrayDimension(typeName, checkPos, length, buffer);
585                            buffer.append(C_BOOLEAN);
586                            return pos;
587                        }
588                    checkPos = checkName(BYTE, typeName, pos, length);
589                    if (checkPos > 0) {
590                        pos = encodeArrayDimension(typeName, checkPos, length, buffer);
591                            buffer.append(C_BYTE);
592                            return pos;
593                        }
594                    break;
595                case 'd':
596                    checkPos = checkName(DOUBLE, typeName, pos, length);
597                    if (checkPos > 0) {
598                        pos = encodeArrayDimension(typeName, checkPos, length, buffer);
599                            buffer.append(C_DOUBLE);
600                            return pos;
601                        }
602                    break;
603                case 'f':
604                    checkPos = checkName(FLOAT, typeName, pos, length);
605                    if (checkPos > 0) {
606                        pos = encodeArrayDimension(typeName, checkPos, length, buffer);
607                            buffer.append(C_FLOAT);
608                            return pos;
609                        }
610                    break;
611                case 'i':
612                    checkPos = checkName(INT, typeName, pos, length);
613                    if (checkPos > 0) {
614                        pos = encodeArrayDimension(typeName, checkPos, length, buffer);
615                            buffer.append(C_INT);
616                            return pos;
617                        }
618                    break;
619                case 'l':
620                    checkPos = checkName(LONG, typeName, pos, length);
621                    if (checkPos > 0) {
622                        pos = encodeArrayDimension(typeName, checkPos, length, buffer);
623                            buffer.append(C_LONG);
624                            return pos;
625                        }
626                    break;
627                case 's':
628                    checkPos = checkName(SHORT, typeName, pos, length);
629                    if (checkPos > 0) {
630                        pos = encodeArrayDimension(typeName, checkPos, length, buffer);
631                            buffer.append(C_SHORT);
632                            return pos;
633                        }
634                    break;
635                case 'v':
636                    checkPos = checkName(VOID, typeName, pos, length);
637                    if (checkPos > 0) {
638                        pos = encodeArrayDimension(typeName, checkPos, length, buffer);
639                            buffer.append(C_VOID);
640                            return pos;
641                        }
642                    break;
643                case 'c':
644                    checkPos = checkName(CHAR, typeName, pos, length);
645                    if (checkPos > 0) {
646                        pos = encodeArrayDimension(typeName, checkPos, length, buffer);
647                            buffer.append(C_CHAR);
648                            return pos;
649                        } else {
650                                break;
651                        }
652    }
653    // non primitive type
654    checkPos = checkArrayDimension(typeName, pos, length);
655        int end;
656        if (checkPos > 0) {
657            end = encodeArrayDimension(typeName, checkPos, length, buffer);
658        } else {
659            end = -1;
660        }
661        buffer.append(isResolved ? C_RESOLVED : C_UNRESOLVED);
662        while (true) { // loop on qualifiedName[<args>][.qualifiedName[<args>]*
663            pos = encodeQualifiedName(typeName, pos, length, buffer);
664                checkPos = checkNextChar(typeName, '.', pos, length, true);
665                if (checkPos > 0) {
666                        buffer.append(C_DOT);
667                        pos = checkPos;
668                } else {
669                        break;
670                }
671        }
672        buffer.append(C_NAME_END);
673        if (end > 0) pos = end; // skip array dimension which were preprocessed
674    return pos;
675}
