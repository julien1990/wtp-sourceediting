397        protected RefactoringStatus doCheckFinalConditions(IProgressMonitor pm, CheckConditionsContext context) throws CoreException {
398                try{
399                        pm.beginTask("", 18); //$NON-NLS-1$
400                        pm.setTaskName(RefactoringCoreMessages.RenameFieldRefactoring_checking); 
401                        RefactoringStatus result= new RefactoringStatus();
402                        result.merge(Checks.checkIfCuBroken(fField));
403                        if (result.hasFatalError())
404                                return result;
405                        result.merge(checkNewElementName(getNewElementName()));
406                        pm.worked(1);
407                        result.merge(checkEnclosingHierarchy());
408                        pm.worked(1);
409                        result.merge(checkNestedHierarchy(fField.getDeclaringType()));
410                        pm.worked(1);
411                        
412                        if (fUpdateReferences){
413                                pm.setTaskName(RefactoringCoreMessages.RenameFieldRefactoring_searching);         
414                                fReferences= getReferences(new SubProgressMonitor(pm, 3), result);
415                                pm.setTaskName(RefactoringCoreMessages.RenameFieldRefactoring_checking); 
416                        } else {
417                                fReferences= new SearchResultGroup[0];
418                                pm.worked(3);
419                        }        
420                        
421                        if (fUpdateReferences)
422                                result.merge(analyzeAffectedCompilationUnits());
423                        else
424                                Checks.checkCompileErrorsInAffectedFile(result, fField.getResource());
425                                
426                        if (getGetter() != null && fRenameGetter){
427                                result.merge(checkAccessor(new SubProgressMonitor(pm, 1), getGetter(), getNewGetterName()));
428                                result.merge(Checks.checkIfConstructorName(getGetter(), getNewGetterName(), fField.getDeclaringType().getElementName()));
429                        } else {
430                                pm.worked(1);
431                        }
432                                
433                        if (getSetter() != null && fRenameSetter){
434                                result.merge(checkAccessor(new SubProgressMonitor(pm, 1), getSetter(), getNewSetterName()));
435                                result.merge(Checks.checkIfConstructorName(getSetter(), getNewSetterName(), fField.getDeclaringType().getElementName()));
436                        } else {
437                                pm.worked(1);
438                        }
439                        
440                        result.merge(createChanges(new SubProgressMonitor(pm, 10)));
441                        if (result.hasFatalError())
442                                return result;
443                        
444                        return result;
445                } finally{
446                        pm.done();
447                }
448        }

562        public Change createChange(IProgressMonitor monitor) throws CoreException {
563                try {
564                        monitor.beginTask(RefactoringCoreMessages.RenameFieldRefactoring_checking, 1);
565                        final TextChange[] changes= fChangeManager.getAllChanges();
566                        final List list= new ArrayList(changes.length);
567                        list.addAll(Arrays.asList(changes));
568                        String project= null;
569                        IJavaScriptProject javaProject= fField.getJavaScriptProject();
570                        if (javaProject != null)
571                                project= javaProject.getElementName();
572                        int flags= JavaScriptRefactoringDescriptor.JAR_MIGRATION | JavaScriptRefactoringDescriptor.JAR_REFACTORING | RefactoringDescriptor.STRUCTURAL_CHANGE;
573                        try {

584                                JavaScriptPlugin.log(exception);
585                        }

588                        final JDTRefactoringDescriptorComment comment= new JDTRefactoringDescriptorComment(project, this, header);

591                        if (fRenameSetter)

593                        final RenameJavaScriptElementDescriptor descriptor= new RenameJavaScriptElementDescriptor(IJavaScriptRefactorings.RENAME_FIELD);
594                        descriptor.setProject(project);

597                        descriptor.setFlags(flags);
598                        descriptor.setJavaElement(fField);
599                        descriptor.setNewName(getNewElementName());
600                        descriptor.setUpdateReferences(fUpdateReferences);
601                        descriptor.setUpdateTextualOccurrences(fUpdateTextualMatches);
602                        descriptor.setRenameGetters(fRenameGetter);
603                        descriptor.setRenameSetters(fRenameSetter);
604                        descriptor.setKeepOriginal(fDelegateUpdating);
605                        descriptor.setDeprecateDelegate(fDelegateDeprecation);
606                        return new DynamicValidationRefactoringChange(descriptor, RefactoringCoreMessages.RenameFieldRefactoring_change_name, (Change[]) list.toArray(new Change[list.size()]));
607                } finally {

609                }

663        private RefactoringStatus addDelegates() throws JavaScriptModelException, CoreException {
664 
665                RefactoringStatus status= new RefactoringStatus();
666                CompilationUnitRewrite rewrite= new CompilationUnitRewrite(fField.getJavaScriptUnit());
667                rewrite.setResolveBindings(true);
668 
669                // add delegate for the field
670                if (RefactoringAvailabilityTester.isDelegateCreationAvailable(fField)) {
671                        FieldDeclaration fieldDeclaration= ASTNodeSearchUtil.getFieldDeclarationNode(fField, rewrite.getRoot());
672                        if (fieldDeclaration.fragments().size() > 1) {
673                                status.addWarning(Messages.format(RefactoringCoreMessages.DelegateCreator_cannot_create_field_delegate_more_than_one_fragment, fField
674                                                .getElementName()), JavaStatusContext.create(fField));
675                        } else if (((VariableDeclarationFragment) fieldDeclaration.fragments().get(0)).getInitializer() == null) {
676                                status.addWarning(Messages.format(RefactoringCoreMessages.DelegateCreator_cannot_create_field_delegate_no_initializer, fField
677                                                .getElementName()), JavaStatusContext.create(fField));
678                        } else {
679                                DelegateFieldCreator creator= new DelegateFieldCreator();
680                                creator.setDeclareDeprecated(fDelegateDeprecation);
681                                creator.setDeclaration(fieldDeclaration);
682                                creator.setNewElementName(getNewElementName());
683                                creator.setSourceRewrite(rewrite);
684                                creator.prepareDelegate();
685                                creator.createEdit();
686                        }
687                }
688 
689                // add delegates for getter and setter methods
690                // there may be getters even if the field is static final
691                if (getGetter() != null && fRenameGetter)
692                        addMethodDelegate(getGetter(), getNewGetterName(), rewrite);
693                if (getSetter() != null && fRenameSetter)
694                        addMethodDelegate(getSetter(), getNewSetterName(), rewrite);
695 
696                final CompilationUnitChange change= rewrite.createChange();
697                if (change != null) {
698                        change.setKeepPreviewEdits(true);
699                        fChangeManager.manage(fField.getJavaScriptUnit(), change);
700                }
701 
702                return status;
703        }

777        private RefactoringStatus analyzeRenameChanges(IProgressMonitor pm) throws CoreException {
778                IJavaScriptUnit[] newWorkingCopies= null;
779                WorkingCopyOwner newWCOwner= new WorkingCopyOwner() { /* must subclass */ };
780                try {
781                        pm.beginTask("", 2); //$NON-NLS-1$
782                        RefactoringStatus result= new RefactoringStatus();
783                        SearchResultGroup[] oldReferences= fReferences;
784 
785                        List compilationUnitsToModify= new ArrayList();
786                        if (fIsComposite) {
787                                // limited change set, no accessors.
788                                for (int i= 0; i < oldReferences.length; i++) 
789                                        compilationUnitsToModify.add(oldReferences[i].getCompilationUnit());
790                                compilationUnitsToModify.add(fField.getJavaScriptUnit());
791                        } else {
792                                // include all cus, including accessors
793                                compilationUnitsToModify.addAll(Arrays.asList(fChangeManager.getAllCompilationUnits()));
794                        }
795                        
796                        newWorkingCopies= RenameAnalyzeUtil.createNewWorkingCopies((IJavaScriptUnit[]) compilationUnitsToModify.toArray(new IJavaScriptUnit[compilationUnitsToModify.size()]),
797                                        fChangeManager, newWCOwner, new SubProgressMonitor(pm, 1));
798                        
799                        SearchResultGroup[] newReferences= getNewReferences(new SubProgressMonitor(pm, 1), result, newWCOwner, newWorkingCopies);
800                        result.merge(RenameAnalyzeUtil.analyzeRenameChanges2(fChangeManager, oldReferences, newReferences, getNewElementName()));
801                        return result;
802                } finally{
803                        pm.done();
804                        if (newWorkingCopies != null){
805                                for (int i= 0; i < newWorkingCopies.length; i++) {
806                                        newWorkingCopies[i].discardWorkingCopy();
807                                }
808                        }
809                }
810        }

812        private SearchResultGroup[] getNewReferences(IProgressMonitor pm, RefactoringStatus status, WorkingCopyOwner owner, IJavaScriptUnit[] newWorkingCopies) throws CoreException {
813                pm.beginTask("", 2); //$NON-NLS-1$
814                IJavaScriptUnit declaringCuWorkingCopy= RenameAnalyzeUtil.findWorkingCopyForCu(newWorkingCopies, fField.getJavaScriptUnit());
815                if (declaringCuWorkingCopy == null)
816                        return new SearchResultGroup[0];
817                
818                IField field= getFieldInWorkingCopy(declaringCuWorkingCopy, getNewElementName());
819                if (field == null || ! field.exists())
820                        return new SearchResultGroup[0];
821                
822                CollectingSearchRequestor requestor= null;
823                if (fDelegateUpdating && RefactoringAvailabilityTester.isDelegateCreationAvailable(getField())) {
824                        // There will be two new matches inside the delegate (the invocation
825                        // and the javadoc) which are OK and must not be reported.
826                        final IField oldField= getFieldInWorkingCopy(declaringCuWorkingCopy, getCurrentElementName());
827                        requestor= new CollectingSearchRequestor() {
828                                public void acceptSearchMatch(SearchMatch match) throws CoreException {
829                                        if (!oldField.equals(match.getElement()))
830                                                super.acceptSearchMatch(match);
831                                }
832                        };
833                } else
834                        requestor= new CollectingSearchRequestor();
835                
836                SearchPattern newPattern= SearchPattern.createPattern(field, IJavaScriptSearchConstants.REFERENCES);                        
837                return RefactoringSearchEngine.search(newPattern, owner, createRefactoringScope(), requestor, new SubProgressMonitor(pm, 1), status);
838        }
