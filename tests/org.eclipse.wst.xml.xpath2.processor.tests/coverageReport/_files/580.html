227        private UrlInfo[] pickupUrl(IStructuredDocument document) {
228                List urls = new ArrayList();
229                IStructuredDocumentRegion documentRegion = document.getFirstStructuredDocumentRegion();
230                while (documentRegion != null) {
231                        ITextRegionList regionList = documentRegion.getRegions();
232                        Iterator i = regionList.iterator();
233                        while (i.hasNext()) {
234                                ITextRegion textRegion = (ITextRegion) i.next();
235                                String type = textRegion.getType();
236                                if (type.equals(CSSRegionContexts.CSS_URI) || type.equals(CSSRegionContexts.CSS_DECLARATION_VALUE_URI)) {
237                                        urls.add(new UrlInfo(type, documentRegion.getFullText(textRegion)));
238                                }
239                        }
240                        documentRegion = documentRegion.getNext();
241                }
242                return (UrlInfo[]) urls.toArray(new UrlInfo[urls.size()]);
243        }
