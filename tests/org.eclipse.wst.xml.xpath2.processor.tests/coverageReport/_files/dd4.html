584        private void computeConstantDeclarationLocation() throws JavaScriptModelException {
585                if (isDeclarationLocationComputed())
586                        return;
587 
588                BodyDeclaration lastStaticDependency= null;
589                Iterator decls= getBodyDeclarations();
590                
591                Assert.isTrue(decls.hasNext()); /* Admissible selected expressions must occur
592                                                   within a body declaration.  Thus, the 
593                                                   class/interface in which such an expression occurs
594                                                   must have at least one body declaration */
595                
596                while (decls.hasNext()) {
597                        BodyDeclaration decl= (BodyDeclaration) decls.next();
598                        
599                        int modifiers;
600                        if (decl instanceof FieldDeclaration)
601                                modifiers= ((FieldDeclaration) decl).getModifiers();
602                        else if (decl instanceof Initializer)
603                                modifiers= ((Initializer) decl).getModifiers();
604                        else {
605                                continue; /* this declaration is not a field declaration
606                                              or initializer, so the placement of the constant
607                                              declaration relative to it does not matter */
608                        }
609                        
610                        if (Modifier.isStatic(modifiers) && depends(getSelectedExpression(), decl))
611                                lastStaticDependency= decl;
612                }
613                
614                if(lastStaticDependency == null)
615                        fInsertFirst= true;
616                else
617                        fToInsertAfter= lastStaticDependency;
618        }

620        /* bd is a static field declaration or static initializer */
621        private static boolean depends(IExpressionFragment selected, BodyDeclaration bd) {
622                /* We currently consider selected to depend on bd only if db includes a declaration
623                 * of a static field on which selected depends.
624                 * 
625                 * A more accurate strategy might be to also check if bd contains (or is) a
626                 * static initializer containing code which changes the value of a static field on 
627                 * which selected depends.  However, if a static is written to multiple times within
628                 * during class initialization, it is difficult to predict which value should be used.
629                 * This would depend on which value is used by expressions instances for which the new 
630                 * constant will be substituted, and there may be many of these; in each, the
631                 * static field in question may have taken on a different value (if some of these uses
632                 * occur within static initializers).
633                 */
634                
635                if(bd instanceof FieldDeclaration) {
636                        FieldDeclaration fieldDecl = (FieldDeclaration) bd;
637                        for(Iterator fragments = fieldDecl.fragments().iterator(); fragments.hasNext();) {
638                                VariableDeclarationFragment fragment = (VariableDeclarationFragment) fragments.next();
639                                SimpleName staticFieldName = fragment.getName();
640                                if(selected.getSubFragmentsMatching(ASTFragmentFactory.createFragmentForFullSubtree(staticFieldName)).length != 0)
641                                        return true;
642                        }
643                }
644                return false;
645        }
