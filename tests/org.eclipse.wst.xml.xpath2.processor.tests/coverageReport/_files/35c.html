568         * put the final java document together
569         */
570        private final void buildResult() {
571                // to build the java document this is the order:
572                // 
573                // + default imports
574                // + user imports
575                // + class header
576                // [+ error page]
577                // + user declarations
578                // + service method header
579                // + try/catch start
580                // + user code
581                // + try/catch end
582                // + service method footer
583                fResult = new StringBuffer(fImplicitImports.length() + fUserImports.length() + fClassHeader.length() + fUserDeclarations.length() + fServiceHeader.length() + fTryCatchStart.length() // try/catch
584                                        // start
585                                        + fUserCode.length() + fTryCatchEnd.length() // try/catch
586                                        // end
587                                        + fFooter.length());
588 
589                int javaOffset = 0;
590 
591                fResult.append(fImplicitImports);
592                javaOffset += fImplicitImports.length();
593 
594                // updateRanges(fIndirectImports, javaOffset);
595                updateRanges(fImportRanges, javaOffset);
596                // user imports
597                append(fUserImports);
598                javaOffset += fUserImports.length();
599 
600                // class header
601                fResult.append(fClassHeader); //$NON-NLS-1$
602                javaOffset += fClassHeader.length();
603                fResult.append(fSuperclass + "{" + ENDL); //$NON-NLS-1$
604                javaOffset += fSuperclass.length() + 2;
605 
606                updateRanges(fDeclarationRanges, javaOffset);
607                // user declarations
608                append(fUserDeclarations);
609                javaOffset += fUserDeclarations.length();
610 
611                updateRanges(fUserELRanges, javaOffset);
612                append(fUserELExpressions);
613                javaOffset += fUserELExpressions.length();
614 
615                fResult.append(fServiceHeader);
616                javaOffset += fServiceHeader.length();
617                // session participant
618                if (fIsInASession) {
619                        fResult.append(fSessionVariableDeclaration);
620                        javaOffset += fSessionVariableDeclaration.length();
621                }
622                // error page
623                if (fIsErrorPage) {
624                        fResult.append(fException);
625                        javaOffset += fException.length();
626                }
627 
628 
629                fResult.append(fTryCatchStart);
630                javaOffset += fTryCatchStart.length();
631 
632                updateRanges(fCodeRanges, javaOffset);
633 
634                // user code
635                append(fUserCode);
636                javaOffset += fUserCode.length();
637 
638 
639                fResult.append(fTryCatchEnd);
640                javaOffset += fTryCatchEnd.length();
641 
642                // footer
643                fResult.append(fFooter);
644                javaOffset += fFooter.length();
645 
646                fJava2JspRanges.putAll(fImportRanges);
647                fJava2JspRanges.putAll(fDeclarationRanges);
648                fJava2JspRanges.putAll(fCodeRanges);
649                fJava2JspRanges.putAll(fUserELRanges);
650 
651        }
