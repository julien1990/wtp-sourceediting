521        private static void fillWithTypeStubs(final StringBuffer bufBefore, final StringBuffer bufAfter, final int focalPosition, List/*<? extends BodyDeclaration>*/ types) {
522                StringBuffer buf;
523                for (Iterator iter= types.iterator(); iter.hasNext();) {
524                        BodyDeclaration bodyDeclaration= (BodyDeclaration) iter.next();
525                        if (! (bodyDeclaration instanceof AbstractTypeDeclaration)) {
526                                //account for local classes:
527                                if (! (bodyDeclaration instanceof FunctionDeclaration))
528                                        continue;
529                                int bodyStart= bodyDeclaration.getStartPosition();
530                                int bodyEnd= bodyDeclaration.getStartPosition() + bodyDeclaration.getLength();
531                                if (! (bodyStart < focalPosition && focalPosition < bodyEnd))
532                                        continue;
533                                FunctionDeclaration methodDeclaration= (FunctionDeclaration) bodyDeclaration;
534                                buf= bufBefore;
535                                appendModifiers(buf, methodDeclaration.modifiers());
536                                appendTypeParameters(buf, methodDeclaration.typeParameters());
537                                buf.append(" void "); //$NON-NLS-1$
538                                buf.append(methodDeclaration.getName().getIdentifier());
539                                buf.append("(){\n"); //$NON-NLS-1$
540                                Block body= methodDeclaration.getBody();
541                                body.accept(new HierarchicalASTVisitor() {
542                                        public boolean visit(AbstractTypeDeclaration node) {
543                                                fillWithTypeStubs(bufBefore, bufAfter, focalPosition, Collections.singletonList(node));
544                                                return false;
545                                        }
546                                        public boolean visit(ClassInstanceCreation node) {
547                                                AnonymousClassDeclaration anonDecl= node.getAnonymousClassDeclaration();
548                                                if (anonDecl == null)
549                                                        return false;
550                                                int anonStart= anonDecl.getStartPosition();
551                                                int anonEnd= anonDecl.getStartPosition() + anonDecl.getLength();
552                                                if (! (anonStart < focalPosition && focalPosition < anonEnd))
553                                                        return false;
554                                                bufBefore.append(" new "); //$NON-NLS-1$
555                                                bufBefore.append(node.getType().toString());
556                                                bufBefore.append("(){\n"); //$NON-NLS-1$
557                                                fillWithTypeStubs(bufBefore, bufAfter, focalPosition, anonDecl.bodyDeclarations());
558                                                bufAfter.insert(0, "};\n"); //$NON-NLS-1$
559                                                return false;
560                                        }
561                                });
562                                buf= bufAfter;
563                                buf.append("}\n"); //$NON-NLS-1$
564                                continue;
565                        }
566                        
567                        AbstractTypeDeclaration decl= (AbstractTypeDeclaration) bodyDeclaration;
568                        buf= decl.getStartPosition() < focalPosition ? bufBefore : bufAfter;
569                        appendModifiers(buf, decl.modifiers());
570                        
571                        if (decl instanceof TypeDeclaration) {
572                                TypeDeclaration type= (TypeDeclaration) decl;
573                                buf.append(type.isInterface() ? "interface " : "class "); //$NON-NLS-1$//$NON-NLS-2$
574                                buf.append(type.getName().getIdentifier());
575                                appendTypeParameters(buf, type.typeParameters());
576                                if (type.getSuperclassType() != null) {
577                                        buf.append(" extends "); //$NON-NLS-1$
578                                        buf.append(ASTNodes.asString(type.getSuperclassType()));
579                                }
580                                List superInterfaces= type.superInterfaceTypes();
581                                appendSuperInterfaces(buf, superInterfaces);
582                                
583                        }
584                        
585                        buf.append("{\n"); //$NON-NLS-1$
586                        fillWithTypeStubs(bufBefore, bufAfter, focalPosition, decl.bodyDeclarations());
587                        buf= decl.getStartPosition() + decl.getLength() < focalPosition ? bufBefore : bufAfter;
588                        buf.append("}\n"); //$NON-NLS-1$
589                }
590        }
