641         * 
642         * @param event
643         *            the event
644         */
645        private void handleChange(final IDiffChangeEvent event) {
646                final Set existing= getVisibleProjects();
647                // Get all existing and open projects that contain changes
648                // and determine what needs to be done to the project
649                // (i.e. add, remove or refresh)
650                final IJavaScriptProject[] changed= getChangedProjects(event);
651                final List refreshes= new ArrayList(changed.length);
652                final List additions= new ArrayList(changed.length);
653                final List removals= new ArrayList(changed.length);
654                for (int index= 0; index < changed.length; index++) {
655                        final IJavaScriptProject project= changed[index];
656                        if (hasVisibleChanges(event.getTree(), project)) {
657                                if (existing.contains(project))
658                                        refreshes.add(project);
659                                else
660                                        additions.add(project);
661                        } else
662                                removals.add(project);
663                }
664                // Remove any java projects that correspond to deleted or closed projects
665                final Set removed= getDeletedProjects(event);
666                for (final Iterator iterator= existing.iterator(); iterator.hasNext();) {
667                        final IJavaScriptProject element= (IJavaScriptProject) iterator.next();
668                        if (removed.contains(element.getResource()))
669                                removals.add(element);
670                }
671 
672                if (!removals.isEmpty() || !additions.isEmpty() || !refreshes.isEmpty()) {
673                        final TreeViewer viewer= (TreeViewer) getViewer();
674                        final Tree tree= viewer.getTree();
675                        try {
676                                tree.setRedraw(false);
677                                if (!additions.isEmpty())
678                                        viewer.add(viewer.getInput(), additions.toArray());
679                                if (!removals.isEmpty())
680                                        viewer.remove(viewer.getInput(), removals.toArray());
681                                if (!refreshes.isEmpty()) {
682                                        for (final Iterator iter= refreshes.iterator(); iter.hasNext();)
683                                                viewer.refresh(iter.next());
684                                }
685                        } finally {
686                                tree.setRedraw(true);
687                        }
688                }
689        }
