469public static String displayString(String inputString, int indent, boolean shift) {
470        if (inputString == null)
471                return "null";
472        int length = inputString.length();
473        StringBuffer buffer = new StringBuffer(length);
474        java.util.StringTokenizer tokenizer = new java.util.StringTokenizer(inputString, "\n\r", true);
475        for (int i = 0; i < indent; i++) buffer.append("\t");
476        if (shift) indent++;
477        buffer.append("\"");
478        while (tokenizer.hasMoreTokens()){
479 
480                String token = tokenizer.nextToken();
481                if (token.equals("\r")) {
482                        buffer.append("\\r");
483                        if (tokenizer.hasMoreTokens()) {
484                                token = tokenizer.nextToken();
485                                if (token.equals("\n")) {
486                                        buffer.append("\\n");
487                                        if (tokenizer.hasMoreTokens()) {
488                                                buffer.append("\" + \n");
489                                                for (int i = 0; i < indent; i++) buffer.append("\t");
490                                                buffer.append("\"");
491                                        }
492                                        continue;
493                                }
494                                buffer.append("\" + \n");
495                                for (int i = 0; i < indent; i++) buffer.append("\t");
496                                buffer.append("\"");
497                        } else {
498                                continue;
499                        }
500                } else if (token.equals("\n")) {
501                        buffer.append("\\n");
502                        if (tokenizer.hasMoreTokens()) {
503                                buffer.append("\" + \n");
504                                for (int i = 0; i < indent; i++) buffer.append("\t");
505                                buffer.append("\"");
506                        }
507                        continue;
508                }        
509 
510                StringBuffer tokenBuffer = new StringBuffer();
511                for (int i = 0; i < token.length(); i++){ 
512                        char c = token.charAt(i);
513                        switch (c) {
514                                case '\r' :
515                                        tokenBuffer.append("\\r");
516                                        break;
517                                case '\n' :
518                                        tokenBuffer.append("\\n");
519                                        break;
520                                case '\b' :
521                                        tokenBuffer.append("\\b");
522                                        break;
523                                case '\t' :
524                                        tokenBuffer.append("\t");
525                                        break;
526                                case '\f' :
527                                        tokenBuffer.append("\\f");
528                                        break;
529                                case '\"' :
530                                        tokenBuffer.append("\\\"");
531                                        break;
532                                case '\'' :
533                                        tokenBuffer.append("\\'");
534                                        break;
535                                case '\\' :
536                                        tokenBuffer.append("\\\\");
537                                        break;
538                                default :
539                                        tokenBuffer.append(c);
540                        }
541                }
542                buffer.append(tokenBuffer.toString());
543        }
544        buffer.append("\"");
545        return buffer.toString();
546}
