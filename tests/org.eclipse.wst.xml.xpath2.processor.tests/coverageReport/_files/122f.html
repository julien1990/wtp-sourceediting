497        public static void insertTag(ListRewrite rewriter, TagElement newElement, Set sameKindLeadingNames, TextEditGroup groupDescription) {
498                List tags= rewriter.getRewrittenList();
499 
500                String insertedTagName= newElement.getTagName();
501 
502                ASTNode after= null;
503                int tagRanking= getTagRanking(insertedTagName);
504                for (int i= tags.size() - 1; i >= 0; i--) {
505                        TagElement curr= (TagElement) tags.get(i);
506                        String tagName= curr.getTagName();
507                        if (tagName == null || tagRanking > getTagRanking(tagName)) {
508                                after= curr;
509                                break;
510                        }
511                        if (sameKindLeadingNames != null && isSameTag(insertedTagName, tagName)) {
512                                String arg= getArgument(curr);
513                                if (arg != null && sameKindLeadingNames.contains(arg)) {
514                                        after= curr;
515                                        break;
516                                }
517                        }
518                }
519                if (after != null) {
520                        rewriter.insertAfter(newElement, after, groupDescription);
521                } else {
522                        rewriter.insertFirst(newElement, groupDescription);
523                }
524        }

560        private static String getArgument(TagElement curr) {
561                List fragments= curr.fragments();
562                if (!fragments.isEmpty()) {
563                        Object first= fragments.get(0);
564                        if (first instanceof Name) {
565                                return ASTNodes.getSimpleNameIdentifier((Name) first);
566                        } else if (first instanceof TextElement && TagElement.TAG_PARAM.equals(curr.getTagName())) {
567                                String text= ((TextElement) first).getText();
568                                if ("<".equals(text) && fragments.size() >= 3) { //$NON-NLS-1$
569                                        Object second= fragments.get(1);
570                                        Object third= fragments.get(2);
571                                        if (second instanceof Name && third instanceof TextElement && ">".equals(((TextElement) third).getText())) { //$NON-NLS-1$
572                                                return '<' + ASTNodes.getSimpleNameIdentifier((Name) second) + '>';
573                                        }
574                                } else if (text.startsWith(String.valueOf('<')) && text.endsWith(String.valueOf('>')) && text.length() > 2) {
575                                        return text.substring(1, text.length() - 1);
576                                }
577                        }
578                }
579                return null;
580        }
