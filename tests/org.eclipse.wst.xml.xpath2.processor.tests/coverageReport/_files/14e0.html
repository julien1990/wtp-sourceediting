445         * no longer need to re-use old document instance.
446         */
447        private void transformInstance(IStructuredDocument oldInstance, IStructuredDocument newInstance) {
448                /**
449                 * https://w3.opensource.ibm.com/bugzilla/show_bug.cgi?id=4920
450                 * 
451                 * JSP taglib support broken, correct by duplicating extended setup

453                 * StructuredDocumentRegionParser extensions)
454                 */

456                RegionParser newParser = newInstance.getParser().newInstance();
457                // Register all of the old StructuredDocumentRegionHandlers on the new
458                // parser
459                if (oldParser instanceof StructuredDocumentRegionParserExtension && newParser instanceof StructuredDocumentRegionParserExtension) {
460                        List oldHandlers = ((StructuredDocumentRegionParserExtension) oldParser).getStructuredDocumentRegionHandlers();
461                        for (int i = 0; i < oldHandlers.size(); i++) {
462                                StructuredDocumentRegionHandler handler = ((StructuredDocumentRegionHandler) oldHandlers.get(i));
463                                if (handler instanceof StructuredDocumentRegionHandlerExtension) {
464                                        /**
465                                         * Skip the transferring here, the handler will do this
466                                         * after everything else but the source is transferred.
467                                         */
468                                }
469                                else {
470                                        ((StructuredDocumentRegionParser) oldParser).removeStructuredDocumentRegionHandler(handler);
471                                        ((StructuredDocumentRegionParser) newParser).addStructuredDocumentRegionHandler(handler);
472                                        handler.resetNodes();
473                                }
474                        }
475                }
476                // Add any global BlockMarkers to the new parser
477                if (oldParser instanceof BlockTagParser && newParser instanceof BlockTagParser) {
478                        List oldBlockMarkers = ((BlockTagParser) oldParser).getBlockMarkers();
479                        for (int i = 0; i < oldBlockMarkers.size(); i++) {
480                                BlockMarker blockMarker = ((BlockMarker) oldBlockMarkers.get(i));
481                                if (blockMarker.isGlobal()) {
482                                        ((BlockTagParser) newParser).addBlockMarker(blockMarker);
483                                }
484                        }
485                }
486 
487                ((BasicStructuredDocument) oldInstance).setParser(newParser);
488 
489                ((BasicStructuredDocument) oldInstance).setReParser(newInstance.getReParser().newInstance());
490 
491                if (newInstance.getDocumentPartitioner() instanceof StructuredTextPartitioner) {
492                        StructuredTextPartitioner partitioner = null;
493                        if (oldInstance instanceof IDocumentExtension3 && newInstance instanceof IDocumentExtension3) {
494                                partitioner = ((StructuredTextPartitioner) ((IDocumentExtension3) newInstance).getDocumentPartitioner(IStructuredPartitioning.DEFAULT_STRUCTURED_PARTITIONING));
495                                if (partitioner != null) {
496                                        partitioner = (StructuredTextPartitioner) partitioner.newInstance();
497                                }
498                                ((IDocumentExtension3) oldInstance).setDocumentPartitioner(IStructuredPartitioning.DEFAULT_STRUCTURED_PARTITIONING, partitioner);
499                        }
500                        if (partitioner == null) {
501                                partitioner = (StructuredTextPartitioner) ((StructuredTextPartitioner) newInstance.getDocumentPartitioner()).newInstance();
502                                oldInstance.setDocumentPartitioner(partitioner);
503                        }
504                        if (partitioner != null) {
505                                partitioner.connect(oldInstance);
506                        }
507                }
508 
509                String existingLineDelimiter = null;
510                try {
511                        existingLineDelimiter = newInstance.getLineDelimiter(0);
512                }
513                catch (BadLocationException e) {
514                        // if empty file, assume platform default
515                        // TODO: should be using user set preference, per content type?
516                        existingLineDelimiter = System.getProperty("line.separator"); //$NON-NLS-1$
517                }
518 
519                oldInstance.setLineDelimiter(existingLineDelimiter); //$NON-NLS-1$);
520                if (newInstance.getEncodingMemento() != null) {
521                        oldInstance.setEncodingMemento((EncodingMemento) newInstance.getEncodingMemento().clone());
522                }
523 
524                /**
525                 * https://w3.opensource.ibm.com/bugzilla/show_bug.cgi?id=4920
526                 *

529                 * StructuredDocumentRegionParser extensions)

537                                        handlerExtension.setStructuredDocument(oldInstance);
538                                }
539                        }
540                }

542                newInstance = null;
