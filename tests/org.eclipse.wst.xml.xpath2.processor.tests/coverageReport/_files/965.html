359 *

361public static final boolean camelCaseMatch(String pattern, int patternStart, int patternEnd, String name, int nameStart, int nameEnd) {
362        if (name == null)
363                return false; // null name cannot match
364        if (pattern == null)
365                return true; // null pattern is equivalent to '*'
366        if (patternEnd < 0)         patternEnd = pattern.length();
367        if (nameEnd < 0) nameEnd = name.length();
368 
369        if (patternEnd <= patternStart) return nameEnd <= nameStart;
370        if (nameEnd <= nameStart) return false;
371        // check first pattern char
372        if (name.charAt(nameStart) != pattern.charAt(patternStart)) {
373                // first char must strictly match (upper/lower)
374                return false;
375        }
376 
377        char patternChar, nameChar;
378        int iPattern = patternStart;
379        int iName = nameStart;
380 
381        // Main loop is on pattern characters
382        while (true) {
383 
384                iPattern++;
385                iName++;
386 
387                if (iPattern == patternEnd) {
388                        // We have exhausted pattern, so it's a match
389                        return true;
390                }
391 
392                if (iName == nameEnd){
393                        // We have exhausted name (and not pattern), so it's not a match
394                        return false;
395                }
396 
397                // For as long as we're exactly matching, bring it on (even if it's a lower case character)
398                if ((patternChar = pattern.charAt(iPattern)) == name.charAt(iName)) {
399                        continue;
400                }
401 
402                // If characters are not equals, then it's not a match if patternChar is lowercase
403                if (patternChar < ScannerHelper.MAX_OBVIOUS) {
404                        if ((ScannerHelper.OBVIOUS_IDENT_CHAR_NATURES[patternChar] & ScannerHelper.C_UPPER_LETTER) == 0) {
405                                return false;
406                        }
407                }
408                else if (Character.isJavaIdentifierPart(patternChar) && !Character.isUpperCase(patternChar)) {
409                        return false;
410                }
411 
412                // patternChar is uppercase, so let's find the next uppercase in name
413                while (true) {
414                        if (iName == nameEnd){
415                    //        We have exhausted name (and not pattern), so it's not a match
416                                return false;
417                        }
418 
419                        nameChar = name.charAt(iName);
420 
421                        if (nameChar < ScannerHelper.MAX_OBVIOUS) {
422                                if ((ScannerHelper.OBVIOUS_IDENT_CHAR_NATURES[nameChar] & (ScannerHelper.C_LOWER_LETTER | ScannerHelper.C_SPECIAL | ScannerHelper.C_DIGIT)) != 0) {
423                                        // nameChar is lowercase
424                                        iName++;
425                                // nameChar is uppercase...
426                                } else  if (patternChar != nameChar) {
427                                        //.. and it does not match patternChar, so it's not a match
428                                        return false;
429                                } else {
430                                        //.. and it matched patternChar. Back to the big loop
431                                        break;
432                                }
433                        }
434                        else if (Character.isJavaIdentifierPart(nameChar) && !Character.isUpperCase(nameChar)) {
435                                // nameChar is lowercase
436                                iName++;
437                        // nameChar is uppercase...
438                        } else  if (patternChar != nameChar) {
439                                //.. and it does not match patternChar, so it's not a match
440                                return false;
441                        } else {
442                                //.. and it matched patternChar. Back to the big loop
443                                break;
444                        }
445                }
446                // At this point, either name has been exhausted, or it is at an uppercase letter.
447                // Since pattern is also at an uppercase letter
448        }
449}

1050 *                <li>{@link IJavaScriptSearchConstants#FIELD}: look for fields</li>
1051 *                <li>{@link IJavaScriptSearchConstants#METHOD}: look for methods</li>
1052 *                <li>{@link IJavaScriptSearchConstants#CONSTRUCTOR}: look for constructors</li>

1631                }

1844/**
1845 * Validate compatibility between given string pattern and match rule.
1846 *<br>
1847 * Optimized (ie. returned match rule is modified) combinations are:
1848 * <ul>
1849 *         <li>{@link #R_PATTERN_MATCH} without any '*' or '?' in string pattern:
1850 *                 pattern match bit is unset,
1851 *         </li>
1852 *         <li>{@link #R_PATTERN_MATCH} and {@link #R_PREFIX_MATCH}  bits simultaneously set:
1853 *                 prefix match bit is unset,
1854 *         </li>

1856 *                 camel case match bit is unset,
1857 *         </li>
1858 *         <li>{@link #R_CAMELCASE_MATCH} with invalid combination of uppercase and lowercase characters:
1859 *                 camel case match bit is unset and replaced with prefix match pattern,
1860 *         </li>

1862 *                 bits is reduced to only {@link #R_CAMELCASE_MATCH} as Camel Case search is already prefix and case sensitive,
1863 *         </li>
1864 * </ul>
1865 *<br>
1866 * Rejected (ie. returned match rule -1) combinations are:
1867 * <ul>
1868 *         <li>{@link #R_REGEXP_MATCH} with any other match mode bit set,
1869 *         </li>
1870 * </ul>

1873 * @param matchRule The match rule
1874 * @return Optimized valid match rule or -1 if an incompatibility was detected.
1875 *  
1876 */

1879        // Verify Regexp match rule
1880        if ((matchRule & R_REGEXP_MATCH) != 0) {
1881                if ((matchRule & R_PATTERN_MATCH) != 0 || (matchRule & R_PREFIX_MATCH) != 0 || (matchRule & R_CAMELCASE_MATCH) != 0) {
1882                        return -1;
1883                }
1884        }
1885 
1886        // Verify Pattern match rule
1887        int starIndex = stringPattern.indexOf('*');

1889        if (starIndex < 0 && questionIndex < 0) {
1890                // reset pattern match bit if any
1891                matchRule &= ~R_PATTERN_MATCH;
1892        } else {
1893                // force Pattern rule
1894                matchRule |= R_PATTERN_MATCH;
1895        }
1896        if ((matchRule & R_PATTERN_MATCH) != 0) {
1897                // remove Camel Case and Prefix match bits if any
1898                matchRule &= ~R_CAMELCASE_MATCH;
1899                matchRule &= ~R_PREFIX_MATCH;
1900        }
1901 
1902        // Verify Camel Case match rule
1903        if ((matchRule & R_CAMELCASE_MATCH) != 0) {
1904                // Verify sting pattern validity
1905                int length = stringPattern.length();
1906                boolean validCamelCase = true;
1907                boolean uppercase = false;
1908                for (int i=0; i<length && validCamelCase; i++) {

1912                        // (see bug https://bugs.eclipse.org/bugs/show_bug.cgi?id=136313)
1913                        if (!uppercase) uppercase = ScannerHelper.isUpperCase(ch);
1914                }
1915                validCamelCase = validCamelCase && uppercase;
1916                // Verify bits compatibility
1917                if (validCamelCase) {
1918                        if ((matchRule & R_PREFIX_MATCH) != 0) {
1919                                if ((matchRule & R_CASE_SENSITIVE) != 0) {
1920                                        // This is equivalent to Camel Case match rule
1921                                        matchRule &= ~R_PREFIX_MATCH;
1922                                        matchRule &= ~R_CASE_SENSITIVE;
1923                                }
1924                        }
1925                } else {
1926                        matchRule &= ~R_CAMELCASE_MATCH;
1927                        if ((matchRule & R_PREFIX_MATCH) == 0) {
1928                                matchRule |= R_PREFIX_MATCH;
1929                                matchRule |= R_CASE_SENSITIVE;

1931                }
1932        }

1934}
