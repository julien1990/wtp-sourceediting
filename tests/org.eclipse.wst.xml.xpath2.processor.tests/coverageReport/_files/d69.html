373                                prevEndPos= getCurrentEndOffset();
374                                prevEndLine= getLineOfOffset(prevEndPos - 1);
375                        }
376                        if (sameLineComment) {
377                                if (linesDifference == 0) {
378                                        res= prevEndPos;
379                                } else {
380                                        sameLineComment= false;
381                                }
382                        }
383                        curr= readNextWithEOF(false);
384                }
385                if (curr == ITerminalSymbols.TokenNameEOF) {
386                        return prevEndPos;
387                }
388                int currStartLine= getLineOfOffset(getCurrentStartOffset());
389                int linesDifference= currStartLine - prevEndLine;
390                if (linesDifference > 1) {
391                        return prevEndPos; // separated comments
392                }
393                return res;
394        }
