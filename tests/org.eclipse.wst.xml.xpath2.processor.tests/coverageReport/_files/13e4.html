471        public void dispose() {
472                if (fRegisteredSelectionListeners != null) {

474                        for (Iterator iter= fRegisteredSelectionListeners.iterator(); iter.hasNext();) {

517        private int addEditorAction(IMenuManager menu, String actionID) {
518                if (fEditor == null)

521                if (action == null)
522                        return 0;

524                        ((IUpdate)action).update();
525                if (action.isEnabled()) {
526                        menu.add(action);
527                        return 1;
528                }
529                return 0;
