373        private IHyperlink createHyperlink(String uriString, IRegion hyperlinkRegion, IDocument document, Node node) {
374                IHyperlink link = null;
375 
376                if (uriString != null) {
377                        String temp = uriString.toLowerCase();
378                        if (temp.startsWith(HTTP_PROTOCOL)) {
379                                // this is a URLHyperlink since this is a web address
380                                link = new URLHyperlink(hyperlinkRegion, uriString);
381                        }
382                        else if (temp.startsWith(JAR_PROTOCOL)) {
383                                // this is a URLFileHyperlink since this is a local address
384                                try {
385                                        link = new URLFileRegionHyperlink(hyperlinkRegion, TAG, node.getLocalName(), new URL(uriString)) {
386                                                public String getHyperlinkText() {
387                                                        return JSPUIMessages.CustomTagHyperlink_hyperlinkText;
388                                                }
389                                        };
390                                }
391                                catch (MalformedURLException e) {
392                                        Logger.log(Logger.WARNING_DEBUG, e.getMessage(), e);
393                                }
394                        }
395                        else {
396                                // try to locate the file in the workspace
397                                IPath path = new Path(uriString);
398                                if (path.segmentCount() > 1) {
399                                        IFile file = ResourcesPlugin.getWorkspace().getRoot().getFile(path);
400                                        if (file.getType() == IResource.FILE && file.isAccessible()) {
401                                                if (node != null) {
402                                                        link = new TLDFileHyperlink(file, TAG, node.getLocalName(), hyperlinkRegion) {
403                                                                public String getHyperlinkText() {
404                                                                        return JSPUIMessages.CustomTagHyperlink_hyperlinkText;
405                                                                }
406                                                        };
407                                                }
408                                                else {
409                                                        link = new WorkspaceFileHyperlink(hyperlinkRegion, file);
410                                                }
411                                        }
412                                }
413                        }
414                        if (link == null) {
415                                // this is an ExternalFileHyperlink since file does not exist
416                                // in workspace
417                                File externalFile = new File(uriString);
418                                link = new ExternalFileHyperlink(hyperlinkRegion, externalFile);
419                        }
420                }
421 
422                return link;
423        }
