426                if (node == null)
427                        return false;
428 
429                Document doc = null;
430                if (node.getNodeType() != Node.DOCUMENT_NODE)
431                        doc = node.getOwnerDocument();
432                else
433                        doc = ((Document) node);
434 
435                if (doc instanceof IDOMDocument)
436                        return ((IDOMDocument) doc).isXMLType();
437 
438 
439                if (doc instanceof INodeNotifier) {
440                        ModelQueryAdapter adapter = (ModelQueryAdapter) ((INodeNotifier) doc).getAdapterFor(ModelQueryAdapter.class);
441                        CMDocument cmdoc = null;
442                        if (adapter != null && adapter.getModelQuery() != null)
443                                cmdoc = adapter.getModelQuery().getCorrespondingCMDocument(doc);
444                        if (cmdoc != null) {
445                                // treat as XHTML unless we've got the in-code HTML content
446                                // model
447                                if (cmdoc instanceof HTMLCMDocument)
448                                        return false;
449                                if (cmdoc.supports(HTMLCMProperties.IS_XHTML))
450                                        return Boolean.TRUE.equals(cmdoc.getProperty(HTMLCMProperties.IS_XHTML));
451                        }
452                }
453                // this should never be reached
454                DocumentType docType = doc.getDoctype();
455                return docType != null && docType.getPublicId() != null && docType.getPublicId().indexOf("-//W3C//DTD XHTML ") == 0; //$NON-NLS-1$
