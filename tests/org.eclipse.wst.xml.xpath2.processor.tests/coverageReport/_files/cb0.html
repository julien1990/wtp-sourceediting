556/*

558 * Note that 'binaryTypeBinding' is null if focus type is a source type.

560private void reportHierarchy(IType focus, CompilationUnitDeclaration parsedUnit, ReferenceBinding binaryTypeBinding) {
561 
562        // set focus type binding
563        if (focus != null) {
564                if (binaryTypeBinding != null) {
565                        // binary type
566                        this.focusType = binaryTypeBinding;
567                } else {
568                        // source type
569                        Member declaringMember = ((Member)focus).getOuterMostLocalContext();

587                                        }
588                                }
589                        }
590                }
591        }

620        }

622        if (!this.hasMissingSuperClass && objectIndex > -1) {

624                this.builder.connect(objectType, this.builder.getHandle(objectType, this.typeBindings[objectIndex]), null, null);
625        }

639/**

641 * Inform the requestor of the resolved supertypes using:
642 *    connect(ISourceType suppliedType, IGenericType superclass, IGenericType[] superinterfaces)
643 * @param suppliedType
644 */
645public void resolve(IGenericType suppliedType) {

647                if (suppliedType.isBinaryType()) {
648                        ReferenceBinding binaryTypeBinding = this.lookupEnvironment.cacheBinaryType((ISourceType) suppliedType, false/*don't need field and method (bug 125067)*/, null /*no access restriction*/);

680/**

682 * Inform the requestor of the resolved supertypes for each
683 * supplied source type using:

687 * additional requested super type which is also a source type
688 * instead of a binary type.

693public void resolve(Openable[] openables, HashSet localTypes, IProgressMonitor monitor) {
694        try {
695                int openablesLength = openables.length;
696                CompilationUnitDeclaration[] parsedUnits = new CompilationUnitDeclaration[openablesLength];
697                boolean[] hasLocalType = new boolean[openablesLength];

703                IType focus = this.builder.getType();

705                if (focus != null) {
706                        if (focus.isBinary()) {
707                                focusOpenable = (Openable)focus.getClassFile();
708                        } else {
709                                focusOpenable = (Openable)focus.getJavaScriptUnit();
710                        }
711                }
712 
713                processedUnits=new HashSet();
714                // build type bindings
715                Parser parser = new Parser(this.lookupEnvironment.problemReporter, true);
716                for (int i = 0; i < openablesLength; i++) {
717                        Openable openable = openables[i];
718                        if (openable instanceof org.eclipse.wst.jsdt.core.IJavaScriptUnit) {
719                                org.eclipse.wst.jsdt.core.IJavaScriptUnit cu = (org.eclipse.wst.jsdt.core.IJavaScriptUnit)openable;
720 
721                                // contains a potential subtype as a local or anonymous type?
722                                boolean containsLocalType = false;
723                                if (localTypes == null) { // case of hierarchy on region
724                                        containsLocalType = true;
725                                } else {
726                                        IPath path = cu.getPath();
727                                        containsLocalType = localTypes.contains(path.toString());
728                                }
729 
730                                // build parsed unit
731                                CompilationUnitDeclaration parsedUnit = null;
732                                if (cu.isOpen()) {
733                                        // create parsed unit from source element infos
734                                        CompilationResult result = new CompilationResult(((ICompilationUnit)cu).getFileName(), ((ICompilationUnit)cu).getPackageName(), i, openablesLength, this.options.maxProblemsPerUnit);
735                                        SourceTypeElementInfo[] typeInfos = null;
736                                        try {
737                                                IType[] topLevelTypes = cu.getTypes();
738                                                int topLevelLength = topLevelTypes.length;
739                                                if (topLevelLength == 0) continue; // empty cu: no need to parse (see https://bugs.eclipse.org/bugs/show_bug.cgi?id=65677)
740                                                typeInfos = new SourceTypeElementInfo[topLevelLength];
741                                                for (int j = 0; j < topLevelLength; j++) {
742                                                        IType topLevelType = topLevelTypes[j];
743                                                        typeInfos[j] = (SourceTypeElementInfo)((JavaElement)topLevelType).getElementInfo();
744                                                }
745                                        } catch (JavaScriptModelException e) {
746                                                // types/cu exist since cu is opened
747                                        }
748                                        int flags = !containsLocalType
749                                                ? SourceTypeConverter.MEMBER_TYPE
750                                                : SourceTypeConverter.FIELD_AND_METHOD | SourceTypeConverter.MEMBER_TYPE | SourceTypeConverter.LOCAL_TYPE;
751                                        parsedUnit =
752                                                SourceTypeConverter.buildCompilationUnit(
753                                                        typeInfos,
754                                                        flags,
755                                                        this.lookupEnvironment.problemReporter,
756                                                        result);
757                                        if (containsLocalType)         parsedUnit.bits |= ASTNode.HasAllMethodBodies;
758                                } else {

760                                        IFile file = (IFile) cu.getResource();

762 
763                                        CompilationResult unitResult = new CompilationResult(sourceUnit, i, openablesLength, this.options.maxProblemsPerUnit);
764                                        parsedUnit = parser.dietParse(sourceUnit, unitResult);
765                                }

768                                        parser.inferTypes(parsedUnit, this.options);
769                                        hasLocalType[unitsIndex] = containsLocalType;

772                                        try {
773                                                if (!processedUnits.contains(openable))
774                                                        this.lookupEnvironment.buildTypeBindings(parsedUnit, null /*no access restriction*/);

947private boolean subTypeOfType(ReferenceBinding subType, ReferenceBinding typeBinding) {
948        if (typeBinding == null || subType == null) return false;
949        if (subType == typeBinding) return true;
950        ReferenceBinding superclass = subType.superclass();
951        if (superclass != null) superclass = (ReferenceBinding) superclass.erasure();
952        if (superclass != null && superclass.id == TypeIds.T_JavaLangObject && subType.isHierarchyInconsistent()) return false;
953        if (this.subTypeOfType(superclass, typeBinding)) return true;
954        ReferenceBinding[] superInterfaces = subType.superInterfaces();
955        if (superInterfaces != null) {
956                for (int i = 0, length = superInterfaces.length; i < length; i++) {
957                        ReferenceBinding superInterface = (ReferenceBinding) superInterfaces[i].erasure();
958                        if (this.subTypeOfType(superInterface, typeBinding)) return true;
959                }
960        }
961        return false;
