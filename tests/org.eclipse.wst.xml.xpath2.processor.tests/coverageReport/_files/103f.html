411                HashSet res= new HashSet();

413                for (int k= 0; k < projects.length; k++) {
414                        IJavaScriptProject iJavaProject= projects[k];

432                        } catch (JavaScriptModelException e) {
433                                JavaScriptPlugin.log(e);

465        /**
466         * Gets a list of elements to generated javadoc for from each project. 
467         * Javadoc can be generated for either a IPackageFragment or a IJavaScriptUnit.
468         */
469        private IJavaScriptElement[] getSourceElements(IJavaScriptProject[] projects) {
470                ArrayList res= new ArrayList();
471                try {
472                        Set allChecked= fInputGroup.getAllCheckedTreeItems();
473 
474                        Set incompletePackages= new HashSet();
475                        for (int h= 0; h < projects.length; h++) {
476                                IJavaScriptProject iJavaProject= projects[h];
477 
478                                IPackageFragmentRoot[] roots= iJavaProject.getPackageFragmentRoots();
479                                for (int i= 0; i < roots.length; i++) {
480                                        IPackageFragmentRoot root= roots[i];
481                                        if (root.getKind() == IPackageFragmentRoot.K_SOURCE) {
482                                                IPath rootLocation= root.getResource().getLocation();
483                                                IJavaScriptElement[] packs= root.getChildren();
484                                                for (int k= 0; k < packs.length; k++) {
485                                                        IJavaScriptElement curr= packs[k];
486                                                        if (curr.getElementType() == IJavaScriptElement.PACKAGE_FRAGMENT) {
487                                                                // default packages are always incomplete
488                                                                if (curr.getElementName().length() == 0 || !allChecked.contains(curr)
489                                                                                || fInputGroup.isTreeItemGreyChecked(curr) || !isAccessibleLocation(curr.getResource().getLocation(), rootLocation)) {
490                                                                        incompletePackages.add(curr.getElementName());
491                                                                }
492                                                        }
493                                                }
494                                        }
495                                }
496                        }
497 
498                        Iterator checkedElements= fInputGroup.getAllCheckedListItems();
499                        while (checkedElements.hasNext()) {
500                                Object element= checkedElements.next();
501                                if (element instanceof IJavaScriptUnit) {
502                                        IJavaScriptUnit unit= (IJavaScriptUnit) element;
503                                        if (incompletePackages.contains(unit.getParent().getElementName())) {
504                                                res.add(unit);
505                                        }
506                                }
507                        }
508 
509                        Set addedPackages= new HashSet();
510 
511                        checkedElements= allChecked.iterator();
512                        while (checkedElements.hasNext()) {
513                                Object element= checkedElements.next();
514                                if (element instanceof IPackageFragment) {
515                                        IPackageFragment fragment= (IPackageFragment) element;
516                                        String name= fragment.getElementName();
517                                        if (!incompletePackages.contains(name) && !addedPackages.contains(name)) {
518                                                res.add(fragment);
519                                                addedPackages.add(name);
520                                        }
521                                }
522                        }
523 
524                } catch (JavaScriptModelException e) {
525                        JavaScriptPlugin.log(e);
526                }
527                return (IJavaScriptElement[]) res.toArray(new IJavaScriptElement[res.size()]);
528        }

578        protected void doValidation(int validate) {
579 
580                
581                switch (validate) {
582                        case PREFERENCESTATUS :
583                                fPreferenceStatus= new StatusInfo();
584                                fDocletStatus= new StatusInfo();
585                                updateStatus(findMostSevereStatus());
586                                break;
587                        case CUSTOMSTATUS :
588 
589                                if (fCustomButton.getSelection()) {
590                                        fDestinationStatus= new StatusInfo();
591                                        fDocletStatus= new StatusInfo();
592                                        String doclet= fDocletTypeText.getText();
593                                        String docletPath= fDocletText.getText();
594                                        if (doclet.length() == 0) {
595                                                fDocletStatus.setError(JavadocExportMessages.JavadocTreeWizardPage_nodocletname_error); 
596 
597                                        } else if (JavaScriptConventions.validateJavaScriptTypeName(doclet).matches(IStatus.ERROR)) {
598                                                fDocletStatus.setError(JavadocExportMessages.JavadocTreeWizardPage_invaliddocletname_error); 
599                                        } else if ((docletPath.length() == 0) || !validDocletPath(docletPath)) {
600                                                fDocletStatus.setError(JavadocExportMessages.JavadocTreeWizardPage_invaliddocletpath_error); 
601                                        }
602                                        updateStatus(findMostSevereStatus());
603                                }
604                                break;
605 
606                        case STANDARDSTATUS :
607                                if (fStandardButton.getSelection()) {
608                                        fDestinationStatus= new StatusInfo();
609                                        fDocletStatus= new StatusInfo();
610                                        String dest= fDestinationText.getText();
611                                        if (dest.length() == 0) {
612                                                fDestinationStatus.setError(JavadocExportMessages.JavadocTreeWizardPage_nodestination_error); 
613                                        }
614                                        File file= new File(dest);
615                                        if (!Path.ROOT.isValidPath(dest) || file.isFile()) {
616                                                fDestinationStatus.setError(JavadocExportMessages.JavadocTreeWizardPage_invaliddestination_error); 
617                                        }
618                                        if (new File(dest, "package-list").exists() || new File(dest, "index.html").exists()) //$NON-NLS-1$//$NON-NLS-2$
619                                                fDestinationStatus.setWarning(JavadocExportMessages.JavadocTreeWizardPage_warning_mayoverwritefiles); 
620                                        updateStatus(findMostSevereStatus());
621                                }
622                                break;
623 
624                        case TREESTATUS :
625 
626                                fTreeStatus= new StatusInfo();
627 
628                                if (!fInputGroup.getAllCheckedListItems().hasNext())
629                                        fTreeStatus.setError(JavadocExportMessages.JavadocTreeWizardPage_invalidtreeselection_error); 
630                                updateStatus(findMostSevereStatus());
631 
632                                break;
633                                
634                        case JAVADOCSTATUS:
635                                fJavadocStatus= new StatusInfo();
636                                String text= fJavadocCommandText.getText();
637                                if (text.length() == 0) {
638                                        fJavadocStatus.setError(JavadocExportMessages.JavadocTreeWizardPage_javadoccmd_error_enterpath);  
639                                } else {
640                                        File file= new File(text);
641                                        if (!file.isFile()) {
642                                                fJavadocStatus.setError(JavadocExportMessages.JavadocTreeWizardPage_javadoccmd_error_notexists);  
643                                        }
644                                }
645                                updateStatus(findMostSevereStatus());
646                                break;
647                } //end switch
