375                if (document == null)
376                        return;
377 
378                String positionCategory= getPositionCategory();
379 
380                List removedPositionsList= Arrays.asList(removedPositions);
381 
382                try {
383                        synchronized (fPositionLock) {
384                                List oldPositions= fPositions;
385                                int newSize= Math.max(fPositions.size() + addedPositions.length - removedPositions.length, 10);
386                                
387                                /*
388                                 * The following loop is a kind of merge sort: it merges two List<Position>, each
389                                 * sorted by position.offset, into one new list. The first of the two is the
390                                 * previous list of positions (oldPositions), from which any deleted positions get
391                                 * removed on the fly. The second of two is the list of added positions. The result
392                                 * is stored in newPositions.
393                                 */
394                                List newPositions= new ArrayList(newSize);
395                                Position position= null;
396                                Position addedPosition= null;
397                                for (int i= 0, j= 0, n= oldPositions.size(), m= addedPositions.length; i < n || position != null || j < m || addedPosition != null;) {
398                                        // loop variant: i + j < old(i + j)
399                                        
400                                        // a) find the next non-deleted Position from the old list
401                                        while (position == null && i < n) {
402                                                position= (Position) oldPositions.get(i++);
403                                                if (position.isDeleted() || contain(removedPositionsList, position)) {
404                                                        document.removePosition(positionCategory, position);
405                                                        position= null;
406                                                }
407                                        }
408                                        
409                                        // b) find the next Position from the added list
410                                        if (addedPosition == null && j < m) {
411                                                addedPosition= addedPositions[j++];
412                                                document.addPosition(positionCategory, addedPosition);
413                                        }
414                                        
415                                        // c) merge: add the next of position/addedPosition with the lower offset
416                                        if (position != null) {
417                                                if (addedPosition != null)
418                                                        if (position.getOffset() <= addedPosition.getOffset()) {
419                                                                newPositions.add(position);
420                                                                position= null;
421                                                        } else {
422                                                                newPositions.add(addedPosition);
423                                                                addedPosition= null;
424                                                        }
425                                                else {
426                                                        newPositions.add(position);
427                                                        position= null;
428                                                }
429                                        } else if (addedPosition != null) {
430                                                newPositions.add(addedPosition);
431                                                addedPosition= null;
432                                        }
433                                }
434                                fPositions= newPositions;
435                        }
436                } catch (BadPositionCategoryException e) {
437                        // Should not happen
438                        JavaScriptPlugin.log(e);
439                } catch (BadLocationException e) {
440                        // Should not happen
441                        JavaScriptPlugin.log(e);
442                }
443//                checkOrdering("new positions: ", fPositions); //$NON-NLS-1$
444 
445                if (textPresentation != null)
446                        fSourceViewer.changeTextPresentation(textPresentation, false);
447                else
448                        fSourceViewer.invalidateTextPresentation();
449        }
