794                                        return mask & (SimilarElementsRequestor.REF_TYPES);
795                                }
796                                mask&= SimilarElementsRequestor.REF_TYPES;
797                        } else if (parent instanceof ParameterizedType) {
798                                if (node.getLocationInParent() == ParameterizedType.TYPE_ARGUMENTS_PROPERTY) {
799                                        return mask & SimilarElementsRequestor.REF_TYPES_AND_VAR;
800                                }
801                                mask&= SimilarElementsRequestor.CLASSES | SimilarElementsRequestor.INTERFACES;
802                        } else if (parent instanceof WildcardType) {
803                                if (node.getLocationInParent() == WildcardType.BOUND_PROPERTY) {
804                                        return mask & SimilarElementsRequestor.REF_TYPES_AND_VAR;
805                                }
806                        }
807                        node= parent;
808                        parent= parent.getParent();
809                }

811                switch (parent.getNodeType()) {
812                        case ASTNode.TYPE_DECLARATION:
813                                if (node.getLocationInParent() == TypeDeclaration.SUPER_INTERFACE_TYPES_PROPERTY) {
814                                        kind= SimilarElementsRequestor.INTERFACES;
815                                } else if (node.getLocationInParent() == TypeDeclaration.SUPERCLASS_TYPE_PROPERTY) {
816                                        kind= SimilarElementsRequestor.CLASSES;
817                                }
818                                break;
819                        case ASTNode.FUNCTION_DECLARATION:
820                                if (node.getLocationInParent() == FunctionDeclaration.THROWN_EXCEPTIONS_PROPERTY) {
821                                        kind= SimilarElementsRequestor.CLASSES;
822                                } else if (node.getLocationInParent() == FunctionDeclaration.RETURN_TYPE2_PROPERTY) {
823                                        kind= SimilarElementsRequestor.ALL_TYPES | SimilarElementsRequestor.VOIDTYPE;
824                                }
825                                break;
826                        case ASTNode.INSTANCEOF_EXPRESSION:
827                                kind= SimilarElementsRequestor.REF_TYPES;
828                                break;
829                        case ASTNode.THROW_STATEMENT:
830                                kind= SimilarElementsRequestor.CLASSES;
831                                break;
832                        case ASTNode.CLASS_INSTANCE_CREATION:
833                                if (((ClassInstanceCreation) parent).getAnonymousClassDeclaration() == null) {
834                                        kind= SimilarElementsRequestor.CLASSES;
835                                } else {
836                                        kind= SimilarElementsRequestor.CLASSES | SimilarElementsRequestor.INTERFACES;
837                                }
838                                break;
839                        case ASTNode.SINGLE_VARIABLE_DECLARATION:
840                                int superParent= parent.getParent().getNodeType();
841                                if (superParent == ASTNode.CATCH_CLAUSE) {
842                                        kind= SimilarElementsRequestor.CLASSES;
843                                }
844                                break;
845                        case ASTNode.TAG_ELEMENT:
846                                kind= SimilarElementsRequestor.REF_TYPES;
847                                break;
848                        case ASTNode.TYPE_PARAMETER:
849                                if (((TypeParameter) parent).typeBounds().indexOf(node) > 0) {
850                                        kind= SimilarElementsRequestor.INTERFACES;
851                                } else {
852                                        kind= SimilarElementsRequestor.REF_TYPES_AND_VAR;
853                                }
854                                break;
855                        case ASTNode.TYPE_LITERAL:
856                                kind= SimilarElementsRequestor.REF_TYPES;
857                                break;
858                        default:
859                }
860                return kind & mask;
861        }
