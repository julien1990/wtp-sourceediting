314/*
315 * @see JavaElement
316 */
317public IJavaScriptElement getHandleFromMemento(String token, MementoTokenizer memento, WorkingCopyOwner workingCopyOwner) {
318        switch (token.charAt(0)) {
319                case JEM_COUNT:
320                        return getHandleUpdatingCountFromMemento(memento, workingCopyOwner);
321                case JEM_FIELD:
322                        if (!memento.hasMoreTokens()) return this;
323                        String fieldName = memento.nextToken();
324                        JavaElement field = (JavaElement)getField(fieldName);
325                        return field.getHandleFromMemento(memento, workingCopyOwner);
326                case JEM_INITIALIZER:
327                        if (!memento.hasMoreTokens()) return this;
328                        String count = memento.nextToken();
329                        JavaElement initializer = (JavaElement)getInitializer(Integer.parseInt(count));
330                        return initializer.getHandleFromMemento(memento, workingCopyOwner);
331                case JEM_METHOD:
332                        if (!memento.hasMoreTokens()) return this;
333                        String selector = memento.nextToken();
334                        ArrayList params = new ArrayList();
335                        nextParam: while (memento.hasMoreTokens()) {
336                                token = memento.nextToken();
337                                switch (token.charAt(0)) {
338                                        case JEM_TYPE:
339                                        case JEM_TYPE_PARAMETER:
340                                                break nextParam;
341                                        case JEM_METHOD:
342                                                if (!memento.hasMoreTokens()) return this;
343                                                String param = memento.nextToken();
344                                                StringBuffer buffer = new StringBuffer();
345                                                while (param.length() == 1 && Signature.C_ARRAY == param.charAt(0)) { // backward compatible with 3.0 mementos
346                                                        buffer.append(Signature.C_ARRAY);
347                                                        if (!memento.hasMoreTokens()) return this;
348                                                        param = memento.nextToken();
349                                                }
350                                                params.add(buffer.toString() + param);
351                                                break;
352                                        default:
353                                                break nextParam;
354                                }
355                        }
356                        String[] parameters = new String[params.size()];
357                        params.toArray(parameters);
358                        JavaElement method = (JavaElement)getFunction(selector, parameters);
359                        switch (token.charAt(0)) {
360                                case JEM_TYPE:
361                                case JEM_TYPE_PARAMETER:
362                                case JEM_LOCALVARIABLE:
363                                        return method.getHandleFromMemento(token, memento, workingCopyOwner);
364                                default:
365                                        return method;
366                        }
367                case JEM_TYPE:
368                        String typeName;
369                        if (memento.hasMoreTokens()) {
370                                typeName = memento.nextToken();
371                                char firstChar = typeName.charAt(0);
372                                if (firstChar == JEM_FIELD || firstChar == JEM_INITIALIZER || firstChar == JEM_METHOD || firstChar == JEM_TYPE || firstChar == JEM_COUNT) {
373                                        token = typeName;
374                                        typeName = ""; //$NON-NLS-1$
375                                } else {
376                                        token = null;
377                                }
378                        } else {
379                                typeName = ""; //$NON-NLS-1$
380                                token = null;
381                        }
382                        JavaElement type = (JavaElement)getType(typeName);
383                        if (token == null) {
384                                return type.getHandleFromMemento(memento, workingCopyOwner);
385                        } else {
386                                return type.getHandleFromMemento(token, memento, workingCopyOwner);
387                        }
388                case JEM_TYPE_PARAMETER:
389                        if (!memento.hasMoreTokens()) return this;
390                        String typeParameterName = memento.nextToken();
391                        JavaElement typeParameter = null;
392                        return typeParameter.getHandleFromMemento(memento, workingCopyOwner);
393        }
394        return null;
395}

851/*
852 * Returns the source file name as defined in the given info.
853 * If not present in the info, infers it from this type.
854 */
855public String sourceFileName(IBinaryType info) {
856        char[] sourceFileName = info.sourceFileName();
857        if (sourceFileName == null) {
858                /*
859                 * We assume that this type has been compiled from a file with its name
860                 * For example, A.class comes from A.js and p.A.class comes from a file A.js
861                 * in the folder p.
862                 */
863                if (info.isMember()) {
864                        IType enclosingType = getDeclaringType();
865                        if (enclosingType == null) return null; // play it safe
866                        while (enclosingType.getDeclaringType() != null) {
867                                enclosingType = enclosingType.getDeclaringType();
868                        }
869                        return enclosingType.getElementName() + Util.defaultJavaExtension();
870                } else if (info.isLocal() || info.isAnonymous()){
871                        String typeQualifiedName = getTypeQualifiedName();
872                        int dollar = typeQualifiedName.indexOf('$');
873                        if (dollar == -1) {
874                                // malformed inner type: name doesn't contain a dollar
875                                return getElementName() + Util.defaultJavaExtension();
876                        }
877                        return typeQualifiedName.substring(0, dollar) + Util.defaultJavaExtension();
878                } else {
879                        return getElementName() + Util.defaultJavaExtension();
880                }
881        } else {
882                return  new String(sourceFileName);
883        }

920        if (contents == null) return null;

932                        // try to find optional enum constant summary start
933                        indexOfNextSummary = contents.indexOf(JavadocConstants.ANNOTATION_TYPE_OPTIONAL_MEMBER_SUMMARY);

940        if (indexOfNextSummary == -1) {
941                // try to find constructor summary start
942                indexOfNextSummary = contents.indexOf(JavadocConstants.CONSTRUCTOR_SUMMARY);
943        }
944        if (indexOfNextSummary == -1) {
945                // try to find method summary start
946                indexOfNextSummary = contents.indexOf(JavadocConstants.METHOD_SUMMARY);

965                start = indexOfFirstParagraph;

969public String getJavadocContents(IProgressMonitor monitor) throws JavaScriptModelException {
