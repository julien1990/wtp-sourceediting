664                        }
665                }

706        protected boolean processChanges(IWorkbenchPreferenceContainer container) {
707                IScopeContext currContext= fLookupOrder[0];
708                
709                List /* <Key>*/ changedOptions= new ArrayList();
710                boolean needsBuild= getChanges(currContext, changedOptions);
711                if (changedOptions.isEmpty()) {
712                        return true;
713                }
714                if (needsBuild) {
715                        int count= getRebuildCount();
716                        if (count > fRebuildCount) {
717                                needsBuild= false; // build already requested
718                                fRebuildCount= count;
719                        }
720                }
721 
722                boolean doBuild= false;
723                if (needsBuild) {
724                        String[] strings= getFullBuildDialogStrings(fProject == null);
725                        if (strings != null) {
726                                MessageDialog dialog= new MessageDialog(getShell(), strings[0], null, strings[1], MessageDialog.QUESTION, new String[] { IDialogConstants.YES_LABEL, IDialogConstants.NO_LABEL, IDialogConstants.CANCEL_LABEL }, 2);
727                                int res= dialog.open();
728                                if (res == 0) {
729                                        doBuild= true;
730                                } else if (res != 1) {
731                                        return false; // cancel pressed
732                                }
733                        }
734                }
735                if (container != null) {
736                        // no need to apply the changes to the original store: will be done by the page container
737                        if (doBuild) { // post build
738                                incrementRebuildCount();
739                                container.registerUpdateJob(CoreUtility.getBuildJob(fProject));
740                        }
741                } else {
742                        // apply changes right away
743                        try {
744                                fManager.applyChanges();
745                        } catch (BackingStoreException e) {
746                                JavaScriptPlugin.log(e);
747                                return false;
748                        }
749                        if (doBuild) {
750                                CoreUtility.getBuildJob(fProject).schedule();
751                        }
752                        
753                }
754                return true;
755        }
