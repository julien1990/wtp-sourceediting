450        /**
451         * Goes through every changed node in event and add to queue of node
452         * changes that will be recalculated for projection annotation model.
453         * 
454         * @param event
455         * @param changeType
456         */
457        private void processNodesEvent(NodesEvent event, int changeType) {
458                List nodes = event.getNodes();
459                Iterator it = nodes.iterator();
460                while (it.hasNext()) {
461                        DTDNode node = (DTDNode) it.next();
462                        if (isNodeProjectable(node)) {
463                                if (fQueuedNodeChanges == null) {
464                                        fQueuedNodeChanges = new ArrayList();
465                                }
466 
467                                int existingIndex = fQueuedNodeChanges.indexOf(node);
468                                if (existingIndex > -1) {
469                                        // node is already queued up
470                                        NodeChange existingChange = (NodeChange) fQueuedNodeChanges.remove(existingIndex);
471                                        // don't add if added then removed node or vice versa
472                                        if (existingChange.getChangeType() == changeType) {
473                                                NodeChange newChange = new NodeChange(node, changeType);
474                                                fQueuedNodeChanges.add(newChange);
475                                        }
476                                }
477                                else {
478                                        // not queued up yet, so queue node
479                                        NodeChange newChange = new NodeChange(node, changeType);
480                                        fQueuedNodeChanges.add(newChange);
481                                }
482                        }
483                }
484                // if document isn't changing, go ahead and apply it
485                if (!fIsDocumentChanging) {
486                        applyAnnotationModelChanges();
487                }
488        }
