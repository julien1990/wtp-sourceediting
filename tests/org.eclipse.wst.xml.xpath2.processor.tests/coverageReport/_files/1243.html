397                private boolean updateJRE( IProgressMonitor monitor) throws CoreException, JavaScriptModelException {
398                        try {
399                                IVMInstall vm50Install= find50OrGreaterVMInstall();
400                                f50JREFound= vm50Install != null;
401                                if (vm50Install != null) {
402                                        IVMInstall install= JavaRuntime.getVMInstall(fProject); // can be null
403                                        if (fChangeOnWorkspace) {
404                                                monitor.beginTask(CorrectionMessages.ReorgCorrectionsSubProcessor_50_compliance_operation, 4);
405                                                IVMInstall defaultVM= JavaRuntime.getDefaultVMInstall(); // can be null
406                                                if (defaultVM != null && !defaultVM.equals(install)) {
407                                                        IPath newPath= new Path(JavaRuntime.JRE_CONTAINER);
408                                                        updateClasspath(newPath, new SubProgressMonitor(monitor, 1));
409                                                } else {
410                                                        monitor.worked(1);
411                                                }
412                                                if (defaultVM == null || !is50orGreaterVMInstall(defaultVM)) {
413                                                        JavaRuntime.setDefaultVMInstall(vm50Install, new SubProgressMonitor(monitor, 3), true);
414                                                        return false;
415                                                }
416                                                return true;
417                                        } else {
418                                                if (install == null || !is50orGreaterVMInstall(install)) {
419                                                        IPath newPath= new Path(JavaRuntime.JRE_CONTAINER).append(vm50Install.getVMInstallType().getId()).append(vm50Install.getName());
420                                                        updateClasspath(newPath, monitor);
421                                                        return false;
422                                                }
423                                        }
424                                }
425                        } finally {
426                                monitor.done();
427                        }
428                        return true;
429                }

431                private void updateClasspath(IPath newPath, IProgressMonitor monitor) throws JavaScriptModelException {
432                        IIncludePathEntry[] classpath= fProject.getRawIncludepath();
433                        IPath jreContainerPath= new Path(JavaRuntime.JRE_CONTAINER);
434                        for (int i= 0; i < classpath.length; i++) {
435                                IIncludePathEntry curr= classpath[i];
436                                if (curr.getEntryKind() == IIncludePathEntry.CPE_CONTAINER && curr.getPath().matchingFirstSegments(jreContainerPath) > 0) {
437                                        classpath[i]= JavaScriptCore.newContainerEntry(newPath, curr.getAccessRules(), curr.getExtraAttributes(), curr.isExported());
438                                }
439                        }
440                        fProject.setRawIncludepath(classpath, monitor);
441                }
