113protected void verifyCompletionOnJavadocTag(char[] tag, char[][] expectedTags, boolean inline) {

116        StringBuffer expected = new StringBuffer("<CompleteOnJavadocTag:");
117        if (inline) expected.append('{');
118        expected.append('@');
119        if (tag != null) expected.append(tag);
120        if (inline) expected.append('}');

130        expected.append(">");
131        if (expectedTags == null) {
132                assertEquals("Invalid completion tag", expected.toString(), completionTag.toString());
133        } else {
134                String completionTagString = completionTag.toString();
135                StringTokenizer completionTagTokenizer = new StringTokenizer(completionTagString, "\n");
136                StringBuffer completionTagBuffer = new StringBuffer(completionTagString.length());
137                boolean possibleLine = false, newLine = false;
138                while (completionTagTokenizer.hasMoreTokens()) {
139                        String line = completionTagTokenizer.nextToken();
140                        if (line.startsWith("possible")) {
141                                if (!possibleLine) {
142                                        possibleLine = true;
143                                        completionTagBuffer.append("\npossible tags:");
144                                }
145                        } else {
146                                if (newLine) completionTagBuffer.append('\n');
147                                completionTagBuffer.append(line);
148                        }
149                        newLine = true;
150                }
151                assertEquals("Invalid completion tag", expected.toString(), completionTagBuffer.toString());
152        }
