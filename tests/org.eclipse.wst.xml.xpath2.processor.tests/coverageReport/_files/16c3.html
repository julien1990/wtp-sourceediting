372                        else if ((type == DOMRegionContext.XML_TAG_NAME) || (type == DOMRegionContext.XML_TAG_ATTRIBUTE_NAME) || (type == DOMRegionContext.XML_TAG_ATTRIBUTE_EQUALS) || (type == DOMRegionContext.XML_TAG_ATTRIBUTE_VALUE) || (type == DOMRegionContext.XML_COMMENT_TEXT) || (type == DOMRegionContext.XML_PI_CONTENT) || (type == DOMRegionContext.XML_DOCTYPE_INTERNAL_SUBSET)) {
373                                currentNode.addRegion(region);
374                                currentNode.setLength(region.getStart() + region.getLength() - currentNode.getStart());
375                                region.adjustStart(-currentNode.getStart());
376                                // DW 4/16/2003 regions no longer have parents
377                                // region.setParent(currentNode);
378                        }
379                        // the following contexts close off StructuredDocumentRegions
380                        // cleanly
381                        else if ((type == DOMRegionContext.XML_PI_CLOSE) || (type == DOMRegionContext.XML_TAG_CLOSE) || (type == DOMRegionContext.XML_EMPTY_TAG_CLOSE) || (type == DOMRegionContext.XML_COMMENT_CLOSE) || (type == DOMRegionContext.XML_DECLARATION_CLOSE) || (type == DOMRegionContext.XML_CDATA_CLOSE)) {
382                                currentNode.setEnded(true);
383                                currentNode.setLength(region.getStart() + region.getLength() - currentNode.getStart());
384                                currentNode.addRegion(region);
385                                region.adjustStart(-currentNode.getStart());
386                                // DW 4/16/2003 regions no longer have parents
387                                // region.setParent(currentNode);
388                        }
389                        // this is extremely rare, but valid
390                        else if (type == DOMRegionContext.WHITE_SPACE) {
391                                ITextRegion lastRegion = currentNode.getLastRegion();
392                                // pack the embedded container with this region
393                                if (lastRegion instanceof ITextRegionContainer) {
394                                        ITextRegionContainer container = (ITextRegionContainer) lastRegion;
395                                        container.getRegions().add(region);
396                                        // containers must have parent set ...
397                                        // setting for EACH subregion is redundent, but not sure
398                                        // where else to do, so will do here for now.
399                                        container.setParent(currentNode);
400                                        // DW 4/16/2003 regions no longer have parents
401                                        // region.setParent(container);
402                                        region.adjustStart(container.getLength() - region.getStart());
403                                }
404                                currentNode.getLastRegion().adjustLength(region.getLength());
405                                currentNode.adjustLength(region.getLength());
406                        }
407                        else if (type == DOMRegionContext.UNDEFINED && currentNode != null) {
408                                // skip on a very-first region situation as the default
409                                // behavior is good enough
410                                // combine with previous if also undefined
411                                if (currentNode.getLastRegion() != null && currentNode.getLastRegion().getType() == DOMRegionContext.UNDEFINED) {
412                                        currentNode.getLastRegion().adjustLength(region.getLength());
413                                        currentNode.adjustLength(region.getLength());
414                                }
415                                // previous wasn't undefined
416                                else {
417                                        currentNode.addRegion(region);
418                                        currentNode.setLength(region.getStart() + region.getLength() - currentNode.getStart());
419                                        region.adjustStart(-currentNode.getStart());
420                                }
421                        }
422                        else {
423                                // if an unknown type is the first region in the document,
424                                // ensure that a node exists
425                                if (currentNode == null) {
426                                        currentNode = createStructuredDocumentRegion(type);
427                                        currentNode.setStart(region.getStart());
428                                }
429                                currentNode.addRegion(region);
430                                currentNode.setLength(region.getStart() + region.getLength() - currentNode.getStart());
431                                region.adjustStart(-currentNode.getStart());
432                                // DW 4/16/2003 regions no longer have parents
433                                // region.setParent(currentNode);
434                                if (Debug.debugTokenizer)
435                                        System.out.println(getClass().getName() + " found region of not specifically handled type " + region.getType() + " @ " + region.getStart() + "[" + region.getLength() + "]"); //$NON-NLS-4$//$NON-NLS-3$//$NON-NLS-2$//$NON-NLS-1$
436                                //$NON-NLS-3$//$NON-NLS-2$//$NON-NLS-1$
437                        }
438 
439                        // these regions also get their own node, so close them cleanly
440                        // NOTE: these regions have new StructuredDocumentRegions created
441                        // for them above; it may
442                        // be more readable if that is handled here as well, but the
443                        // current layout
444                        // ensures that they open StructuredDocumentRegions the same way
445                        if ((type == DOMRegionContext.XML_CONTENT) || (type == DOMRegionContext.XML_CHAR_REFERENCE) || (type == DOMRegionContext.XML_ENTITY_REFERENCE)) {
446                                currentNode.setEnded(true);
447                        }
448                        if (headNode == null && currentNode != null) {
449                                headNode = currentNode;
450                        }
451                }
452                if (currentNode != null) {
453                        fireNodeParsed(currentNode);
454                        currentNode.setPrevious(lastNode);
455                }
456                // fStringInput = null;
457                primReset();
458                return headNode;
459        }
