331                        case ASTNode.FUNCTION_INVOCATION:
332                                FunctionInvocation decl= (FunctionInvocation) parent;
333                                if (selector == decl.getName()) {
334                                        return getBinding(decl.getExpression());
335                                }
336                                return null;
337                        case ASTNode.QUALIFIED_NAME:
338                                QualifiedName qualifiedName= (QualifiedName) parent;
339                                if (selector == qualifiedName.getName()) {
340                                        return getBinding(qualifiedName.getQualifier());
341                                }
342                                return null;
343                        case ASTNode.FIELD_ACCESS:
344                                FieldAccess fieldAccess= (FieldAccess) parent;
345                                if (selector == fieldAccess.getName()) {
346                                        return getBinding(fieldAccess.getExpression());
347                                }
348                                return null;                        
349                        case ASTNode.SUPER_FIELD_ACCESS: {
350                                ITypeBinding curr= Bindings.getBindingOfParentType(parent);
351                                return curr.getSuperclass();
352                        }
353                        case ASTNode.SUPER_METHOD_INVOCATION: {
354                                SuperMethodInvocation superInv= (SuperMethodInvocation) parent;
355                                if (selector == superInv.getName()) {
356                                        ITypeBinding curr= Bindings.getBindingOfParentType(parent);
357                                        return curr.getSuperclass();
358                                }
359                                return null;
360                        }
361                        default:
362                                if (parent instanceof Type) {
363                                        // bug 67644: in 'a.new X()', all member types of A are visible as location of X. 
364                                        ASTNode normalizedNode= ASTNodes.getNormalizedNode(parent);
365                                        if (normalizedNode.getLocationInParent() == ClassInstanceCreation.TYPE_PROPERTY) {
366                                                ClassInstanceCreation creation= (ClassInstanceCreation) normalizedNode.getParent();
367                                                return getBinding(creation.getExpression());
368                                        }
369                                }
370                                return null;
371                }
372        }

420                public boolean acceptBinding(IBinding binding) {
421                        if (fFound)
422                                return true;
423                        
424                        if (binding == null)
425                                return false;
426                        
427                        if (fToSearch.getKind() != binding.getKind()) {
428                                return false;
429                        }

431                        boolean checkVisibility= hasFlag(CHECK_VISIBILITY, fFlags);
432                        if (binding == fToSearch) {
433                                fFound= true;
434                        } else {
435                                IBinding bindingDeclaration= Bindings.getDeclaration(binding);
436                                if (bindingDeclaration == fToSearch) {
437                                        fFound= true;
438                                } else if (bindingDeclaration.getName().equals(fToSearch.getName())) {
439                                        String signature= getSignature(bindingDeclaration);
440                                        if (signature != null && signature.equals(getSignature(fToSearch))) {
441                                                if (checkVisibility) {
442                                                        fIsVisible= false;
443                                                }
444                                                return true; // found element that hides the binding to find
445                                        }
446                                }
447                        }
448 
449                        if (fFound && checkVisibility) {
450                                fIsVisible= ScopeAnalyzer.isVisible(binding, fParentTypeBinding);
451                        }
452                        return fFound;
453                }

561        /**
562         * Evaluates if the declaration is visible in a certain context. 
563         * @param binding The binding of the declaration to examine
564         * @param context The context to test in
565         * @return Returns 
566         */
567        public static boolean isVisible(IBinding binding, ITypeBinding context) {
568                if (binding.getKind() == IBinding.VARIABLE && !((IVariableBinding) binding).isField()) {
569                        return true; // all local variables found are visible
570                }
571                ITypeBinding declaring= getDeclaringType(binding);
572                if (declaring == null) {
573                        return false;
574                }
575        
576                int modifiers= binding.getModifiers();
577                if (Modifier.isPublic(modifiers) || declaring.isInterface()) {
578                        return true;
579                } else if (Modifier.isProtected(modifiers) || !Modifier.isPrivate(modifiers)) {
580                        if (declaring.getPackage() == context.getPackage()) {
581                                return true;
582                        }
583                        return isTypeInScope(declaring, context, Modifier.isProtected(modifiers));
584                }
585                // private visibility
586                return isTypeInScope(declaring, context, false);
587        }

589        private static boolean isTypeInScope(ITypeBinding declaring, ITypeBinding context, boolean includeHierarchy) {
590                ITypeBinding curr= context;
591                while (curr != null && curr != declaring) {
592                        if (includeHierarchy && Bindings.isSuperType(declaring, curr)) {
593                                return true;
594                        }
595                        curr= curr.getDeclaringClass();
596                }
597                return curr == declaring;
598        }
