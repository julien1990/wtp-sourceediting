317 * @param nameStart the start index of the name, inclusive
318 * @param nameEnd the end index of the name, exclusive
319 * @return true if a sub-pattern matches the subpart of the given name, false otherwise
320 */
321public static final boolean camelCaseMatch(char[] pattern, int patternStart, int patternEnd, char[] name, int nameStart, int nameEnd) {
322        if (name == null)
323                return false; // null name cannot match
324        if (pattern == null)
325                return true; // null pattern is equivalent to '*'
326        if (patternEnd < 0)         patternEnd = pattern.length;
327        if (nameEnd < 0) nameEnd = name.length;
328 
329        if (patternEnd <= patternStart) return nameEnd <= nameStart;
330        if (nameEnd <= nameStart) return false;
331        // check first pattern char
332        if (name[nameStart] != pattern[patternStart]) {
333                // first char must strictly match (upper/lower)
334                return false;
335        }
336 
337        char patternChar, nameChar;
338        int iPattern = patternStart;
339        int iName = nameStart;
340 
341        // Main loop is on pattern characters
342        while (true) {
343 
344                iPattern++;
345                iName++;
346 
347                if (iPattern == patternEnd) {
348                        // We have exhausted pattern, so it's a match
349                        return true;
350                }
351 
352                if (iName == nameEnd){
353                        // We have exhausted name (and not pattern), so it's not a match
354                        return false;
355                }
356 
357                // For as long as we're exactly matching, bring it on (even if it's a lower case character)
358                if ((patternChar = pattern[iPattern]) == name[iName]) {
359                        continue;
360                }
361 
362                // If characters are not equals, then it's not a match if patternChar is lowercase
363                if (patternChar < ScannerHelper.MAX_OBVIOUS) {
364                        if ((ScannerHelper.OBVIOUS_IDENT_CHAR_NATURES[patternChar] & ScannerHelper.C_UPPER_LETTER) == 0) {
365                                return false;
366                        }
367                }
368                else if (Character.isJavaIdentifierPart(patternChar) && !Character.isUpperCase(patternChar)) {
369                        return false;
370                }
371 
372                // patternChar is uppercase, so let's find the next uppercase in name
373                while (true) {
374                        if (iName == nameEnd){
375                    //        We have exhausted name (and not pattern), so it's not a match
376                                return false;
377                        }
378 
379                        nameChar = name[iName];
380                        if (nameChar < ScannerHelper.MAX_OBVIOUS) {
381                                if ((ScannerHelper.OBVIOUS_IDENT_CHAR_NATURES[nameChar] & (ScannerHelper.C_LOWER_LETTER | ScannerHelper.C_SPECIAL | ScannerHelper.C_DIGIT)) != 0) {
382                                        // nameChar is lowercase
383                                        iName++;
384                                // nameChar is uppercase...
385                                } else  if (patternChar != nameChar) {
386                                        //.. and it does not match patternChar, so it's not a match
387                                        return false;
388                                } else {
389                                        //.. and it matched patternChar. Back to the big loop
390                                        break;
391                                }
392                        }
393                        else if (Character.isJavaIdentifierPart(nameChar) && !Character.isUpperCase(nameChar)) {
394                                // nameChar is lowercase
395                                iName++;
396                        // nameChar is uppercase...
397                        } else  if (patternChar != nameChar) {
398                                //.. and it does not match patternChar, so it's not a match
399                                return false;
400                        } else {
401                                //.. and it matched patternChar. Back to the big loop
402                                break;
403                        }
404                }
405                // At this point, either name has been exhausted, or it is at an uppercase letter.
406                // Since pattern is also at an uppercase letter
407        }
408}
