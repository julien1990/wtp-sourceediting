430                public String encodeSettings(Map settings) throws CoreException {
431                        ByteArrayOutputStream stream= new ByteArrayOutputStream(2000);
432                        try {
433                                CleanUpProfileVersioner versioner= new CleanUpProfileVersioner();
434                                CustomProfile profile= new ProfileManager.CustomProfile("custom", settings, versioner.getCurrentVersion(), versioner.getProfileKind()); //$NON-NLS-1$
435                                ArrayList profiles= new ArrayList();
436                                profiles.add(profile);
437                                ProfileStore.writeProfilesToStream(profiles, stream, ENCODING, versioner);
438                                try {
439                                        return stream.toString(ENCODING);
440                                } catch (UnsupportedEncodingException e) {
441                                        return stream.toString(); 
442                                }
443                        } finally {

445                        }

449                        byte[] bytes;
450                        try {
451                                bytes= settings.getBytes(ENCODING);
452                        } catch (UnsupportedEncodingException e) {
453                                bytes= settings.getBytes();
454                        }
455                        InputStream is= new ByteArrayInputStream(bytes);
456                        try {
457                                List res= ProfileStore.readProfilesFromStream(new InputSource(is));
458                                if (res == null || res.size() == 0)
459                                        return CleanUpConstants.getEclipseDefaultSettings();
460                                
461                                CustomProfile profile= (CustomProfile)res.get(0);
462                                new CleanUpProfileVersioner().update(profile);
463                                return profile.getSettings();
464                        } finally {
465                                try { is.close(); } catch (IOException e) { /* ignore */ }
466                        }
