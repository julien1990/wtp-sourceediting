715        private String getNamedStyleAtOffset(int offset) {
716                // ensure the offset is clean
717                if (offset >= document.getLength())
718                        return getNamedStyleAtOffset(document.getLength() - 1);
719                else if (offset < 0)
720                        return getNamedStyleAtOffset(0);
721                IStructuredDocumentRegion documentRegion = document
722                                .getFirstStructuredDocumentRegion();
723                while (documentRegion != null && !documentRegion.containsOffset(offset)) {
724                        documentRegion = documentRegion.getNext();
725                }
726                if (documentRegion != null) {
727                        // find the ITextRegion's Context at this offset
728                        ITextRegion interest = documentRegion
729                                        .getRegionAtCharacterOffset(offset);
730                        if (interest == null)
731                                return null;
732                        if (offset > documentRegion.getTextEndOffset(interest))
733                                return null;
734                        String regionContext = interest.getType();
735                        if (regionContext == null)
736                                return null;
737                        // find the named style (internal/selectable name) for that
738                        // context
739                        String namedStyle = (String) contextToXSLStyleMap.get(regionContext);
740                        if (namedStyle != null) {
741                                return namedStyle;
742                        }
743                }
744                return null;
