398 
399                // Read first char
400                // readChar() code is inlined to balance additional method call in checkDeprectation(int)
401                char first = this.source[this.index++];
402                if (first == '\\' && this.source[this.index] == 'u') {
403                        int c1, c2, c3, c4;
404                        int pos = this.index;
405                        this.index++;
406                        while (this.source[this.index] == 'u')
407                                this.index++;
408                        if (!(((c1 = ScannerHelper.getNumericValue(this.source[this.index++])) > 15 || c1 < 0)
409                                        || ((c2 = ScannerHelper.getNumericValue(this.source[this.index++])) > 15 || c2 < 0)
410                                        || ((c3 = ScannerHelper.getNumericValue(this.source[this.index++])) > 15 || c3 < 0) || ((c4 = ScannerHelper.getNumericValue(this.source[this.index++])) > 15 || c4 < 0))) {
411                                first = (char) (((c1 * 16 + c2) * 16 + c3) * 16 + c4);
412                        } else {
413                                this.index = pos;
414                        }
415                }
416 
417                // switch on first tag char
418                switch (first) {
419                        case 'd':
420                        if ((readChar() == 'e') &&
421                                                (readChar() == 'p') && (readChar() == 'r') &&

423                                                (readChar() == 'a') && (readChar() == 't') &&
424                                                (readChar() == 'e') && (readChar() == 'd')) {

426                                        char c = readChar();
427                                        if (ScannerHelper.isWhitespace(c) || c == '*') {
428                                                this.abort = true;
429                                            this.deprecated = true;
430                                                this.tagValue = TAG_DEPRECATED_VALUE;
431                                        }
432                        }
433                                break;
434                }

437        protected boolean parseTag(int previousPosition) throws InvalidInputException {

467                                        case '}':

474                                        case '\'':

483                                                break;
484                                        case '-': // allowed in tag names as this character is often used in doclets (bug 68087)

486                                                tagName[length] = this.scanner.currentCharacter;
487                                                this.tagSourceEnd = this.scanner.getCurrentTokenEndPosition();
488                                                this.index = this.scanner.currentPosition;
489                                                break;
490                                        default:

492                                                        break tagNameToken;
