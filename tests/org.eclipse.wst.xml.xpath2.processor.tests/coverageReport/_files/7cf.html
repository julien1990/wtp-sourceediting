327        private String getErrorMessage(ErrorInfo info) {
328                ErrorTable tab = getErrorTable(info.getTargetType());
329                if (tab == null)
330                        return MSG_UNKNOWN_ERROR;
331 
332                String template = tab.getMessage(info.getState());
333                Object[] arguments = {info.getHint()};
334                String s = null;
335                try {
336                        s = NLS.bind(template, arguments);
337                }
338                catch (IllegalArgumentException e) {
339                        Logger.logException(e);
340                        s = template + ":" + arguments.toString(); //$NON-NLS-1$
341                }
342                return s;
343        }

346         */
347        private int getErrorSeverity(ErrorInfo info) {
348                ErrorTable tab = getErrorTable(info.getTargetType());
349                if (tab == null)
350                        return 0;
351                return tab.getSeverity(info.getState());
352        }

354        private ErrorTable getErrorTable(short nodetype) {

356                switch (nodetype) {
357                        case Node.ATTRIBUTE_NODE :
358                                tab = errTables[NodeType.ATTRIBUTE];
359                                break;
360                        case Node.ELEMENT_NODE :
361                                tab = errTables[NodeType.ELEMENT];
362                                break;
363                        case Node.DOCUMENT_TYPE_NODE :
364                                tab = errTables[NodeType.DOCUMENT_TYPE];
365                                break;
366                        case Node.TEXT_NODE :
367                                tab = errTables[NodeType.TEXT];
368                                break;
369                        case Node.COMMENT_NODE :
370                                tab = errTables[NodeType.COMMENT];
371                                break;
372                        case Node.CDATA_SECTION_NODE :
373                                tab = errTables[NodeType.CDATA_SECTION];
374                                break;
375                        case Node.PROCESSING_INSTRUCTION_NODE :
376                                tab = errTables[NodeType.PROCESSING_INSTRUCTION];
377                                break;
378                        case Node.ENTITY_REFERENCE_NODE :
379                                tab = errTables[NodeType.ENTITY_REFERENCE];
380                                break;
381                        default :
382                                return null;
383                }
