445                public void update(Observable o, Object arg) {
446 
447                        final IWorkspace workspace= JavaScriptPlugin.getWorkspace();
448 
449                        final String name= fNameGroup.getName();
450 
451                        // check whether the project name field is empty
452                        if (name.length() == 0) { 
453                                setErrorMessage(null);
454                                setMessage(NewWizardMessages.JavaProjectWizardFirstPage_Message_enterProjectName); 
455                                setPageComplete(false);
456                                return;
457                        }
458 
459                        // check whether the project name is valid
460                        final IStatus nameStatus= workspace.validateName(name, IResource.PROJECT);
461                        if (!nameStatus.isOK()) {
462                                setErrorMessage(nameStatus.getMessage());
463                                setPageComplete(false);
464                                return;
465                        }
466 
467                        // check whether project already exists
468                        final IProject handle= getProjectHandle();
469                        if (handle.exists()) {
470                                setErrorMessage(NewWizardMessages.JavaProjectWizardFirstPage_Message_projectAlreadyExists); 
471                                setPageComplete(false);
472                                return;
473                        }
474 
475                        final String location= fLocationGroup.getLocation().toOSString();
476 
477                        // check whether location is empty
478                        if (location.length() == 0) {
479                                setErrorMessage(null);
480                                setMessage(NewWizardMessages.JavaProjectWizardFirstPage_Message_enterLocation); 
481                                setPageComplete(false);
482                                return;
483                        }
484 
485                        // check whether the location is a syntactically correct path
486                        if (!Path.EMPTY.isValidPath(location)) { 
487                                setErrorMessage(NewWizardMessages.JavaProjectWizardFirstPage_Message_invalidDirectory); 
488                                setPageComplete(false);
489                                return;
490                        }
491 
492                        IPath projectPath= Path.fromOSString(location);
493                        // check external location
494                        if (!fLocationGroup.isInWorkspace()) {                                
495                                if (!canCreate(projectPath.toFile())) {
496                                        setErrorMessage(NewWizardMessages.JavaProjectWizardFirstPage_Message_cannotCreateAtExternalLocation); 
497                                        setPageComplete(false);
498                                        return;
499                                }
500 
501                                if (!Platform.getLocation().equals(projectPath) && Platform.getLocation().isPrefixOf(projectPath)) {
502                                        if (!Platform.getLocation().equals(projectPath.removeLastSegments(1))) {
503                                                setErrorMessage(NewWizardMessages.JavaProjectWizardFirstPage_Message_notOnWorkspaceRoot);
504                                                setPageComplete(false);
505                                                return;
506                                        }
507                                        
508                                        if (!projectPath.toFile().exists()) {
509                                                setErrorMessage(NewWizardMessages.JavaProjectWizardFirstPage_Message_notExisingProjectOnWorkspaceRoot);
510                                                setPageComplete(false);
511                                                return;
512                                        }
513                                        
514                                        String existingName= projectPath.lastSegment();
515                                        if (!existingName.equals(fNameGroup.getName())) {
516                                                setErrorMessage(Messages.format(NewWizardMessages.JavaProjectWizardFirstPage_Message_invalidProjectNameForWorkspaceRoot, existingName));
517                                                setPageComplete(false);
518                                                return;
519                                        }
520                                } else {
521                                        // If we do not place the contents in the workspace validate the
522                                        // location.
523                                        final IStatus locationStatus= workspace.validateProjectLocation(handle, projectPath);
524                                        if (!locationStatus.isOK()) {
525                                                setErrorMessage(locationStatus.getMessage());
526                                                setPageComplete(false);
527                                                return;
528                                        }
529                                }
530                        } else {
531                                IPath projectFolder= projectPath.append(fNameGroup.getName());
532                                if (projectFolder.toFile().exists()) {
533                                        setErrorMessage(NewWizardMessages.JavaProjectWizardFirstPage_Message_existingFolderInWorkspace); 
534                                        setPageComplete(false);
535                                        return;
536                                }
537                        }
538                        
539                        setPageComplete(true);
540 
541                        setErrorMessage(null);
542                        setMessage(null);
543                }

547                                file= file.getParentFile();
548                                if (file == null)
549                                        return false;

552                        return file.canWrite();
553                }
