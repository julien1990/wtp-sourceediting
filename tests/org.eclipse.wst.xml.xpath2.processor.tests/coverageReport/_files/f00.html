422        public void addImport(String type) {
423                if (isReadOnly())
424                        return;

426                IJavaScriptUnit cu= getCompilationUnit();
427                if (cu == null)
428                        return;

430                try {
431                        boolean qualified= type.indexOf('.') != -1;
432                        if (!qualified) {
433                                IJavaScriptSearchScope searchScope= SearchEngine.createJavaSearchScope(new IJavaScriptElement[] { cu.getJavaScriptProject() });
434                                SimpleName nameNode= null;

449                        try {
450

455                                        context= null;
456                                else
457                                        context= new ContextSensitiveImportRewriteContext(root, getCompletionOffset(), rewrite);
458                                rewrite.addImport(type, type, context);

462                                setCompletionLength(position.getLength());
463                                
464                        } catch (CoreException e) {
465                                handleException(null, e);
466                        } finally {

469                                document.removePositionCategory(category);
470                        }
471                        
472                } catch (BadLocationException e) {
473                        handleException(null, e);
474                } catch (BadPositionCategoryException e) {

488        private TypeNameMatch[] findAllTypes(String simpleTypeName, IJavaScriptSearchScope searchScope, SimpleName nameNode, IProgressMonitor monitor, IJavaScriptUnit cu) throws JavaScriptModelException {
489                boolean is50OrHigher= JavaModelUtil.is50OrHigher(cu.getJavaScriptProject());
490                
491                int typeKinds= SimilarElementsRequestor.ALL_TYPES;
492                if (nameNode != null) {
493                        typeKinds= ASTResolving.getPossibleTypeKinds(nameNode, is50OrHigher);
494                }
495                
496                ArrayList typeInfos= new ArrayList();
497                TypeNameMatchCollector requestor= new TypeNameMatchCollector(typeInfos);
498                new SearchEngine().searchAllTypeNames(null, 0, simpleTypeName.toCharArray(), SearchPattern.R_EXACT_MATCH | SearchPattern.R_CASE_SENSITIVE, getSearchForConstant(typeKinds), searchScope, requestor, IJavaScriptSearchConstants.FORCE_IMMEDIATE_SEARCH, monitor);
499 
500                ArrayList typeRefsFound= new ArrayList(typeInfos.size());
501                for (int i= 0, len= typeInfos.size(); i < len; i++) {
502                        TypeNameMatch curr= (TypeNameMatch) typeInfos.get(i);
503                        if (curr.getPackageName().length() > 0) { // do not suggest imports from the default package
504                                if (isOfKind(curr, typeKinds, is50OrHigher) && isVisible(curr, cu)) {
505                                        typeRefsFound.add(curr);
506                                }
507                        }
508                }
509                return (TypeNameMatch[]) typeRefsFound.toArray(new TypeNameMatch[typeRefsFound.size()]);
510        }

512        private int getSearchForConstant(int typeKinds) {
513                final int CLASSES= SimilarElementsRequestor.CLASSES;
514                final int INTERFACES= SimilarElementsRequestor.INTERFACES;
515                final int ENUMS= SimilarElementsRequestor.ENUMS;
516                final int ANNOTATIONS= SimilarElementsRequestor.ANNOTATIONS;
517 
518                switch (typeKinds & (CLASSES | INTERFACES | ENUMS | ANNOTATIONS)) {
519                        case CLASSES: return IJavaScriptSearchConstants.CLASS;
520                        case INTERFACES: return IJavaScriptSearchConstants.INTERFACE;
521                        case ENUMS: return IJavaScriptSearchConstants.ENUM;
522                        case ANNOTATIONS: return IJavaScriptSearchConstants.ANNOTATION_TYPE;
523                        case CLASSES | INTERFACES: return IJavaScriptSearchConstants.CLASS_AND_INTERFACE;
524                        case CLASSES | ENUMS: return IJavaScriptSearchConstants.CLASS_AND_ENUM;
525                        default: return IJavaScriptSearchConstants.TYPE;
526                }
