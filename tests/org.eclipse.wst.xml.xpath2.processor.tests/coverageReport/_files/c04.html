449        if (charContents == null) {
450                buffer.append("<null>"); //$NON-NLS-1$
451        } else {
452                int length = charContents.length;
453                for (int i = 0; i < length; i++) {
454                        char c = charContents[i];
455                        switch (c) {
456                                case '\n':

458                                        break;
459                                case '\r':

462                                                i++;
463                                        } else {
464                                                buffer.append("\\r\n"); //$NON-NLS-1$
465                                        }

467                                default:
