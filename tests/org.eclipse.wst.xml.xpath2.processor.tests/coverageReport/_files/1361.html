371    public void setSelection(final List elements) {
372        if (elements == null || elements.size() == 0)
373            return;
374                try {
375                ResourcesPlugin.getWorkspace().run(new IWorkspaceRunnable() {
376                        public void run(IProgressMonitor monitor) throws CoreException {
377                                fPackageViewer.refresh();
378                        IStructuredSelection selection= new StructuredSelection(elements);
379                        fPackageViewer.setSelection(selection, true);
380                        fPackageViewer.getTree().setFocus();
381                        
382                        if (elements.size() == 1 && elements.get(0) instanceof IJavaScriptProject)
383                            fPackageViewer.expandToLevel(elements.get(0), 1);
384                    }
385                }, ResourcesPlugin.getWorkspace().getRoot(), IWorkspace.AVOID_UPDATE, new NullProgressMonitor());
386        } catch (CoreException e) {
387                JavaScriptPlugin.log(e);
388        }
389    }
