527  /**
528   * From a list of used schemas, update the unusedImports list for the given schema
529   * 
530   * @param schema
531   * @param unusedImports
532   * @param usedSchemas
533   */
534  protected void addToUnusedImports(XSDSchema schema, List usedSchemas)
535  {
536    // now that we have the list of usedSchemas, get the list of unused
537    // schemas by comparing this list to what is actually in the schema
538    Iterator iter = schema.getContents().iterator();
539    while(iter.hasNext())
540    {
541      Object o = iter.next();
542      if (o instanceof XSDSchemaDirective)
543      {
544        XSDSchemaDirective directive = (XSDSchemaDirective) o;
545        boolean isUsed = false;
546        Iterator iter2 = usedSchemas.iterator();
547        while (iter2.hasNext())
548        {
549          XSDSchema usedSchema = (XSDSchema) iter2.next();
550          if (directive instanceof XSDImport && directive.getResolvedSchema() == usedSchema)
551          {
552            isUsed = true;
553            break;
554          }
555          if (directive instanceof XSDInclude && ((XSDInclude)directive).getIncorporatedSchema() == usedSchema)
556          {
557            isUsed = true;
558            usedDirectIncludes.add(usedSchema);
559            break;
560          }
561          // blindly accept redefines as used
562          if (directive instanceof XSDRedefine)
563          {
564            isUsed = true;
565            break;
566          }
567        }
568        
569        // If it is an include, we need to check if it is used indirectly
570        if (directive instanceof XSDInclude && !isUsed)
571        {
572          XSDInclude inc = (XSDInclude)directive;
573          XSDSchema incSchema = inc.getIncorporatedSchema();
574          if (incSchema != null)
575          {
576            XSDSchema usedSchema = getUsedIncludeSchema(incSchema, inc);
577            if (usedSchema != null)
578            {
579              usedIndirectIncludes.add(directive);
580              usedIndirectIncludesMap.put(directive, usedSchema);
581              isUsed = true;
582            }
583            else
584            {
585              isUsed = false;
586            }
587          }
588        }
589        
590        // If resolved directives are determined unused
591        // If resolved directives are not already in the unused list
592        // Also any redefines should be considered used
593        if (!isUsed && !unusedDirectives.contains(directive) && !(directive instanceof XSDRedefine))
594        {
595          unusedDirectives.add(directive);
596        }
597      }
598    }
599    Iterator iter3 = usedIndirectIncludes.iterator();
600    while (iter3.hasNext())
601    {
602      Object o = iter3.next();
603      if (o instanceof XSDInclude)
604      {
605        XSDInclude inc = (XSDInclude)o;
606        XSDSchema targetSchema = (XSDSchema)usedIndirectIncludesMap.get(inc);
607        if (usedIncludeSchemas.contains(targetSchema) && usedDirectIncludes.contains(targetSchema))
608        {
609          unusedDirectives.add(inc);
610        }
611        else
612        {
613          usedDirectIncludes.add(targetSchema);
614        }
615      }
616    }
617  }
618 
619  /**
620   * Includes can be used indirectly.   If the schema includes A which includes B, but the schema
621   * references something in B, then A is indirectly used, and hence A cannot be removed.
622   * 
623   * @param schema
624   * @param xsdInclude
625   * @return the referenced schema if used, null if not used
626   */
627  private XSDSchema getUsedIncludeSchema(XSDSchema schema, XSDInclude xsdInclude)
628  {
629    XSDSchema refSchema = null;
630    boolean isUsed = false;
631    Iterator iter = schema.getContents().iterator();
632    while (iter.hasNext())
633    {
634      Object o = iter.next();
635      if (o instanceof XSDInclude)
636      {
637        XSDInclude inc = (XSDInclude)o;
638        XSDSchema incSchema = inc.getIncorporatedSchema();
639        if (incSchema != null)
640        {
641          Iterator iter2 = usedIncludeSchemas.iterator();
642          while (iter2.hasNext())
643          {
644            XSDSchema xsdSch = (XSDSchema)iter2.next();
645            if (incSchema == xsdSch)
646            {
647              isUsed = true;
648              refSchema = incSchema;
649              break;
650            }
651          }
652          
653          if (!isUsed)
654          {
655            if (!analyzedIncludeSchemas.contains(incSchema)) // To prevent infinite cycle
656            {
657              analyzedIncludeSchemas.add(incSchema);
658              refSchema = getUsedIncludeSchema(incSchema, inc);
659            }
660          }
661          if (isUsed || refSchema != null)
662          {
663            return refSchema;
664          }
665        }
666      }
667      else
668      {
669        break;
670      }
671    }
672    return refSchema;
673  }
