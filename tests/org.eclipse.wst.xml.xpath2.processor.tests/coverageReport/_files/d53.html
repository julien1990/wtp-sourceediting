402                        monitor.beginTask(Messages.format(CodeGenerationMessages.OrganizeImportsOperation_description, fCompilationUnit.getElementName()), 9);
403                        
404                        JavaScriptUnit astRoot= fASTRoot;
405                        if (astRoot == null) {
406                                astRoot= ASTProvider.getASTProvider().getAST(fCompilationUnit, ASTProvider.WAIT_YES, new SubProgressMonitor(monitor, 2));
407                                if (monitor.isCanceled())
408                                        throw new OperationCanceledException();
409                        } else {
410                                monitor.worked(2);
411                        }
412                        
413                        ImportRewrite importsRewrite= StubUtility.createImportRewrite(astRoot, false);
414 
415                        Set/*<String>*/ oldSingleImports= new HashSet();
416                        Set/*<String>*/  oldDemandImports= new HashSet();
417                        List/*<SimpleName>*/ typeReferences= new ArrayList();
418                        List/*<SimpleName>*/ staticReferences= new ArrayList();
419                        
420                        if (!collectReferences(astRoot, typeReferences, staticReferences, oldSingleImports, oldDemandImports))
421                                return null;
422                                                
423                        monitor.worked(1);
424                
425                        TypeReferenceProcessor processor= new TypeReferenceProcessor(oldSingleImports, oldDemandImports, astRoot, importsRewrite, fIgnoreLowerCaseNames);
426                        
427                        Iterator refIterator= typeReferences.iterator();
428                        while (refIterator.hasNext()) {
429                                SimpleName typeRef= (SimpleName) refIterator.next();
430                                processor.add(typeRef);
431                        }
432                        
433                        boolean hasOpenChoices= processor.process(new SubProgressMonitor(monitor, 3));
434                        addStaticImports(staticReferences, importsRewrite);
435                        
436                        if (hasOpenChoices && fChooseImportQuery != null) {
437                                TypeNameMatch[][] choices= processor.getChoices();
438                                ISourceRange[] ranges= processor.getChoicesSourceRanges();
439                                TypeNameMatch[] chosen= fChooseImportQuery.chooseImports(choices, ranges);
440                                if (chosen == null) {
441                                        // cancel pressed by the user
442                                        throw new OperationCanceledException();
443                                }
444                                for (int i= 0; i < chosen.length; i++) {
445                                        TypeNameMatch typeInfo= chosen[i];
446                                        importsRewrite.addImport(typeInfo.getFullyQualifiedName());
447                                }                                
448                        }
449 
450                        TextEdit result= importsRewrite.rewriteImports(new SubProgressMonitor(monitor, 3));
451                        
452                        determineImportDifferences(importsRewrite, oldSingleImports, oldDemandImports);
453                        
454                        return result;
455                } finally {
456                        monitor.done();
457                }

460        private void determineImportDifferences(ImportRewrite importsStructure, Set oldSingleImports, Set oldDemandImports) {
461                  ArrayList importsAdded= new ArrayList();
462                  importsAdded.addAll(Arrays.asList(importsStructure.getCreatedImports()));
463                  importsAdded.addAll(Arrays.asList(importsStructure.getCreatedStaticImports()));
464                
465                Object[] content= oldSingleImports.toArray();
466            for (int i= 0; i < content.length; i++) {
467                String importName= (String) content[i];
468                if (importsAdded.remove(importName))
469                    oldSingleImports.remove(importName);
470            }
471            content= oldDemandImports.toArray();
472            for (int i= 0; i < content.length; i++) {
473                String importName= (String) content[i]; 
474                if (importsAdded.remove(importName + ".*")) //$NON-NLS-1$
475                    oldDemandImports.remove(importName);
476            }
477            fNumberOfImportsAdded= importsAdded.size();
478            fNumberOfImportsRemoved= oldSingleImports.size() + oldDemandImports.size();            
479        }

494        private boolean collectReferences(JavaScriptUnit astRoot, List typeReferences, List staticReferences, Set oldSingleImports, Set oldDemandImports) {
495                if (!fAllowSyntaxErrors) {
496                        IProblem[] problems= astRoot.getProblems();
497                        for (int i= 0; i < problems.length; i++) {
498                                IProblem curr= problems[i];
499                                if (curr.isError() && (curr.getID() & IProblem.Syntax) != 0) {
500                                        fParsingError= problems[i];
501                                        return false;
502                                }
503                        }
504                }
505                List imports= astRoot.imports();
506                for (int i= 0; i < imports.size(); i++) {
507                        ImportDeclaration curr= (ImportDeclaration) imports.get(i);
508                        String id= ASTResolving.getFullName(curr.getName());
509                        if (curr.isOnDemand()) {
510                                oldDemandImports.add(id);
511                        } else {
512                                oldSingleImports.add(id);
513                        }
514                }
515                
516                IJavaScriptProject project= fCompilationUnit.getJavaScriptProject();
517                ImportReferencesCollector.collect(astRoot, project, null, typeReferences, staticReferences);
518 
519                return true;
520        }
