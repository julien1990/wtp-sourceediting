644         * Adds elements at the given index
645         */        
646        public boolean addElements(List elements, int index) {
647                
648                int nElements= elements.size();
649                
650                if (nElements > 0 && index >= 0 && index <= fElements.size()) {
651                        // filter duplicated
652                        ArrayList elementsToAdd= new ArrayList(nElements);
653                        
654                        for (int i= 0; i < nElements; i++) {
655                                Object elem= elements.get(i);
656                                if (!fElements.contains(elem)) {
657                                        elementsToAdd.add(elem);
658                                }        
659                        }
660                        if (!elementsToAdd.isEmpty()) {
661                                fElements.addAll(index, elementsToAdd);
662                                if (isOkToUse(fTableControl)) {
663                                        if (index == fElements.size()) {
664                                                fTable.add(elementsToAdd.toArray());
665                                        } else {
666                                                for (int i= elementsToAdd.size() - 1; i >= 0 ; i--) {
667                                                        fTable.insert(elementsToAdd.get(i), index);
668                                                }
669                                        }
670                                        fTable.setSelection(new StructuredSelection(elementsToAdd));
671                                }
672                                dialogFieldChanged();
673                                return true;
674                        }
675                }
676                return false;
677        }

794        private List moveUp(List elements, List move) {
795                int nElements= elements.size();
796                List res= new ArrayList(nElements);
797                Object floating= null;
798                for (int i= 0; i < nElements; i++) {
799                        Object curr= elements.get(i);
800                        if (move.contains(curr)) {
801                                res.add(curr);
802                        } else {
803                                if (floating != null) {
804                                        res.add(floating);
805                                }
806                                floating= curr;
807                        }
808                }
809                if (floating != null) {
810                        res.add(floating);
811                }
812                return res;
813        }
