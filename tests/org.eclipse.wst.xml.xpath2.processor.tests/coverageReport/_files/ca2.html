301                private boolean sameMatchLenTest(String currName) {
302                        int matchLen= this.bestMatchLen;
303                        // known: bestName and currName differ from newName at position 'matchLen'
304                        // currName and bestName don't have to differ at position 'matchLen'
305 
306                        // determine the order and return true if currName is closer to newName
307                        char newChar= getCharAt(this.newName, matchLen);
308                        char currChar= getCharAt(currName, matchLen);
309                        char bestChar= getCharAt(this.bestName, matchLen);
310 
311                        if (newChar < currChar) {
312                                if (bestChar < newChar) {                                                                // b < n < c
313                                        return (currChar - newChar) < (newChar - bestChar);        // -> (c - n) < (n - b)
314                                } else {                                                                                                // n < b  && n < c
315                                        if (currChar == bestChar) { // longer match between curr and best
316                                                return false; // keep curr and best together, new should be before both
317                                        } else {
318                                                return currChar < bestChar; // -> (c < b)
319                                        }
320                                }
321                        } else {
322                                if (bestChar > newChar) {                                                                // c < n < b
323                                        return (newChar - currChar) < (bestChar - newChar);        // -> (n - c) < (b - n)
324                                } else {                                                                                                // n > b  && n > c
325                                        if (currChar == bestChar) {  // longer match between curr and best
326                                                return true; // keep curr and best together, new should be ahead of both
327                                        } else {
328                                                return currChar > bestChar; // -> (c > b)
329                                        }
330                                }
331                        }
332                }

635                        return -1;

649                                        return pos; // found
650                                }

652                                        pos+= k;
653                                } else {
654                                        pos+= step;
655                                }
656                        } else {
657                                pos++;
658                        }
659                }
660                return -1;
