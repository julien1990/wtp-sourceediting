298                        if (!Checks.isAvailable(fMethod)) {
299                                result.addFatalError(RefactoringCoreMessages.RenameMethodProcessor_is_binary, JavaStatusContext.create(fMethod));

301                        }

303                        if (result.hasFatalError())
304                                return result;
305                        pm.setTaskName(RefactoringCoreMessages.RenameMethodRefactoring_taskName_checkingPreconditions); 
306                        result.merge(checkNewElementName(getNewElementName()));
307                        if (result.hasFatalError())
308                                return result;
309                        
310                        boolean mustAnalyzeShadowing;
311                        IFunction[] newNameMethods= searchForDeclarationsOfClashingMethods(new SubProgressMonitor(pm, 1));
312                        if (newNameMethods.length == 0) {
313                                mustAnalyzeShadowing= false;
314                                pm.worked(1);
315                        } else {
316                                IType[] outerTypes= searchForOuterTypesOfReferences(newNameMethods, new SubProgressMonitor(pm, 1));
317                                if (outerTypes.length > 0) {
318                                        //There exists a reference to a clashing method, where the reference is in a nested type.
319                                        //That nested type could be a type in a ripple method's hierarchy, which could
320                                        //cause the reference to bind to the new ripple method instead of to
321                                        //its old binding (a method of an enclosing scope).
322                                        //-> Getting *more* references than before -> Semantics not preserved.
323                                        //Example: RenamePrivateMethodTests#testFail6()
324                                        //TODO: could pass declaringTypes to the RippleMethodFinder and check whether
325                                        //a hierarchy contains one of outerTypes (or an outer type of an outerType, recursively).
326                                        mustAnalyzeShadowing= true;
327                                        
328                                } else {
329                                        boolean hasOldRefsInInnerTypes= true;
330                                                //TODO: to implement this optimization:
331                                                //- move search for references to before this check.
332                                                //- collect references in inner types.
333                                                //- for each reference, check for all supertypes and their enclosing types
334                                                //(recursively), whether they declare a rippleMethod
335                                        if (hasOldRefsInInnerTypes) {
336                                                //There exists a reference to a ripple method in a nested type
337                                                //of a type in the hierarchy of any ripple method.
338                                                //When that reference is renamed, and one of the supertypes of the
339                                                //nested type declared a method matching the new name, then
340                                                //the renamed reference will bind to the method in its supertype,
341                                                //since inherited methods bind stronger than methods from enclosing scopes.
342                                                //Getting *less* references than before -> Semantics not preserved.
343                                                //Examples: RenamePrivateMethodTests#testFail2(), RenamePrivateMethodTests#testFail5()
344                                                mustAnalyzeShadowing= true;
345                                        } else {
346                                                mustAnalyzeShadowing= false;
347                                        }
348                                }
349                        }
350                        
351                        initializeMethodsToRename(new SubProgressMonitor(pm, 1));
352                        pm.setTaskName(RefactoringCoreMessages.RenameMethodRefactoring_taskName_searchingForReferences); 
353                        fOccurrences= getOccurrences(new SubProgressMonitor(pm, 3), result);        
354                        pm.setTaskName(RefactoringCoreMessages.RenameMethodRefactoring_taskName_checkingPreconditions); 
355                        
356                        if (fUpdateReferences)
357                                result.merge(checkRelatedMethods());
358                        
359                        result.merge(analyzeCompilationUnits()); //removes CUs with syntax errors
360                        pm.worked(1);
361                        
362                        if (result.hasFatalError())
363                                return result;
364                        
365                        createChanges(new SubProgressMonitor(pm, 1), result);
366                        if (fUpdateReferences & mustAnalyzeShadowing)
367                                result.merge(analyzeRenameChanges(new SubProgressMonitor(pm, 1)));
368                        else
369                                pm.worked(1);
370                        
371                        return result;
372                } finally{
373                        pm.done();
374                }        
375        }

564        private SearchResultGroup[] batchFindNewOccurrences(IFunction[] wcNewMethods, final IFunction[] wcOldMethods, IJavaScriptUnit[] newDeclarationWCs, IProgressMonitor pm, RefactoringStatus status) throws CoreException {
565                pm.beginTask("", 2); //$NON-NLS-1$
566                
567                SearchPattern refsPattern= RefactoringSearchEngine.createOrPattern(wcNewMethods, IJavaScriptSearchConstants.REFERENCES);
568                SearchParticipant[] searchParticipants= SearchUtils.getDefaultSearchParticipants();
569                IJavaScriptSearchScope scope= RefactoringScopeFactory.create(wcNewMethods);

571                MethodOccurenceCollector requestor;
572                if (getDelegateUpdating()) {
573                        // There will be two new matches inside the delegate(s) (the invocation
574                        // and the javadoc) which are OK and must not be reported.
575                        // Note that except these ocurrences, the delegate bodies are empty 
576                        // (as they were created this way).
577                        requestor= new MethodOccurenceCollector(getNewElementName()) {
578                                public void acceptSearchMatch(IJavaScriptUnit unit, SearchMatch match) throws CoreException {
579                                        for (int i= 0; i < wcOldMethods.length; i++) 
580                                                if (wcOldMethods[i].equals(match.getElement()))
581                                                        return;
582                                        super.acceptSearchMatch(unit, match);
583                                }
584                        };
585                } else
586                        requestor= new MethodOccurenceCollector(getNewElementName());
587

589

591                HashSet declaringCUs= new HashSet(newDeclarationWCs.length);
592                for (int i= 0; i < newDeclarationWCs.length; i++)
593                        declaringCUs.add(newDeclarationWCs[i].getPrimary());

598                }

612                }

615        }

632        private static IFunction[] classesDeclareMethodName(ITypeHierarchy hier, List classes, IFunction method, String newName)  throws CoreException {
633                Set result= new HashSet();
634                IType type= method.getDeclaringType();
635                List subtypes= Arrays.asList(hier.getAllSubtypes(type));
636                
637                int parameterCount= method.getParameterTypes().length;
638                boolean isMethodPrivate= JdtFlags.isPrivate(method);
639                
640                for (Iterator iter= classes.iterator(); iter.hasNext(); ){
641                        IType clazz= (IType) iter.next();
642                        IFunction[] methods= clazz.getFunctions();
643                        boolean isSubclass= subtypes.contains(clazz);
644                        for (int j= 0; j < methods.length; j++) {
645                                IFunction foundMethod= Checks.findMethod(newName, parameterCount, false, new IFunction[] {methods[j]});
646                                if (foundMethod == null)
647                                        continue;
648                                if (isSubclass || type.equals(clazz))
649                                        result.add(foundMethod);
650                                else if ((! isMethodPrivate) && (! JdtFlags.isPrivate(methods[j])))
651                                        result.add(foundMethod);
652                        }
653                }
654                return (IFunction[]) result.toArray(new IFunction[result.size()]);
655        }

699                        final String header= Messages.format(RefactoringCoreMessages.RenameMethodProcessor_descriptor_description, new String[] { JavaScriptElementLabels.getTextLabel(fMethod, JavaScriptElementLabels.ALL_FULLY_QUALIFIED), getNewElementName()});

705                        descriptor.setFlags(flags);
706                        descriptor.setJavaElement(fMethod);
707                        descriptor.setNewName(getNewElementName());
708                        descriptor.setUpdateReferences(fUpdateReferences);
709                        descriptor.setKeepOriginal(fDelegateUpdating);
710                        descriptor.setDeprecateDelegate(fDelegateDeprecation);
711                        return new DynamicValidationRefactoringChange(descriptor, RefactoringCoreMessages.RenameMethodProcessor_change_name, (Change[]) list.toArray(new Change[list.size()]));

724        void addOccurrences(TextChangeManager manager, IProgressMonitor pm, RefactoringStatus status) throws CoreException/*thrown in subtype*/{
725                pm.beginTask("", fOccurrences.length);                                 //$NON-NLS-1$
726                for (int i= 0; i < fOccurrences.length; i++){
727                        IJavaScriptUnit cu= fOccurrences[i].getCompilationUnit();
728                        if (cu == null)        
729                                continue;
730                        
731                        SearchMatch[] results= fOccurrences[i].getSearchResults();
732 
733                        // Split matches into declaration and non-declaration matches
734                        
735                        List declarationsInThisCu= new ArrayList();
736                        List referencesInThisCu= new ArrayList();
737                         
738                        for (int j= 0; j < results.length; j++) {
739                                if (results[j] instanceof MethodDeclarationMatch)
740                                        declarationsInThisCu.add(results[j]);
741                                else
742                                        referencesInThisCu.add(results[j]);
743                        }
744 
745                        // First, handle the declarations
746                        if (declarationsInThisCu.size() > 0) {
747 
748                                if (fDelegateUpdating) {
749                                        // Update with delegates
750                                        CompilationUnitRewrite rewrite= new CompilationUnitRewrite(cu);
751                                        rewrite.setResolveBindings(true);
752 
753                                        for (Iterator iter= declarationsInThisCu.iterator(); iter.hasNext();) {
754                                                SearchMatch element= (SearchMatch) iter.next();
755                                                FunctionDeclaration method= ASTNodeSearchUtil.getMethodDeclarationNode((IFunction) element.getElement(), rewrite.getRoot());
756                                                DelegateCreator creator= new DelegateMethodCreator();
757                                                creator.setDeclareDeprecated(fDelegateDeprecation);
758                                                creator.setDeclaration(method);
759                                                creator.setSourceRewrite(rewrite);
760                                                creator.setNewElementName(getNewElementName());
761                                                creator.prepareDelegate();
762                                                creator.createEdit();
763                                        }
764                                        // Need to handle all delegates first as this
765                                        // creates a completely new change object.
766                                        TextChange changeForThisCu= rewrite.createChange();
767                                        changeForThisCu.setKeepPreviewEdits(true);
768                                        manager.manage(cu, changeForThisCu);
769                                }
770 
771                                // Update the normal methods
772                                for (Iterator iter= declarationsInThisCu.iterator(); iter.hasNext();) {
773                                        SearchMatch element= (SearchMatch) iter.next();
774                                        simpleUpdate(element, cu, manager.get(cu));
775                                }
776                        }
777 
778                        // Second, handle references
779                        if (fUpdateReferences) {
780                                for (Iterator iter= referencesInThisCu.iterator(); iter.hasNext();) {
781                                        SearchMatch element= (SearchMatch) iter.next();
782                                        simpleUpdate(element, cu, manager.get(cu));
783                                }
784                        }
785                        
786                        pm.worked(1);
787                        if (pm.isCanceled())
788                                throw new OperationCanceledException();
789                }
790                pm.done();
791        }
