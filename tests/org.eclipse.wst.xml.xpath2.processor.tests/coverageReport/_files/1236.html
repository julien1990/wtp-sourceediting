988        private static Expression getAssignedValue(ProposalParameter context) {
989                ASTNode parent= context.accessNode.getParent();
990                AST ast= context.astRewrite.getAST();
991                switch (parent.getNodeType()) {
992                case ASTNode.ASSIGNMENT:
993                        Assignment assignment= ((Assignment) parent);
994                        Expression rightHandSide= assignment.getRightHandSide();
995                        Expression copiedRightOp= (Expression) context.astRewrite.createCopyTarget(rightHandSide);
996                        if (isNotInBlock(parent))
997                                break;
998                        if (assignment.getOperator() == Operator.ASSIGN) {
999                                ITypeBinding rightHandSideType= rightHandSide.resolveTypeBinding();
1000                                copiedRightOp= checkForNarrowCast(context, copiedRightOp, true, rightHandSideType);
1001                                return copiedRightOp;
1002                        }
1003                        IFunctionBinding getter= findGetter(context);
1004                        if (getter != null) {
1005                                InfixExpression infix= ast.newInfixExpression();
1006                                infix.setLeftOperand(createMethodInvocation(context, getter, null));
1007                                infix.setOperator(ASTNodes.convertToInfixOperator(assignment.getOperator()));
1008                                infix.setRightOperand(copiedRightOp);
1009                                ITypeBinding infixType= infix.resolveTypeBinding();
1010                                return checkForNarrowCast(context, infix, true, infixType);
1011                        }
1012                        break;
1013                case ASTNode.POSTFIX_EXPRESSION:
1014                        PostfixExpression po= (PostfixExpression) parent;
1015                        if (isNotInBlock(parent))
1016                                break;
1017                        InfixExpression.Operator postfixOp= null;
1018                        if (po.getOperator() == PostfixExpression.Operator.INCREMENT)
1019                                postfixOp= InfixExpression.Operator.PLUS;
1020                        if (po.getOperator() == PostfixExpression.Operator.DECREMENT)
1021                                postfixOp= InfixExpression.Operator.MINUS;
1022                        if (postfixOp == null)
1023                                break;
1024                        return createInfixInvocationFromPostPrefixExpression(context, postfixOp);
1025                case ASTNode.PREFIX_EXPRESSION:
1026                        PrefixExpression pe= (PrefixExpression) parent;
1027                        if (isNotInBlock(parent))
1028                                break;
1029                        InfixExpression.Operator prefixOp= null;
1030                        if (pe.getOperator() == PrefixExpression.Operator.INCREMENT)
1031                                prefixOp= InfixExpression.Operator.PLUS;
1032                        if (pe.getOperator() == PrefixExpression.Operator.DECREMENT)
1033                                prefixOp= InfixExpression.Operator.MINUS;
1034                        if (prefixOp == null)
1035                                break;
1036                        return createInfixInvocationFromPostPrefixExpression(context, prefixOp);
1037                }
1038 
1039                return null;
1040        }

1064         * 
1065         * @param context general context
1066         * @param expression the right handside
1067         * @param parenthesize if true places () around expression
1068         * @param expressionType the type of the right handside. Can be null
1069         * @return the casted expression if necessary
1070         */
1071        private static Expression checkForNarrowCast(ProposalParameter context, Expression expression, boolean parenthesize, ITypeBinding expressionType) {
1072                PrimitiveType castTo= null;
1073                ITypeBinding type= context.variableBinding.getType();
1074                if (type.isEqualTo(expressionType))
1075                        return expression; //no cast for same type
1076                AST ast= context.astRewrite.getAST();
1077                if (JavaModelUtil.is50OrHigher(context.compilationUnit.getJavaScriptProject())) {
1078                        if (ast.resolveWellKnownType("java.lang.Character").isEqualTo(type)) //$NON-NLS-1$
1079                                castTo= ast.newPrimitiveType(PrimitiveType.CHAR);
1080                        if (ast.resolveWellKnownType("java.lang.Byte").isEqualTo(type)) //$NON-NLS-1$
1081                                castTo= ast.newPrimitiveType(PrimitiveType.BYTE);
1082                        if (ast.resolveWellKnownType("java.lang.Short").isEqualTo(type)) //$NON-NLS-1$
1083                                castTo= ast.newPrimitiveType(PrimitiveType.SHORT);
1084                }
1085                if (ast.resolveWellKnownType("char").isEqualTo(type)) //$NON-NLS-1$
1086                        castTo= ast.newPrimitiveType(PrimitiveType.CHAR);
1087                if (ast.resolveWellKnownType("byte").isEqualTo(type)) //$NON-NLS-1$
1088                        castTo= ast.newPrimitiveType(PrimitiveType.BYTE);
1089                if (ast.resolveWellKnownType("short").isEqualTo(type)) //$NON-NLS-1$
1090                        castTo= ast.newPrimitiveType(PrimitiveType.SHORT);
1091                if (castTo != null) {
1092                        CastExpression cast= ast.newCastExpression();
1093                        if (parenthesize) {
1094                                ParenthesizedExpression parenthesized= ast.newParenthesizedExpression();
1095                                parenthesized.setExpression(expression);
1096                                cast.setExpression(parenthesized);
1097                        } else
1098                                cast.setExpression(expression);
1099                        cast.setType(castTo);
1100                        return cast;
1101                }
1102                return expression;
1103        }
