306                                                                // [JLS 5.5] T <: S
307                                                                if (expressionType.isRawType() || !expressionType.isEquivalentTo(match)) {
308                                                                        this.bits |= UnsafeCast;
309                                                                        return true;
310                                                                }
311                                                                // [JLS 5.5] S has no subtype X != T, such that |X| == |T|
312                                                                TypeBinding genericCastType = castType.erasure(); // jump to generic type
313                                                                TypeBinding genericMatch = genericCastType.findSuperTypeWithSameErasure(expressionType);
314                                                                if (genericMatch == match) {
315                                                                        this.bits |= UnsafeCast;
316                                                                }
317                                                                return true;
318                                                        } else {
319                                                                // [JLS 5.5] T >: S
320                                                                if (!match.isEquivalentTo(castType)) {
321                                                                        this.bits |= UnsafeCast;
322                                                                        return true;
323                                                                }
324                                                        }
325                                                        break;
326                                                case Binding.RAW_TYPE :
327                                                        this.bits |= UnsafeCast; // upcast since castType is known to be bound paramType
328                                                        return true;
329                                                default :
330                                                        if (isNarrowing){
331                                                                // match is not parameterized or raw, then any other subtype of match will erase  to |T|
332                                                                this.bits |= UnsafeCast;
333                                                                return true;
334                                                        }
335                                                        break;
336                                        }
337                                }
338                                break;
339                        case Binding.ARRAY_TYPE :
340                                TypeBinding leafType = castType.leafComponentType();
341                                if (isNarrowing && (leafType.isBoundParameterizedType() || leafType.isTypeVariable())) {
342                                        this.bits |= UnsafeCast;
343                                        return true;
344                                }
345                                break;
346                        case Binding.TYPE_PARAMETER :
347                                this.bits |= UnsafeCast;
348                                return true;
349                }
350                if (!isNarrowing && match == this.resolvedType.leafComponentType()) { // do not tag as unnecessary when recursing through upper bounds
351                        tagAsUnnecessaryCast(scope, castType);
352                }
353                return true;
354        }

394        public TypeBinding resolveType(BlockScope scope) {
395                // compute a new constant if the cast is effective
396 
397                // due to the fact an expression may start with ( and that a cast can also start with (
398                // the field is an expression....it can be a TypeReference OR a NameReference Or
399                // any kind of Expression <-- this last one is invalid.......
400 
401                constant = Constant.NotAConstant;
402                implicitConversion = T_undefined;
403 
404                if ((type instanceof TypeReference) || (type instanceof NameReference)
405                                && ((type.bits & ASTNode.ParenthesizedMASK) >> ASTNode.ParenthesizedSHIFT) == 0) { // no extra parenthesis around type: ((A))exp
406 
407                        TypeBinding castType = this.resolvedType = type.resolveType(scope);
408                        //expression.setExpectedType(this.resolvedType); // needed in case of generic method invocation
409                        TypeBinding expressionType = expression.resolveType(scope);
410                        if (castType != null) {
411                                if (expressionType != null) {
412                                        boolean isLegal = checkCastTypesCompatibility(scope, castType, expressionType, this.expression);
413                                        if (isLegal) {
414                                                this.expression.computeConversion(scope, castType, expressionType);
415                                                if ((this.bits & UnsafeCast) != 0) { // unsafe cast
416                                                        scope.problemReporter().unsafeCast(this, scope);
417                                                } else {
418                                                        if ((this.bits & (UnnecessaryCast|DisableUnnecessaryCastCheck)) == UnnecessaryCast) { // unnecessary cast
419                                                                if (!isIndirectlyUsed()) // used for generic type inference or boxing ?
420                                                                        scope.problemReporter().unnecessaryCast(this);
421                                                        }
422                                                }
423                                        } else { // illegal cast
424                                                scope.problemReporter().typeCastError(this, castType, expressionType);
425                                                this.bits |= DisableUnnecessaryCastCheck; // disable further secondary diagnosis
426                                        }
427                                }
428                                this.resolvedType = castType.capture(scope, this.sourceEnd);
429                        }
430                        return this.resolvedType;
431                } else { // expression as a cast
432                        TypeBinding expressionType = expression.resolveType(scope);
433                        if (expressionType == null) return null;
434                        scope.problemReporter().invalidTypeReference(type);
435                        return null;
436                }
437        }

448         * perform return type inference of generic method invocation or boxing.
449         */
450        private boolean isIndirectlyUsed() {
451                if (this.expression instanceof MessageSend) {
452                        MethodBinding method = ((MessageSend)this.expression).binding;
453                        if (method instanceof ParameterizedGenericMethodBinding
454                                                && ((ParameterizedGenericMethodBinding)method).inferredReturnType) {
455                                if (this.expectedType == null)
456                                        return true;
457                                if (this.resolvedType != this.expectedType)
458                                        return true;
459                        }
460                }
461                if (this.expectedType != null && this.resolvedType.isBaseType() && !this.resolvedType.isCompatibleWith(this.expectedType)) {
462                        // boxing: Short s = (short) _byte
463                        return true;
464                }
465                return false;
466        }
