485        public static void checkMethodInHierarchy(ITypeBinding type, String methodName, ITypeBinding returnType, ITypeBinding[] parameters, RefactoringStatus result, boolean reUseMethod) {
486                IFunctionBinding method= Bindings.findMethodInHierarchy(type, methodName, parameters);
487                if (method != null) {
488                        boolean returnTypeClash= false;
489                        ITypeBinding methodReturnType= method.getReturnType();
490                        if (returnType != null && methodReturnType != null) {
491                                String returnTypeKey= returnType.getKey();
492                                String methodReturnTypeKey= methodReturnType.getKey();
493                                if (returnTypeKey == null && methodReturnTypeKey == null) {
494                                        returnTypeClash= returnType != methodReturnType;        
495                                } else if (returnTypeKey != null && methodReturnTypeKey != null) {
496                                        returnTypeClash= !returnTypeKey.equals(methodReturnTypeKey);
497                                }
498                        }
499                        ITypeBinding dc= method.getDeclaringClass();
500                        if (returnTypeClash) {
501                                result.addError(Messages.format(RefactoringCoreMessages.Checks_methodName_returnTypeClash, 
502                                        new Object[] {methodName, dc.getName()}),
503                                        JavaStatusContext.create(method));
504                        } else {
505                                if (!reUseMethod)
506                                        result.addError(Messages.format(RefactoringCoreMessages.Checks_methodName_overrides, 
507                                                new Object[] {methodName, dc.getName()}),
508                                                JavaStatusContext.create(method));
509                        }
510                } else {
511                        if (reUseMethod){
512                                result.addError(Messages.format(RefactoringCoreMessages.SelfEncapsulateFieldRefactoring_nosuchmethod_status_fatalError, 
513                                                new Object[] {methodName}),
514                                                JavaStatusContext.create(method));
515                        }
516                }
517        }
