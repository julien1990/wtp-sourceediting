406        protected ASTNode[] findNodes(JavaScriptUnit unit, final MarkerInfo markerInfo) {
407                class Visitor extends ASTVisitor {
408                        ArrayList found = new ArrayList();
409                        public void preVisit(ASTNode node) {
410                                if (node instanceof JavaScriptUnit) return;
411                                int index = markerInfo.indexOfASTStart(node.getStartPosition());
412                                if (index != -1 && node.getStartPosition() + node.getLength() == markerInfo.astEnds[index]) {
413                                        this.found.add(node);
414                                        markerInfo.astStarts[index] = -1; // so that 2 nodes with the same start and end will not be found
415                                }
416                        }
417                }
418                Visitor visitor = new Visitor();
419                unit.accept(visitor);
420                int size = visitor.found.size();
421                ASTNode[] result = new ASTNode[size];
422                visitor.found.toArray(result);
423                return result;
424        }

444                        case ASTNode.FUNCTION_DECLARATION:

446                        case ASTNode.FUNCTION_INVOCATION:

448                        case ASTNode.TYPE_PARAMETER:
449                                return ((TypeParameter) node).resolveBinding();
450                        case ASTNode.PARAMETERIZED_TYPE:
451                                return ((ParameterizedType) node).resolveBinding();
452                        case ASTNode.WILDCARD_TYPE:
453                                return ((WildcardType) node).resolveBinding();
454                        case ASTNode.SIMPLE_NAME:
