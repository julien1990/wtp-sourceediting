259         * Returns a canonicalized path from the given external path.
260         * Note that the return path contains the same number of segments
261         * and it contains a device only if the given path contained one.
262         * @param externalPath IPath
263         * @see java.io.File for the definition of a canonicalized path
264         * @return IPath
265         */
266        public static IPath canonicalizedPath(IPath externalPath) {
267 
268                if (externalPath == null)
269                        return null;
270 
271//                if (JavaModelManager.VERBOSE) {
272//                        System.out.println("JAVA MODEL - Canonicalizing " + externalPath.toString());
273//                }
274 
275                if (IS_CASE_SENSITIVE) {
276//                        if (JavaModelManager.VERBOSE) {
277//                                System.out.println("JAVA MODEL - Canonical path is original path (file system is case sensitive)");
278//                        }
279                        return externalPath;
280                }
281 
282                // if not external path, return original path
283                IWorkspace workspace = ResourcesPlugin.getWorkspace();
284                if (workspace == null) return externalPath; // protection during shutdown (30487)
285                if (workspace.getRoot().findMember(externalPath) != null) {
286//                        if (JavaModelManager.VERBOSE) {
287//                                System.out.println("JAVA MODEL - Canonical path is original path (member of workspace)");
288//                        }
289                        return externalPath;
290                }
291 
292                IPath canonicalPath = null;
293                try {
294                        canonicalPath =
295                                new Path(new File(externalPath.toOSString()).getCanonicalPath());
296                } catch (IOException e) {
297                        // default to original path
298//                        if (JavaModelManager.VERBOSE) {
299//                                System.out.println("JAVA MODEL - Canonical path is original path (IOException)");
300//                        }
301                        return externalPath;
302                }
303 
304                IPath result;
305                int canonicalLength = canonicalPath.segmentCount();
306                if (canonicalLength == 0) {
307                        // the java.io.File canonicalization failed
308//                        if (JavaModelManager.VERBOSE) {
309//                                System.out.println("JAVA MODEL - Canonical path is original path (canonical path is empty)");
310//                        }
311                        return externalPath;
312                } else if (externalPath.isAbsolute()) {
313                        result = canonicalPath;
314                } else {
315                        // if path is relative, remove the first segments that were added by the java.io.File canonicalization
316                        // e.g. 'lib/classes.zip' was converted to 'd:/myfolder/lib/classes.zip'
317                        int externalLength = externalPath.segmentCount();
318                        if (canonicalLength >= externalLength) {
319                                result = canonicalPath.removeFirstSegments(canonicalLength - externalLength);
320                        } else {
321//                                if (JavaModelManager.VERBOSE) {
322//                                        System.out.println("JAVA MODEL - Canonical path is original path (canonical path is " + canonicalPath.toString() + ")");
323//                                }
324                                return externalPath;
325                        }
326                }
327 
328                // keep device only if it was specified (this is because File.getCanonicalPath() converts '/lib/classed.zip' to 'd:/lib/classes/zip')
329                if (externalPath.getDevice() == null) {
330                        result = result.setDevice(null);
331                }
332//                if (JavaModelManager.VERBOSE) {
333//                        System.out.println("JAVA MODEL - Canonical path is " + result.toString());
334//                }
335                return result;
336        }
