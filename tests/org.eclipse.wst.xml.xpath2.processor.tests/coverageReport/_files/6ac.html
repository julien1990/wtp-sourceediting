591                boolean insertRequiredAttrs = getCleanupPreferences().getInsertRequiredAttrs();
592                IDOMNode newNode = node;
593 
594                if (insertRequiredAttrs) {
595                        List requiredAttrs = getRequiredAttrs(newNode);
596                        if (requiredAttrs.size() > 0) {
597                                NamedNodeMap currentAttrs = node.getAttributes();
598                                List insertAttrs = new ArrayList();
599                                if (currentAttrs.getLength() == 0)
600                                        insertAttrs.addAll(requiredAttrs);
601                                else {
602                                        for (int i = 0; i < requiredAttrs.size(); i++) {
603                                                String requiredAttrName = ((CMAttributeDeclaration) requiredAttrs.get(i)).getAttrName();
604                                                boolean found = false;
605                                                for (int j = 0; j < currentAttrs.getLength(); j++) {
606                                                        String currentAttrName = currentAttrs.item(j).getNodeName();
607                                                        if (requiredAttrName.compareToIgnoreCase(currentAttrName) == 0) {
608                                                                found = true;
609                                                                break;
610                                                        }
611                                                }
612                                                if (!found)
613                                                        insertAttrs.add(requiredAttrs.get(i));
614                                        }
615                                }
616                                if (insertAttrs.size() > 0) {
617                                        IStructuredDocumentRegion startStructuredDocumentRegion = newNode.getStartStructuredDocumentRegion();
618                                        int index = startStructuredDocumentRegion.getEndOffset();
619                                        ITextRegion lastRegion = startStructuredDocumentRegion.getLastRegion();
620                                        if (lastRegion.getType() == DOMRegionContext.XML_TAG_CLOSE) {
621                                                index--;
622                                                lastRegion = startStructuredDocumentRegion.getRegionAtCharacterOffset(index - 1);
623                                        }
624                                        else if (lastRegion.getType() == DOMRegionContext.XML_EMPTY_TAG_CLOSE) {
625                                                index = index - 2;
626                                                lastRegion = startStructuredDocumentRegion.getRegionAtCharacterOffset(index - 1);
627                                        }
628                                        MultiTextEdit multiTextEdit = new MultiTextEdit();
629                                        try {
630                                                for (int i = insertAttrs.size() - 1; i >= 0; i--) {
631                                                        CMAttributeDeclaration attrDecl = (CMAttributeDeclaration) insertAttrs.get(i);
632                                                        String requiredAttributeName = attrDecl.getAttrName();
633                                                        String defaultValue = attrDecl.getDefaultValue();
634                                                        if (defaultValue == null)
635                                                                defaultValue = ""; //$NON-NLS-1$
636                                                        String nameAndDefaultValue = " "; //$NON-NLS-1$
637                                                        if (i == 0 && lastRegion.getLength() > lastRegion.getTextLength())
638                                                                nameAndDefaultValue = ""; //$NON-NLS-1$
639                                                        nameAndDefaultValue += requiredAttributeName + "=\"" + defaultValue + "\""; //$NON-NLS-1$ //$NON-NLS-2$
640                                                        multiTextEdit.addChild(new InsertEdit(index, nameAndDefaultValue));
641                                                        // BUG3381: MultiTextEdit applies all child
642                                                        // TextEdit's basing on offsets
643                                                        // in the document before the first TextEdit, not

645                                                        // child TextEdit. Therefore, do not need to

647                                                        // index += nameAndDefaultValue.length();
648                                                }

650                                        }
