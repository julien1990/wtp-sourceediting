449        /**
450         * Hooks up the viewer to follow the selection made in the active editor
451         */
452        private void followSelection() {
453                ITextEditor editor = getActiveEditor();
454                if (editor != null) {
455                        setupViewerForEditor(editor);
456                        if (fHighlightRangeListener == null)
457                                fHighlightRangeListener = new NodeRangeSelectionListener();
458 
459                        fContentOutlinePage = ((IContentOutlinePage) editor.getAdapter(IContentOutlinePage.class));
460                        if (fContentOutlinePage != null) {
461                                fContentOutlinePage.addSelectionChangedListener(fHighlightRangeListener);
462 
463                                if (!fContentOutlinePage.getSelection().isEmpty() && fContentOutlinePage.getSelection() instanceof IStructuredSelection) {
464                                        fSourceViewer.resetVisibleRegion();
465 
466                                        Object[] nodes = ((IStructuredSelection) fContentOutlinePage.getSelection()).toArray();
467                                        IndexedRegion startNode = (IndexedRegion) nodes[0];
468                                        IndexedRegion endNode = (IndexedRegion) nodes[nodes.length - 1];
469 
470                                        if (startNode instanceof Attr)
471                                                startNode = (IndexedRegion) ((Attr) startNode).getOwnerElement();
472                                        if (endNode instanceof Attr)
473                                                endNode = (IndexedRegion) ((Attr) endNode).getOwnerElement();
474 
475                                        int start = startNode.getStartOffset();
476                                        int end = endNode.getEndOffset();
477 
478                                        fSourceViewer.setVisibleRegion(start, end - start);
479                                        fSourceViewer.setSelectedRange(start, 0);
480                                }
481                        }
482                }
483        }
