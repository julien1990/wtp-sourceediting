434        private void restoreFromPreferences() {
435                String compiledTextHoverModifiers= fStore.getString(PreferenceConstants.EDITOR_TEXT_HOVER_MODIFIERS);
436                
437                StringTokenizer tokenizer= new StringTokenizer(compiledTextHoverModifiers, JavaEditorTextHoverDescriptor.VALUE_SEPARATOR);
438                HashMap idToModifier= new HashMap(tokenizer.countTokens() / 2);
439 
440                while (tokenizer.hasMoreTokens()) {
441                        String id= tokenizer.nextToken();
442                        if (tokenizer.hasMoreTokens())
443                                idToModifier.put(id, tokenizer.nextToken());
444                }
445 
446                String compiledTextHoverModifierMasks= JavaScriptPlugin.getDefault().getPreferenceStore().getString(PreferenceConstants.EDITOR_TEXT_HOVER_MODIFIER_MASKS);
447 
448                tokenizer= new StringTokenizer(compiledTextHoverModifierMasks, JavaEditorTextHoverDescriptor.VALUE_SEPARATOR);
449                HashMap idToModifierMask= new HashMap(tokenizer.countTokens() / 2);
450 
451                while (tokenizer.hasMoreTokens()) {
452                        String id= tokenizer.nextToken();
453                        if (tokenizer.hasMoreTokens())
454                                idToModifierMask.put(id, tokenizer.nextToken());
455                }
456 
457                for (int i= 0; i < fHoverConfigs.length; i++) {
458                        String modifierString= (String)idToModifier.get(getContributedHovers()[i].getId());
459                        boolean enabled= true;
460                        if (modifierString == null)
461                                modifierString= JavaEditorTextHoverDescriptor.DISABLED_TAG;
462                        
463                        if (modifierString.startsWith(JavaEditorTextHoverDescriptor.DISABLED_TAG)) {
464                                enabled= false;
465                                modifierString= modifierString.substring(1);
466                        }
467 
468                        if (modifierString.equals(JavaEditorTextHoverDescriptor.NO_MODIFIER))
469                                modifierString= ""; //$NON-NLS-1$
470 
471                        fHoverConfigs[i].fModifierString= modifierString;
472                        fHoverConfigs[i].fIsEnabled= enabled;
473                        fHoverConfigs[i].fStateMask= JavaEditorTextHoverDescriptor.computeStateMask(modifierString);
474 
475                        if (fHoverConfigs[i].fStateMask == -1) {
476                                try {
477                                        fHoverConfigs[i].fStateMask= Integer.parseInt((String)idToModifierMask.get(getContributedHovers()[i].getId()));
478                                } catch (NumberFormatException ex) {
479                                        fHoverConfigs[i].fStateMask= -1;
480                                }
481                        }
482                }
483        }

524        private void updateStatus(HoverConfig hoverConfig) {
525                if (hoverConfig != null && hoverConfig.fIsEnabled && hoverConfig.fStateMask == -1)
526                        fStatus= new StatusInfo(IStatus.ERROR, Messages.format(PreferencesMessages.JavaEditorHoverConfigurationBlock_modifierIsNotValid, hoverConfig.fModifierString)); 
527                else
528                        fStatus= new StatusInfo();
529                
530                int i= 0;
531                HashMap stateMasks= new HashMap(fHoverConfigs.length);
532                while (fStatus.isOK() && i < fHoverConfigs.length) {
533                        if (fHoverConfigs[i].fIsEnabled) {
534                                String label= getContributedHovers()[i].getLabel();
535                                Integer stateMask= new Integer(fHoverConfigs[i].fStateMask);
536                                if (fHoverConfigs[i].fStateMask == -1)
537                                        fStatus= new StatusInfo(IStatus.ERROR, Messages.format(PreferencesMessages.JavaEditorHoverConfigurationBlock_modifierIsNotValidForHover, new String[] {fHoverConfigs[i].fModifierString, label})); 
538                                else if (stateMasks.containsKey(stateMask))
539                                        fStatus= new StatusInfo(IStatus.ERROR, Messages.format(PreferencesMessages.JavaEditorHoverConfigurationBlock_duplicateModifier, new String[] {label, (String)stateMasks.get(stateMask)})); 
540                                else
541                                        stateMasks.put(stateMask, label);
542                        }
543                        i++;
544                }
545 
546                fMainPreferencePage.setValid(fStatus.isOK());
547                StatusUtil.applyToStatusLine(fMainPreferencePage, fStatus);
548        }
