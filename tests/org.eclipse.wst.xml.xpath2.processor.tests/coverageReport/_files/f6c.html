402        private boolean getMenuCategories(List result) {
403                final HashSet/*<String>*/ categories= new HashSet();
404                final HashSet/*<String>*/ foundLRUCategories= new HashSet();
405                final boolean hasUncategorizedMember[]= new boolean[] {false};
406                for (int i= 0; i < fInputElement.length && !(hasUncategorizedMember[0] && foundLRUCategories.size() >= MAX_NUMBER_OF_CATEGORIES_IN_MENU); i++) {
407                        collectCategories(fInputElement[i], new IResultCollector() {
408                                public boolean accept(String[] cats) {
409                                        if (cats.length > 0) {
410                                                for (int j= 0; j < cats.length; j++) {
411                                                        String category= cats[j];
412                                                        categories.add(category);
413                                                        if (fLRUList.containsKey(category)) {
414                                                                foundLRUCategories.add(category);
415                                                        }        
416                                                }
417                                        } else {
418                                                hasUncategorizedMember[0]= true;
419                                        }
420                                        return hasUncategorizedMember[0] && foundLRUCategories.size() >= MAX_NUMBER_OF_CATEGORIES_IN_MENU;
421                                }
422                        });
423                }
424                int count= 0;
425                for (Iterator iter= foundLRUCategories.iterator(); iter.hasNext();) {
426                        String element= (String)iter.next();
427                        result.add(element);
428                        count++;
429                }
430                if (count < MAX_NUMBER_OF_CATEGORIES_IN_MENU) {
431                        List sortedCategories= new ArrayList(categories);
432                        Collections.sort(sortedCategories, Collator.getInstance());
433                        for (Iterator iter= sortedCategories.iterator(); iter.hasNext() && count < MAX_NUMBER_OF_CATEGORIES_IN_MENU;) {
434                                String element= (String)iter.next();
435                                if (!foundLRUCategories.contains(element)) {
436                                        result.add(element);
437                                        count++;
438                                }
439                        }
440                }
441                return hasUncategorizedMember[0];
442        }

444        private boolean collectCategories(IJavaScriptElement element, IResultCollector collector) {//HashSet result, int max, LinkedHashMap lruList) {
445                try {
446                        if (element instanceof IMember) {
447                                IMember member= (IMember)element;
448                                collector.accept(member.getCategories());
449                                return processChildren(member.getChildren(), collector);
450                        } else if (element instanceof IJavaScriptUnit) {
451                                return processChildren(((IJavaScriptUnit)element).getChildren(), collector);
452                        } else if (element instanceof IClassFile) {
453                                return processChildren(((IClassFile)element).getChildren(), collector);
454                        } else if (element instanceof IJavaScriptModel) {
455                                return processChildren(((IJavaScriptModel)element).getChildren(), collector);
456                        } else if (element instanceof IJavaScriptProject) {
457                                return processChildren(((IJavaScriptProject)element).getChildren(), collector);
458                        } else if (element instanceof IPackageFragment) {
459                                return processChildren(((IPackageFragment)element).getChildren(), collector);
460                        } else if (element instanceof IPackageFragmentRoot)         {
461                                return processChildren(((IPackageFragmentRoot)element).getChildren(), collector);
462                        }
463                        return false;
