814        private void updateClasspathList() {
815                List srcelements= fFoldersList.getElements();
816                
817                List cpelements= fClassPathList.getElements();
818                int nEntries= cpelements.size();
819                // backwards, as entries will be deleted
820                int lastRemovePos= nEntries;
821                int afterLastSourcePos= 0;
822                for (int i= nEntries - 1; i >= 0; i--) {
823                        CPListElement cpe= (CPListElement)cpelements.get(i);
824                        int kind= cpe.getEntryKind();
825                        if (isEntryKind(kind)) {
826                                if (!srcelements.remove(cpe)) {
827                                        cpelements.remove(i);
828                                        lastRemovePos= i;
829                                } else if (lastRemovePos == nEntries) {
830                                        afterLastSourcePos= i + 1;
831                                }
832                        }
833                }
834 
835                if (!srcelements.isEmpty()) {
836                        int insertPos= Math.min(afterLastSourcePos, lastRemovePos);
837                        cpelements.addAll(insertPos, srcelements);
838                }
839                
840                if (lastRemovePos != nEntries || !srcelements.isEmpty()) {
841                        fClassPathList.setElements(cpelements);
842                }
843        }
