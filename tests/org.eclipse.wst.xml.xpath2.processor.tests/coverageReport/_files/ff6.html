140            public int compare(Object left, Object right) {
141                    TypeNameMatch leftInfo= (TypeNameMatch)left;
142                    TypeNameMatch rightInfo= (TypeNameMatch)right;
143                     int leftCategory= getCamelCaseCategory(leftInfo);
144                     int rightCategory= getCamelCaseCategory(rightInfo);
145                     if (leftCategory < rightCategory)
146                             return -1;
147                     if (leftCategory > rightCategory)
148                             return +1;
149                     int result= compareName(leftInfo.getSimpleTypeName(), rightInfo.getSimpleTypeName());
150                     if (result != 0)
151                             return result;
152                     result= compareTypeContainerName(leftInfo.getTypeContainerName(), rightInfo.getTypeContainerName());
153                     if (result != 0)
154                             return result;
155                     
156                     leftCategory= getElementTypeCategory(leftInfo);
157                     rightCategory= getElementTypeCategory(rightInfo);
158                     if (leftCategory < rightCategory)
159                             return -1;
160                     if (leftCategory > rightCategory)
161                             return +1;
162                     return compareContainerName(leftInfo, rightInfo);
163            }
164                private int compareName(String leftString, String rightString) {
165                        int result= leftString.compareToIgnoreCase(rightString);
166                        if (result != 0 || rightString.length() == 0) {
167                                return result;
168                        } else if (Strings.isLowerCase(leftString.charAt(0)) && 
169                                !Strings.isLowerCase(rightString.charAt(0))) {
170                             return +1;
171                        } else if (Strings.isLowerCase(rightString.charAt(0)) &&
172                             !Strings.isLowerCase(leftString.charAt(0))) {
173                             return -1;
174                        } else {
175                                return leftString.compareTo(rightString);
176                        }
177                }
178                private int compareTypeContainerName(String leftString, String rightString) {
179                        int leftLength= leftString.length();
180                        int rightLength= rightString.length();
181                        if (leftLength == 0 && rightLength > 0)
182                                return -1;
183                        if (leftLength == 0 && rightLength == 0)
184                                return 0;
185                        if (leftLength > 0 && rightLength == 0)
186                                return +1;
187                        return compareName(leftString, rightString);
188                }
189                private int compareContainerName(TypeNameMatch leftType, TypeNameMatch rightType) {
190                        return fLabelProvider.getContainerName(leftType).compareTo(
191                                fLabelProvider.getContainerName(rightType));
192                }
193                private int getCamelCaseCategory(TypeNameMatch type) {
194                        if (fFilter == null)
195                                return 0;
196                        if (!fFilter.isCamelCasePattern())
197                                return 0;
198                        return fFilter.matchesRawNamePattern(type) ? 0 : 1;
199                }
200                private int getElementTypeCategory(TypeNameMatch type) {
201                        try {
202                                if (type.getPackageFragmentRoot().getKind() == IPackageFragmentRoot.K_SOURCE)
203                                        return 0;
204                        } catch (JavaScriptModelException e) {
205                                // TODO Auto-generated catch block
206                                e.printStackTrace();
207                        }
208                        return 1;
209                }

238                private void processVMInstallType(IVMInstallType installType, List locations, List labels) {

240                                IVMInstall[] installs= installType.getVMInstalls();
241                                boolean isMac= Platform.OS_MACOSX.equals(Platform.getOS());
242                                final String HOME_SUFFIX= "/Home"; //$NON-NLS-1$
243                                for (int i= 0; i < installs.length; i++) {

249                                                String filePath= installs[i].getInstallLocation().getAbsolutePath();
250                                                // on MacOS X install locations end in an additional "/Home" segment; remove it
251                                                if (isMac && filePath.endsWith(HOME_SUFFIX))
252                                                        filePath= filePath.substring(0, filePath.length()- HOME_SUFFIX.length() + 1);
253                                                locations.add(filePath);
254                                                labels.add(label);
255                                        }
256                                }
257                        }

301                        if (last != null) {
302                                String lastTN= last.getSimpleTypeName();
303                                String lastTCN= getTypeContainerName(last);
304                                if (currentTCN.equals(lastTCN)) {
305                                        if (currentTN.equals(lastTN)) {

310                                                return result.toString();
311                                        }
312                                } else if (currentTN.equals(lastTN)) {
313                                        qualifications= 1;
314                                }
315                        }
316                        if (next != null) {
317                                String nextTN= next.getSimpleTypeName();
318                                String nextTCN= getTypeContainerName(next);
319                                if (currentTCN.equals(nextTCN)) {
320                                        if (currentTN.equals(nextTN)) {
321                                                result.append(JavaScriptElementLabels.CONCAT_STRING);
322                                                result.append(currentTCN);
323                                                result.append(JavaScriptElementLabels.CONCAT_STRING);
324                                                result.append(getContainerName(current));
325                                                return result.toString();
326                                        }
327                                } else if (currentTN.equals(nextTN)) {
328                                        qualifications= 1;
329                                }
330                        }
331                        if (qualifications > 0) {
332                                result.append(JavaScriptElementLabels.CONCAT_STRING);
333                                result.append(currentTCN);
334                                if (fFullyQualifyDuplicates) {
335                                        result.append(JavaScriptElementLabels.CONCAT_STRING);
336                                        result.append(getContainerName(current));
337                                }
338                        }

529                        fViewer.clear(fTicket);
530

589                                        imageDescriptors.add(fLabelProvider.getImageDescriptor(type));

594                                }
