396                                this.leadingIndexes[this.leadingPtr] = (((long)startIdx)<<32) + endIdx;
397                                extended = this.comments[endIdx].getStartPosition();
398                        }
399                }
400                return extended;
401        }

403        /**
404         * Search and store node trailing comments. Comments are searched in position range
405         * from node end position to specified next start. If one or several comment are found,
406         * returns last comment end position, otherwise returns node end position.
407         * <p>
408         * Starts to search for first comment after node end position and return if none was found...
409         *</p><p>
410         * When first comment is found after node, goes down in comment list until one of
411         * following conditions becomes true:
412         * <ol>
413         *         <li>comment start is after next start</li>
414         *         <li>there's other than white characters between current node and comment</li>
415         *         <li>there's more than 1 line between current node and comment</li>
416         *</ol>
417         * If at least potential comments have been found, then all of them has to be separated
418         * from following node. So, remove all comments which do not verify this assumption.
419         * Note that this verification is not applicable on last node.
420         * </p><p>
421         * If finally there's still trailing comments, then stores indexes of the first and last one
422         * in trailing comments table.
423         */
424        int storeTrailingComments(ASTNode node, int nextStart,  boolean lastChild, int[] parentLineRange) {
425 
426                // Init extended position
427                int nodeEnd = node.getStartPosition()+node.getLength()-1;
428                if (nodeEnd == nextStart) {
429                        // special case for last child of its parent
430                        if (++this.trailingPtr == 0) {
431                                this.trailingNodes = new ASTNode[STORAGE_INCREMENT];
432                                this.trailingIndexes = new long[STORAGE_INCREMENT];
433                                this.lastTrailingPtr = -1;
434                        } else if (this.trailingPtr == this.trailingNodes.length) {
435                                int newLength = (this.trailingPtr*3/2)+STORAGE_INCREMENT;
436                                System.arraycopy(this.trailingNodes, 0, this.trailingNodes = new ASTNode[newLength], 0, this.trailingPtr);
437                                System.arraycopy(this.trailingIndexes, 0, this.trailingIndexes = new long[newLength], 0, this.trailingPtr);
438                        }
439                        this.trailingNodes[this.trailingPtr] = node;
440                        this.trailingIndexes[this.trailingPtr] = -1;
441                        return nodeEnd;
442                }
443                int extended = nodeEnd;
444 
445                // Get line number
446                int nodeEndLine = getLineNumber(nodeEnd, parentLineRange);
447 
448                // Find comments range index
449                int idx = getCommentIndex(0, nodeEnd, 1);
450                if (idx == -1) {
451                        return nodeEnd;
452                }
453 
454                // Look after potential comments
455                int startIdx = idx;
456                int endIdx = -1;
457                int length = this.comments.length;
458                int commentStart = extended+1;
459                int previousEnd = nodeEnd+1;
460                int sameLineIdx = -1;
461                while (idx<length && commentStart < nextStart) {
462                        // get comment and leave if next starting position has been reached
463                        Comment comment = this.comments[idx];
464                        commentStart = comment.getStartPosition();
465                        // verify that there's nothing else than white spaces between node/comments
466                        if (commentStart >= nextStart) {
467                                // stop search on condition 1)
468                                break;
469                        } else if (previousEnd < commentStart) {
470                                this.scanner.resetTo(previousEnd, commentStart);
471                                try {
472                                        int token = this.scanner.getNextToken();
473                                        if (token != TerminalTokens.TokenNameWHITESPACE || this.scanner.currentPosition != commentStart) {
474                                                // stop search on condition 2)
475                                                // if first index fails, then there's no extended position in fact...
476                                                if (idx == startIdx) {
477                                                        return nodeEnd;
478                                                }
479                                                // otherwise we get the last index of trailing comment => break
480                                                break;
481                                        }
482                                } catch (InvalidInputException e) {
483                                        // Should not happen, but return no extended position...
484                                        return nodeEnd;
485                                }
486                                // verify that there's no more than one line between node/comments
487                                char[] gap = this.scanner.getCurrentIdentifierSource();
488                                int nbrLine = 0;
489                                int pos = -1;
490                                while ((pos=CharOperation.indexOf('\n', gap,pos+1)) >= 0) {
491                                        nbrLine++;
492                                }
493                                if (nbrLine > 1) {
494                                        // stop search on condition 3)
495                                        break;
496                                }
497                        }
498                        // Store index if we're on the same line than node end
499                        int commentLine = getLineNumber(commentStart, parentLineRange);
500                        if (commentLine == nodeEndLine) {
501                                sameLineIdx = idx;
502                        }
503                        // Store previous infos
504                        previousEnd = commentStart+comment.getLength();
505                        endIdx = idx++;
506                }
507                if (endIdx != -1) {
508                        // Verify that following node start is separated
509                        if (!lastChild) {
510                                int nextLine = getLineNumber(nextStart, parentLineRange);
511                                int previousLine = getLineNumber(previousEnd, parentLineRange);
512                                if((nextLine - previousLine) <= 1) {
513                                        if (sameLineIdx == -1) return nodeEnd;
514                                        endIdx = sameLineIdx;
515                                }
516                        }
517                        // Store trailing comments indexes
518                        if (++this.trailingPtr == 0) {
519                                this.trailingNodes = new ASTNode[STORAGE_INCREMENT];
520                                this.trailingIndexes = new long[STORAGE_INCREMENT];
521                                this.lastTrailingPtr = -1;
522                        } else if (this.trailingPtr == this.trailingNodes.length) {
523                                int newLength = (this.trailingPtr*3/2)+STORAGE_INCREMENT;
524                                System.arraycopy(this.trailingNodes, 0, this.trailingNodes = new ASTNode[newLength], 0, this.trailingPtr);
525                                System.arraycopy(this.trailingIndexes, 0, this.trailingIndexes = new long[newLength], 0, this.trailingPtr);
526                        }
527                        this.trailingNodes[this.trailingPtr] = node;
528                        long nodeRange = (((long)startIdx)<<32) + endIdx;
529                        this.trailingIndexes[this.trailingPtr] = nodeRange;
530                        // Compute new extended end
531                        extended = this.comments[endIdx].getStartPosition()+this.comments[endIdx].getLength()-1;
532                        // Look for children unresolved extended end
533                        ASTNode previousNode = node;
534                        int ptr = this.trailingPtr - 1; // children extended end were stored before
535                        while (ptr >= 0) {
536                                long range = this.trailingIndexes[ptr];
537                                if (range != -1) break; // there's no more unresolved nodes
538                                ASTNode unresolved = this.trailingNodes[ptr];
539                                if (previousNode != unresolved.getParent()) break; // we're no longer in node ancestor hierarchy
540                                this.trailingIndexes[ptr] = nodeRange;
541                                previousNode = unresolved;
542                                ptr--; // get previous node
543                        }
544                        // Remove remaining unresolved nodes
545                        if (ptr > this.lastTrailingPtr) {
546                                int offset = ptr - this.lastTrailingPtr;
547                                for (int i=ptr+1; i<=this.trailingPtr; i++) {
548                                        this.trailingNodes[i-offset] = this.trailingNodes[i];
549                                        this.trailingIndexes[i-offset] = this.trailingIndexes[i];
550                                }
551                                this.trailingPtr -= offset;
552                        }
553                        this.lastTrailingPtr = this.trailingPtr;
554                }
555                return extended;
556        }

558        class CommentMapperVisitor extends DefaultASTVisitor {
559 
560                ASTNode topSiblingParent = null;
561                ASTNode[] siblings = new ASTNode[10];
562                int[][] parentLineRange = new int[10][];
563                int siblingPtr = -1;
564 
565                protected boolean visitNode(ASTNode node) {
566                        if (node instanceof InferredType)
567                                return true;
568 
569                        // Get default previous end
570                        ASTNode parent = node.getParent();
571                        int previousEnd = parent.getStartPosition();
572 
573                        // Look for sibling node
574                         ASTNode sibling = parent == this.topSiblingParent ? (ASTNode) this.siblings[this.siblingPtr] : null;
575                        if (sibling != null) {
576                                // Found one previous sibling, so compute its trailing comments using current node start position
577                                try {
578                                        previousEnd = storeTrailingComments(sibling, node.getStartPosition(), false, this.parentLineRange[this.siblingPtr]);
579                                } catch (Exception ex) {
580                                        // Give up extended ranges at this level if unexpected exception happens...
581                                }
582                        }
583 
584                        // Stop visit for malformed node (see bug https://bugs.eclipse.org/bugs/show_bug.cgi?id=84049)
585                        if ((node.typeAndFlags & ASTNode.MALFORMED) != 0) {
586                                return false;
587                        }
588 
589                        // Compute leading comments for current node
590                        int[] previousLineRange = this.siblingPtr > -1 ? this.parentLineRange[this.siblingPtr] : new int[] {1, DefaultCommentMapper.this.scanner.linePtr+1};
591                        try {
592                                storeLeadingComments(node, previousEnd, previousLineRange);
593                        } catch (Exception ex) {
594                                // Give up extended ranges at this level if unexpected exception happens...
595                        }
596 
597                        // Store current node as waiting sibling for its parent
598                        if (this.topSiblingParent != parent) {
599                                if (this.siblings.length == ++this.siblingPtr) {
600                                        System.arraycopy(this.siblings, 0, this.siblings = new ASTNode[this.siblingPtr*2], 0, this.siblingPtr);
601                                        System.arraycopy(this.parentLineRange, 0, this.parentLineRange = new int[this.siblingPtr*2][], 0, this.siblingPtr);
602                                }
603                                if (this.topSiblingParent == null) {
604                                        // node is a JavaScriptUnit
605                                        this.parentLineRange[this.siblingPtr] = previousLineRange;
606                                } else {
607                                        int parentStart = parent.getStartPosition();
608                                        int firstLine = getLineNumber(parentStart, previousLineRange);
609                                        int lastLine = getLineNumber(parentStart + parent.getLength() - 1, previousLineRange);
610                                        if (this.parentLineRange[this.siblingPtr] == null) {
611                                                this.parentLineRange[this.siblingPtr] = new int[] {firstLine, lastLine};
612                                        } else {
613                                                int[] lineRange = this.parentLineRange[this.siblingPtr];
614                                                lineRange[0] = firstLine;
615                                                lineRange[1] = lastLine;
616                                        }
617                                }
618                                this.topSiblingParent = parent;
619                        }
620                        this.siblings[this.siblingPtr] = node;
621 
622                        // We're always ok to visit sub-levels
623                        return true;
624                }

626                protected void endVisitNode(ASTNode node) {
627 
628                        // Look if a child node is waiting for trailing comments computing
629                        ASTNode sibling = this.topSiblingParent == node ? (ASTNode) this.siblings[this.siblingPtr] : null;
630                        if (sibling != null) {
631                                try {
632                                        storeTrailingComments(sibling, node.getStartPosition()+node.getLength()-1, true, this.parentLineRange[this.siblingPtr]);
633                                } catch (Exception ex) {
634                                        // Give up extended ranges at this level if unexpected exception happens...
635                                }
636                        }
637                        // Remove sibling if needed
638                        if (this.topSiblingParent != null /*not a JavaScriptUnit*/
639                                        && this.topSiblingParent == node) {
640                                this.siblingPtr--;
641                                this.topSiblingParent = node.getParent();
642                        }
643                }
