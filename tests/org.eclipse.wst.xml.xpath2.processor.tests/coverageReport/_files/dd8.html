460                private void inlineReference(Expression reference) throws CoreException {
461                        ASTNode importDecl= ASTNodes.getParent(reference, ImportDeclaration.class);

748                                for (int i= 0; i < searchResultGroups.length; i++) {
749                                        if (pm.isCanceled())
750                                                throw new OperationCanceledException();
751                                        SearchResultGroup group= searchResultGroups[i];
752                                        IJavaScriptUnit cu= group.getCompilationUnit();
753 
754                                        CompilationUnitRewrite cuRewrite= getCuRewrite(cu);
755                                        Name[] references= extractReferenceNodes(group.getSearchResults(), cuRewrite.getRoot());
756                                        InlineTargetCompilationUnit targetCompilationUnit= new InlineTargetCompilationUnit(
757                                                        cuRewrite, references, this, staticImportsInInitializer);
758                                        changes.add(targetCompilationUnit.getChange());
759                                }
760 
761                        } else {
762                                Assert.isTrue(! isDeclarationSelected());
763                                InlineTargetCompilationUnit targetForOnlySelectedReference= new InlineTargetCompilationUnit(
764                                                fSelectionCuRewrite, new Name[] { fSelectedConstantName }, this, staticImportsInInitializer);
765                                changes.add(targetForOnlySelectedReference.getChange());
766                        }
767 
768                        if (result.hasFatalError())
769                                return result;
770 
771                        if (getRemoveDeclaration() && getReplaceAllReferences()) {
772                                boolean declarationRemoved= false;
773                                for (Iterator iter= changes.iterator(); iter.hasNext();) {
774                                        CompilationUnitChange change= (CompilationUnitChange) iter.next();
775                                        if (change.getCompilationUnit().equals(fDeclarationCuRewrite.getCu())) {
776                                                declarationRemoved= true;
777                                                break;
778                                        }
779                                }
780                                if (! declarationRemoved) {
781                                        InlineTargetCompilationUnit targetForDeclaration= new InlineTargetCompilationUnit(fDeclarationCuRewrite, new Name[0], this, staticImportsInInitializer);
782                                        CompilationUnitChange change= targetForDeclaration.getChange();
783                                        if (change != null)
784                                                changes.add(change);
785                                }
786                        }
787 
788                        IJavaScriptUnit[] cus= new IJavaScriptUnit[changes.size()];
789                        for (int i= 0; i < changes.size(); i++) {
790                                CompilationUnitChange change= (CompilationUnitChange) changes.get(i);
791                                cus[i]= change.getCompilationUnit();
792                        }
793                        result.merge(Checks.validateModifiesFiles(ResourceUtil.getFiles(cus), getValidationContext()));
794 
795                        pm.worked(1);
796 
797                        fChanges= (CompilationUnitChange[]) changes.toArray(new CompilationUnitChange[changes.size()]);
798 
799                        return result;
800                        
801                } finally {
802                        fSelectionCuRewrite= null;
803                        fSelectedConstantName= null;
804                        fDeclarationCuRewrite= null;
805                        fDeclaration= null;
806                        pm.done();
807                }
808        }

843        public Change createChange(IProgressMonitor pm) throws CoreException {
844                try {
845                        pm.beginTask(RefactoringCoreMessages.InlineConstantRefactoring_preview, 2);
846                        final Map arguments= new HashMap();
847                        String project= null;
848                        IJavaScriptProject javaProject= fSelectionCu.getJavaScriptProject();
849                        if (javaProject != null)

851                        int flags= RefactoringDescriptor.STRUCTURAL_CHANGE | JavaScriptRefactoringDescriptor.JAR_REFACTORING | JavaScriptRefactoringDescriptor.JAR_SOURCE_ATTACHMENT;
852                        try {
853                                if (!Flags.isPrivate(fField.getFlags()))
854                                        flags|= RefactoringDescriptor.MULTI_CHANGE;
855                        } catch (JavaScriptModelException exception) {
856                                JavaScriptPlugin.log(exception);
857                        }
858                        final String description= Messages.format(RefactoringCoreMessages.InlineConstantRefactoring_descriptor_description_short, fField.getElementName());
859                        final String header= Messages.format(RefactoringCoreMessages.InlineConstantRefactoring_descriptor_description, new String[] { JavaScriptElementLabels.getElementLabel(fField, JavaScriptElementLabels.ALL_FULLY_QUALIFIED), JavaScriptElementLabels.getElementLabel(fField.getParent(), JavaScriptElementLabels.ALL_FULLY_QUALIFIED)});
860                        final JDTRefactoringDescriptorComment comment= new JDTRefactoringDescriptorComment(project, this, header);

866                        final JDTRefactoringDescriptor descriptor= new JDTRefactoringDescriptor(IJavaScriptRefactorings.INLINE_CONSTANT, project, description, comment.asString(), arguments, flags);
867                        arguments.put(JDTRefactoringDescriptor.ATTRIBUTE_INPUT, descriptor.elementToHandle(fSelectionCu));
868                        arguments.put(JDTRefactoringDescriptor.ATTRIBUTE_SELECTION, new Integer(fSelectionStart).toString() + " " + new Integer(fSelectionLength).toString()); //$NON-NLS-1$
869                        arguments.put(ATTRIBUTE_REMOVE, Boolean.valueOf(fRemoveDeclaration).toString());
870                        arguments.put(ATTRIBUTE_REPLACE, Boolean.valueOf(fReplaceAllReferences).toString());
871                        return new DynamicValidationRefactoringChange(descriptor, RefactoringCoreMessages.InlineConstantRefactoring_inline, fChanges);
872                } finally {

911        public RefactoringStatus initialize(final RefactoringArguments arguments) {

914                        final String selection= extended.getAttribute(JDTRefactoringDescriptor.ATTRIBUTE_SELECTION);
915                        if (selection != null) {
916                                int offset= -1;
917                                int length= -1;
918                                final StringTokenizer tokenizer= new StringTokenizer(selection);
919                                if (tokenizer.hasMoreTokens())
920                                        offset= Integer.valueOf(tokenizer.nextToken()).intValue();
921                                if (tokenizer.hasMoreTokens())
922                                        length= Integer.valueOf(tokenizer.nextToken()).intValue();
923                                if (offset >= 0 && length >= 0) {
924                                        fSelectionStart= offset;

927                                        return RefactoringStatus.createFatalErrorStatus(Messages.format(RefactoringCoreMessages.InitializableRefactoring_illegal_argument, new Object[] { selection, JDTRefactoringDescriptor.ATTRIBUTE_SELECTION}));

930                        if (handle != null) {
931                                final IJavaScriptElement element= JDTRefactoringDescriptor.handleToElement(extended.getProject(), handle, false);
932                                if (element == null || !element.exists())
933                                        return createInputFatalStatus(element, IJavaScriptRefactorings.INLINE_CONSTANT);
934                                else {

936                                                fSelectionCu= (IJavaScriptUnit) element;

940                                                final IField field= (IField) element;
941                                                try {
942                                                        final ISourceRange range= field.getNameRange();

944                                                                fSelectionStart= range.getOffset();

964                        final String replace= extended.getAttribute(ATTRIBUTE_REPLACE);
