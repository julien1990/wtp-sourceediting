410                case CAN_ONLY_NULL | IN_INSTANCEOF:
411                        if (flowInfo.isPotentiallyNonNull(local)
412                                        || flowInfo.isPotentiallyUnknown(local)) {
413                                return;
414                        }
415                        if (flowInfo.isDefinitelyNull(local)) {
416                                switch(checkType & CONTEXT_MASK) {
417                                        case FlowContext.IN_COMPARISON_NULL:
418                                                scope.problemReporter().localVariableRedundantCheckOnNull(local, reference);
419                                                return;
420                                        case FlowContext.IN_COMPARISON_NON_NULL:
421                                                scope.problemReporter().localVariableNullComparedToNonNull(local, reference);
422                                                return;
423                                        case FlowContext.IN_ASSIGNMENT:
424                                                scope.problemReporter().localVariableRedundantNullAssignment(local, reference);
425                                                return;
426                                        case FlowContext.IN_INSTANCEOF:
427                                                scope.problemReporter().localVariableNullInstanceof(local, reference);
428                                                return;
429                                }
430                        }
431                        recordNullReference(local, reference, checkType);
432                        return;
433                case MAY_NULL :
434                        if (flowInfo.isDefinitelyNonNull(local)) {
435                                return;
436                        }
437                        if (flowInfo.isDefinitelyNull(local)) {
438                                scope.problemReporter().localVariableNullReference(local, reference);
439                                return;
440                        }
441                        if (flowInfo.isPotentiallyNull(local)) {
442                                scope.problemReporter().localVariablePotentialNullReference(local, reference);
443                                return;
444                        }
445                        recordNullReference(local, reference, checkType);
446                        return;
447                default:
448                        // never happens
449        }
450}
451 
452        void removeFinalAssignmentIfAny(Reference reference) {
453                for (int i = 0; i < assignCount; i++) {
454                        if (finalAssignments[i] == reference) {
455                                finalAssignments[i] = null;
456                                finalVariables[i] = null;
457                                return;
458                        }
459                }
460        }
