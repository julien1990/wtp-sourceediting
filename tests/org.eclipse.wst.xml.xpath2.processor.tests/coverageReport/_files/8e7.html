343                try {
344                        CompilerOptions compilerOptions = new CompilerOptions(options);
345                        Parser parser = new CommentRecorderParser(
346                                new ProblemReporter(
347                                                DefaultErrorHandlingPolicies.proceedWithAllProblems(),
348                                                compilerOptions,
349                                                new DefaultProblemFactory()),
350                                false);
351                        int length = compilationUnits.length;
352                        if (monitor != null) monitor.beginTask("", length); //$NON-NLS-1$
353                        for (int i = 0; i < length; i++) {
354                                org.eclipse.wst.jsdt.internal.compiler.env.ICompilationUnit sourceUnit = (org.eclipse.wst.jsdt.internal.compiler.env.ICompilationUnit) compilationUnits[i];
355                                CompilationResult compilationResult = new CompilationResult(sourceUnit, 0, 0, compilerOptions.maxProblemsPerUnit);
356                                CompilationUnitDeclaration compilationUnitDeclaration = parser.dietParse(sourceUnit, compilationResult);
357                                parser.inferTypes(compilationUnitDeclaration, compilerOptions);
358 
359                                if (compilationUnitDeclaration.ignoreMethodBodies) {
360                                        compilationUnitDeclaration.ignoreFurtherInvestigation = true;
361                                        // if initial diet parse did not work, no need to dig into method bodies.
362                                        continue;
363                                }
364 
365                                //fill the methods bodies in order for the code to be generated
366                                //real parse of the method....
367                                org.eclipse.wst.jsdt.internal.compiler.ast.TypeDeclaration[] types = compilationUnitDeclaration.types;
368                                if (types != null) {
369                                        for (int j = types.length; --j >= 0;)
370                                                types[j].parseMethod(parser, compilationUnitDeclaration);
371                                }
372 
373                                // convert AST
374                                JavaScriptUnit node = convert(compilationUnitDeclaration, parser.scanner.getSource(), apiLevel, options, false/*don't resolve binding*/, null/*no owner needed*/, null/*no binding table needed*/, flags /* flags */, monitor);
375                                node.setTypeRoot(compilationUnits[i]);
376 
377                                // accept AST
378                                astRequestor.acceptAST(compilationUnits[i], node);
379 
380                                if (monitor != null) monitor.worked(1);
381                        }
382                } finally {
383                        if (monitor != null) monitor.done();
384                }

387        public static CompilationUnitDeclaration parse(

389                        NodeSearcher nodeSearcher,
390                        Map settings,
391                        int flags) {
392                if (sourceUnit == null) {
393                        throw new IllegalStateException();
394                }
395                CompilerOptions compilerOptions = new CompilerOptions(settings);
396                boolean statementsRecovery = (flags & IJavaScriptUnit.ENABLE_STATEMENTS_RECOVERY) != 0;
397                compilerOptions.performMethodsFullRecovery = statementsRecovery;
398                compilerOptions.performStatementsRecovery = statementsRecovery;
399                Parser parser = new CommentRecorderParser(
400                        new ProblemReporter(

402                                        compilerOptions,

406                CompilationUnitDeclaration compilationUnitDeclaration = parser.dietParse(sourceUnit, compilationResult);
407 
408                parser.inferTypes(compilationUnitDeclaration, compilerOptions);
409                if (compilationUnitDeclaration.ignoreMethodBodies) {
410                        compilationUnitDeclaration.ignoreFurtherInvestigation = true;
411                        // if initial diet parse did not work, no need to dig into method bodies.
412                        return null;
413                }

421                        }

423                        compilationUnitDeclaration.traverse(nodeSearcher, compilationUnitDeclaration.scope);
424 
425                        org.eclipse.wst.jsdt.internal.compiler.ast.ASTNode node = nodeSearcher.found;
426                         if (node == null) {
427                                 return compilationUnitDeclaration;
428                         }
429 
430                         org.eclipse.wst.jsdt.internal.compiler.ast.TypeDeclaration enclosingTypeDeclaration = nodeSearcher.enclosingType;
431 
432                        if (node instanceof AbstractMethodDeclaration) {
433                                ((AbstractMethodDeclaration)node).parseStatements(parser, compilationUnitDeclaration);
434                        } else if (enclosingTypeDeclaration != null) {
435                                if (node instanceof org.eclipse.wst.jsdt.internal.compiler.ast.Initializer) {
436                                        ((org.eclipse.wst.jsdt.internal.compiler.ast.Initializer) node).parseStatements(parser, enclosingTypeDeclaration, compilationUnitDeclaration);
437                                } else {
438                                        ((org.eclipse.wst.jsdt.internal.compiler.ast.TypeDeclaration)node).parseMethod(parser, compilationUnitDeclaration);
439                                }
440                        }
441                } else {
442                        //fill the methods bodies in order for the code to be generated
443                        //real parse of the method....
444                        org.eclipse.wst.jsdt.internal.compiler.ast.TypeDeclaration[] types = compilationUnitDeclaration.types;
445                        if (types != null) {
446                                for (int i = types.length; --i >= 0;)
447                                        types[i].parseMethod(parser, compilationUnitDeclaration);
448                        }
449                }
450                return compilationUnitDeclaration;
451        }

680        private void resolve(IJavaScriptUnit[] compilationUnits, String[] bindingKeys, ASTRequestor astRequestor, int apiLevel, Map compilerOptions, WorkingCopyOwner owner, int flags) {
681 
682                // temporararily connect ourselves to the ASTResolver - must disconnect when done
683                astRequestor.compilationUnitResolver = this;
684                this.bindingTables = new DefaultBindingResolver.BindingTables();
685                CompilationUnitDeclaration unit = null;
686                int i = 0;
687                try {
688                        int length = compilationUnits.length;
689                        org.eclipse.wst.jsdt.internal.compiler.env.ICompilationUnit[] sourceUnits = new org.eclipse.wst.jsdt.internal.compiler.env.ICompilationUnit[length];
690                        System.arraycopy(compilationUnits, 0, sourceUnits, 0, length);
691                        beginToCompile(sourceUnits, bindingKeys);
692                        // process all units (some more could be injected in the loop by the lookup environment)
693                        for (; i < this.totalUnits; i++) {
694                                if (this.requestedSources.size() == 0 && this.requestedKeys.size() == 0) {
695                                        // no need to keep resolving if no more ASTs and no more binding keys are needed
696                                        // see https://bugs.eclipse.org/bugs/show_bug.cgi?id=114935
697                                        // cleanup remaining units
698                                        for (; i < this.totalUnits; i++) {
699                                                this.unitsToProcess[i].cleanUp();
700                                                this.unitsToProcess[i] = null;
701                                        }
702                                        break;
703                                }
704                                unit = this.unitsToProcess[i];
705                                try {
706                                        super.process(unit, i); // this.process(...) is optimized to not process already known units
707 
708                                        // requested AST
709                                        char[] fileName = unit.compilationResult.getFileName();
710                                        IJavaScriptUnit source = (IJavaScriptUnit) this.requestedSources.get(fileName);
711                                        if (source != null) {
712                                                // convert AST
713                                                CompilationResult compilationResult = unit.compilationResult;
714                                                org.eclipse.wst.jsdt.internal.compiler.env.ICompilationUnit sourceUnit = compilationResult.compilationUnit;
715                                                char[] contents = sourceUnit.getContents();
716                                                AST ast = AST.newAST(apiLevel);
717                                                ast.setFlag(flags | AST.RESOLVED_BINDINGS);
718                                                ast.setDefaultNodeFlag(ASTNode.ORIGINAL);
719                                                ASTConverter converter = new ASTConverter(compilerOptions, true/*need to resolve bindings*/, this.monitor);
720                                                BindingResolver resolver = new DefaultBindingResolver(unit.scope, owner, this.bindingTables, (flags & IJavaScriptUnit.ENABLE_BINDINGS_RECOVERY) != 0);
721                                                ast.setBindingResolver(resolver);
722                                                converter.setAST(ast);
723                                                JavaScriptUnit compilationUnit = converter.convert(unit, contents);
724                                                compilationUnit.setTypeRoot(source);
725                                                compilationUnit.setLineEndTable(compilationResult.getLineSeparatorPositions());
726                                                ast.setDefaultNodeFlag(0);
727                                                ast.setOriginalModificationCount(ast.modificationCount());
728 
729                                                // pass it to requestor
730                                                astRequestor.acceptAST(source, compilationUnit);
731 
732                                                worked(1);
733                                        }
734 
735                                        // requested binding
736                                        Object key = this.requestedKeys.get(fileName);
737                                        if (key instanceof BindingKeyResolver) {
738                                                reportBinding(key, astRequestor, owner, unit);
739                                                worked(1);
740                                        } else if (key instanceof ArrayList) {
741                                                Iterator iterator = ((ArrayList) key).iterator();
742                                                while (iterator.hasNext()) {
743                                                        reportBinding(iterator.next(), astRequestor, owner, unit);
744                                                        worked(1);
745                                                }
746                                        }
747 
748                                        // remove at the end so that we don't resolve twice if a source and a key for the same file name have been requested
749                                        this.requestedSources.removeKey(fileName);
750                                        this.requestedKeys.removeKey(fileName);
751                                } finally {
752                                        // cleanup javaScript unit result
753                                        unit.cleanUp();
754                                }
755                                this.unitsToProcess[i] = null; // release reference to processed unit declaration
756                                this.requestor.acceptResult(unit.compilationResult.tagAsAccepted());
757                        }
758 
759                        // remaining binding keys
760                        DefaultBindingResolver resolver = new DefaultBindingResolver(this.lookupEnvironment, owner, this.bindingTables, (flags & IJavaScriptUnit.ENABLE_BINDINGS_RECOVERY) != 0);
761                        Object[] keys = this.requestedKeys.valueTable;
762                        for (int j = 0, keysLength = keys.length; j < keysLength; j++) {
763                                BindingKeyResolver keyResolver = (BindingKeyResolver) keys[j];
764                                if (keyResolver == null) continue;
765                                Binding compilerBinding = keyResolver.getCompilerBinding();
766                                IBinding binding = compilerBinding == null ? null : resolver.getBinding(compilerBinding);
767                                // pass it to requestor
768                                astRequestor.acceptBinding(((BindingKeyResolver) this.requestedKeys.valueTable[j]).getKey(), binding);
769                                worked(1);
770                        }
771                } catch (OperationCanceledException e) {
772                        throw e;
773                } catch (AbortCompilation e) {
774                        this.handleInternalException(e, unit);
775                } catch (Error e) {
776                        this.handleInternalException(e, unit, null);
777                        throw e; // rethrow
778                } catch (RuntimeException e) {
779                        this.handleInternalException(e, unit, null);
780                        throw e; // rethrow
781                } finally {
782                        // disconnect ourselves from ast requestor
783                        astRequestor.compilationUnitResolver = null;
784                }
785        }

787        private void reportBinding(Object key, ASTRequestor astRequestor, WorkingCopyOwner owner, CompilationUnitDeclaration unit) {
788                BindingKeyResolver keyResolver = (BindingKeyResolver) key;
789                Binding compilerBinding = keyResolver.getCompilerBinding();
790                if (compilerBinding != null) {
791                        DefaultBindingResolver resolver = new DefaultBindingResolver(unit.scope, owner, this.bindingTables, false);
792                        IBinding binding = resolver.getBinding(compilerBinding);
793                        if (binding != null)
794                                astRequestor.acceptBinding(keyResolver.getKey(), binding);
795                }

798        private CompilationUnitDeclaration resolve(
799                        CompilationUnitDeclaration unit,
800                        org.eclipse.wst.jsdt.internal.compiler.env.ICompilationUnit sourceUnit,
801                        NodeSearcher nodeSearcher,
802                        boolean verifyMethods,
803                        boolean analyzeCode,
804                        boolean generateCode) {
805 
806                try {
807 
808                        if (unit == null) {
809                                // build and record parsed units
810                                this.parseThreshold = 0; // will request a full parse
811                                beginToCompile(new org.eclipse.wst.jsdt.internal.compiler.env.ICompilationUnit[] { sourceUnit });
812                                // process all units (some more could be injected in the loop by the lookup environment)
813                                unit = this.unitsToProcess[0];
814                        } else {
815                                // initial type binding creation
816                                this.lookupEnvironment.buildTypeBindings(unit, null /*no access restriction*/);
817 
818                                // binding resolution
819                                this.lookupEnvironment.completeTypeBindings();
820                        }
821 
822                        if (nodeSearcher == null) {
823                                this.parser.getMethodBodies(unit); // no-op if method bodies have already been parsed
824                        } else {
825                                int searchPosition = nodeSearcher.position;
826                                char[] source = sourceUnit.getContents();
827                                int length = source.length;
828                                if (searchPosition >= 0 && searchPosition <= length) {
829                                        unit.traverse(nodeSearcher, unit.scope);
830 
831                                        org.eclipse.wst.jsdt.internal.compiler.ast.ASTNode node = nodeSearcher.found;
832 
833                                        this.parser.scanner.setSource(source, unit.compilationResult);
834 
835                                         if (node != null) {
836                                                org.eclipse.wst.jsdt.internal.compiler.ast.TypeDeclaration enclosingTypeDeclaration = nodeSearcher.enclosingType;
837                                                  if (node instanceof AbstractMethodDeclaration) {
838                                                        ((AbstractMethodDeclaration)node).parseStatements(this.parser, unit);
839                                                 } else if (enclosingTypeDeclaration != null) {
840                                                        if (node instanceof org.eclipse.wst.jsdt.internal.compiler.ast.Initializer) {
841                                                                 ((org.eclipse.wst.jsdt.internal.compiler.ast.Initializer) node).parseStatements(this.parser, enclosingTypeDeclaration, unit);
842                                                         } else if (node instanceof org.eclipse.wst.jsdt.internal.compiler.ast.TypeDeclaration) {
843                                                                ((org.eclipse.wst.jsdt.internal.compiler.ast.TypeDeclaration)node).parseMethod(this.parser, unit);
844                                                        }
845                                                 }
846                                         }
847                                }
848                        }
849 
850                        if (unit.scope != null) {
851                                // fault in fields & methods
852                                unit.scope.faultInTypes();
853                                if (unit.scope != null && verifyMethods) {
854                                        // http://dev.eclipse.org/bugs/show_bug.cgi?id=23117
855                                         // verify inherited methods
856                                        unit.scope.verifyMethods(this.lookupEnvironment.methodVerifier());
857                                }
858                                // type checking
859                                unit.resolve();
860 
861                                // flow analysis
862                                if (analyzeCode) unit.analyseCode();
863 
864                        }
865                        if (this.unitsToProcess != null) this.unitsToProcess[0] = null; // release reference to processed unit declaration
866                        this.requestor.acceptResult(unit.compilationResult.tagAsAccepted());
867                        return unit;
868                } catch (AbortCompilation e) {
869                        this.handleInternalException(e, unit);
870                        return unit == null ? this.unitsToProcess[0] : unit;
871                } catch (Error e) {
872                        this.handleInternalException(e, unit, null);
873                        throw e; // rethrow
874                } catch (RuntimeException e) {
875                        this.handleInternalException(e, unit, null);
876                        throw e; // rethrow
877                } finally {
878                        // No reset is performed there anymore since,
879                        // within the CodeAssist (or related tools),
880                        // the validator may be called *after* a call
881                        // to this resolve(...) method. And such a call
882                        // needs to have a validator with a non-empty
883                        // environment.
884                        // this.reset();
885                }
886        }
