633        /*
634         * Registers the given action to be run when the outer most java model operation has finished.
635         * The insertion mode controls whether:
636         * - the action should discard all existing actions with the same id, and be queued at the end (REMOVEALL_APPEND),
637         * - the action should be ignored if there is already an action with the same id (KEEP_EXISTING),
638         * - the action should be queued at the end without looking at existing actions (APPEND)
639         */
640        protected void postAction(IPostAction action, int insertionMode) {
641                if (POST_ACTION_VERBOSE) {
642                        System.out.print("(" + Thread.currentThread() + ") [JavaModelOperation.postAction(IPostAction, int)] Posting action " + action.getID()); //$NON-NLS-1$ //$NON-NLS-2$
643                        switch(insertionMode) {
644                                case REMOVEALL_APPEND:
645                                        System.out.println(" (REMOVEALL_APPEND)"); //$NON-NLS-1$
646                                        break;
647                                case KEEP_EXISTING:
648                                        System.out.println(" (KEEP_EXISTING)"); //$NON-NLS-1$
649                                        break;
650                                case APPEND:
651                                        System.out.println(" (APPEND)"); //$NON-NLS-1$
652                                        break;
653                        }
654                }
655 
656                JavaModelOperation topLevelOp = (JavaModelOperation)getCurrentOperationStack().get(0);
657                IPostAction[] postActions = topLevelOp.actions;
658                if (postActions == null) {
659                        topLevelOp.actions = postActions = new IPostAction[1];
660                        postActions[0] = action;
661                        topLevelOp.actionsEnd = 0;
662                } else {
663                        String id = action.getID();
664                        switch (insertionMode) {
665                                case REMOVEALL_APPEND :
666                                        int index = this.actionsStart-1;
667                                        while ((index = topLevelOp.firstActionWithID(id, index+1)) >= 0) {
668                                                // remove action[index]
669                                                System.arraycopy(postActions, index+1, postActions, index, topLevelOp.actionsEnd - index);
670                                                postActions[topLevelOp.actionsEnd--] = null;
671                                        }
672                                        topLevelOp.addAction(action);
673                                        break;
674                                case KEEP_EXISTING:
675                                        if (topLevelOp.firstActionWithID(id, 0) < 0) {
676                                                topLevelOp.addAction(action);
677                                        }
678                                        break;
679                                case APPEND:
680                                        topLevelOp.addAction(action);
681                                        break;
682                        }
683                }
684        }

713                if (postActions == null) return;

722        /**
723         * Runs this operation and registers any deltas created.
724         *
725         * @see IWorkspaceRunnable
726         * @exception CoreException if the operation fails
727         */
728        public void run(IProgressMonitor monitor) throws CoreException {
729                JavaModelManager manager = JavaModelManager.getJavaModelManager();
730                DeltaProcessor deltaProcessor = manager.getDeltaProcessor();
731                int previousDeltaCount = deltaProcessor.javaModelDeltas.size();
732                try {
733                        progressMonitor = monitor;
734                        pushOperation(this);
735                        try {
736                                if (canModifyRoots()) {
737                                        // computes the root infos before executing the operation
738                                        // noop if aready initialized
739                                        JavaModelManager.getJavaModelManager().deltaState.initializeRoots();
740                                }
741 
742                                executeOperation();
743                        } finally {
744                                if (this.isTopLevelOperation()) {
745                                        this.runPostActions();
746                                }
747                        }
748                } finally {
749                        try {
750                                // reacquire delta processor as it can have been reset during executeOperation()
751                                deltaProcessor = manager.getDeltaProcessor();
752 
753                                // update JavaModel using deltas that were recorded during this operation
754                                for (int i = previousDeltaCount, size = deltaProcessor.javaModelDeltas.size(); i < size; i++) {
755                                        deltaProcessor.updateJavaModel((IJavaScriptElementDelta)deltaProcessor.javaModelDeltas.get(i));
756                                }
757 
758                                // close the parents of the created elements and reset their project's cache (in case we are in an
759                                // IWorkspaceRunnable and the clients wants to use the created element's parent)
760                                // see https://bugs.eclipse.org/bugs/show_bug.cgi?id=83646
761                                for (int i = 0, length = this.resultElements.length; i < length; i++) {
762                                        IJavaScriptElement element = this.resultElements[i];
763                                        Openable openable = (Openable) element.getOpenable();
764                                        if (!(openable instanceof CompilationUnit) || !((CompilationUnit) openable).isWorkingCopy()) { // a working copy must remain a child of its parent even after a move
765                                                ((JavaElement) openable.getParent()).close();
766                                        }
767                                        switch (element.getElementType()) {
768                                                case IJavaScriptElement.PACKAGE_FRAGMENT_ROOT:
769                                                case IJavaScriptElement.PACKAGE_FRAGMENT:
770                                                        deltaProcessor.projectCachesToReset.add(element.getJavaScriptProject());
771                                                        break;
772                                        }
773                                }
774                                deltaProcessor.resetProjectCaches();
775 
776                                // fire only iff:
777                                // - the operation is a top level operation
778                                // - the operation did produce some delta(s)
779                                // - but the operation has not modified any resource
780                                if (this.isTopLevelOperation()) {
781                                        if ((deltaProcessor.javaModelDeltas.size() > previousDeltaCount || !deltaProcessor.reconcileDeltas.isEmpty())
782                                                        && !this.hasModifiedResource()) {
783                                                deltaProcessor.fire(null, DeltaProcessor.DEFAULT_CHANGE_EVENT);
784                                        } // else deltas are fired while processing the resource delta
785                                }
786                        } finally {
787                                popOperation();
788                        }
789                }
790        }
