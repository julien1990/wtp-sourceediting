321        //---- Helper to process switch statement ----------------------------------------
322        
323        protected SwitchData createSwitchData(SwitchStatement node) {
324                SwitchData result= new SwitchData();
325                List statements= node.statements();
326                if (statements.isEmpty())
327                        return result;
328                        
329                int start= -1, end= -1;
330                GenericSequentialFlowInfo info= null;
331                
332                for (Iterator iter= statements.iterator(); iter.hasNext(); ) {
333                        Statement statement= (Statement)iter.next();
334                        if (statement instanceof SwitchCase) {
335                                SwitchCase switchCase= (SwitchCase)statement;
336                                if (switchCase.isDefault()) {
337                                        result.setHasDefaultCase();
338                                }
339                                if (info == null) {
340                                        info= createSequential();
341                                        start= statement.getStartPosition();
342                                } else {
343                                        if (info.isReturn() || info.isPartialReturn() || info.branches()) {
344                                                result.add(new Region(start, end - start + 1), info);
345                                                info= createSequential();
346                                                start= statement.getStartPosition();
347                                        }
348                                }
349                        } else {
350                                info.merge(getFlowInfo(statement), fFlowContext);
351                        }
352                        end= statement.getStartPosition() + statement.getLength() - 1;
353                }
354                result.add(new Region(start, end - start + 1), info);
355                return result;
356        }

911        private void endVisitIncDecOperation(Expression node, Expression operand) {
912                if (skipNode(node))
913                        return;
914                FlowInfo info= getFlowInfo(operand);
915                if (info instanceof LocalFlowInfo) {
916                        // Normally we should do this in the parent node since the write access take place later.
917                        // But I couldn't come up with a case where this influences the flow analysis. So I kept
918                        // it here to simplify the code.
919                        GenericSequentialFlowInfo result= createSequential(node);
920                        result.merge(info, fFlowContext);
921                        result.merge(
922                                new LocalFlowInfo((LocalFlowInfo)info, FlowInfo.WRITE, fFlowContext), 
923                                fFlowContext);
924                } else {
