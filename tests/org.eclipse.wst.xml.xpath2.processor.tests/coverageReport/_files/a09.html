379                constructor = evaluationTestClass.getConstructor(paramTypes);
380        }
381        catch (Exception e) {
382                // cannot get constructor, skip suite
383                return tests;
384        }
385 
386        // Get all tests from "test%" methods
387        Method[] methods = evaluationTestClass.getDeclaredMethods();
388        Class evaluationTestSuperclass = evaluationTestClass.getSuperclass();
389        for (int i=0; i<inheritedDepth && !Flags.isAbstract(evaluationTestSuperclass.getModifiers()); i++) {
390                Method[] superMethods = evaluationTestSuperclass.getDeclaredMethods();
391                Method[] mergedMethods = new Method[methods.length+superMethods.length];
392                System.arraycopy(superMethods, 0, mergedMethods, 0, superMethods.length);
393                System.arraycopy(methods, 0, mergedMethods, superMethods.length, methods.length);
394                methods = mergedMethods;
395                evaluationTestSuperclass = evaluationTestSuperclass.getSuperclass();
396        }
397 
398        // Build test names list
399        final int methodPrefixLength = METHOD_PREFIX.length();
400        nextMethod: for (int m = 0, max = methods.length; m < max; m++) {
401                int modifiers = methods[m].getModifiers();
402                if (Flags.isPublic(modifiers) && !Flags.isStatic(modifiers)) {
403                        String methName = methods[m].getName();
404                        if (methName.startsWith(METHOD_PREFIX)) {
405 
406                                // look if this is a run only method
407                                boolean isOnly = RUN_ONLY_ID != null && methName.substring(methodPrefixLength).startsWith(RUN_ONLY_ID);
408                                if (isOnly) {
409                                        if (!onlyNames.contains(methName)) {
410                                                onlyNames.add(methName);
411                                        }
412                                        continue;
413                                }
414 
415                                // no prefix, no subsets => add method
416                                if (TESTS_PREFIX == null && TESTS_NAMES == null && TESTS_NUMBERS == null && TESTS_RANGE == null) {
417                                        if (!testNames.contains(methName)) {
418                                                testNames.add(methName);
419                                        }
420                                        continue nextMethod;
421                                }
422 
423                                // no prefix or method matches prefix
424                                if (TESTS_PREFIX == null || methName.startsWith(TESTS_PREFIX)) {
425                                        int numStart = TESTS_PREFIX==null ? methodPrefixLength : TESTS_PREFIX.length();
426                                        // tests names subset
427                                        if (TESTS_NAMES != null) {
428                                                for (int i = 0, imax= TESTS_NAMES.length; i<imax; i++) {
429                                                        if (methName.indexOf(TESTS_NAMES[i]) >= 0) {
430                                                                if (!testNames.contains(methName)) {
431                                                                        testNames.add(methName);
432                                                                }
433                                                                continue nextMethod;
434                                                        }
435                                                }
436                                        }
437                                        // look for test number
438                                        int length = methName.length();
439                                        if (numStart < length) {
440                                                // get test number
441                                                while (numStart<length && !Character.isDigit(methName.charAt(numStart))) numStart++; // skip to first digit
442                                                while (numStart<length && methName.charAt(numStart) == '0') numStart++; // skip to first non-nul digit
443                                                int n = numStart;
444                                                while (n<length && Character.isDigit(methName.charAt(n))) n++; // skip to next non-digit
445                                                if (n>numStart && n <= length) {
446                                                        try {
447                                                                int num = Integer.parseInt(methName.substring(numStart, n));
448                                                                // tests numbers subset
449                                                                if (TESTS_NUMBERS != null && !testNames.contains(methName)) {
450                                                                        for (int i = 0; i < TESTS_NUMBERS.length; i++) {
451                                                                                if (TESTS_NUMBERS[i] == num) {
452                                                                                        testNames.add(methName);
453                                                                                        continue nextMethod;
454                                                                                }
455                                                                        }
456                                                                }
457                                                                // tests range subset
458                                                                if (TESTS_RANGE != null && TESTS_RANGE.length == 2 && !testNames.contains(methName)) {
459                                                                        if ((TESTS_RANGE[0]==-1 || num>=TESTS_RANGE[0]) && (TESTS_RANGE[1]==-1 || num<=TESTS_RANGE[1])) {
460                                                                                testNames.add(methName);
461                                                                                continue nextMethod;
462                                                                        }
463                                                                }
464                                                        } catch (NumberFormatException e) {
465                                                                System.out.println("Method "+methods[m]+" has an invalid number format: "+e.getMessage());
466                                                        }
467                                                }
468                                        }
469 
470                                        // no subset, add all tests
471                                        if (TESTS_NAMES==null && TESTS_NUMBERS==null && TESTS_RANGE==null) {
472                                                if (!testNames.contains(methName)) {
473                                                        testNames.add(methName);
474                                                }
475                                        }
476                                }
477                        }
478                }
479        }
480 
481        // Order tests
482        List names = onlyNames.size() > 0 ? onlyNames : testNames;
483        if (ordering == ALPHA_REVERSE_SORT) {
484                Collections.sort(names, Collections.reverseOrder());
485        } else if (ordering == ALPHABETICAL_SORT) {
486                Collections.sort(names);
487        } else if (ordering != NO_ORDER) {
488                Collections.shuffle(names, new Random(ordering));
489        }
490 
491        // Add corresponding tests
492        Iterator iterator = names.iterator();
493        while (iterator.hasNext()) {
494                String testName = (String) iterator.next();
495                try {
496                        tests.add(constructor.newInstance(new Object[] { testName } ));
497                }
498                catch (Exception e) {
499                        System.err.println("Method "+testName+" removed from suite due to exception: "+e.getMessage());
500                }
501        }
502        return tests;
503}

577protected static String showLineSeparators(String string) {
578        if (string == null) return null;
579        StringBuffer buffer = new StringBuffer();
580        int length = string.length();
581        for (int i = 0; i < length; i++) {
582                char car = string.charAt(i);
583                switch (car) {
584                        case '\n': 
585                                buffer.append("\\n\n"); //$NON-NLS-1$
586                                break;
587                        case '\r':
588                                if (i < length-1 && string.charAt(i+1) == '\n') {
589                                        buffer.append("\\r\\n\n"); //$NON-NLS-1$
590                                        i++;
591                                } else {
592                                        buffer.append("\\r\n"); //$NON-NLS-1$
593                                }
594                                break;
595                        default:
596                                buffer.append(car);
597                                break;
598                }
599        }
600        return buffer.toString();
601}
