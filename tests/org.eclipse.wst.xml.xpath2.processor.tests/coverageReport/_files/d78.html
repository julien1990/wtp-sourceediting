535            Hashtable nonStaticAccessOps= new Hashtable();

542                                        ToStaticAccessOperation[] nonStaticAccessInformation= createToStaticAccessOperations(compilationUnit, problem);

567        private static ToStaticAccessOperation[] createToStaticAccessOperations(JavaScriptUnit astRoot, IProblemLocation problem) {
568                ASTNode selectedNode= problem.getCoveringNode(astRoot);
569                if (selectedNode == null) {
570                        return null;
571                }
572 
573                Expression qualifier= null;
574                IBinding accessBinding= null;
575 
576        if (selectedNode instanceof QualifiedName) {
577                QualifiedName name= (QualifiedName) selectedNode;
578                qualifier= name.getQualifier();
579                accessBinding= name.resolveBinding();
580        } else if (selectedNode instanceof SimpleName) {
581                ASTNode parent= selectedNode.getParent();
582                if (parent instanceof FieldAccess) {
583                        FieldAccess fieldAccess= (FieldAccess) parent;
584                        qualifier= fieldAccess.getExpression();
585                        accessBinding= fieldAccess.getName().resolveBinding();
586                } else if (parent instanceof QualifiedName) {
587                        QualifiedName qualifiedName= (QualifiedName) parent;
588                        qualifier= qualifiedName.getQualifier();
589                        accessBinding= qualifiedName.getName().resolveBinding();
590                }
591        } else if (selectedNode instanceof FunctionInvocation) {
592                FunctionInvocation methodInvocation= (FunctionInvocation) selectedNode;
593                qualifier= methodInvocation.getExpression();
594                SimpleName name = methodInvocation.getName();
595                if (name!=null)
596                        accessBinding= name.resolveBinding();
597        } else if (selectedNode instanceof FieldAccess) {
598                        FieldAccess fieldAccess= (FieldAccess) selectedNode;
599                        qualifier= fieldAccess.getExpression();
600                        accessBinding= fieldAccess.getName().resolveBinding();
601                }
602        
603                if (accessBinding != null && qualifier != null) {
604                        ToStaticAccessOperation declaring= null;
605                        ITypeBinding declaringTypeBinding= getDeclaringTypeBinding(accessBinding);
606                        if (declaringTypeBinding != null) {
607                                declaringTypeBinding= declaringTypeBinding.getTypeDeclaration(); // use generic to avoid any type arguments
608                                
609                                declaring= new ToStaticAccessOperation(declaringTypeBinding, qualifier);
610                        }
611                        ToStaticAccessOperation instance= null;
612                        ITypeBinding instanceTypeBinding= Bindings.normalizeTypeBinding(qualifier.resolveTypeBinding());
613                        if (instanceTypeBinding != null) {
614                                instanceTypeBinding= instanceTypeBinding.getTypeDeclaration();  // use generic to avoid any type arguments
615                                if (instanceTypeBinding.getTypeDeclaration() != declaringTypeBinding) {
616                                        instance= new ToStaticAccessOperation(instanceTypeBinding, qualifier);
617                                }
618                        }
619                        if (declaring != null && instance != null) {
620                                return new ToStaticAccessOperation[] {declaring, instance};
621                        } else {
622                                return new ToStaticAccessOperation[] {declaring};
623                        }
624                }
625                return null;
626        }

669        private static String getNonStaticQualifier(ITypeBinding declaringClass, ImportRewrite imports, SimpleName name) {
670                ITypeBinding parentType= Bindings.getBindingOfParentType(name);
671                ITypeBinding currType= parentType;
672                while (currType != null && !Bindings.isSuperType(declaringClass, currType)) {
673                        currType= currType.getDeclaringClass();
674                }
675                if (currType == null) {
676                        declaringClass= declaringClass.getTypeDeclaration();
677                        currType= parentType;
678                        while (currType != null && !Bindings.isSuperType(declaringClass, currType)) {
679                                currType= currType.getDeclaringClass();
680                        }
681                }

683                        if (currType == null)
684                                return null;
685                        
686                        if (currType.isAnonymous())
687                                //If we access a field of a super class of an anonymous class

689                                //see bug 115277
