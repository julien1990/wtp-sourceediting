443public void traverse(ASTVisitor visitor, BlockScope blockScope) {
444        if (visitor.visit(this, blockScope)) {
445                if(this.scope==null) this.scope=blockScope;
446                this.tryBlock.traverse(visitor, this.scope);
447                if (this.catchArguments != null) {
448                        for (int i = 0, max = this.catchBlocks.length; i < max; i++) {
449                                this.catchArguments[i].traverse(visitor, this.scope);
450                                this.catchBlocks[i].traverse(visitor, this.scope);
451                        }
452                }
453                if (this.finallyBlock != null)
454                        this.finallyBlock.traverse(visitor, this.scope);
455        }
456        visitor.endVisit(this, blockScope);
