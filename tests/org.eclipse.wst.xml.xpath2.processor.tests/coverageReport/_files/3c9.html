268        private void verifyAttributeDeclaration(CMElementDeclaration elemDecl, CMNode attr) {
269                assertTrue(attr.getNodeType() == CMNode.ATTRIBUTE_DECLARATION);
270                assertNotNull("no name on an attribute declaration", attr.getNodeName());
271                CMAttributeDeclaration attrDecl = (CMAttributeDeclaration) attr;
272                assertNotNull("no attribute 'type' on an attribute declaration " + elemDecl.getNodeName() + "/" + attr.getNodeName(), attrDecl.getAttrType());
273        }

275        private void verifyElementDeclarationHasName(CMNode item) {
276                assertTrue(item.getNodeType() == CMNode.ELEMENT_DECLARATION);
277                assertNotNull("no name on an element declaration", item.getNodeName());
278                CMNamedNodeMap attrs = ((CMElementDeclaration) item).getAttributes();
279                for (int i = 0; i < attrs.getLength(); i++) {
280                        CMNode attr = attrs.item(i);
281                        verifyAttributeDeclaration(((CMElementDeclaration) item), attr);
282                }
