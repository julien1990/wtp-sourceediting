380                // in 2.0, the Javadoc locations were stored as one big string in the persistent properties
381                // note that it is wrong to use a stream reader with XML declaring to be UTF-8
382                try {
383                        final QualifiedName QUALIFIED_NAME= new QualifiedName(JavaScriptUI.ID_PLUGIN, "jdoclocation"); //$NON-NLS-1$
384                        
385                        IWorkspaceRoot root= ResourcesPlugin.getWorkspace().getRoot();
386                        String xmlString= root.getPersistentProperty(QUALIFIED_NAME); 
387                        if (xmlString != null) { // only set when workspace is old
388                                Reader reader= new StringReader(xmlString);
389                                try {
390                                        loadFromStream(new InputSource(reader), resultingOldLocations);
391                                        root.setPersistentProperty(QUALIFIED_NAME, null); // clear property
392                                        return resultingOldLocations;
393                                } finally {
394 
395                                        try {
396                                                reader.close();
397                                        } catch (IOException e) {
398                                                // error closing reader: ignore
399                                        }
400                                }
401                        }
402                } catch (CoreException e) {
403                        JavaScriptPlugin.log(e); // log but ignore
404                }
405                return resultingOldLocations;
406        }

408        private static void loadFromStream(InputSource inputSource, Map/*<Path, String>*/ oldLocations) throws CoreException {
409                Element cpElement;
410                try {
411                        DocumentBuilder parser = DocumentBuilderFactory.newInstance().newDocumentBuilder();
412                        cpElement = parser.parse(inputSource).getDocumentElement();
413                } catch (SAXException e) {
414                        throw createException(e, CorextMessages.JavaDocLocations_error_readXML); 
415                } catch (ParserConfigurationException e) {
416                        throw createException(e, CorextMessages.JavaDocLocations_error_readXML); 
417                } catch (IOException e) {
418                        throw createException(e, CorextMessages.JavaDocLocations_error_readXML); 
419                }
420                
421                if (cpElement == null) return;
422                if (!cpElement.getNodeName().equalsIgnoreCase(NODE_ROOT)) {
423                        return;
424                }
425                NodeList list= cpElement.getChildNodes();
426                int length= list.getLength();
427                for (int i= 0; i < length; ++i) {
428                        Node node= list.item(i);
429                        short type= node.getNodeType();
430                        if (type == Node.ELEMENT_NODE) {
431                                Element element= (Element) node;
432                                if (element.getNodeName().equalsIgnoreCase(NODE_ENTRY)) {
433                                        String varPath = element.getAttribute(NODE_PATH);
434                                        String varURL = element.getAttribute(NODE_URL);
435                                        
436                                        oldLocations.put(Path.fromPortableString(varPath), varURL);
437                                }
438                        }
439                }

442        public static URL getJavadocLocation(IJavaScriptElement element, boolean includeMemberReference) throws JavaScriptModelException {
443                URL baseLocation= getJavadocBaseLocation(element);
444                if (baseLocation == null) {
445                        return null;
446                }
447 
448                String urlString= baseLocation.toExternalForm();
449 
450                StringBuffer pathBuffer= new StringBuffer(urlString);
451                if (!urlString.endsWith("/")) { //$NON-NLS-1$
452                        pathBuffer.append('/');
453                }
454 
455                switch (element.getElementType()) {
456                        case IJavaScriptElement.PACKAGE_FRAGMENT:
457                                appendPackageSummaryPath((IPackageFragment) element, pathBuffer);
458                                break;
459                        case IJavaScriptElement.JAVASCRIPT_PROJECT:
460                        case IJavaScriptElement.PACKAGE_FRAGMENT_ROOT :
461                                appendIndexPath(pathBuffer);
462                                break;
463                        case IJavaScriptElement.IMPORT_CONTAINER :
464                                element= element.getParent();
465                                // fall through
466                        case IJavaScriptElement.JAVASCRIPT_UNIT :
467                                IType mainType= ((IJavaScriptUnit) element).findPrimaryType();
468                                if (mainType == null) {
469                                        return null;
470                                }
471                                appendTypePath(mainType, pathBuffer);
472                                break;
473                        case IJavaScriptElement.CLASS_FILE :
474                                appendTypePath(((IClassFile) element).getType(), pathBuffer);
475                                break;
476                        case IJavaScriptElement.TYPE :
477                                appendTypePath((IType) element, pathBuffer);
478                                break;
479                        case IJavaScriptElement.FIELD :
480                                IField field= (IField) element;
481                                appendTypePath(field.getDeclaringType(), pathBuffer);
482                                if (includeMemberReference) {
483                                        appendFieldReference(field, pathBuffer);
484                                }
485                                break;
486                        case IJavaScriptElement.METHOD :
487                                IFunction method= (IFunction) element;
488                                appendTypePath(method.getDeclaringType(), pathBuffer);
489                                if (includeMemberReference) {
490                                        appendMethodReference(method, pathBuffer);
491                                }
492                                break;
493                        case IJavaScriptElement.INITIALIZER :
494                                appendTypePath(((IMember) element).getDeclaringType(), pathBuffer);
495                                break;
496                        case IJavaScriptElement.IMPORT_DECLARATION :
497                                IImportDeclaration decl= (IImportDeclaration) element;
498 
499                                if (decl.isOnDemand()) {
500                                        IJavaScriptElement cont= JavaModelUtil.findTypeContainer(element.getJavaScriptProject(), Signature.getQualifier(decl.getElementName()));
501                                        if (cont instanceof IType) {
502                                                appendTypePath((IType) cont, pathBuffer);
503                                        } else if (cont instanceof IPackageFragment) {
504                                                appendPackageSummaryPath((IPackageFragment) cont, pathBuffer);
505                                        }
506                                } else {
507                                        IType imp= element.getJavaScriptProject().findType(decl.getElementName());
508                                        appendTypePath(imp, pathBuffer);
509                                }
510                                break;
511                        case IJavaScriptElement.PACKAGE_DECLARATION :
512                                IJavaScriptElement pack= element.getAncestor(IJavaScriptElement.PACKAGE_FRAGMENT);
513                                if (pack != null) {
514                                        appendPackageSummaryPath((IPackageFragment) pack, pathBuffer);
515                                } else {
516                                        return null;
517                                }
518                                break;
519                        default :
520                                return null;
521                }
522 
523                try {
524                        return new URL(pathBuffer.toString());
525                } catch (MalformedURLException e) {
526                        JavaScriptPlugin.log(e);
527                }
528                return null;
529        }

556        private static void appendMethodReference(IFunction meth, StringBuffer buf) throws JavaScriptModelException {
557                buf.append('#');
558                buf.append(meth.getElementName());        
559                
560                buf.append('(');
561                String[] params= meth.getParameterTypes();
562                IType declaringType= meth.getDeclaringType();
563                boolean isVararg= Flags.isVarargs(meth.getFlags());
564                int lastParam= params.length - 1;
565                for (int i= 0; i <= lastParam; i++) {
566                        if (i != 0) {
567                                buf.append(", "); //$NON-NLS-1$
568                        }
569                        String curr= params[i];
570                        String fullName= JavaModelUtil.getResolvedTypeName(curr, declaringType);
571                        if (fullName != null) {
572                                buf.append(fullName);
573                                int dim= Signature.getArrayCount(curr);
574                                if (i == lastParam && isVararg) {
575                                        dim--;
576                                }
577                                while (dim > 0) {
578                                        buf.append("[]"); //$NON-NLS-1$
579                                        dim--;
580                                }
581                                if (i == lastParam && isVararg) {
582                                        buf.append("..."); //$NON-NLS-1$
583                                }
584                        }
585                }
586                buf.append(')');
587        }
