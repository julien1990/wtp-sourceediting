407         * is <strong>not</strong> filtered by partition type.
408         * </em></p>
409         */
410        public void sessionEnded() {
411                if (!isEnabled())
412                        return;
413 
414                IStatus status;
415                try {
416                        IJavaCompletionProposalComputer computer= getComputer();
417                        if (computer == null) // not active yet
418                                return;

422                        stopMeter(stats, SESSION_ENDED);

425                } catch (InvalidRegistryObjectException x) {

436        private PerformanceStats startMeter(Object context, IJavaCompletionProposalComputer computer) {
437                final PerformanceStats stats;
438                if (MEASURE_PERFORMANCE) {
439                        stats= PerformanceStats.getStats(PERFORMANCE_EVENT, computer);
440                        stats.startRun(context.toString());
441                } else {
442                        stats= null;
443                }
444                
445                if (fIsReportingDelay) {
446                        fStart= System.currentTimeMillis();
447                }
448                
449                return stats;
450        }

452        private void stopMeter(final PerformanceStats stats, String operation) {
453                if (MEASURE_PERFORMANCE) {
454                        stats.endRun();
455                        if (stats.isFailure()) {
456                                IStatus status= createPerformanceStatus(operation);
457                                fRegistry.informUser(this, status);
458                                return;
459                        }
460                }
461                
462                if (fIsReportingDelay) {
463                        long current= System.currentTimeMillis();
464                        if (current - fStart > MAX_DELAY) {
465                                IStatus status= createPerformanceStatus(operation);
466                                fRegistry.informUser(this, status);
467                        }
468                }
