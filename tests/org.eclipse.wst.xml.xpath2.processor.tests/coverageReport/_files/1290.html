747        private StyleRange[] setViewerBackground(Annotation annotation) {
748                StyledText text= fInput.fViewer.getTextWidget();
749                if (text == null || text.isDisposed())
750                        return null;
751 
752                Display disp= text.getDisplay();
753 
754                Position pos= fInput.model.getPosition(annotation);
755                if (pos == null)
756                        return null;
757 
758                IRegion region= ((TextViewer)fInput.fViewer).modelRange2WidgetRange(new Region(pos.offset, pos.length));
759                if (region == null)
760                        return null;
761 
762                StyleRange[] ranges= text.getStyleRanges(region.getOffset(), region.getLength());
763 
764                List undoRanges= new ArrayList(ranges.length);
765                for (int i= 0; i < ranges.length; i++) {
766                        undoRanges.add(ranges[i].clone());
767                }
768 
769                int offset= region.getOffset();
770                StyleRange current= undoRanges.size() > 0 ? (StyleRange) undoRanges.get(0) : null;
771                int curStart= current != null ? current.start : region.getOffset() + region.getLength();
772                int curEnd= current != null ? current.start + current.length : -1;
773                int index= 0;
774 
775                // fill no-style regions
776                while (curEnd < region.getOffset() + region.getLength()) {
777                        // add empty range
778                        if (curStart > offset) {
779                                StyleRange undoRange= new StyleRange(offset, curStart - offset, null, null);
780                                undoRanges.add(index, undoRange);
781                                index++;
782                        }
783 
784                        // step
785                        index++;
786                        if (index < undoRanges.size()) {
787                                offset= curEnd;
788                                current= (StyleRange) undoRanges.get(index);
789                                curStart= current.start;
790                                curEnd= current.start + current.length;
791                        } else if (index == undoRanges.size()) {
792                                // last one
793                                offset= curEnd;
794                                current= null;
795                                curStart= region.getOffset() + region.getLength();
796                                curEnd= -1;
797                        } else
798                                curEnd= region.getOffset() + region.getLength();
799                }
800 
801                // create modified styles (with background)
802                List shadedRanges= new ArrayList(undoRanges.size());
803                for (Iterator it= undoRanges.iterator(); it.hasNext(); ) {
804                        StyleRange range= (StyleRange) ((StyleRange) it.next()).clone();
805                        shadedRanges.add(range);
806                        range.background= getHighlightColor(disp);
807                }
808 
809                // set the ranges one by one
810                for (Iterator iter= shadedRanges.iterator(); iter.hasNext(); ) {
811                        text.setStyleRange((StyleRange) iter.next());
812 
813                }
814 
815                return (StyleRange[]) undoRanges.toArray(undoRanges.toArray(new StyleRange[0]));
816        }
