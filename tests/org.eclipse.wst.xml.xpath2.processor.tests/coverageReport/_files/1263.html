392                if (rhsCache != null) {
393                        int count= rhsCache.size();
394                        LinkedHashMap history= new LinkedHashMap((int) (count / 0.75));
395                        int rank= 1;
396                        for (Iterator it= rhsCache.iterator(); it.hasNext(); rank++) {
397                                String type= (String) it.next();
398                                history.put(type, new Integer(rank));
399                        }
400                        return new RHSHistory(history);
401                }
402                return EMPTY_HISTORY;
403        }

415                        String lhs= (String) entry.getKey();

419        }
