493                dialog.setText(Messages.format(PreferencesMessages.CodeTemplateBlock_export_title, String.valueOf(templates.length))); 
494                dialog.setFilterExtensions(new String[] {PreferencesMessages.CodeTemplateBlock_export_extension}); 
495                dialog.setFileName(PreferencesMessages.CodeTemplateBlock_export_filename); 
496                String path= dialog.open();
497 
498                if (path == null)
499                        return;
500                
501                File file= new File(path);

503                if (file.isHidden()) {
