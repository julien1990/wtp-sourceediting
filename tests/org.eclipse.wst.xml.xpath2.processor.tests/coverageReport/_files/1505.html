402                                                if (!fActiveScanners.contains(fileScanners[j]) && !monitor.isCanceled()) {
403                                                        fileScanners[j].startup(file.getProject());
404                                                        fActiveScanners.add(fileScanners[j]);
405                                                }

407                                                Map[] taskMarkerAttributes = fileScanners[j].scan(file, taskTags, new SubProgressMonitor(scannerMonitor, 1, SubProgressMonitor.PREPEND_MAIN_LABEL_TO_SUBTASK));
408                                                /*
409                                                 * TODO: pool the marker results so there's only one
410                                                 * operation creating them
411                                                 */
412                                                for (int i = 0; i < taskMarkerAttributes.length; i++) {
413                                                        if (markerAttributes == null) {
414                                                                markerAttributes = new ArrayList();
415                                                        }
416                                                        if (!taskMarkerAttributes[i].containsKey(IMarker.TASK)) {

432                }

435                        return;

437                if (fileScanners != null && fileScanners.length > 0) {

440                                replaceTaskMarkers(file, (String[]) markerTypes.toArray(new String[markerTypes.size()]), (Map[]) markerAttributes.toArray(new Map[markerAttributes.size()]), markerUpdateMonitor);
441                        }
442                }
443                else {

445                }

447        }
