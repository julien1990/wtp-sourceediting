695    private void addLocalDeclarationSplit(ASTRewrite rewrite) throws JavaScriptModelException {
696            VariableDeclarationStatement tempDeclarationStatement= getTempDeclarationStatement();
697            Block block= (Block)tempDeclarationStatement.getParent();//XXX can it be anything else?
698            int statementIndex= block.statements().indexOf(tempDeclarationStatement);
699                      Assert.isTrue(statementIndex != -1);
700            List fragments= tempDeclarationStatement.fragments();
701        int fragmentIndex= fragments.indexOf(fTempDeclarationNode);
702            Assert.isTrue(fragmentIndex != -1);
703 
704        for (int i1= fragmentIndex, n = fragments.size(); i1 < n; i1++) {
705                VariableDeclarationFragment fragment= (VariableDeclarationFragment)fragments.get(i1);
706                rewrite.remove(fragment, null);
707        }
708        if (fragmentIndex == 0)
709                   rewrite.remove(tempDeclarationStatement, null);
710        
711        Assert.isTrue(tempHasInitializer());
712        rewrite.getListRewrite(block, Block.STATEMENTS_PROPERTY).insertAt(createExpressionStatementThatInitializesField(rewrite), statementIndex + 1, null);
713        
714        if (fragmentIndex + 1 < fragments.size()){

730                VariableDeclarationFragment fragmentCopy= (VariableDeclarationFragment)rewrite.createCopyTarget(fragment);
731                statement.fragments().add(fragmentCopy);
732            }
733            rewrite.getListRewrite(block, Block.STATEMENTS_PROPERTY).insertAt(statement, statementIndex + 2, null);
734        }

837                public boolean visit(SimpleName node) {
838                        ITypeBinding typeBinding= node.resolveTypeBinding();
839                        if (typeBinding != null && typeBinding.isLocal()) {
840                                if (node.isDeclaration()) {
841                                        fLocalDefinitions.add(typeBinding);
842                                } else if (! fLocalDefinitions.contains(typeBinding)) {
843                                        fLocalReferencesToEnclosing.add(node);
844                                }
845                        }
846                        if (typeBinding != null && typeBinding.isTypeVariable()) {
847                                if (node.isDeclaration()) {
848                                        fLocalDefinitions.add(typeBinding);
849                                } else if (! fLocalDefinitions.contains(typeBinding)) {
850                                        if (fMethodTypeVariables.contains(typeBinding)) {
851                                                fLocalReferencesToEnclosing.add(node);
852                                        } else {
853                                                fClassTypeVariablesUsed= true;
854                                        }
855                                }
856                        }
857                        IBinding binding= node.resolveBinding();
858                        if (binding != null && binding.getKind() == IBinding.VARIABLE && ! ((IVariableBinding)binding).isField()) {
859                                if (node.isDeclaration()) {
860                                        fLocalDefinitions.add(binding);
861                                } else if (! fLocalDefinitions.contains(binding)) {
862                                        fLocalReferencesToEnclosing.add(node);
863                                }
864                        }
865                        return super.visit(node);

873                        final String selection= extended.getAttribute(JDTRefactoringDescriptor.ATTRIBUTE_SELECTION);
