424        public RefactoringStatus initialize(RefactoringArguments arguments) {
425                if (!(arguments instanceof JavaRefactoringArguments)) {
426                        return RefactoringStatus.createFatalErrorStatus(RefactoringCoreMessages.InitializableRefactoring_inacceptable_arguments);
427                }
428                
429                final JavaRefactoringArguments extended= (JavaRefactoringArguments) arguments;
430                final String handle= extended.getAttribute(JDTRefactoringDescriptor.ATTRIBUTE_INPUT);
431                if (handle == null) {
432                        return RefactoringStatus.createFatalErrorStatus(Messages.format(RefactoringCoreMessages.InitializableRefactoring_argument_not_exist, JDTRefactoringDescriptor.ATTRIBUTE_INPUT));
433                }
434                        
435                final IJavaScriptElement element= JDTRefactoringDescriptor.handleToElement(extended.getProject(), handle, false);
436                if (element == null || !element.exists() || element.getElementType() != IJavaScriptElement.JAVASCRIPT_UNIT)
437                        return ScriptableRefactoring.createInputFatalStatus(element, getRefactoring().getName(), IJavaScriptRefactorings.RENAME_JAVASCRIPT_UNIT);
438                
439                final String name= extended.getAttribute(JDTRefactoringDescriptor.ATTRIBUTE_NAME);
440                if (name == null || name.length() == 0)
441                        return RefactoringStatus.createFatalErrorStatus(Messages.format(RefactoringCoreMessages.InitializableRefactoring_argument_not_exist, JDTRefactoringDescriptor.ATTRIBUTE_NAME));
442 
443                fCu= (IJavaScriptUnit) element;
444                try {
445                        computeRenameTypeRefactoring();
446                        setNewElementName(name);
447                } catch (CoreException exception) {
448                        JavaScriptPlugin.log(exception);
449                        return ScriptableRefactoring.createInputFatalStatus(element, getRefactoring().getName(), IJavaScriptRefactorings.RENAME_JAVASCRIPT_UNIT);
450                }
451                return new RefactoringStatus();
452        }
