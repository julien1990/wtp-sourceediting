525                                        path = manager.performStringSubstitution(path, false);

529                                {
530                                        IContainer[] containers = root.findContainersForLocationURI(f.toURI()); 
531                                        if (containers.length > 0)
532                                        {
533                                                res = containers[0];
534                                        }
535                                }

537                                {

539                                        if (files.length > 0)
540                                        {
