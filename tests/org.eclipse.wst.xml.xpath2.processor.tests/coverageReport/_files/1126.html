145        private void updateTabPreferences(String tabPolicy, NumberPreference tabPreference, NumberPreference indentPreference, CheckboxPreference onlyForLeading) {
146                /*
147                 * If the tab-char is SPACE (or TAB), INDENTATION_SIZE
148                 * preference is not used by the core formatter. We piggy back the
149                 * visual tab length setting in that preference in that case. If the
150                 * user selects MIXED, we use the previous TAB_SIZE preference as the
151                 * new INDENTATION_SIZE (as this is what it really is) and set the 
152                 * visual tab size to the value piggy backed in the INDENTATION_SIZE
153                 * preference. See also CodeFormatterUtil. 
154                 */
155                if (DefaultCodeFormatterConstants.MIXED.equals(tabPolicy)) {
156                        if (JavaScriptCore.SPACE.equals(fOldTabChar) || JavaScriptCore.TAB.equals(fOldTabChar))
157                                swapTabValues();
158                        tabPreference.setEnabled(true);
159                        tabPreference.setKey(DefaultCodeFormatterConstants.FORMATTER_TAB_SIZE);
160                        indentPreference.setEnabled(true);
161                        indentPreference.setKey(DefaultCodeFormatterConstants.FORMATTER_INDENTATION_SIZE);
162                        onlyForLeading.setEnabled(true);
163                } else if (JavaScriptCore.SPACE.equals(tabPolicy)) {
164                        if (DefaultCodeFormatterConstants.MIXED.equals(fOldTabChar))
165                                swapTabValues();
166                        tabPreference.setEnabled(true);
167                        tabPreference.setKey(DefaultCodeFormatterConstants.FORMATTER_INDENTATION_SIZE);
168                        indentPreference.setEnabled(true);
169                        indentPreference.setKey(DefaultCodeFormatterConstants.FORMATTER_TAB_SIZE);
170                        onlyForLeading.setEnabled(false);
171                } else if (JavaScriptCore.TAB.equals(tabPolicy)) {
172                        if (DefaultCodeFormatterConstants.MIXED.equals(fOldTabChar))
173                                swapTabValues();
174                        tabPreference.setEnabled(true);
175                        tabPreference.setKey(DefaultCodeFormatterConstants.FORMATTER_TAB_SIZE);
176                        indentPreference.setEnabled(false);
177                        indentPreference.setKey(DefaultCodeFormatterConstants.FORMATTER_TAB_SIZE);
178                        onlyForLeading.setEnabled(true);
179                } else {
180                        Assert.isTrue(false);
181                }
182                fOldTabChar= tabPolicy;
183        }
