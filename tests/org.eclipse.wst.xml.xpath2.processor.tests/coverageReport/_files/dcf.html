331        private void checkInvocationContext(RefactoringStatus result, int severity) {
332                if (fInvocation.getNodeType() == ASTNode.FUNCTION_INVOCATION) {
333                        Expression exp= ((FunctionInvocation)fInvocation).getExpression();
334                        if (exp != null && exp.resolveTypeBinding() == null) {
335                                addEntry(result, RefactoringCoreMessages.CallInliner_receiver_type, 
336                                        RefactoringStatusCodes.INLINE_METHOD_NULL_BINDING, severity);
337                                return;
338                        }
339                }
340                int nodeType= fTargetNode.getNodeType();
341                if (nodeType == ASTNode.EXPRESSION_STATEMENT) {
342                        if (fSourceProvider.isExecutionFlowInterrupted()) {
343                                addEntry(result, RefactoringCoreMessages.CallInliner_execution_flow,  
344                                        RefactoringStatusCodes.INLINE_METHOD_EXECUTION_FLOW, severity);
345                                return;
346                        }
347                } else if (nodeType == ASTNode.FUNCTION_INVOCATION) {
348                        ASTNode parent= fTargetNode.getParent();
349                        if (isReturnStatement(parent)) {
350                                //support inlining even if the execution flow is interrupted
351                                return;
352                        }
353                        if (fSourceProvider.isExecutionFlowInterrupted()) {
354                                addEntry(result, RefactoringCoreMessages.CallInliner_execution_flow,  
355                                        RefactoringStatusCodes.INLINE_METHOD_EXECUTION_FLOW, severity);
356                                return;
357                        }
358                        if (isAssignment(parent) || isSingleDeclaration(parent)) {
359                                // we support inlining expression in assigment and initializers as
360                                // long as the execution flow isn't interrupted.
361                                return;
362                        } else {
363                                boolean isFieldDeclaration= ASTNodes.getParent(fInvocation, FieldDeclaration.class) != null;
364                                if (!fSourceProvider.isSimpleFunction()) {
365                                        if (isMultiDeclarationFragment(parent)) {
366                                                addEntry(result, RefactoringCoreMessages.CallInliner_multiDeclaration, 
367                                                        RefactoringStatusCodes.INLINE_METHOD_INITIALIZER_IN_FRAGEMENT, severity);
368                                        } else if (isFieldDeclaration) {
369                                                addEntry(result,
370                                                        RefactoringCoreMessages.CallInliner_field_initializer_simple, 
371                                                        RefactoringStatusCodes.INLINE_METHOD_FIELD_INITIALIZER, severity);
372                                        } else {
373                                                addEntry(result, RefactoringCoreMessages.CallInliner_simple_functions, 
374                                                        RefactoringStatusCodes.INLINE_METHOD_ONLY_SIMPLE_FUNCTIONS, severity);
375                                        }
376                                        return;
377                                }
378                                if (isFieldDeclaration) {
379                                        int argumentsCount= fContext.arguments.length;
380                                        for (int i= 0; i < argumentsCount; i++) {
381                                                ParameterData parameter= fSourceProvider.getParameterData(i);
382                                                if(parameter.isWrite()) {
383                                                        addEntry(result,
384                                                                RefactoringCoreMessages.CallInliner_field_initialize_write_parameter, 
385                                                                RefactoringStatusCodes.INLINE_METHOD_FIELD_INITIALIZER, severity);
386                                                        return;
387                                                }
388                                        }
389                                        if(fLocals.size() > 0) {
390                                                addEntry(result,
391                                                        RefactoringCoreMessages.CallInliner_field_initialize_new_local, 
392                                                        RefactoringStatusCodes.INLINE_METHOD_FIELD_INITIALIZER, severity);
393                                                return;
394                                        }
395                                        // verify that the field is not referenced by the initializer method 
396                                        VariableDeclarationFragment variable= (VariableDeclarationFragment)ASTNodes.getParent(fInvocation, ASTNode.VARIABLE_DECLARATION_FRAGMENT);
397                                        if(fSourceProvider.isVariableReferenced(variable.resolveBinding())) {
398                                                addEntry(result,
399                                                        RefactoringCoreMessages.CallInliner_field_initialize_self_reference, 
400                                                        RefactoringStatusCodes.INLINE_METHOD_FIELD_INITIALIZER, severity);
401                                                return;
402                                        }
403                                }
404                        }
405                }

484        private void computeRealArguments() throws BadLocationException {
485                List arguments= Invocations.getArguments(fInvocation);
486                Set canNotInline= crossCheckArguments(arguments);
487                boolean needsVarargBoxing= needsVarargBoxing(arguments);
488                int varargIndex= fSourceProvider.getVarargIndex();
489                String[] realArguments= new String[needsVarargBoxing ? varargIndex + 1 : arguments.size()];
490                for (int i= 0; i < (needsVarargBoxing ? varargIndex : arguments.size()); i++) {
491                        Expression expression= (Expression)arguments.get(i);
492                        ParameterData parameter= fSourceProvider.getParameterData(i);
493                        if (canInline(expression, parameter) && !canNotInline.contains(expression)) {
494                                realArguments[i] = getContent(expression);
495                                // fixes bugs #35905, #38471
496                                if (argumentNeedsParenthesis(expression, parameter)) {
497                                        realArguments[i] = "(" + realArguments[i] + ")"; //$NON-NLS-1$ //$NON-NLS-2$
498                                }
499                        } else {
500                                String name= fInvocationScope.createName(parameter.getName(), true);
501                                realArguments[i]= name;
502                                fLocals.add(createLocalDeclaration(
503                                        parameter.getTypeBinding(), name, 
504                                        (Expression)fRewrite.createCopyTarget(expression)));
505                        }
506                }
507                if (needsVarargBoxing) {
508                        ParameterData parameter= fSourceProvider.getParameterData(varargIndex);
509                        String name= fInvocationScope.createName(parameter.getName(), true);
510                        realArguments[varargIndex]= name;
511                        AST ast= fInvocation.getAST();
512                        Type type= fImportRewrite.addImport(parameter.getTypeBinding(), ast);
513                        VariableDeclarationFragment fragment= ast.newVariableDeclarationFragment();
514                        fragment.setName(ast.newSimpleName(name));
515                        ArrayInitializer initializer= ast.newArrayInitializer();
516                        for (int i= varargIndex; i < arguments.size(); i++) {
517                                initializer.expressions().add(fRewrite.createCopyTarget((ASTNode)arguments.get(i)));
518                        }
519                        fragment.setInitializer(initializer);
520                        VariableDeclarationStatement decl= ast.newVariableDeclarationStatement(fragment);
521                        decl.setType(type);
522                        fLocals.add(decl);
523                }
524                fContext.arguments= realArguments;
525        }

527        private boolean needsVarargBoxing(List arguments) {
528                if (!fSourceProvider.isVarargs())
529                        return false;
530                /*
531                if (!fSourceProvider.hasArrayAccess())
532                        return false;
533                */
534                int index= fSourceProvider.getVarargIndex();
535                // we have varags but the call doesn't pass any arguments
536                if (index >= arguments.size())
537                        return true;
538                // parameter is array type
539                // one arg
540                if (index == arguments.size() - 1) {
541                        ITypeBinding argument= ((Expression)arguments.get(index)).resolveTypeBinding();
542                        if (argument == null)
543                                return false;
544                        ITypeBinding parameter= fSourceProvider.getParameterData(index).getTypeBinding();
545                        return !fTypeEnvironment.create(argument).canAssignTo(fTypeEnvironment.create(parameter));
546                }
547                return true;
548        }

604        private void replaceCall(RefactoringStatus status, String[] blocks, TextEditGroup textEditGroup) {
605                // Inline empty body
606                if (blocks.length == 0) {
607                        if (fNeedsStatement) {
608                                fRewrite.replace(fTargetNode, fTargetNode.getAST().newEmptyStatement(), textEditGroup);
609                        } else {
610                                fRewrite.remove(fTargetNode, textEditGroup);
611                        }
612                } else {
613                        ASTNode node= null;
614                        for (int i= 0; i < blocks.length - 1; i++) {
615                                node= fRewrite.createStringPlaceholder(blocks[i], ASTNode.RETURN_STATEMENT);
616                                fListRewrite.insertAt(node, fInsertionIndex++, textEditGroup);
617                        }
618                        String block= blocks[blocks.length - 1];
619                        // We can inline a call where the declaration is a function and the call itself
620                        // is a statement. In this case we have to create a temporary variable if the
621                        // returned expression must be evaluated.
622                        if (fContext.callMode == ASTNode.EXPRESSION_STATEMENT && fSourceProvider.hasReturnValue()) {
623                                if (fSourceProvider.mustEvaluateReturnedExpression()) {
624                                        if (fSourceProvider.returnValueNeedsLocalVariable()) {
625                                                IFunctionBinding invocation= Invocations.resolveBinding(fInvocation);
626                                                node= createLocalDeclaration(
627                                                        invocation.getReturnType(), 
628                                                        fInvocationScope.createName(fSourceProvider.getMethodName(), true), 
629                                                        (Expression)fRewrite.createStringPlaceholder(block, ASTNode.FUNCTION_INVOCATION));
630                                        } else {
631                                                node= fTargetNode.getAST().newExpressionStatement(
632                                                        (Expression)fRewrite.createStringPlaceholder(block, ASTNode.FUNCTION_INVOCATION));
633                                        }
634                                } else {
635                                        node= null;
636                                }
637                        } else if (fTargetNode instanceof Expression) {
638                                node= fRewrite.createStringPlaceholder(block, ASTNode.FUNCTION_INVOCATION);
639                                
640                                // fixes bug #24941
641                                if(needsExplicitCast(status)) {
642                                        AST ast= node.getAST();
643                                        CastExpression castExpression= ast.newCastExpression();
644                                        Type returnType= fImportRewrite.addImport(fSourceProvider.getReturnType(), ast);
645                                        castExpression.setType(returnType);
646                                        castExpression.setExpression((Expression)node);
647                                        node= castExpression;
648                                }
649                                
650                                if (needsParenthesis()) {
651                                        ParenthesizedExpression pExp= fTargetNode.getAST().newParenthesizedExpression();
652                                        pExp.setExpression((Expression)node);
653                                        node= pExp;
654                                }
655                        } else {
656                                node= fRewrite.createStringPlaceholder(block, ASTNode.RETURN_STATEMENT);
657                        }
658                        
659                        // Now replace the target node with the source node
660                        if (node != null) {
661                                if (fTargetNode == null) {
662                                        fListRewrite.insertAt(node, fInsertionIndex++, textEditGroup);
663                                } else {
664                                        fRewrite.replace(fTargetNode, node, textEditGroup);
665                                }
666                        } else {
667                                if (fTargetNode != null) {
668                                        fRewrite.remove(fTargetNode, textEditGroup);
669                                }
670                        }
671                }
672        }

674        /**
675         * @return <code>true</code> if explicit cast is needed otherwise <code>false</code>
676         */
677        private boolean needsExplicitCast(RefactoringStatus status) {
678                // if the return type of the method is the same as the type of the
679                // returned expression then we don't need an explicit cast.
680                if (fSourceProvider.returnTypeMatchesReturnExpressions())
681                                return false;                 
682                ASTNode parent= fTargetNode.getParent();
683                int nodeType= parent.getNodeType();
684                if (nodeType == ASTNode.FUNCTION_INVOCATION) {
685                        FunctionInvocation methodInvocation= (FunctionInvocation)parent;
686                        if(methodInvocation.getExpression() == fTargetNode)
687                                return false;
688                        IFunctionBinding method= methodInvocation.resolveMethodBinding();
689                        if (method == null) {
690                                status.addError(RefactoringCoreMessages.CallInliner_cast_analysis_error,  
691                                        JavaStatusContext.create(fCUnit, methodInvocation));
692                                return false;
693                        }
694                        ITypeBinding[] parameters= method.getParameterTypes();
695                        int argumentIndex= methodInvocation.arguments().indexOf(fInvocation);
696                        List returnExprs= fSourceProvider.getReturnExpressions();
697                        // it is inferred that only methods consisting of a single 
698                        // return statement can be inlined as parameters in other 
699                        // method invocations
700                        if (returnExprs.size() != 1)
701                                return false;
702                        parameters[argumentIndex]= ((Expression)returnExprs.get(0)).resolveTypeBinding();
703 
704                        ITypeBinding type= ASTNodes.getReceiverTypeBinding(methodInvocation);
705                        TypeBindingVisitor visitor= new AmbiguousMethodAnalyzer(
706                                fTypeEnvironment, method, fTypeEnvironment.create(parameters));
707                        if(!visitor.visit(type)) {
708                                return true;
709                        }
710                        else if(type.isInterface()) {
711                                return !Bindings.visitInterfaces(type, visitor);
712                        }
713                        else if(Modifier.isAbstract(type.getModifiers())) {
714                                return !Bindings.visitHierarchy(type, visitor);
715                        }
716                        else {
717                                // it is not needed to visit interfaces if receiver is a concrete class
718                                return !Bindings.visitSuperclasses(type, visitor);
719                        }
720                }
721                return false;
722        }

725                if (!fSourceProvider.needsReturnedExpressionParenthesis())
726                        return false;
727                ASTNode parent= fTargetNode.getParent();
728                int type= parent.getNodeType();
729                return 
730                        type == ASTNode.FUNCTION_INVOCATION || 
731                        (parent instanceof Expression && type != ASTNode.ASSIGNMENT) ||
732                        (fSourceProvider.returnsConditionalExpression() &&
733                                type == ASTNode.VARIABLE_DECLARATION_FRAGMENT &&  
734                                ((VariableDeclarationFragment)parent).getInitializer() == fTargetNode);
735        }

739                VariableDeclarationStatement decl= (VariableDeclarationStatement)ASTNodeFactory.newStatement(

745    /**
746     * Checks whether arguments are passed to the method which do some assignments
747     * inside the expression. If so these arguments can't be inlined into the
748     * calling method since the assignments might be reorder. An example is:
749     * <code>
750     *   add((field=args).length,field.hashCode());
751     * </code>
752     * Field might not be initialized when the arguments are reorder in the called
753     * method.
754     */
755        private Set crossCheckArguments(List arguments) {
756                final Set assigned= new HashSet();
757                final Set result= new HashSet();
758                for (Iterator iter= arguments.iterator(); iter.hasNext();) {
759                        final Expression expression= (Expression) iter.next();
760                        expression.accept(new ASTVisitor() {
761                                public boolean visit(Assignment node) {
762                                        Expression lhs= node.getLeftHandSide();
763                                        if (lhs instanceof Name) {
764                                                IBinding binding= ((Name)lhs).resolveBinding();
765                                                if (binding instanceof IVariableBinding) {
766                                                        assigned.add(binding);
767                                                        result.add(expression);
768                                                }
769                                        }
770                                        return true;
771                                }
772                        });
773                }
774                for (Iterator iter= arguments.iterator(); iter.hasNext();) {
775                        final Expression expression= (Expression) iter.next();
776                        if (!result.contains(expression)) {
777                                expression.accept(new HierarchicalASTVisitor() {
778                                        public boolean visit(Name node) {
779                                                IBinding binding= node.resolveBinding();
780                                                if (binding != null && assigned.contains(binding))
781                                                        result.add(expression);
782                                                return false;
783                                        }
784                                });
785                        }
786                }
787                return result;
788        }

796        private void initializeInsertionPoint(int nos) {
797                fInsertionIndex= -1;
798                fNeedsStatement= false;
799                // if we have a constructor invocation the invocation itself is already a statement
800                ASTNode parentStatement= fInvocation instanceof Statement 
801                        ? fInvocation 
802                        : ASTNodes.getParent(fInvocation, Statement.class);
803                ASTNode container= parentStatement.getParent();
804                int type= container.getNodeType();
805                if (type == ASTNode.BLOCK) { 
806                        Block block= (Block)container;
807                        fListRewrite= fRewrite.getListRewrite(block, Block.STATEMENTS_PROPERTY);
808                        fInsertionIndex= fListRewrite.getRewrittenList().indexOf(parentStatement);
809                } else if (type == ASTNode.SWITCH_STATEMENT) {
810                        SwitchStatement switchStatement= (SwitchStatement)container;
811                        fListRewrite= fRewrite.getListRewrite(switchStatement, SwitchStatement.STATEMENTS_PROPERTY);
812                        fInsertionIndex= fListRewrite.getRewrittenList().indexOf(parentStatement);
813                } else if (isControlStatement(container) || type == ASTNode.LABELED_STATEMENT) {
814                        fNeedsStatement= true;
815                        if (nos > 1 || needsBlockAroundDanglingIf()) {
816                                Block block= fInvocation.getAST().newBlock();
817                                fInsertionIndex= 0;
818                                Statement currentStatement= null;
819                                switch(type) {
820                                        case ASTNode.LABELED_STATEMENT:
821                                                currentStatement= ((LabeledStatement)container).getBody();
822                                                break;
823                                        case ASTNode.FOR_STATEMENT:
824                                                currentStatement= ((ForStatement)container).getBody();
825                                                break;
826                                        case ASTNode.FOR_IN_STATEMENT:
827                                                currentStatement= ((ForInStatement)container).getBody(); 
828                                                break;
829                                        case ASTNode.ENHANCED_FOR_STATEMENT:
830                                                currentStatement= ((EnhancedForStatement)container).getBody();
831                                                break;
832                                        case ASTNode.WHILE_STATEMENT:
833                                                currentStatement= ((WhileStatement)container).getBody();
834                                                break;
835                                        case ASTNode.WITH_STATEMENT:
836                                                currentStatement= ((WithStatement)container).getBody();
837                                                break;
838                                        case ASTNode.DO_STATEMENT:
839                                                currentStatement= ((DoStatement)container).getBody();
840                                                break;
841                                        case ASTNode.IF_STATEMENT:
842                                                IfStatement node= (IfStatement)container;
843                                                Statement thenPart= node.getThenStatement();
844                                                if (fTargetNode == thenPart || ASTNodes.isParent(fTargetNode, thenPart)) {
845                                                        currentStatement= thenPart;
846                                                } else {
847                                                        currentStatement= node.getElseStatement();
848                                                }
849                                                break;
850                                }
851                                Assert.isNotNull(currentStatement);
852                                fRewrite.replace(currentStatement, block, null);
853                                fListRewrite= fRewrite.getListRewrite(block, Block.STATEMENTS_PROPERTY);
854                                // The method to be inlined is not the body of the control statement.
855                                if (currentStatement != fTargetNode) {
856                                        fListRewrite.insertLast(fRewrite.createCopyTarget(currentStatement), null);
857                                } else {
858                                        // We can't replace a copy with something else. So we
859                                        // have to insert all statements to be inlined.
860                                        fTargetNode= null;
861                                }
862                        }
863                }
864                // We only insert one new statement or we delete the existing call. 
865                // So there is no need to have an insertion index.
866        }

868        private boolean needsBlockAroundDanglingIf() {
869                /* see https://bugs.eclipse.org/bugs/show_bug.cgi?id=169331
870                 * 
871                 * Situation:
872                 * boolean a, b;
873                 * void toInline() {
874                 *     if (a)
875                 *         hashCode();
876                 * }
877                 * void m() {
878                 *     if (b)
879                 *         toInline();
880                 *     else
881                 *         toString();
882                 * }
883                 * => needs block around inlined "if (a)..." to avoid attaching else to wrong if.
884                 */
885                return fTargetNode.getLocationInParent() == IfStatement.THEN_STATEMENT_PROPERTY
886                                && fTargetNode.getParent().getStructuralProperty(IfStatement.ELSE_STATEMENT_PROPERTY) != null
887                                && fSourceProvider.isDangligIf();
888        }
