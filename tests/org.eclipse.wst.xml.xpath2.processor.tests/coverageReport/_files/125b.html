309                        for (int i= 0; requiredProposals != null &&  i < requiredProposals.length; i++) {
310                                int oldLen= document.getLength();
311                                if (requiredProposals[i].getKind() == CompletionProposal.TYPE_REF) {
312                                        LazyJavaCompletionProposal proposal= new LazyJavaTypeCompletionProposal(requiredProposals[i], fInvocationContext);
313                                        proposal.apply(document);
314                                        setReplacementOffset(getReplacementOffset() + document.getLength() - oldLen);
315                                } else if (requiredProposals[i].getKind() == CompletionProposal.TYPE_IMPORT) {
316                                        ImportCompletionProposal proposal= new ImportCompletionProposal(requiredProposals[i], fInvocationContext, coreProposal.getKind());
317                                        proposal.setReplacementOffset(getReplacementOffset());
318                                        proposal.apply(document);
319                                        setReplacementOffset(getReplacementOffset() + document.getLength() - oldLen);
320                                } else if (requiredProposals[i].getKind() == CompletionProposal.METHOD_IMPORT) {
321                                        ImportCompletionProposal proposal= new ImportCompletionProposal(requiredProposals[i], fInvocationContext, coreProposal.getKind());
322                                        proposal.setReplacementOffset(getReplacementOffset());
323                                        proposal.apply(document);
324                                        setReplacementOffset(getReplacementOffset() + document.getLength() - oldLen);
325                                } else if (requiredProposals[i].getKind() == CompletionProposal.FIELD_IMPORT) {
326                                        ImportCompletionProposal proposal= new ImportCompletionProposal(requiredProposals[i], fInvocationContext, coreProposal.getKind());
327                                        proposal.setReplacementOffset(getReplacementOffset());
328                                        proposal.apply(document);
329                                        setReplacementOffset(getReplacementOffset() + document.getLength() - oldLen);
330                                } else {
331                                        /*
332                                         * In 3.3 we only support the above required proposals, see
333                                         * CompletionProposal#getRequiredProposals()
334                                         */
335                                         Assert.isTrue(false);
336                                }
337                        }
338                }
339                
340                try {
341                        // patch replacement length
342                        int delta= offset - (getReplacementOffset() + getReplacementLength());
343                        if (delta > 0)
344                                setReplacementLength(getReplacementLength() + delta);
345        
346                        boolean isSmartTrigger= isSmartTrigger(trigger);
347        
348                        String replacement;
349                        if (isSmartTrigger || trigger == (char) 0) {
350                                replacement= getReplacementString();
351                        } else {
352                                StringBuffer buffer= new StringBuffer(getReplacementString());
353        
354                                // fix for PR #5533. Assumes that no eating takes place.
355                                if ((getCursorPosition() > 0 && getCursorPosition() <= buffer.length() && buffer.charAt(getCursorPosition() - 1) != trigger)) {
356                                        buffer.insert(getCursorPosition(), trigger);
357                                        setCursorPosition(getCursorPosition() + 1);
358                                }
359        
360                                replacement= buffer.toString();
361                                setReplacementString(replacement);
362                        }
363        
364                        // reference position just at the end of the document change.
365                        int referenceOffset= getReplacementOffset() + getReplacementLength();
366                        final ReferenceTracker referenceTracker= new ReferenceTracker();
367                        referenceTracker.preReplace(document, referenceOffset);
368        
369                        replace(document, getReplacementOffset(), getReplacementLength(), replacement);
370        
371                        referenceOffset= referenceTracker.postReplace(document);
372                        setReplacementOffset(referenceOffset - (replacement == null ? 0 : replacement.length()));
373        
374                        // PR 47097
375                        if (isSmartTrigger)
376                                handleSmartTrigger(document, trigger, referenceOffset);
377        
378                } catch (BadLocationException x) {
379                        // ignore
380                }
381        }

746         * @return <code>true</code> if the string begins with the given prefix and
747         *                        <code>false</code> if <code>prefix</code> is longer than <code>string</code>
748         *                        or the string doesn't start with the given prefix 
749         *

751        protected boolean isPrefix(String prefix, String string) {
752                if (prefix == null || string ==null || prefix.length() > string.length())
753                        return false;
754                String start= string.substring(0, prefix.length());

758        /**
759         * Matches <code>prefix</code> against <code>string</code> and replaces the matched region
760         * by prefix. Case is preserved as much as possible. This method returns <code>string</code> if camel case completion
761         * is disabled. Examples when camel case completion is enabled:
762         * <ul>
763         * <li>getCamelCompound("NuPo", "NullPointerException") -> "NuPointerException"</li>
764         * <li>getCamelCompound("NuPoE", "NullPointerException") -> "NuPoException"</li>
765         * <li>getCamelCompound("hasCod", "hashCode") -> "hasCode"</li>
766         * </ul>
767         * 
768         * @param prefix the prefix to match against
769         * @param string the string to match
770         * @return a compound of prefix and any postfix taken from <code>string</code>
771         * 
772         */
773        protected final String getCamelCaseCompound(String prefix, String string) {
774                if (prefix.length() > string.length())
775                        return string;
776 
777                // a normal prefix - no camel case logic at all
778                String start= string.substring(0, prefix.length());
779                if (start.equalsIgnoreCase(prefix))
780                        return string;
781 
782                final char[] patternChars= prefix.toCharArray();
783                final char[] stringChars= string.toCharArray();
784 
785                for (int i= 1; i <= stringChars.length; i++)
786                        if (CharOperation.camelCaseMatch(patternChars, 0, patternChars.length, stringChars, 0, i))
787                                return prefix + string.substring(i);
788 
789                // Not a camel case match at all.
790                // This should not happen -> stay with the default behavior
791                return string;
792        }

860        private void updateStyle(ITextViewer viewer) {
861 
862                StyledText text= viewer.getTextWidget();
863                if (text == null || text.isDisposed())
864                        return;
865 
866                int widgetCaret= text.getCaretOffset();
867 
868                int modelCaret= 0;
869                if (viewer instanceof ITextViewerExtension5) {
870                        ITextViewerExtension5 extension= (ITextViewerExtension5) viewer;
871                        modelCaret= extension.widgetOffset2ModelOffset(widgetCaret);
872                } else {
873                        IRegion visibleRegion= viewer.getVisibleRegion();
874                        modelCaret= widgetCaret + visibleRegion.getOffset();
875                }
876 
877                if (modelCaret >= getReplacementOffset() + getReplacementLength()) {
878                        repairPresentation(viewer);
879                        return;
880                }
881 
882                int offset= widgetCaret;
883                int length= getReplacementOffset() + getReplacementLength() - modelCaret;
884 
885                Color foreground= getForegroundColor(text);
886                Color background= getBackgroundColor(text);
887 
888                StyleRange range= text.getStyleRangeAtOffset(offset);
889                int fontStyle= range != null ? range.fontStyle : SWT.NORMAL;
890 
891                repairPresentation(viewer);
892                fRememberedStyleRange= new StyleRange(offset, length, foreground, background, fontStyle);
893                if (range != null) {
894                        fRememberedStyleRange.strikeout= range.strikeout;
895                        fRememberedStyleRange.underline= range.underline;
896                }
897 
898                // http://dev.eclipse.org/bugs/show_bug.cgi?id=34754
899                try {
900                        text.setStyleRange(fRememberedStyleRange);
901                } catch (IllegalArgumentException x) {
902                        // catching exception as offset + length might be outside of the text widget
903                        fRememberedStyleRange= null;
904                }
905        }
