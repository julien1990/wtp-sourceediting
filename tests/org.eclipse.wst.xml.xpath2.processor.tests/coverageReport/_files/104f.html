422                                        typeBinding= typeBinding.getElementType();
423                                }
424                                if (typeBinding.isTypeVariable() || typeBinding.isCapture() || typeBinding.isWildcardType()) { // can be removed when bug 98473 is fixed
425                                        continue; 
426                                }
427                                
428                                if (typeBinding.isMember() || typeBinding.isTopLevel()) {
429                                        String name= Bindings.getRawQualifiedName(typeBinding);
430                                        if (name.length() > 0) {
431                                                namesToImport.add(name);
432                                        }
433                                }
434                        }
435                }
436 
437                HashSet staticsToImport= new HashSet(staticImportsRefs.size());
438                for (int i= 0; i < staticImportsRefs.size(); i++) {
439                        Name curr= (Name) staticImportsRefs.get(i);
440                        IBinding binding= curr.resolveBinding();
441                        if (binding != null) {
442                                StringBuffer buf= new StringBuffer(Bindings.getImportName(binding));
443                                if (binding.getKind() == IBinding.METHOD) {
444                                        buf.append("()"); //$NON-NLS-1$
445                                }
446                                staticsToImport.add(buf.toString());
447                        }
448                }
449 
450 
451                if (namesToImport.isEmpty() && staticsToImport.isEmpty()) {
452                        return null;
453                }
454 
455                String[] typeImports= (String[]) namesToImport.toArray(new String[namesToImport.size()]);
456                String[] staticImports= (String[]) staticsToImport.toArray(new String[staticsToImport.size()]);
457                return new ClipboardData(inputElement, typeImports, staticImports);
458        }

462                ITextEditor editor= getTextEditor();
463                IJavaScriptElement inputElement= (IJavaScriptElement) editor.getEditorInput().getAdapter(IJavaScriptElement.class);
464 
465                Clipboard clipboard= new Clipboard(getDisplay());

467                if (importsData != null && inputElement instanceof IJavaScriptUnit && !importsData.isFromSame(inputElement)) {
468                        // combine operation and adding of imports
469                        IRewriteTarget target= editor != null ? (IRewriteTarget) editor.getAdapter(IRewriteTarget.class) : null;
470                        if (target != null) {

478                        } finally {
479                                if (target != null) {
480                                        target.endCompoundChange();
481                                }
482                        }
483                } else {
484                        fOperationTarget.doOperation(fOperationCode);

489        private void addImports(IJavaScriptUnit unit, ClipboardData data) throws CoreException {
490                ImportRewrite rewrite= StubUtility.createImportRewrite(unit, true);
491                String[] imports= data.getTypeImports();
492                for (int i= 0; i < imports.length; i++) {
493                        rewrite.addImport(imports[i]);
494                }
495                String[] staticImports= data.getStaticImports();
496                for (int i= 0; i < staticImports.length; i++) {
497                        String name= Signature.getSimpleName(staticImports[i]);
498                        boolean isField= !name.endsWith("()"); //$NON-NLS-1$
499                        if (!isField) {
500                                name= name.substring(0, name.length() - 2);
501                        }
502                        String qualifier= Signature.getQualifier(staticImports[i]);
503                        rewrite.addStaticImport(qualifier, name, isField);
504                }
505 
506                JavaModelUtil.applyEdit(unit, rewrite.rewriteImports(null), false, null);
507        }
