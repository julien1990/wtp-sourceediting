382 
383                boolean allSameState= true;

390                        }

400 
401                BusyIndicator.showWhile(getShell().getDisplay(), new Runnable() {
402                        public void run() {
403                                Object[] elements;
404                                if (fWorkingSet == null) {
405                                        // Use current part's selection for initialization
406                                        IWorkbenchPage page= JavaScriptPlugin.getActivePage();
407                                        if (page == null)

409                                        
410                                        IWorkbenchPart part= JavaScriptPlugin.getActivePage().getActivePart();
411                                        if (part == null)
412                                                return;
413                                        
414                                        try {
415                                                elements= SelectionConverter.getStructuredSelection(part).toArray();
416                                                for (int i= 0; i < elements.length; i++) {
417                                                        if (elements[i] instanceof IResource) {
418                                                                IJavaScriptElement je= (IJavaScriptElement)((IResource)elements[i]).getAdapter(IJavaScriptElement.class);
419                                                                if (je != null && je.exists() &&  je.getJavaScriptProject().isOnIncludepath((IResource)elements[i]))
420                                                                        elements[i]= je;
421                                                        }
422                                                }
423                                        } catch (JavaScriptModelException e) {
424                                                return;
425                                        }
426                                }
427                                else
428                                        elements= fWorkingSet.getElements();
429 
430                                // Use closed project for elements in closed project

433                                        if (element instanceof IResource) {

442                                        }

459                                                parents.add(parent);
460                                }
461                                
462                                for (Iterator i = parents.iterator(); i.hasNext();)

464                        }
465                });
