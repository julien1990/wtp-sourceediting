285    /**
286     * Check if for the current type of selection, no context specific actions can 
287     * be applied. Note: this does not mean, that there are NO actions available at all.<p>
288     * 
289     * For example: if the default package is selected, there is no specific action for this kind 
290     * of selection as no operations are allowed on the default package. Nevertheless, the 
291     * <code>PackageExplorerActionEvent</code> will return at least one action that allows to 
292     * link to an existing folder in the file system, but this operation is always available 
293     * and does not add any supporting information to the current selection. Therefore, 
294     * it can be filtered and the correct answer to the user is that there is no specific 
295     * action for the default package.
296     * 
297     * @param actions an array of provided actions
298     * @return <code>true</code> if there is at least one action that allows context 
299     * sensitive operations, <code>false</code> otherwise.
300     */
301    private boolean noContextHelpAvailable(BuildpathModifierAction[] actions) {
302        if (actions.length == 0)
303            return true;
304        if (actions.length == 1) {
305            int id= Integer.parseInt(actions[0].getId());
306            if (id == BuildpathModifierAction.CREATE_LINK)
307                return true;
308        }
309        if (actions.length == 2) {
310            int idLink= Integer.parseInt(actions[0].getId());
311            int idReset= Integer.parseInt(actions[1].getId());
312            if (idReset == BuildpathModifierAction.RESET_ALL && 
313                idLink == BuildpathModifierAction.CREATE_LINK)
314                return true;
315        }
316        return false;
