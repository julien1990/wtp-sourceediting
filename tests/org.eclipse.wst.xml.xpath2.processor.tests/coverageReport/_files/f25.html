372        private TextEdit probeFormatting(String source, int indentationLevel, String lineSeparator, int offset, int length) {
373                if (ProbingScanner == null) {
374                        // scanner use to check if the kind could be K_JAVA_DOC, K_MULTI_LINE_COMMENT or K_SINGLE_LINE_COMMENT
375                        ProbingScanner = new Scanner(true, true, false/*nls*/, ClassFileConstants.JDK1_3, ClassFileConstants.JDK1_3, null/*taskTags*/, null/*taskPriorities*/, true/*taskCaseSensitive*/);
376                }
377                ProbingScanner.setSource(source.toCharArray());
378                ProbingScanner.resetTo(offset, offset + length);
379                try {
380                        switch(ProbingScanner.getNextToken()) {
381                                case ITerminalSymbols.TokenNameCOMMENT_BLOCK :
382                                        if (ProbingScanner.getCurrentTokenEndPosition() == offset + length - 1) {
383                                                return formatComment(K_MULTI_LINE_COMMENT, source, indentationLevel, lineSeparator, offset, length);
384                                        }
385                                        break;
386                                case ITerminalSymbols.TokenNameCOMMENT_LINE :
387                                        if (ProbingScanner.getCurrentTokenEndPosition() == offset + length - 1) {
388                                                return formatComment(K_SINGLE_LINE_COMMENT, source, indentationLevel, lineSeparator, offset, length);
389                                        }
390                                        break;
391                                case ITerminalSymbols.TokenNameCOMMENT_JAVADOC :
392                                        if (ProbingScanner.getCurrentTokenEndPosition() == offset + length - 1) {
393                                                return formatComment(K_JAVA_DOC, source, indentationLevel, lineSeparator, offset, length);
394                                        }
395                        }
396                } catch (InvalidInputException e) {
397                        // ignore
398                }
399                ProbingScanner.setSource((char[]) null);
400 
401                // probe for expression
402                Expression expression = this.codeSnippetParsingUtil.parseExpression(source.toCharArray(), getDefaultCompilerOptions(), true);
403                if (expression != null) {
404                        return internalFormatExpression(source, indentationLevel, lineSeparator, expression, offset, length);
405                }
406 
407                // probe for body declarations (fields, methods, constructors)
408                ASTNode[] bodyDeclarations = this.codeSnippetParsingUtil.parseClassBodyDeclarations(source.toCharArray(), getDefaultCompilerOptions(), true);
409                if (bodyDeclarations != null) {
410                        return internalFormatClassBodyDeclarations(source, indentationLevel, lineSeparator, bodyDeclarations, offset, length);
411                }
412 
413                // probe for statements
414                ConstructorDeclaration constructorDeclaration = this.codeSnippetParsingUtil.parseStatements(source.toCharArray(), getDefaultCompilerOptions(), true, false);
415                if (constructorDeclaration.statements != null) {
416                        return internalFormatStatements(source, indentationLevel, lineSeparator, constructorDeclaration, offset, length);
417                }
418 
419                // this has to be a compilation unit
420                return formatCompilationUnit(source, indentationLevel, lineSeparator, offset, length);
421        }
