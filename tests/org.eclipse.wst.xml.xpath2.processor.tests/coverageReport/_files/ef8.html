342        private boolean isValidComment(String template) {
343                IScanner scanner= ToolFactory.createScanner(true, false, false, false);
344                scanner.setSource(template.toCharArray());
345                try {
346                        int next= scanner.getNextToken();
347                        while (TokenScanner.isComment(next)) {
348                                next= scanner.getNextToken();
349                        }
350                        return next == ITerminalSymbols.TokenNameEOF;
351                } catch (InvalidInputException e) {
352                }
353                return false;
354        }
