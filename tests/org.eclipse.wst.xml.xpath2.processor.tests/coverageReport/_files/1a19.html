381   *  sets the Wizard's error message accordingly.
382   *
383   * @return Whether the regex is valid.
384   */
385  private boolean validateRegex()
386  {
387 
388    boolean isValid;
389    try
390    {
391      // We validate the regex by checking whether we get a ParseException.
392      // By default, we assume that it's valid unless we determine otherwise.
393      isValid = true;
394      displayRegexErrorMessage(null);
395      value.setStyleRange(null);
396 
397      Pattern.compile(value.getText());
398    }
399    catch (PatternSyntaxException pe)
400    {
401      isValid = false;
402      displayRegexErrorMessage(pe.getMessage());
403 
404      // An off-by-one bug in the xerces regex parser will sometimes return a location for the parseError that
405      //  is off the end of the string.  If this is the case, then we want to highlight the last character.
406      if (pe.getIndex() >= value.getText().length())
407      {
408        currentError.start = value.getText().length() - 1;
409      }
410      else
411      {
412        currentError.start = pe.getIndex();
413      }
414 
415      if (debug)
416      {
417        System.out.println("Parse Error location: " + pe.getIndex());  //$NON-NLS-1$
418        System.out.println("currentError.start: " + currentError.start); //$NON-NLS-1$
419      }
420 
421      value.setStyleRange(currentError);
422 
423    }
424 
425    // Another bug in the xerces parser will sometimes throw a RuntimeException instead of a ParseException.
426    //  When we get a RuntimeException, we aren't provided with the additional information we need to highlight
427    //  the parse error.  So, we merely report that there is an error.
428    catch (RuntimeException re)
429    {
430      displayRegexErrorMessage("");
431      value.setStyleRange(null);
432      isValid = false;
433    }
434    
435    setPageComplete(isValid);    
436    return isValid;
437  }
