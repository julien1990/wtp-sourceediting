379 *                If the first name provided is a field (or local) then the field (or local) is answered
380 *                Otherwise, package names and type names are consumed until a field is found.
381 *                In this case, the field is answered.
382 *
383 *        if the TYPE mask is set,
384 *                package names and type names are consumed until the end of the input.
385 *                Only if all of the input is consumed is the type answered
386 *
387 *        All other conditions are errors, and a problem binding is returned.
388 *
389 *        NOTE: If a problem binding is returned, senders should extract the compound name
390 *        from the binding & not assume the problem applies to the entire compoundName.
391 *
392 *        The VARIABLE mask has precedence over the TYPE mask.
393 *
394 *        InvocationSite implements
395 *                isSuperAccess(); this is used to determine if the discovered field is visible.
396 *                setFieldIndex(int); this is used to record the number of names that were consumed.
397 *
398 *        For example, getBinding({"foo","y","q", VARIABLE, site) will answer
399 *        the binding for the field or local named "foo" (or an error binding if none exists).
400 *        In addition, setFieldIndex(1) will be sent to the invocation site.
401 *        If a type named "foo" exists, it will not be detected (and an error binding will be answered)
402 *
403 *        IMPORTANT NOTE: This method is written under the assumption that compoundName is longer than length 1.
404 */

637/*
638 * This retrieves the argument that maps to an enclosing instance of the suitable type,
639 *         if not found then answers nil -- do not create one
640 *
641 *                #implicitThis                                                   : the implicit this will be ok
642 *                #((arg) this$n)                                                : available as a constructor arg
643 *                 #((arg) this$n ... this$p)                         : available as as a constructor arg + a sequence of fields
644 *                 #((fieldDescr) this$n ... this$p)         : available as a sequence of fields
645 *                 nil                                                                                                          : not found
646 *
647 *         Note that this algorithm should answer the shortest possible sequence when
648 *                 shortcuts are available:
649 *                                 this$0 . this$0 . this$0
650 *                 instead of
651 *                                 this$2 . this$1 . this$0 . this$1 . this$0
652 *                 thus the code generation will be more compact and runtime faster
653 */
654public VariableBinding[] getEmulationPath(LocalVariableBinding outerLocalVariable) {
655        MethodScope currentMethodScope = this.methodScope();
656        SourceTypeBinding sourceType = currentMethodScope.enclosingSourceType();
657 
658        // identity check
659        BlockScope variableScope = outerLocalVariable.declaringScope;
660        if (variableScope == null /*val$this$0*/ || currentMethodScope == variableScope.methodScope()) {
661                return new VariableBinding[] { outerLocalVariable };
662                // implicit this is good enough
663        }
664        // use synthetic constructor arguments if possible
665        if (currentMethodScope.isInsideInitializerOrConstructor()
666                && (sourceType.isNestedType())) {
667                SyntheticArgumentBinding syntheticArg;
668                if ((syntheticArg = ((NestedTypeBinding) sourceType).getSyntheticArgument(outerLocalVariable)) != null) {
669                        return new VariableBinding[] { syntheticArg };
670                }
671        }
672        // use a synthetic field then
673        if (!currentMethodScope.isStatic) {
674                FieldBinding syntheticField;
675                if ((syntheticField = sourceType.getSyntheticField(outerLocalVariable)) != null) {
676                        return new VariableBinding[] { syntheticField };
677                }
678        }
679        return null;

682/*

684 *         if not found then answers nil -- do not create one
685 *
686 *                #implicitThis                                                                                                                   :  the implicit this will be ok

690 *         null                                                                                                                                          : not found

693public Object[] getEmulationPath(ReferenceBinding targetEnclosingType, boolean onlyExactMatch, boolean denyEnclosingArgInConstructorCall) {
694        MethodScope currentMethodScope = this.methodScope();
695        SourceTypeBinding sourceType = currentMethodScope.enclosingSourceType();
696 
697        // use 'this' if possible
698        if (!currentMethodScope.isStatic && !currentMethodScope.isConstructorCall) {
699                if (sourceType == targetEnclosingType || (!onlyExactMatch && sourceType.findSuperTypeWithSameErasure(targetEnclosingType) != null)) {
700                        return BlockScope.EmulationPathToImplicitThis; // implicit this is good enough
701                }
702        }
703        if (!sourceType.isNestedType() || sourceType.isStatic()) { // no emulation from within non-inner types
704                if (currentMethodScope.isConstructorCall) {
705                        return BlockScope.NoEnclosingInstanceInConstructorCall;
706                } else if (currentMethodScope.isStatic){
707                        return BlockScope.NoEnclosingInstanceInStaticContext;
708                }
709                return null;
710        }
711        boolean insideConstructor = currentMethodScope.isInsideInitializerOrConstructor();
712        // use synthetic constructor arguments if possible
713        if (insideConstructor) {
714                SyntheticArgumentBinding syntheticArg;
715                if ((syntheticArg = ((NestedTypeBinding) sourceType).getSyntheticArgument(targetEnclosingType, onlyExactMatch)) != null) {
716                        // reject allocation and super constructor call
717                        if (denyEnclosingArgInConstructorCall
718                                        && currentMethodScope.isConstructorCall
719                                        && (sourceType == targetEnclosingType || (!onlyExactMatch && sourceType.findSuperTypeWithSameErasure(targetEnclosingType) != null))) {
720                                return BlockScope.NoEnclosingInstanceInConstructorCall;
721                        }
722                        return new Object[] { syntheticArg };
723                }
724        }
725 
726        // use a direct synthetic field then
727        if (currentMethodScope.isStatic) {
728                return BlockScope.NoEnclosingInstanceInStaticContext;
729        }
730        if (sourceType.isAnonymousType()) {
731                ReferenceBinding enclosingType = sourceType.enclosingType();
732                if (enclosingType.isNestedType()) {
733                        NestedTypeBinding nestedEnclosingType = (NestedTypeBinding) enclosingType;
734                        SyntheticArgumentBinding enclosingArgument = nestedEnclosingType.getSyntheticArgument(nestedEnclosingType.enclosingType(), onlyExactMatch);
735                        if (enclosingArgument != null) {
736                                FieldBinding syntheticField = sourceType.getSyntheticField(enclosingArgument);
737                                if (syntheticField != null) {
738                                        if (syntheticField.type == targetEnclosingType || (!onlyExactMatch && ((ReferenceBinding)syntheticField.type).findSuperTypeWithSameErasure(targetEnclosingType) != null))
739                                                return new Object[] { syntheticField };
740                                }
741                        }
742                }
743        }
744        FieldBinding syntheticField = sourceType.getSyntheticField(targetEnclosingType, onlyExactMatch);
745        if (syntheticField != null) {
746                if (currentMethodScope.isConstructorCall){
747                        return BlockScope.NoEnclosingInstanceInConstructorCall;
748                }
749                return new Object[] { syntheticField };
750        }
751 
752        // could be reached through a sequence of enclosing instance link (nested members)
753        Object[] path = new Object[2]; // probably at least 2 of them
754        ReferenceBinding currentType = sourceType.enclosingType();
755        if (insideConstructor) {
756                path[0] = ((NestedTypeBinding) sourceType).getSyntheticArgument(currentType, onlyExactMatch);
757        } else {
758                if (currentMethodScope.isConstructorCall){
759                        return BlockScope.NoEnclosingInstanceInConstructorCall;
760                }

762        }
763        if (path[0] != null) { // keep accumulating

767                while ((currentEnclosingType = currentType.enclosingType()) != null) {
768 
769                        //done?

771                                || (!onlyExactMatch && currentType.findSuperTypeWithSameErasure(targetEnclosingType) != null))        break;
772 
773                        if (currentMethodScope != null) {
774                                currentMethodScope = currentMethodScope.enclosingMethodScope();
775                                if (currentMethodScope != null && currentMethodScope.isConstructorCall){
776                                        return BlockScope.NoEnclosingInstanceInConstructorCall;
777                                }
778                                if (currentMethodScope != null && currentMethodScope.isStatic){
779                                        return BlockScope.NoEnclosingInstanceInStaticContext;

784                        if (syntheticField == null) break;

787                        if (count == path.length) {
788                                System.arraycopy(path, 0, (path = new Object[count + 1]), 0, count);
789                        }
790                        // private access emulation is necessary since synthetic field is private
791                        path[count++] = ((SourceTypeBinding) syntheticField.declaringClass).addSyntheticMethod(syntheticField, true);
792                        currentType = currentEnclosingType;
