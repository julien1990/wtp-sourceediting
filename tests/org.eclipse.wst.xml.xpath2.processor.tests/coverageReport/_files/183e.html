384        /**
385         * Hooks up the viewer to follow the selection made in the active editor
386         */
387        private void followSelection() {
388                ITextEditor editor = getActiveEditor();
389                if (editor != null) {
390                        setupViewerForEditor(editor);
391                        if (fHighlightRangeListener == null)
392                                fHighlightRangeListener = new NodeRangeSelectionListener();
393 
394                        fContentOutlinePage = ((IContentOutlinePage) editor.getAdapter(IContentOutlinePage.class));
395                        if (fContentOutlinePage != null) {
396                                fContentOutlinePage.addSelectionChangedListener(fHighlightRangeListener);
397 
398                                if (!fContentOutlinePage.getSelection().isEmpty() && fContentOutlinePage.getSelection() instanceof IStructuredSelection) {
399                                        fSourceViewer.resetVisibleRegion();
400 
401                                        Object[] nodes = ((IStructuredSelection) fContentOutlinePage.getSelection()).toArray();
402                                        IndexedRegion startNode = (IndexedRegion) nodes[0];
403                                        IndexedRegion endNode = (IndexedRegion) nodes[nodes.length - 1];
404 
405                                        if (startNode instanceof Attr)
406                                                startNode = (IndexedRegion) ((Attr) startNode).getOwnerElement();
407                                        if (endNode instanceof Attr)
408                                                endNode = (IndexedRegion) ((Attr) endNode).getOwnerElement();
409 
410                                        int start = startNode.getStartOffset();
411                                        int end = endNode.getEndOffset();
412 
413                                        fSourceViewer.setVisibleRegion(start, end - start);
414                                        fSourceViewer.setSelectedRange(start, 0);
415                                }
416                        }
417                }
418        }
