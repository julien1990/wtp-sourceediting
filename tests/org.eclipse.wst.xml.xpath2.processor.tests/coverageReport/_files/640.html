468   * @return         the index of the last entry
469   */
470  private static int yy_unpack(String packed, int offset, int [] trans) {
471    int i = 0;       /* index in packed string  */
472    int j = offset;  /* index in unpacked array */
473    int l = packed.length();

475      int count = packed.charAt(i++);

477      value--;
478      do trans[j++] = value; while (--count > 0);
479    }
480    return j;
481  }

502  /**
503   * Refills the input buffer.
504   *
505   * @return      <code>false</code>, iff there was new input.
506   * 
507   * @exception   java.io.IOException  if any I/O-Error occurs
508   */
509  private boolean yy_refill() throws java.io.IOException {
510 
511    /* first: make room (if you can) */
512    if (yy_startRead > 0) {
513      System.arraycopy(yy_buffer, yy_startRead, 
514                       yy_buffer, 0, 
515                       yy_endRead-yy_startRead);
516 
517      /* translate stored positions */
518      yy_endRead-= yy_startRead;
519      yy_currentPos-= yy_startRead;
520      yy_markedPos-= yy_startRead;
521      yy_pushbackPos-= yy_startRead;
522      yy_startRead = 0;
523    }
524 
525    /* is the buffer big enough? */
526    if (yy_currentPos >= yy_buffer.length) {
527      /* if not: blow it up */
528      char newBuffer[] = new char[yy_currentPos*2];
529      System.arraycopy(yy_buffer, 0, newBuffer, 0, yy_buffer.length);
530      yy_buffer = newBuffer;
531    }
532 
533    /* finally: fill the buffer with new input */
534    int numRead = yy_reader.read(yy_buffer, yy_endRead, 
535                                            yy_buffer.length-yy_endRead);
536 
537    if (numRead < 0) {
538      return true;
539    }
540    else {
541      yy_endRead+= numRead;  
542      return false;
543    }
544  }
