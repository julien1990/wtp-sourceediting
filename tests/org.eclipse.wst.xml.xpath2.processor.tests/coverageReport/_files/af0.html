1297                } else {

1299                                new ArrayTypeReference(
1300                                        identifierStack[localIdentifierPtr],
1301                                        dim,
1302                                        identifierPositionStack[localIdentifierPtr--]);
1303                        ref.sourceEnd = endPosition;
1304                }
1305        } else {

1314                        }

1317                        localIdentifierPtr -= length;

1320                        System.arraycopy(
1321                                identifierPositionStack,
