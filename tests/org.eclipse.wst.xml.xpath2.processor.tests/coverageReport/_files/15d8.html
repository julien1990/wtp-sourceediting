703                                TextPresentation presentation= new TextPresentation(damage, 1);
704                                presentation.setDefaultStyleRange(new StyleRange(damage.getOffset(), validLength, null, null));
705                                return presentation;
706                        }
707 
708                        TextPresentation presentation= new TextPresentation(damage, 1000);
709 
710                        ITypedRegion[] partitions= TextUtilities.computePartitioning(document, getDocumentPartitioning(), damage.getOffset(), validLength, false);
711                        for (int i= 0; i < partitions.length; i++) {
712                                ITypedRegion r= partitions[i];
713                                IPresentationRepairer repairer= getRepairer(r.getType());
714                                if (repairer != null)
715                                        repairer.createPresentation(presentation, r);

719

721                        /* ignored in platform PresentationReconciler, too */
722                }
723

725        }

728        /**
729         * Checks for the first and the last affected partition affected by a
730         * document event and calls their damagers. Invalidates everything from the
731         * start of the damage for the first partition until the end of the damage
732         * for the last partition.
733         *
734         * @param e the event describing the document change
735         * @param optimize <code>true</code> if partition changes should be
736         *        considered for optimization
737         * @return the damaged caused by the change or <code>null</code> if
738         *         computing the partitioning failed
739         * @since 3.0
740         */
741        IRegion getDamage(DocumentEvent e, boolean optimize) {
742                int length= e.getText() == null ? 0 : e.getText().length();
743                
744                if (fDamagers == null || fDamagers.isEmpty()) {
745                        length= Math.max(e.getLength(), length);
746                        length= Math.min(e.getDocument().getLength() - e.getOffset(), length);
747                        return new Region(e.getOffset(), length);
748                }
749 
750                boolean isDeletion= length == 0;
751                IRegion damage= null;
752                try {
753                        int offset= e.getOffset();
754                        if (isDeletion)
755                                offset= Math.max(0, offset - 1);
756                        ITypedRegion partition= getPartition(e.getDocument(), offset);
757                        IPresentationDamager damager= getDamager(partition.getType());
758                        if (damager == null)
759                                return null;
760 
761                        IRegion r= damager.getDamageRegion(partition, e, fDocumentPartitioningChanged);
762 
763                        if (!fDocumentPartitioningChanged && optimize && !isDeletion) {
764                                damage= r;
765                        } else {
766 
767                                int damageEnd= getDamageEndOffset(e);
768 
769                                int parititionDamageEnd= -1;
770                                if (fChangedDocumentPartitions != null)
771                                        parititionDamageEnd= fChangedDocumentPartitions.getOffset() + fChangedDocumentPartitions.getLength();
772 
773                                int end= Math.max(damageEnd, parititionDamageEnd);
774 
775                                damage= end == -1 ? r : new Region(r.getOffset(), end - r.getOffset());
776                        }
777 
778                } catch (BadLocationException x) {
779                        /* ignored in platform PresentationReconciler, too */

785        /**
786         * Returns the end offset of the damage. If a partition has been split by
787         * the given document event also the second half of the original
788         * partition must be considered. This is achieved by using the remembered
789         * partition range.
790         *
791         * @param e the event describing the change
792         * @return the damage end offset (excluding)
793         * @exception BadLocationException if method accesses invalid offset
794         */
795        int getDamageEndOffset(DocumentEvent e) throws BadLocationException {
796 
797                IDocument d= e.getDocument();
798 
799                int length= 0;
800                if (e.getText() != null) {
801                        length= e.getText().length();
802                        if (length > 0)
803                                -- length;
804                }
805 
806                ITypedRegion partition= getPartition(d, e.getOffset() + length);
807                int endOffset= partition.getOffset() + partition.getLength();
808                if (endOffset == e.getOffset())
809                        return -1;
810 
811                int end= fRememberedPosition == null ? -1 : fRememberedPosition.getOffset() + fRememberedPosition.getLength();
812                if (endOffset < end && end < d.getLength())
813                        partition= getPartition(d, end);
814 
815                IPresentationDamager damager= getDamager(partition.getType());
816                if (damager == null)
817                        return -1;
818 
819                IRegion r= damager.getDamageRegion(partition, e, fDocumentPartitioningChanged);
820 
821                return r.getOffset() + r.getLength();
822        }

843        private boolean isOverlappingRegion(IRegion base, IRegion damage) {
844                if(base == null || damage == null)
845                        return false;
846                
847                int baseEnd = base.getOffset() + base.getLength();
848                int damageEnd = damage.getOffset() + damage.getLength();
849                
850                return damage.getOffset() <= base.getOffset() && (damageEnd >= baseEnd);
851        }
