470MethodBinding computeSubstituteMethod(MethodBinding inheritedMethod, MethodBinding currentMethod) {
471        if (inheritedMethod == null) return null;
472        if (currentMethod.parameters.length != inheritedMethod.parameters.length) return null; // no match
473 
474        // due to hierarchy & compatibility checks, we need to ensure these 2 methods are resolved
475        if (currentMethod.declaringClass instanceof BinaryTypeBinding)
476                ((BinaryTypeBinding) currentMethod.declaringClass).resolveTypesFor(currentMethod);
477        if (inheritedMethod.declaringClass instanceof BinaryTypeBinding)
478                ((BinaryTypeBinding) inheritedMethod.declaringClass).resolveTypesFor(inheritedMethod);
479 
480        TypeVariableBinding[] inheritedTypeVariables = inheritedMethod.typeVariables;
481        if (inheritedTypeVariables == Binding.NO_TYPE_VARIABLES) return inheritedMethod;
482        int inheritedLength = inheritedTypeVariables.length;
483        TypeVariableBinding[] typeVariables = currentMethod.typeVariables;
484        int length = typeVariables.length;
485        if (length > 0 && inheritedLength != length) return inheritedMethod; // no match JLS 8.4.2
486        TypeBinding[] arguments = new TypeBinding[inheritedLength];
487        if (inheritedLength <= length) {
488                System.arraycopy(typeVariables, 0, arguments, 0, inheritedLength);
489        } else {
490                System.arraycopy(typeVariables, 0, arguments, 0, length);
491                for (int i = length; i < inheritedLength; i++)
492                        arguments[i] = inheritedTypeVariables[i].upperBound();
493        }
494        ParameterizedGenericMethodBinding substitute =
495                this.environment.createParameterizedGenericMethod(inheritedMethod, arguments);
496 
497        // interface I { <T> void foo(T t); }
498        // class X implements I { public <T extends I> void foo(T t) {} }
499        // for the above case, we do not want to answer the substitute method since its not a match
500        for (int i = 0; i < inheritedLength; i++) {
501                TypeVariableBinding inheritedTypeVariable = inheritedTypeVariables[i];
502                TypeBinding argument = arguments[i];
503                if (argument instanceof TypeVariableBinding) {
504                        TypeVariableBinding typeVariable = (TypeVariableBinding) argument;
505                        if (typeVariable.firstBound == inheritedTypeVariable.firstBound) {
506                                if (typeVariable.firstBound == null)
507                                        continue; // both are null
508                        } else if (typeVariable.firstBound != null && inheritedTypeVariable.firstBound != null) {
509                                if (typeVariable.firstBound.isClass() != inheritedTypeVariable.firstBound.isClass())
510                                        return inheritedMethod; // not a match
511                        }
512                        if (Scope.substitute(substitute, inheritedTypeVariable.superclass) != typeVariable.superclass)
513                                return inheritedMethod; // not a match
514                        int interfaceLength = inheritedTypeVariable.superInterfaces.length;
515                        ReferenceBinding[] interfaces = typeVariable.superInterfaces;
516                        if (interfaceLength != interfaces.length)
517                                return inheritedMethod; // not a match
518                        // TODO (kent) another place where we expect the superinterfaces to be in the exact same order
519                        next : for (int j = 0; j < interfaceLength; j++) {
520                                TypeBinding superType = Scope.substitute(substitute, inheritedTypeVariable.superInterfaces[j]);
521                                for (int k = 0; k < interfaceLength; k++)
522                                        if (superType == interfaces[k])
523                                                continue next;
524                                return inheritedMethod; // not a match
525                        }
526                } else if (inheritedTypeVariable.boundCheck(substitute, argument) != TypeConstants.OK) {
527                            return inheritedMethod;
528                }
529        }
530   return substitute;

550boolean doesSubstituteMethodOverride(MethodBinding method, MethodBinding substituteMethod) {
551        if (doTypeVariablesClash(method, substituteMethod)) return false;
552        if (areParametersEqual(method, substituteMethod)) return true;
553        if (method.declaringClass == substituteMethod.declaringClass) return false;
554 
555        TypeBinding[] params = method.parameters;
556        TypeBinding[] inheritedParams = substituteMethod.parameters;

558        if (length != inheritedParams.length)
559                return false;

562        if (method.typeVariables != Binding.NO_TYPE_VARIABLES || !substituteMethod.hasSubstitutedParameters())
563                return false;
564 
565        for (int i = 0; i < length; i++) {
566                if (inheritedParams[i].kind() == Binding.TYPE_PARAMETER) {
567                        if (params[i] != ((TypeVariableBinding) inheritedParams[i]).upperBound())
568                                return false;
569                } else if (params[i] != inheritedParams[i]) {
570                        return false;
571                }
572        }
573        return true;

589SimpleSet findSuperinterfaceCollisions(ReferenceBinding superclass, ReferenceBinding[] superInterfaces) {

615                                }
616                        }
617                }
618                superType = superType.superclass();
619        }
620 
621        for (int i = 0; i < nextPosition; i++) {

626                                int itsLength = itsInterfaces.length;
