780protected IType resolveType(char[] packageName, char[] fileName, char[] typeName, int acceptFlags) {
781 
782        IType type= null;
783 
784        if (fileName!=null)
785        {
786                ITypeRoot compilationUnit = (ITypeRoot)resolveCompilationUnit(packageName, fileName);
787                if (compilationUnit!=null && compilationUnit.exists())
788                {
789                         String fulltypeName=null;
790                         if (packageName==null || packageName.length==0)
791                            fulltypeName=new String(typeName);
792                         else {
793                                fulltypeName=new String(CharOperation.concat(packageName, typeName, '.'));
794                        }
795                         type=compilationUnit.getType(fulltypeName);
796                         if (type!=null && type.exists())
797                                 return type;
798                }
799                
800        }
801        
802        if (this.openable instanceof CompilationUnit && ((CompilationUnit)this.openable).isWorkingCopy()) {
803                CompilationUnit wc = (CompilationUnit) this.openable;
804                try {
805                        if(((packageName == null || packageName.length == 0) && wc.getPackageDeclarations().length == 0) ||
806                                (!(packageName == null || packageName.length == 0) && wc.getPackageDeclaration(new String(packageName)).exists())) {
807 
808                                char[][] compoundName = CharOperation.splitOn('.', typeName);
809                                if(compoundName.length > 0) {
810                                        type = wc.getType(new String(compoundName[0]));
811                                        for (int i = 1, length = compoundName.length; i < length; i++) {
812                                                type = type.getType(new String(compoundName[i]));
813                                        }
814                                }
815 
816                                if(type != null && !type.exists()) {
817                                        type = null;
818                                }
819                        }
820                }catch (JavaScriptModelException e) {
821                        // type is null
822                }
823        }
824 
825        if(type == null) {
826                IPackageFragment[] pkgs = this.nameLookup.findPackageFragments(
827                        (packageName == null || packageName.length == 0) ? IPackageFragment.DEFAULT_PACKAGE_NAME : new String(packageName),
828                        false);
829                // iterate type lookup in each package fragment
830                for (int i = 0, length = pkgs == null ? 0 : pkgs.length; i < length; i++) {
831                        type= this.nameLookup.findType(new String(typeName), pkgs[i], false, acceptFlags, true/*consider secondary types*/);
832                        if (type != null) break;
833                }
834                if (type == null) {
835                        String pName= IPackageFragment.DEFAULT_PACKAGE_NAME;
836                        if (packageName != null) {
837                                pName = new String(packageName);
838                        }
839                        if (this.openable != null && this.openable.getParent().getElementName().equals(pName)) {
840                                // look inside the type in which we are resolving in
841                                String tName= new String(typeName);
842                                tName = tName.replace('.','$');
843                                IType[] allTypes= null;
844                                try {
845                                        ArrayList list = this.openable.getChildrenOfType(IJavaScriptElement.TYPE);
846                                        allTypes = new IType[list.size()];
847                                        list.toArray(allTypes);
848                                } catch (JavaScriptModelException e) {
849                                        return null;
850                                }
851                                for (int i= 0; i < allTypes.length; i++) {
852                                        if (allTypes[i].getTypeQualifiedName().equals(tName)) {
853                                                return allTypes[i];
854                                        }
855                                }
856                        }
857                }
858        }
859        return type;
860}
