420/**
421 * Find the set of candidate subtypes of a given type.
422 *
423 * The requestor is notified of super type references (with actual path of
424 * its occurrence) for all types which are potentially involved inside a particular
425 * hierarchy.
426 * The match locator is not used here to narrow down the results, the type hierarchy
427 * resolver is rather used to compute the whole hierarchy at once.
428 * @param type
429 * @param scope
430 * @param binariesFromIndexMatches
431 * @param pathRequestor
432 * @param waitingPolicy
433 * @param progressMonitor
434 */
435public static void searchAllPossibleSubTypes(
436        IType type,
437        IJavaScriptSearchScope scope,
438        final Map binariesFromIndexMatches,
439        final IPathRequestor pathRequestor,
440        int waitingPolicy,        // WaitUntilReadyToSearch | ForceImmediateSearch | CancelIfNotReadyToSearch
441        final IProgressMonitor progressMonitor) {
442 
443        /* embed constructs inside arrays so as to pass them to (inner) collector */
444        final Queue queue = new Queue();
445        final HashtableOfObject foundSuperNames = new HashtableOfObject(5);
446 
447        IndexManager indexManager = JavaModelManager.getJavaModelManager().getIndexManager();
448 
449        /* use a special collector to collect paths and queue new subtype names */
450        IndexQueryRequestor searchRequestor = new IndexQueryRequestor() {
451                public boolean acceptIndexMatch(String documentPath, SearchPattern indexRecord, SearchParticipant participant, AccessRuleSet access) {
452                        SuperTypeReferencePattern record = (SuperTypeReferencePattern)indexRecord;
453                        boolean isLocalOrAnonymous = record.enclosingTypeName == IIndexConstants.ONE_ZERO;
454                        pathRequestor.acceptPath(documentPath, isLocalOrAnonymous);
455                        char[] typeName = record.simpleName;
456                        int suffix = documentPath.toLowerCase().lastIndexOf(SUFFIX_STRING_java);
457                        if (suffix != -1){
458                                HierarchyBinaryType binaryType = (HierarchyBinaryType)binariesFromIndexMatches.get(documentPath);
459                                if (binaryType == null){
460                                        char[] enclosingTypeName = record.enclosingTypeName;
461                                        if (isLocalOrAnonymous) {
462                                                int lastSlash = documentPath.lastIndexOf('/');
463                                                int lastDollar = documentPath.lastIndexOf('$');
464                                                if (lastDollar == -1) {
465                                                        // malformed local or anonymous type: it doesn't contain a $ in its name
466                                                        // treat it as a top level type
467                                                        enclosingTypeName = null;
468                                                        typeName = documentPath.substring(lastSlash+1, suffix).toCharArray();
469                                                } else {
470                                                        enclosingTypeName = documentPath.substring(lastSlash+1, lastDollar).toCharArray();
471                                                        typeName = documentPath.substring(lastDollar+1, suffix).toCharArray();
472                                                }
473                                        }
474                                        binaryType = new HierarchyBinaryType(record.modifiers, record.pkgName, typeName, enclosingTypeName, record.typeParameterSignatures, record.classOrInterface);
475                                        binariesFromIndexMatches.put(documentPath, binaryType);
476                                }
477                                binaryType.recordSuperType(record.superSimpleName, record.superQualification, record.superClassOrInterface);
478                        }
479                        if (!isLocalOrAnonymous // local or anonymous types cannot have subtypes outside the cu that define them
480                                        && !foundSuperNames.containsKey(typeName)){
481                                foundSuperNames.put(typeName, typeName);
482                                queue.add(typeName);
483                        }
484                        return true;
485                }
486        };
487 
488        int superRefKind;
489        try {
490                superRefKind = type.isClass() ? SuperTypeReferencePattern.ONLY_SUPER_CLASSES : SuperTypeReferencePattern.ALL_SUPER_TYPES;
491        } catch (JavaScriptModelException e) {
492                superRefKind = SuperTypeReferencePattern.ALL_SUPER_TYPES;
493        }
494        SuperTypeReferencePattern pattern =
495                new SuperTypeReferencePattern(null, null, superRefKind, SearchPattern.R_EXACT_MATCH | SearchPattern.R_CASE_SENSITIVE);
496        MatchLocator.setFocus(pattern, type);
497        SubTypeSearchJob job = new SubTypeSearchJob(
498                pattern,
499                new JavaSearchParticipant(), // java search only
500                scope,
501                searchRequestor);
502 
503        int ticks = 0;
504        queue.add(type.getElementName().toCharArray());
505        try {
506                while (queue.start <= queue.end) {
507                        if (progressMonitor != null && progressMonitor.isCanceled()) return;
508 
509                        // all subclasses of OBJECT are actually all types
510                        char[] currentTypeName = queue.retrieve();
511                        if (CharOperation.equals(currentTypeName, IIndexConstants.OBJECT))
512                                currentTypeName = null;
513 
514                        // search all index references to a given supertype
515                        pattern.superSimpleName = currentTypeName;
516                        indexManager.performConcurrentJob(job, waitingPolicy, progressMonitor == null ? null : new NullProgressMonitor() {
517                                // don't report progress since this is too costly for deep hierarchies
518                                // just handle isCanceled() (seehttps://bugs.eclipse.org/bugs/show_bug.cgi?id=179511)
519                                public void setCanceled(boolean value) {
520                                        progressMonitor.setCanceled(value);
521                                }
522                                public boolean isCanceled() {
523                                        return progressMonitor.isCanceled();
524                                }
525                        });
526                        if (progressMonitor != null && ++ticks <= MAXTICKS)
527                                progressMonitor.worked(1);
528 
529                        // in case, we search all subtypes, no need to search further
530                        if (currentTypeName == null) break;
531                }
532        } finally {
533                job.finished();
534        }
535}
