507public IJavaScriptElement getHandleFromMemento(String token, MementoTokenizer memento, WorkingCopyOwner owner) {
508        switch (token.charAt(0)) {
509                case JEM_PACKAGEFRAGMENT:
510                        String pkgName;
511                        if (memento.hasMoreTokens()) {
512                                pkgName = memento.nextToken();
513                                char firstChar = pkgName.charAt(0);
514                                if (firstChar == JEM_CLASSFILE || firstChar == JEM_COMPILATIONUNIT || firstChar == JEM_COUNT) {
515                                        token = pkgName;
516                                        pkgName = IPackageFragment.DEFAULT_PACKAGE_NAME;
517                                } else {
518                                        token = null;
519                                }
520                        } else {
521                                pkgName = IPackageFragment.DEFAULT_PACKAGE_NAME;
522                                token = null;
523                        }
524                        JavaElement pkg = (JavaElement)getPackageFragment(pkgName);
525                        if (token == null) {
526                                return pkg.getHandleFromMemento(memento, owner);
527                        } else {
528                                return pkg.getHandleFromMemento(token, memento, owner);
529                        }
530        }
531        return null;
532}

653public IPath getSourceAttachmentPath() throws JavaScriptModelException {
654        if (getKind() != K_BINARY) return null;
655 
656        // 1) look source attachment property (set iff attachSource(...) was called
657        IPath path = getPath();

659        if (serverPathString != null) {
660                int index= serverPathString.lastIndexOf(ATTACHMENT_PROPERTY_DELIMITER);
661                if (index < 0) {
662                        // no root path specified
663                        return new Path(serverPathString);
664                } else {
665                        String serverSourcePathString= serverPathString.substring(0, index);
666                        return new Path(serverSourcePathString);
667                }
668        }
669 
670        // 2) look at classpath entry
671        IIncludePathEntry entry = ((JavaProject) getParent()).getClasspathEntryFor(path);
672        IPath sourceAttachmentPath;
673        if (entry != null && (sourceAttachmentPath = entry.getSourceAttachmentPath()) != null)
674                return sourceAttachmentPath;
675 
676        // 3) look for a recommendation
677        entry = findSourceAttachmentRecommendation();
678        if (entry != null && (sourceAttachmentPath = entry.getSourceAttachmentPath()) != null) {
679                return sourceAttachmentPath;
680        }
681 
682        return null;
683}
