751        private void handleUpdateSelection(Map items) {
752                Iterator keyIterator= items.keySet().iterator();
753 
754                //Update the store before the hierarchy to prevent updating parents before all of the children are done
755                while (keyIterator.hasNext()) {
756                        Object key= keyIterator.next();
757                        //Replace the items in the checked state store with those from the supplied items
758                        List selections= (List) items.get(key);
759                        if (selections.size() == 0)
760                                //If it is empty remove it from the list
761                                fCheckedStateStore.remove(key);
762                        else {
763                                fCheckedStateStore.put(key, selections);
764                                // proceed up the tree element hierarchy
765                                Object parent= fTreeContentProvider.getParent(key);
766                                if (parent != null) {
767                                        addToHierarchyToCheckedStore(parent);
768                                }
769                        }
770                }
771 
772                //Now update hierarchies
773                keyIterator= items.keySet().iterator();
774 
775                while (keyIterator.hasNext()) {
776                        Object key= keyIterator.next();
777                        updateHierarchy(key);
778                        if (fCurrentTreeSelection != null && fCurrentTreeSelection.equals(key)) {
779                                fListViewer.setAllChecked(false);
780                                fListViewer.setCheckedElements(((List) items.get(key)).toArray());
781                        }
782                }
783        }
