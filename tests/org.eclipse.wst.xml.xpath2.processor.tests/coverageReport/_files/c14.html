456public IJavaScriptElement[] findElements(IJavaScriptElement element) {
457        ArrayList children = new ArrayList();
458        while (element != null && element.getElementType() != IJavaScriptElement.JAVASCRIPT_UNIT) {
459                children.add(element);
460                element = element.getParent();
461        }
462        if (element == null) return null;
463        IJavaScriptElement currentElement = this;
464        for (int i = children.size()-1; i >= 0; i--) {

467                        case IJavaScriptElement.PACKAGE_DECLARATION:

469                                break;
470                        case IJavaScriptElement.IMPORT_CONTAINER:

472                                break;

474                                currentElement = ((IImportContainer)currentElement).getImport(child.getElementName());
475                                break;
476                        case IJavaScriptElement.TYPE:
477                                switch (currentElement.getElementType()) {
478                                        case IJavaScriptElement.JAVASCRIPT_UNIT:
479                                                currentElement = ((IJavaScriptUnit)currentElement).getType(child.getElementName());
480                                                break;
481                                        case IJavaScriptElement.TYPE:
482                                                currentElement = ((IType)currentElement).getType(child.getElementName());
483                                                break;
484                                        case IJavaScriptElement.FIELD:
485                                        case IJavaScriptElement.INITIALIZER:
486                                        case IJavaScriptElement.METHOD:
487                                                currentElement =  ((IMember)currentElement).getType(child.getElementName(), child.occurrenceCount);
488                                                break;
489                                }
490                                break;
491                        case IJavaScriptElement.INITIALIZER:
492                                currentElement = ((IType)currentElement).getInitializer(child.occurrenceCount);
493                                break;
494                        case IJavaScriptElement.FIELD:

508                                break;
509                }
510 
511        }

513                return new IJavaScriptElement[] {currentElement};
514        } else {
515                return null;

663public IJavaScriptElement getHandleFromMemento(String token, MementoTokenizer memento, WorkingCopyOwner workingCopyOwner) {
664        switch (token.charAt(0)) {
665                case JEM_IMPORTDECLARATION:
666                        JavaElement container = (JavaElement)getImportContainer();
667                        return container.getHandleFromMemento(token, memento, workingCopyOwner);
668                case JEM_PACKAGEDECLARATION:
669                        if (!memento.hasMoreTokens()) return this;
670                        String pkgName = memento.nextToken();
671                        JavaElement pkgDecl = (JavaElement)getPackageDeclaration(pkgName);
672                        return pkgDecl.getHandleFromMemento(memento, workingCopyOwner);
673                case JEM_TYPE:
674                        if (!memento.hasMoreTokens()) return this;
675                        String typeName = memento.nextToken();
676                        JavaElement type = (JavaElement)getType(typeName);
677                        return type.getHandleFromMemento(memento, workingCopyOwner);
678 
679 
680                case JEM_FIELD:
681                        if (!memento.hasMoreTokens()) return this;
682                        String fieldName = memento.nextToken();
683                        JavaElement field = (JavaElement)getField(fieldName);
684                        return field.getHandleFromMemento(memento, workingCopyOwner);
685 
686                case JEM_METHOD:
687                        if (!memento.hasMoreTokens()) return this;
688                        String selector = memento.nextToken();
689                        ArrayList params = new ArrayList();
690                        nextParam: while (memento.hasMoreTokens()) {
691                                token = memento.nextToken();
692                                switch (token.charAt(0)) {
693                                        case JEM_TYPE:
694                                        case JEM_TYPE_PARAMETER:
695                                                break nextParam;
696                                        case JEM_METHOD:
697                                                if (!memento.hasMoreTokens()) return this;
698                                                String param = memento.nextToken();
699                                                StringBuffer buffer = new StringBuffer();
700                                                while (param.length() == 1 && Signature.C_ARRAY == param.charAt(0)) { // backward compatible with 3.0 mementos
701                                                        buffer.append(Signature.C_ARRAY);
702                                                        if (!memento.hasMoreTokens()) return this;
703                                                        param = memento.nextToken();
704                                                }
705                                                params.add(buffer.toString() + param);
706                                                break;
707                                        default:
708                                                break nextParam;
709                                }
710                        }
711                        String[] parameters = new String[params.size()];
712                        params.toArray(parameters);
713                        JavaElement method = (JavaElement)getFunction(selector, parameters);
714                        switch (token.charAt(0)) {
715                                case JEM_TYPE:
716                                case JEM_TYPE_PARAMETER:
717                                case JEM_LOCALVARIABLE:
718                                        return method.getHandleFromMemento(token, memento, workingCopyOwner);
719                                default:
720                                        return method;
721                        }
722 
723        }
724        return null;

1103                        : BufferManager.createBuffer(this);

1107        // see https://bugs.eclipse.org/bugs/show_bug.cgi?id=146331
1108        synchronized(bufManager) {
1109                IBuffer existingBuffer = bufManager.getBuffer(this);
1110                if (existingBuffer != null)

1114                if (buffer.getCharacters() == null) {
1115                        if (isWorkingCopy) {

1118                                                && (original = new CompilationUnit((PackageFragment)getParent(), getElementName(), DefaultWorkingCopyOwner.PRIMARY)).isOpen()) {
1119                                        buffer.setContents(original.getSource());

1122                                        if (file == null || !file.exists()) {
1123                                                // initialize buffer with empty contents
1124                                                buffer.setContents(CharOperation.NO_CHAR);
1125                                        } else {
1126                                                buffer.setContents(Util.getResourceContentsAsCharArray(file));
1127                                        }
1128                                }
1129                        } else {
1130                                IFile file = (IFile)this.getResource();
1131                                if (file == null || !file.exists()) throw newNotPresentException();
1132                                buffer.setContents(Util.getResourceContentsAsCharArray(file));
1133                        }
1134                }
1135 
1136                // add buffer to buffer cache
1137                // note this may cause existing buffers to be removed from the buffer cache, but only primary compilation unit's buffer
1138                // can be closed, thus no call to a client's IBuffer#close() can be done in this synchronized block.
