350        private static class SimilarElementComparator extends JavaScriptElementComparator {
351 
352                /*
353                 * (non-Javadoc)
354                 * @see org.eclipse.wst.jsdt.ui.JavaElementSorter#category(java.lang.Object)
355                 */
356                public int category(Object element) {
357 
358                        /*
359                         * We'd like to present the elements in the same order as they
360                         * appear in the source. This can be achieved by assigning a
361                         * distinct category to every element; the category being derived
362                         * from the source position of the element.
363                         */
364 
365                        ISourceRange sourceRange= null;
366                        if (element instanceof IMember) {
367                                IMember member= (IMember) element;
368                                try {
369                                        sourceRange= member.getNameRange();
370                                } catch (JavaScriptModelException e) {
371                                        // fall through
372                                }
373                        }
374                        if (element instanceof ILocalVariable) {
375                                ILocalVariable var= (ILocalVariable) element;
376                                sourceRange= var.getNameRange();
377                        }
378 
379                        if (sourceRange != null)
380                                return 100 + sourceRange.getOffset(); // +100: safe distance from all other categories.
381 
382                        return super.category(element);
383                }
