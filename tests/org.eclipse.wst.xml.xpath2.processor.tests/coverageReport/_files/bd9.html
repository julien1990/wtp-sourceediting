356                int argumentCount = argumentTypeSignatures == null ? 0 : argumentTypeSignatures.length;
357                if (argumentCount > 0) {
358                        long position = ((long) start << 32) + end;
359                        method.arguments = new Argument[argumentCount];
360                        for (int i = 0; i < argumentCount; i++) {
361                                TypeReference typeReference = createTypeReference(argumentTypeSignatures[i], start, end);
362                                if (isVarargs && i == argumentCount-1) {
363                                        typeReference.bits |= ASTNode.IsVarArgs;
364                                }
365                                method.arguments[i] =
366                                        new Argument(
367                                                argumentNames[i],
368                                                position,
369                                                typeReference,
370                                                ClassFileConstants.AccDefault);
371                                // do not care whether was final or not
372                        }
373                }
374 
375                /* convert thrown exceptions */
376                char[][] exceptionTypeNames = methodInfo.getExceptionTypeNames();
377                int exceptionCount = exceptionTypeNames == null ? 0 : exceptionTypeNames.length;
378                if (exceptionCount > 0) {
379                        method.thrownExceptions = new TypeReference[exceptionCount];
380                        for (int i = 0; i < exceptionCount; i++) {
381                                method.thrownExceptions[i] =
382                                        createTypeReference(exceptionTypeNames[i], start, end);
383                        }
384                }
385 
386                /* convert local and anonymous types */
387                if ((this.flags & LOCAL_TYPE) != 0) {
388                        IJavaScriptElement[] children = methodInfo.getChildren();
389                        int typesLength = children.length;
390                        if (typesLength != 0) {
391                                Statement[] statements = new Statement[typesLength];
392                                for (int i = 0; i < typesLength; i++) {
393                                        SourceType type = (SourceType) children[i];
394                                        TypeDeclaration localType = convert(type, compilationResult);
395                                        if ((localType.bits & ASTNode.IsAnonymousType) != 0) {
396                                                QualifiedAllocationExpression expression = new QualifiedAllocationExpression(localType);
397                                                expression.type = localType.superclass;
398                                                localType.superclass = null;
399                                                localType.superInterfaces = null;
400                                                localType.allocation = expression;
401                                                statements[i] = expression;
402                                        } else {
403                                                statements[i] = localType;
404                                        }
405                                }
406                                method.statements = statements;
407                        }
408                }
409 
410                return method;
411        }

413        /*
414         * Convert a source element type into a parsed type declaration
415         */
416        private TypeDeclaration convert(SourceType typeHandle, CompilationResult compilationResult) throws JavaScriptModelException {
417                SourceTypeElementInfo typeInfo = (SourceTypeElementInfo) typeHandle.getElementInfo();
418                if (typeInfo.isAnonymousMember())
419                        throw new AnonymousMemberFound();
420                /* create type declaration - can be member type */
421                TypeDeclaration type = new TypeDeclaration(compilationResult);
422                if (typeInfo.getEnclosingType() == null) {
423                        if (typeHandle.isAnonymous()) {
424                                type.name = CharOperation.NO_CHAR;
425                                type.bits |= (ASTNode.IsAnonymousType|ASTNode.IsLocalType);
426                        } else {
427                                if (typeHandle.isLocal()) {
428                                        type.bits |= ASTNode.IsLocalType;
429                                }
430                        }
431                }  else {
432                        type.bits |= ASTNode.IsMemberType;
433                }
434                if ((type.bits & ASTNode.IsAnonymousType) == 0) {
435                        type.name = typeInfo.getName();
436                }
437                type.name = typeInfo.getName();
438                int start, end; // only positions available
439                type.sourceStart = start = typeInfo.getNameSourceStart();
440                type.sourceEnd = end = typeInfo.getNameSourceEnd();
441                type.modifiers = typeInfo.getModifiers();
442                type.declarationSourceStart = typeInfo.getDeclarationSourceStart();
443                type.declarationSourceEnd = typeInfo.getDeclarationSourceEnd();
444                type.bodyEnd = type.declarationSourceEnd;
445 
446                /* set superclass and superinterfaces */
447                if (typeInfo.getSuperclassName() != null) {
448                        type.superclass = createTypeReference(typeInfo.getSuperclassName(), start, end);
449                        type.superclass.bits |= ASTNode.IsSuperType;
450                }
451                char[][] interfaceNames = typeInfo.getInterfaceNames();
452                int interfaceCount = interfaceNames == null ? 0 : interfaceNames.length;
453                if (interfaceCount > 0) {
454                        type.superInterfaces = new TypeReference[interfaceCount];
455                        for (int i = 0; i < interfaceCount; i++) {
456                                type.superInterfaces[i] = createTypeReference(interfaceNames[i], start, end);
457                                type.superInterfaces[i].bits |= ASTNode.IsSuperType;
458                        }
459                }
460                /* convert member types */
461                if ((this.flags & MEMBER_TYPE) != 0) {
462                        SourceType[] sourceMemberTypes = typeInfo.getMemberTypeHandles();
463                        int sourceMemberTypeCount = sourceMemberTypes.length;
464                        type.memberTypes = new TypeDeclaration[sourceMemberTypeCount];
465                        for (int i = 0; i < sourceMemberTypeCount; i++) {
466                                type.memberTypes[i] = convert(sourceMemberTypes[i], compilationResult);
467                        }
468                }
469 
470                /* convert intializers and fields*/
471                InitializerElementInfo[] initializers = null;
472                int initializerCount = 0;
473                if ((this.flags & LOCAL_TYPE) != 0) {
474                        initializers = typeInfo.getInitializers();
475                        initializerCount = initializers.length;
476                }
477                SourceField[] sourceFields = null;
478                int sourceFieldCount = 0;
479                if ((this.flags & FIELD) != 0) {
480                        sourceFields = typeInfo.getFieldHandles();
481                        sourceFieldCount = sourceFields.length;
482                }
483                int length = initializerCount + sourceFieldCount;
484                if (length > 0) {
485                        type.fields = new FieldDeclaration[length];
486                        for (int i = 0; i < initializerCount; i++) {
487                                type.fields[i] = convert(initializers[i], compilationResult);
488                        }
489                        int index = 0;
490                        for (int i = initializerCount; i < length; i++) {
491                                type.fields[i] = convert(sourceFields[index++], type, compilationResult);
492                        }
493                }
494 
495                /* convert methods - need to add default constructor if necessary */
496                boolean needConstructor = (this.flags & CONSTRUCTOR) != 0;
497                boolean needMethod = (this.flags & METHOD) != 0;
498                if (needConstructor || needMethod) {
499 
500                        SourceMethod[] sourceMethods = typeInfo.getMethodHandles();
501                        int sourceMethodCount = sourceMethods.length;
502 
503                        /* source type has a constructor ?           */
504                        /* by default, we assume that one is needed. */
505                        int extraConstructor = 0;
506                        int methodCount = 0;
507                        int kind = TypeDeclaration.kind(type.modifiers);
508                        boolean isAbstract = kind == TypeDeclaration.INTERFACE_DECL || kind == TypeDeclaration.ANNOTATION_TYPE_DECL;
509                        if (!isAbstract) {
510                                extraConstructor = needConstructor ? 1 : 0;
511                                for (int i = 0; i < sourceMethodCount; i++) {
512                                        if (sourceMethods[i].isConstructor()) {
513                                                if (needConstructor) {
514                                                        extraConstructor = 0; // Does not need the extra constructor since one constructor already exists.
515                                                        methodCount++;
516                                                }
517                                        } else if (needMethod) {
518                                                methodCount++;
519                                        }
520                                }
521                        } else {
522                                methodCount = needMethod ? sourceMethodCount : 0;
523                        }
524                        type.methods = new AbstractMethodDeclaration[methodCount + extraConstructor];
525                        if (extraConstructor != 0) { // add default constructor in first position
526                                type.methods[0] = type.createDefaultConstructor(false, false);
527                        }
528                        int index = 0;
529                        boolean hasAbstractMethods = false;
530                        for (int i = 0; i < sourceMethodCount; i++) {
531                                SourceMethod sourceMethod = sourceMethods[i];
532                                SourceMethodElementInfo methodInfo = (SourceMethodElementInfo)sourceMethod.getElementInfo();
533                                boolean isConstructor = methodInfo.isConstructor();
534                                if ((methodInfo.getModifiers() & ClassFileConstants.AccAbstract) != 0) {
535                                        hasAbstractMethods = true;
536                                }
537                                if ((isConstructor && needConstructor) || (!isConstructor && needMethod)) {
538                                        AbstractMethodDeclaration method = convert(sourceMethod, methodInfo, compilationResult);
539                                        if (isAbstract || method.isAbstract()) { // fix-up flag
540                                                method.modifiers |= ExtraCompilerModifiers.AccSemicolonBody;
541                                        }
542                                        type.methods[extraConstructor + index++] = method;
543                                }
544                        }
545                        if (hasAbstractMethods) type.bits |= ASTNode.HasAbstractMethods;
546                }
547 
548                return type;
549        }

619                this.namePos = 0;

621        }
622 
623        private TypeReference decodeType(String typeSignature, int length, int start, int end) {
624                int identCount = 1;
625                int dim = 0;
626                int nameFragmentStart = this.namePos, nameFragmentEnd = -1;
627                boolean nameStarted = false;
628                ArrayList fragments = null;
629                typeLoop: while (this.namePos < length) {
630                        char currentChar = typeSignature.charAt(this.namePos);
631                        switch (currentChar) {
632                                case Signature.C_BOOLEAN :
633                                        if (!nameStarted) {
634                                                this.namePos++;
635                                                if (dim == 0)
636                                                        return new SingleTypeReference(TypeBinding.BOOLEAN.simpleName, ((long) start << 32) + end);
637                                                else
638                                                        return new ArrayTypeReference(TypeBinding.BOOLEAN.simpleName, dim, ((long) start << 32) + end);
639                                        }
640                                        break;
641                                case Signature.C_BYTE :
642                                        if (!nameStarted) {
643                                                this.namePos++;
644                                                if (dim == 0)
645                                                        return new SingleTypeReference(TypeBinding.BYTE.simpleName, ((long) start << 32) + end);
646                                                else
647                                                        return new ArrayTypeReference(TypeBinding.BYTE.simpleName, dim, ((long) start << 32) + end);
648                                        }
649                                        break;
650                                case Signature.C_CHAR :
651                                        if (!nameStarted) {
652                                                this.namePos++;
653                                                if (dim == 0)
654                                                        return new SingleTypeReference(TypeBinding.CHAR.simpleName, ((long) start << 32) + end);
655                                                else
656                                                        return new ArrayTypeReference(TypeBinding.CHAR.simpleName, dim, ((long) start << 32) + end);
657                                        }
658                                        break;
659                                case Signature.C_DOUBLE :
660                                        if (!nameStarted) {
661                                                this.namePos++;
662                                                if (dim == 0)
663                                                        return new SingleTypeReference(TypeBinding.DOUBLE.simpleName, ((long) start << 32) + end);
664                                                else
665                                                        return new ArrayTypeReference(TypeBinding.DOUBLE.simpleName, dim, ((long) start << 32) + end);
666                                        }
667                                        break;
668                                case Signature.C_FLOAT :
669                                        if (!nameStarted) {
670                                                this.namePos++;
671                                                if (dim == 0)
672                                                        return new SingleTypeReference(TypeBinding.FLOAT.simpleName, ((long) start << 32) + end);
673                                                else
674                                                        return new ArrayTypeReference(TypeBinding.FLOAT.simpleName, dim, ((long) start << 32) + end);
675                                        }
676                                        break;
677                                case Signature.C_INT :
678                                        if (!nameStarted) {
679                                                this.namePos++;
680                                                if (dim == 0)
681                                                        return new SingleTypeReference(TypeBinding.INT.simpleName, ((long) start << 32) + end);
682                                                else
683                                                        return new ArrayTypeReference(TypeBinding.INT.simpleName, dim, ((long) start << 32) + end);
684                                        }
685                                        break;
686                                case Signature.C_LONG :
687                                        if (!nameStarted) {
688                                                this.namePos++;
689                                                if (dim == 0)
690                                                        return new SingleTypeReference(TypeBinding.LONG.simpleName, ((long) start << 32) + end);
691                                                else
692                                                        return new ArrayTypeReference(TypeBinding.LONG.simpleName, dim, ((long) start << 32) + end);
693                                        }
694                                        break;
695                                case Signature.C_SHORT :
696                                        if (!nameStarted) {
697                                                this.namePos++;
698                                                if (dim == 0)
699                                                        return new SingleTypeReference(TypeBinding.SHORT.simpleName, ((long) start << 32) + end);
700                                                else
701                                                        return new ArrayTypeReference(TypeBinding.SHORT.simpleName, dim, ((long) start << 32) + end);
702                                        }
703                                        break;
704                                case Signature.C_VOID :
705                                        if (!nameStarted) {
706                                                this.namePos++;
707                                                new SingleTypeReference(TypeBinding.VOID.simpleName, ((long) start << 32) + end);
708                                        }
709                                        break;
710                                case Signature.C_RESOLVED :
711                                case Signature.C_UNRESOLVED :
712                                        if (!nameStarted) {
713                                                nameFragmentStart = this.namePos+1;
714                                                nameStarted = true;
715                                        }
716                                        break;
717                                case Signature.C_ARRAY :
718                                        dim++;
719                                        break;
720                                case Signature.C_SEMICOLON :
721                                        nameFragmentEnd = this.namePos-1;
722                                        this.namePos++;
723                                        break typeLoop;
724                                case Signature.C_DOT :
725                                case Signature.C_DOLLAR:
726                                        if (!nameStarted) {
727                                                nameFragmentStart = this.namePos+1;
728                                                nameStarted = true;
729                                        } else if (this.namePos > nameFragmentStart) // handle name starting with a $ (see https://bugs.eclipse.org/bugs/show_bug.cgi?id=91709)
730                                                identCount ++;
731                                        break;
732                        }
733                        this.namePos++;
734                }
735                if (fragments == null) { // non parameterized
736                        /* rebuild identifiers and dimensions */
737                        if (identCount == 1) { // simple type reference
738                                if (dim == 0) {
739                                        char[] nameFragment = new char[nameFragmentEnd - nameFragmentStart + 1];
740                                        typeSignature.getChars(nameFragmentStart, nameFragmentEnd +1, nameFragment, 0);
741                                        return new SingleTypeReference(nameFragment, ((long) start << 32) + end);
742                                } else {
743                                        char[] nameFragment = new char[nameFragmentEnd - nameFragmentStart + 1];
744                                        typeSignature.getChars(nameFragmentStart, nameFragmentEnd +1, nameFragment, 0);
745                                        return new ArrayTypeReference(nameFragment, dim, ((long) start << 32) + end);
746                                }
747                        } else { // qualified type reference
748                                long[] positions = new long[identCount];
749                                long pos = ((long) start << 32) + end;
750                                for (int i = 0; i < identCount; i++) {
751                                        positions[i] = pos;
752                                }
753                                char[][] identifiers = extractIdentifiers(typeSignature, nameFragmentStart, nameFragmentEnd, identCount);
754                                if (dim == 0) {
755                                        return new QualifiedTypeReference(identifiers, positions);
756                                } else {
757                                        return new ArrayQualifiedTypeReference(identifiers, dim, positions);
758                                }
759                        }
760                } else { // parameterized
761                        // rebuild type reference from available fragments: char[][], arguments, char[][], arguments...
762                        // check trailing qualified name
763                        if (nameStarted) {
764                                addIdentifiers(typeSignature, nameFragmentStart, nameFragmentEnd + 1, identCount, fragments);
765                        }
766                        int fragmentLength = fragments.size();
767                        if (fragmentLength == 2) {
768                                Object firstFragment = fragments.get(0);
769                                if (firstFragment instanceof char[]) {
770                                        // parameterized single type
771                                        return new ParameterizedSingleTypeReference((char[]) firstFragment, (TypeReference[]) fragments.get(1), dim, ((long) start << 32) + end);
772                                }
773                        }
774                        // parameterized qualified type
775                        identCount = 0;
776                        for (int i = 0; i < fragmentLength; i ++) {
777                                Object element = fragments.get(i);
778                                if (element instanceof char[][]) {
779                                        identCount += ((char[][])element).length;
780                                } else if (element instanceof char[])
781                                        identCount++;
782                        }
783                        char[][] tokens = new char[identCount][];
784                        TypeReference[][] arguments = new TypeReference[identCount][];
785                        int index = 0;
786                        for (int i = 0; i < fragmentLength; i ++) {
787                                Object element = fragments.get(i);
788                                if (element instanceof char[][]) {
789                                        char[][] fragmentTokens = (char[][]) element;
