419        private void addExclusionPatterns(IIncludePathEntry newEntry, List existing, Set modifiedEntries) {
420                IPath entryPath= newEntry.getPath();
421                for (int i= 0; i < existing.size(); i++) {
422                        IIncludePathEntry curr= (IIncludePathEntry) existing.get(i);
423                        IPath currPath= curr.getPath();
424                        if (curr.getEntryKind() == IIncludePathEntry.CPE_SOURCE && currPath.isPrefixOf(entryPath)) {
425                                IPath[] exclusionFilters= curr.getExclusionPatterns();
426                                if (!JavaModelUtil.isExcludedPath(entryPath, exclusionFilters)) {
427                                        IPath pathToExclude= entryPath.removeFirstSegments(currPath.segmentCount()).addTrailingSeparator();
428                                        IPath[] newExclusionFilters= new IPath[exclusionFilters.length + 1];
429                                        System.arraycopy(exclusionFilters, 0, newExclusionFilters, 0, exclusionFilters.length);
430                                        newExclusionFilters[exclusionFilters.length]= pathToExclude;
431                                        
432                                        IIncludePathEntry updated= JavaScriptCore.newSourceEntry(currPath, newExclusionFilters, null);
433                                        existing.set(i, updated);
434                                        modifiedEntries.add(updated);
435                                }
436                        }
437                }
