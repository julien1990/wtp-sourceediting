643        public void checkParameterizedSuperTypeCollisions() {
644                // check for parameterized interface collisions (when different parameterizations occur)
645                SourceTypeBinding sourceType = getReferenceBinding();
646                ReferenceBinding[] interfaces = sourceType.superInterfaces;
647                Map invocations = new HashMap(2);
648                ReferenceBinding itsSuperclass = sourceType.isInterface() ? null : sourceType.superclass;
649                nextInterface: for (int i = 0, length = interfaces.length; i < length; i++) {
650                        ReferenceBinding one =  interfaces[i];
651                        if (one == null) continue nextInterface;
652                        if (itsSuperclass != null && hasErasedCandidatesCollisions(itsSuperclass, one, invocations, sourceType, referenceContext))
653                                continue nextInterface;
654                        nextOtherInterface: for (int j = 0; j < i; j++) {
655                                ReferenceBinding two = interfaces[j];
656                                if (two == null) continue nextOtherInterface;
657                                if (hasErasedCandidatesCollisions(one, two, invocations, sourceType, referenceContext))
658                                        continue nextInterface;
659                        }
660                }
661 
662                TypeParameter[] typeParameters = this.referenceContext.typeParameters;
663                nextVariable : for (int i = 0, paramLength = typeParameters == null ? 0 : typeParameters.length; i < paramLength; i++) {
664                        TypeParameter typeParameter = typeParameters[i];
665                        TypeVariableBinding typeVariable = typeParameter.binding;
666                        if (typeVariable == null || !typeVariable.isValidBinding()) continue nextVariable;
667 
668                        TypeReference[] boundRefs = typeParameter.bounds;
669                        if (boundRefs != null) {
670                                boolean checkSuperclass = typeVariable.firstBound == typeVariable.superclass;
671                                for (int j = 0, boundLength = boundRefs.length; j < boundLength; j++) {
672                                        TypeReference typeRef = boundRefs[j];
673                                        TypeBinding superType = typeRef.resolvedType;
674                                        if (superType == null || !superType.isValidBinding()) continue;
675 
676                                        // check against superclass
677                                        if (checkSuperclass)
678                                                if (hasErasedCandidatesCollisions(superType, typeVariable.superclass, invocations, typeVariable, typeRef))
679                                                        continue nextVariable;
680                                        // check against superinterfaces
681                                        for (int index = typeVariable.superInterfaces.length; --index >= 0;)
682                                                if (hasErasedCandidatesCollisions(superType, typeVariable.superInterfaces[index], invocations, typeVariable, typeRef))
683                                                        continue nextVariable;
684                                }
685                        }
686                }
687 
688                ReferenceBinding[] memberTypes = getReferenceBinding().memberTypes;
689                if (memberTypes != null && memberTypes != Binding.NO_MEMBER_TYPES)
690                        for (int i = 0, size = memberTypes.length; i < size; i++)
691                                 ((SourceTypeBinding) memberTypes[i]).classScope.checkParameterizedSuperTypeCollisions();
692        }

694        private void checkForInheritedMemberTypes(SourceTypeBinding sourceType) {

696                // when no member types are defined, tag the sourceType & each superType with the HasNoMemberTypes bit

699                ReferenceBinding[] interfacesToVisit = null;
