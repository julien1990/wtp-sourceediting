465        protected void handleInternalException(
466                AbortCompilation abortException,
467                CompilationUnitDeclaration unit) {
468 
469                if (!this.options.enableSemanticValidation)
470                        return;
471 
472                /* special treatment for SilentAbort: silently cancelling the compilation process */
473                if (abortException.isSilent) {
474                        if (abortException.silentException == null) {
475                                return;
476                        }
477                        throw abortException.silentException;
478                }
479 
480                /* uncomment following line to see where the abort came from */
481                // abortException.printStackTrace();
482 
483                // Exception may tell which compilation result it is related, and which problem caused it
484                CompilationResult result = abortException.compilationResult;
485                if ((result == null) && (unit != null)) {
486                        result = unit.compilationResult; // current unit being processed ?
487                }
488                // Lookup environment may be in middle of connecting types
489                if ((result == null) && lookupEnvironment.unitBeingCompleted != null) {
490                    result = lookupEnvironment.unitBeingCompleted.compilationResult;
491                }
492                if ((result == null) && (unitsToProcess != null) && (totalUnits > 0))
493                        result = unitsToProcess[totalUnits - 1].compilationResult;
494                // last unit in beginToCompile ?
495                if (result != null && !result.hasBeenAccepted) {
496                        /* distant problem which could not be reported back there? */
497                        if (abortException.problem != null) {
498                                recordDistantProblem: {
499                                CategorizedProblem distantProblem = abortException.problem;
500                                CategorizedProblem[] knownProblems = result.problems;
501                                        for (int i = 0; i < result.problemCount; i++) {
502                                                if (knownProblems[i] == distantProblem) { // already recorded
503                                                        break recordDistantProblem;
504                                                }
505                                        }
506                                        if (distantProblem instanceof DefaultProblem) { // fixup filename TODO (philippe) should improve API to make this official
507                                                ((DefaultProblem) distantProblem).setOriginatingFileName(result.getFileName());
508                                        }
509                                        result.record(distantProblem, unit);
510                                }
511                        } else {
512                                /* distant internal exception which could not be reported back there */
513                                if (abortException.exception != null) {
514                                        this.handleInternalException(abortException.exception, null, result);
515                                        return;
516                                }
517                        }
518                        /* hand back the compilation result */
519                        if (!result.hasBeenAccepted) {
520                                requestor.acceptResult(result.tagAsAccepted());
521                        }
522                } else {
523                        abortException.printStackTrace();
524                }

611        protected void processAnnotations() {
612                int newUnitSize = 0;
613                int newClassFilesSize = 0;
614                int bottom = 0;
615                int top = this.unitsToProcess.length;
616                ReferenceBinding[] binaryTypeBindingsTemp = this.referenceBindings;

623                        int index = 0;
624                        for (int i = bottom; i < top; i++) {

627                                        currentUnits[index++] = currentUnit;
628                                }
629                        }
