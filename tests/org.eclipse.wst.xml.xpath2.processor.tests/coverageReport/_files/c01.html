299private char[][] splitParameters(char[] parametersSource, int paramCount) {

301        char[][] params = new char[paramCount][];

306        int start = 0;
307        while(index < length) {
308                switch (parametersSource[index]) {
309                        case '<':
310                                balance++;
311                                index++;
312                                while(index < length && parametersSource[index] != '>') {
313                                        index++;
314                                }
315                                break;
316                        case '>' :
317                                balance--;
318                                index++;
319                                break;
320                        case ',' :

322                                        params[paramIndex++] = CharOperation.subarray(parametersSource, start, index);
323                                        start = index + 1;
324                                }
325                                index++;
326                                break;
327                        case '&' :
328                                if ((index + 4) < length) {
329                                        if (parametersSource[index + 1] == 'l'
330                                                        && parametersSource[index + 2] == 't'
331                                                        && parametersSource[index + 3] == ';') {

333                                                index += 4;
334                                        } else if (parametersSource[index + 1] == 'g'

533private String extractJavadoc(IType declaringType, String contents) throws JavaScriptModelException {
534        if (contents == null) return null;
535 
536        String typeQualifiedName = null;
537        final boolean declaringTypeIsMember = declaringType.isMember();
538        if (declaringTypeIsMember) {
539                IType currentType = declaringType;
540                StringBuffer buffer = new StringBuffer();
541                while (currentType != null) {
542                        buffer.insert(0, currentType.getElementName());
543                        currentType = currentType.getDeclaringType();
544                        if (currentType != null) {
545                                buffer.insert(0, '.');
546                        }
547                }
548                typeQualifiedName = new String(buffer.toString());
549        } else {
550                typeQualifiedName = declaringType.getElementName();
551        }
552        String methodName = this.getElementName();
553        if (this.isConstructor()) {
554                methodName = typeQualifiedName;
555        }
556        IBinaryMethod info = (IBinaryMethod) getElementInfo();
557        char[] genericSignature = info.getGenericSignature();
558        String anchor = null;
559        if (genericSignature != null) {
560                genericSignature = CharOperation.replaceOnCopy(genericSignature, '/', '.');
561                anchor = Util.toAnchor(genericSignature, methodName, Flags.isVarargs(this.getFlags()));
562                if (anchor == null) throw new JavaScriptModelException(new JavaModelStatus(IJavaScriptModelStatusConstants.UNKNOWN_JSDOC_FORMAT, this));
563        } else {
564                anchor = Signature.toString(this.getSignature().replace('/', '.'), methodName, null, true, false, Flags.isVarargs(this.getFlags()));
565        }
566        if (declaringTypeIsMember) {
567                int depth = 0;
568                final String packageFragmentName = declaringType.getPackageFragment().getElementName();
569                // might need to remove a part of the signature corresponding to the synthetic argument
570                final IJavaScriptProject javaProject = declaringType.getJavaScriptProject();
571                char[][] typeNames = CharOperation.splitOn('.', typeQualifiedName.toCharArray());
572                if (!Flags.isStatic(declaringType.getFlags())) depth++;
573                StringBuffer typeName = new StringBuffer();
574                for (int i = 0, max = typeNames.length; i < max; i++) {
575                        if (typeName.length() == 0) {
576                                typeName.append(typeNames[i]);
577                        } else {
578                                typeName.append('.').append(typeNames[i]);
579                        }
580                        IType resolvedType = javaProject.findType(packageFragmentName, String.valueOf(typeName));
581                        if (resolvedType != null && resolvedType.isMember() && !Flags.isStatic(resolvedType.getFlags())) depth++;
582                }
583                if (depth != 0) {
584                        int indexOfOpeningParen = anchor.indexOf('(');
585                        if (indexOfOpeningParen == -1) return null;
586                        int index = indexOfOpeningParen;
587                        indexOfOpeningParen++;
588                        for (int i = 0; i < depth; i++) {
589                                int indexOfComma = anchor.indexOf(',', index);
590                                if (indexOfComma != -1) {
591                                        index = indexOfComma + 2;
592                                }
593                        }
594                        anchor = anchor.substring(0, indexOfOpeningParen) + anchor.substring(index);
595                }
596        }
597        int indexAnchor = contents.indexOf(JavadocConstants.ANCHOR_PREFIX_START + anchor + JavadocConstants.ANCHOR_PREFIX_END);
598        if (indexAnchor == -1) {
599                return null; // method without javadoc
600        }
601        int indexOfEndLink = contents.indexOf(JavadocConstants.ANCHOR_SUFFIX, indexAnchor);
602        if (indexOfEndLink == -1) throw new JavaScriptModelException(new JavaModelStatus(IJavaScriptModelStatusConstants.UNKNOWN_JSDOC_FORMAT, this));
603        int indexOfNextMethod = contents.indexOf(JavadocConstants.ANCHOR_PREFIX_START, indexOfEndLink);
604        // find bottom
605        int indexOfBottom = -1;
606        if (this.isConstructor()) {
607                indexOfBottom = contents.indexOf(JavadocConstants.METHOD_DETAIL, indexOfEndLink);
608                if (indexOfBottom == -1) {
609                        indexOfBottom = contents.indexOf(JavadocConstants.END_OF_CLASS_DATA, indexOfEndLink);
610                }
611        } else {
612                indexOfBottom = contents.indexOf(JavadocConstants.END_OF_CLASS_DATA, indexOfEndLink);
613        }
614        if (indexOfBottom == -1) throw new JavaScriptModelException(new JavaModelStatus(IJavaScriptModelStatusConstants.UNKNOWN_JSDOC_FORMAT, this));
615        indexOfNextMethod = Math.min(indexOfNextMethod, indexOfBottom);
616        if (indexOfNextMethod == -1) throw new JavaScriptModelException(new JavaModelStatus(IJavaScriptModelStatusConstants.UNKNOWN_JSDOC_FORMAT, this));
617        return contents.substring(indexOfEndLink + JavadocConstants.ANCHOR_SUFFIX_LENGTH, indexOfNextMethod);
618}
