380                        } else if (!commentLines[lineIndex] && startingPartition.getOffset() == offset && startingPartition.getType().equals(IJavaScriptPartitions.JAVA_SINGLE_LINE_COMMENT)) {
381                                return false;                                
382                        }
383                } 
384                
385                // standard java indentation
386                if (indent == null) {
387                        StringBuffer computed= indenter.computeIndentation(offset);
388                        if (computed != null)
389                                indent= computed.toString();
390                        else
391                                indent= new String();
392                }
393                
394                // change document:
395                // get current white space
396                int lineLength= currentLine.getLength();
397                int end= scanner.findNonWhitespaceForwardInAnyPartition(wsStart, offset + lineLength);
398                if (end == JavaHeuristicScanner.NOT_FOUND)
399                        end= offset + lineLength;
400                int length= end - offset;
401                String currentIndent= document.get(offset, length);
402                
403                // memorize the fact that a line is a single line comment (but not at column 0) and should be treated like code
404                // as opposed to commented out code, which should keep its slashes at column 0
405                if (length > 0) {
406                        ITypedRegion partition= TextUtilities.getPartition(document, IJavaScriptPartitions.JAVA_PARTITIONING, end, false);
407                        if (partition.getOffset() == end && IJavaScriptPartitions.JAVA_SINGLE_LINE_COMMENT.equals(partition.getType())) {
408                                commentLines[lineIndex]= true;
409                        }
410                }
411                
412                // only change the document if it is a real change
413                if (!indent.equals(currentIndent)) {
414                        document.replace(offset, length, indent);
415                        return true;
416                }
417                
418                return false;
419        }
420        
421        /**
422         * Computes and returns the indentation for a javadoc line. The line
423         * must be inside a javadoc comment.
424         * 
425         * @param document the document
426         * @param line the line in document
427         * @param scanner the scanner
428         * @param partition the comment partition
429         * @return the indent, or <code>null</code> if not computable
430         * @throws BadLocationException
431         */
432        private static String computeJavadocIndent(IDocument document, int line, JavaHeuristicScanner scanner, ITypedRegion partition) throws BadLocationException {
433                if (line == 0) // impossible - the first line is never inside a javadoc comment
434                        return null;
435                
436                // don't make any assumptions if the line does not start with \s*\* - it might be
437                // commented out code, for which we don't want to change the indent
438                final IRegion lineInfo= document.getLineInformation(line);
439                final int lineStart= lineInfo.getOffset();
440                final int lineLength= lineInfo.getLength();
441                final int lineEnd= lineStart + lineLength;
442                int nonWS= scanner.findNonWhitespaceForwardInAnyPartition(lineStart, lineEnd);
443                if (nonWS == JavaHeuristicScanner.NOT_FOUND || document.getChar(nonWS) != '*') {
444                        if (nonWS == JavaHeuristicScanner.NOT_FOUND)
445                                return document.get(lineStart, lineLength);
446                        return document.get(lineStart, nonWS - lineStart);
447                }
448                
449                // take the indent from the previous line and reuse
450                IRegion previousLine= document.getLineInformation(line - 1);
451                int previousLineStart= previousLine.getOffset();
452                int previousLineLength= previousLine.getLength();
453                int previousLineEnd= previousLineStart + previousLineLength;
454                
455                StringBuffer buf= new StringBuffer();
456                int previousLineNonWS= scanner.findNonWhitespaceForwardInAnyPartition(previousLineStart, previousLineEnd);
457                if (previousLineNonWS == JavaHeuristicScanner.NOT_FOUND || document.getChar(previousLineNonWS) != '*') {
458                        // align with the comment start if the previous line is not an asterix line
459                        previousLine= document.getLineInformationOfOffset(partition.getOffset());
460                        previousLineStart= previousLine.getOffset();
461                        previousLineLength= previousLine.getLength();
462                        previousLineEnd= previousLineStart + previousLineLength;
463                        previousLineNonWS= scanner.findNonWhitespaceForwardInAnyPartition(previousLineStart, previousLineEnd);
464                        if (previousLineNonWS == JavaHeuristicScanner.NOT_FOUND)
465                                previousLineNonWS= previousLineEnd;
466                        
467                        // add the initial space 
468                        // TODO this may be controlled by a formatter preference in the future
469                        buf.append(' ');
470                }
471                
472                String indentation= document.get(previousLineStart, previousLineNonWS - previousLineStart);
473                buf.insert(0, indentation);
474                return buf.toString();
475        }
