493  public static XMLInputSource _internalResolveEntity(URIResolver uriResolver, XMLResourceIdentifier rid, NestedValidatorContext context) throws  IOException
494  {
495    XMLInputSource is = null;
496    
497    if (uriResolver != null)
498    {         
499      String id = rid.getPublicId();
500      if(id == null)
501      {
502        id = rid.getNamespace();
503      }
504      
505      String location = null;
506      if (id != null || rid.getLiteralSystemId() != null)
507      {  
508        location = uriResolver.resolve(rid.getBaseSystemId(), id, rid.getLiteralSystemId());
509      }  
510      
511      if (location != null)
512      {                     
513        String physical = uriResolver.resolvePhysicalLocation(rid.getBaseSystemId(), id, location);
514 
515        // if physical is already a known bad uri, just go ahead and throw an exception
516        if (context instanceof XMLNestedValidatorContext)
517        {
518          XMLNestedValidatorContext xmlContext = ((XMLNestedValidatorContext)context);
519 
520          if (xmlContext.isURIMarkedInaccessible(physical))
521          {
522                 throw new FileNotFoundException(physical);
523          }
524        }
525        
526        is = new XMLInputSource(rid.getPublicId(), location, location);
527        
528        // This block checks that the file exists. If it doesn't we need to throw
529        // an exception so Xerces will report an error. note: This may not be
530        // necessary with all versions of Xerces but has specifically been 
531        // experienced with the version included in IBM's 1.4.2 JDK.
532        InputStream isTemp = null;
533        try
534        {
535          isTemp = new URL(physical).openStream();
536        }
537        catch (IOException e)
538        {
539          // physical was a bad url, so cache it so we know next time
540          if (context instanceof XMLNestedValidatorContext)
541          {
542            XMLNestedValidatorContext xmlContext = ((XMLNestedValidatorContext)context);
543            xmlContext.markURIInaccessible(physical);
544          }
545          throw e;
546        }
547        finally
548        {
549          if(isTemp != null)
550          {
551            isTemp.close();
552          }
553        }
554        is.setByteStream(new LazyURLInputStream(physical));      
555      }
556    }
557    return is;    
558  }
