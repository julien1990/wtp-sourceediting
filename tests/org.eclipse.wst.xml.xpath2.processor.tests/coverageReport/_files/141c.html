554         * @see SourceViewerConfiguration#getConfiguredTextHoverStateMasks(ISourceViewer, String)
555         * 
556         */
557        public int[] getConfiguredTextHoverStateMasks(ISourceViewer sourceViewer, String contentType) {
558                JavaEditorTextHoverDescriptor[] hoverDescs= JavaScriptPlugin.getDefault().getJavaEditorTextHoverDescriptors();
559                int stateMasks[]= new int[hoverDescs.length];
560                int stateMasksLength= 0;
561                for (int i= 0; i < hoverDescs.length; i++) {
562                        if (hoverDescs[i].isEnabled()) {
563                                int j= 0;
564                                int stateMask= hoverDescs[i].getStateMask();
565                                while (j < stateMasksLength) {
566                                        if (stateMasks[j] == stateMask)
567                                                break;
568                                        j++;
569                                }
570                                if (j == stateMasksLength)
571                                        stateMasks[stateMasksLength++]= stateMask;
572                        }
573                }
574                if (stateMasksLength == hoverDescs.length)
575                        return stateMasks;
576 
577                int[] shortenedStateMasks= new int[stateMasksLength];
578                System.arraycopy(stateMasks, 0, shortenedStateMasks, 0, stateMasksLength);
579                return shortenedStateMasks;
580        }
