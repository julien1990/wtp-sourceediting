358public int resolveLevel(Binding binding) {
359        if (binding == null) return INACCURATE_MATCH;
360        if (!(binding instanceof MethodBinding)) return IMPOSSIBLE_MATCH;
361 
362        MethodBinding constructor = (MethodBinding) binding;
363        int level= matchConstructor(constructor);
364        if (level== IMPOSSIBLE_MATCH) {
365                if (constructor != constructor.original()) {
366                        level= matchConstructor(constructor.original());
367                }
368        }
369        return level;
370}
371protected int resolveLevel(ConstructorDeclaration constructor, boolean checkDeclarations) {
372        int referencesLevel = IMPOSSIBLE_MATCH;
373        if (this.pattern.findReferences) {
374                ExplicitConstructorCall constructorCall = constructor.constructorCall;
375                if (constructorCall != null && constructorCall.accessMode == ExplicitConstructorCall.ImplicitSuper) {
376                        // eliminate explicit super call as it will be treated with matchLevel(ExplicitConstructorCall, boolean)
377                        int callCount = (constructorCall.arguments == null) ? 0 : constructorCall.arguments.length;
378                        int patternCount = (this.pattern.parameterSimpleNames == null) ? 0 : this.pattern.parameterSimpleNames.length;
379                        if (patternCount != callCount) {
380                                referencesLevel = IMPOSSIBLE_MATCH;
381                        } else {
382                                referencesLevel = resolveLevel(constructorCall.binding);
383                                if (referencesLevel == ACCURATE_MATCH) return ACCURATE_MATCH; // cannot get better
384                        }
385                }
386        }
387        if (!checkDeclarations) return referencesLevel;
388 
389        int declarationsLevel = this.pattern.findDeclarations ? resolveLevel(constructor.binding) : IMPOSSIBLE_MATCH;
390        return referencesLevel >= declarationsLevel ? referencesLevel : declarationsLevel; // answer the stronger match
391}
392protected int resolveLevel(TypeDeclaration type) {
393        // find default constructor
394        AbstractMethodDeclaration[] methods = type.methods;
395        if (methods != null) {
396                for (int i = 0, length = methods.length; i < length; i++) {
397                        AbstractMethodDeclaration method = methods[i];
398                        if (method.isDefaultConstructor() && method.sourceStart < type.bodyStart) // if synthetic
399                                return resolveLevel((ConstructorDeclaration) method, false);
400                }
401        }
402        return IMPOSSIBLE_MATCH;
403}
