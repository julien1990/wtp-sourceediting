536                protected static void saveLibraries(List libraries, File file, String encoding, IProgressMonitor monitor) throws IOException {
537                        OutputStream stream= new FileOutputStream(file);
538                        try {
539                                DocumentBuilder docBuilder= null;
540                                DocumentBuilderFactory factory= DocumentBuilderFactory.newInstance();
541                                factory.setValidating(false);
542                                docBuilder= factory.newDocumentBuilder();
543                                Document document= docBuilder.newDocument();
544        
545                                // Create the document
546                                Element rootElement= document.createElement(TAG_ROOT);
547                                document.appendChild(rootElement);
548        
549                                rootElement.setAttribute(TAG_VERSION, CURRENT_VERSION); 
550                                
551                                for (int i= 0; i < libraries.size(); i++) {
552                                        Element libraryElement= document.createElement(TAG_LIBRARY);
553                                        rootElement.appendChild(libraryElement);
554                                        
555                                        CPUserLibraryElement curr= (CPUserLibraryElement) libraries.get(i);
556                                        libraryElement.setAttribute(TAG_NAME, curr.getName());
557                                        
558                                        CPListElement[] children= curr.getChildren();
559                                        for (int k= 0; k < children.length; k++) {
560                                                CPListElement child= children[k];
561                                                
562                                                Element childElement= document.createElement(TAG_ARCHIVE); 
563                                                libraryElement.appendChild(childElement);
564                                                
565                                                childElement.setAttribute(TAG_ARCHIVE_PATH, child.getPath().toPortableString());
566                                                
567                                                String javadocLocation= (String) child.getAttribute(CPListElement.JAVADOC);
568                                                if (javadocLocation != null) {
569                                                        childElement.setAttribute(TAG_JSDOC, javadocLocation);
570                                                }
571 
572                                                IAccessRule[] accessRules= (IAccessRule[]) child.getAttribute(CPListElement.ACCESSRULES);
573                                                if (accessRules != null && accessRules.length > 0) {
574                                                        Element rulesElement= document.createElement(TAG_ACCESSRULES); 
575                                                        childElement.appendChild(rulesElement);
576                                                        for (int n= 0; n < accessRules.length; n++) {
577                                                                IAccessRule rule= accessRules[n];
578                                                                Element ruleElement= document.createElement(TAG_ACCESSRULE); 
579                                                                rulesElement.appendChild(ruleElement);
580                                                                ruleElement.setAttribute(TAG_RULE_KIND, String.valueOf(rule.getKind()));
581                                                                ruleElement.setAttribute(TAG_RULE_PATTERN, rule.getPattern().toPortableString());
582                                                        }
583                                                }
584                                        }
585                                }

587                                // Write the document to the stream
588                                Transformer transformer=TransformerFactory.newInstance().newTransformer();
589                                transformer.setOutputProperty(OutputKeys.METHOD, "xml"); //$NON-NLS-1$
590                                transformer.setOutputProperty(OutputKeys.ENCODING, encoding);
591                                transformer.setOutputProperty(OutputKeys.INDENT, "yes"); //$NON-NLS-1$
592                                transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount","4"); //$NON-NLS-1$ //$NON-NLS-2$
593 
594                                DOMSource source = new DOMSource(document);
595                                StreamResult result = new StreamResult(stream);
596                                transformer.transform(source, result);        
597                        } catch (ParserConfigurationException e) {
598                                throw new IOException(e.getMessage());
599                        } catch (TransformerException e) {
600                                throw new IOException(e.getMessage());                        
601                        } finally {
602                                try {
603                                        stream.close();
604                                } catch (IOException e) {
605                                        // ignore
606                                }

609                                }

613                private static List loadLibraries(File file) throws IOException {
614                        InputStream stream= new FileInputStream(file);
615                        Element cpElement;
616                        try {
617                                DocumentBuilder parser = DocumentBuilderFactory.newInstance().newDocumentBuilder();
618                                cpElement = parser.parse(new InputSource(stream)).getDocumentElement();
619                        } catch (SAXException e) {
620                                throw new IOException(PreferencesMessages.UserLibraryPreferencePage_LoadSaveDialog_load_badformat); 
621                        } catch (ParserConfigurationException e) {
622                                throw new IOException(PreferencesMessages.UserLibraryPreferencePage_LoadSaveDialog_load_badformat); 
623                        } finally {
624                                stream.close();
625                        }
626                        
627                        if (!cpElement.getNodeName().equalsIgnoreCase(TAG_ROOT)) {
628                                throw new IOException(PreferencesMessages.UserLibraryPreferencePage_LoadSaveDialog_load_badformat); 
629                        }
630                        
631                        String version= cpElement.getAttribute(TAG_VERSION); 
632                        
633                        NodeList libList= cpElement.getElementsByTagName(TAG_LIBRARY);
634                        int length = libList.getLength();
635                        
636                        ArrayList result= new ArrayList(length);
637                        for (int i= 0; i < length; i++) {
638                                Node lib= libList.item(i);
639                                if (!(lib instanceof Element)) {
640                                        continue;
641                                }
642                                Element libElement= (Element) lib;
643                                String name= libElement.getAttribute(TAG_NAME);
644                                
645                                CPUserLibraryElement newLibrary= new CPUserLibraryElement(name, null);
646                                result.add(newLibrary);
647                                
648                                NodeList archiveList= libElement.getElementsByTagName(TAG_ARCHIVE);
649                                for (int k= 0; k < archiveList.getLength(); k++) {
650                                        Node archiveNode= archiveList.item(k);
651                                        if (!(archiveNode instanceof Element)) {
652                                                continue;
653                                        }
654                                        Element archiveElement= (Element) archiveNode;
655                                        
656                                        String pathString= archiveElement.getAttribute(TAG_ARCHIVE_PATH);
657                                        IPath path= version.equals(VERSION1) ? Path.fromOSString(pathString) : Path.fromPortableString(pathString);
658                                        CPListElement newArchive= new CPListElement(newLibrary, null, IIncludePathEntry.CPE_LIBRARY, path, null);
659                                        newLibrary.add(newArchive);
660                                        
661                                        if (archiveElement.hasAttribute(TAG_JSDOC)) {
662                                                String javadoc= archiveElement.getAttribute(TAG_JSDOC);
663                                                newArchive.setAttribute(CPListElement.JAVADOC, javadoc);
664                                        }
665                                        
666                                        NodeList rulesParentNodes= archiveElement.getElementsByTagName(TAG_ACCESSRULES);
667                                        if (rulesParentNodes.getLength() > 0 && rulesParentNodes.item(0) instanceof Element) {
668                                                Element ruleParentElement= (Element) rulesParentNodes.item(0); // take first, ignore others
669                                                NodeList ruleElements= ruleParentElement.getElementsByTagName(TAG_ACCESSRULE);
670                                                int nRuleElements= ruleElements.getLength();
671                                                if (nRuleElements > 0) {
672                                                        ArrayList resultingRules= new ArrayList(nRuleElements);
673                                                        for (int n= 0; n < nRuleElements; n++) {
674                                                                Node node= ruleElements.item(n);
675                                                                if (node instanceof Element) {
676                                                                        Element ruleElement= (Element) node;
677                                                                        try {
678                                                                                int kind= Integer.parseInt(ruleElement.getAttribute(TAG_RULE_KIND));
679                                                                                IPath pattern= Path.fromPortableString(ruleElement.getAttribute(TAG_RULE_PATTERN));
680                                                                                resultingRules.add(JavaScriptCore.newAccessRule(pattern, kind));
681                                                                        } catch (NumberFormatException e) {
682                                                                                // ignore
683                                                                        }
684                                                                }
685                                                        }
686                                                        newArchive.setAttribute(CPListElement.ACCESSRULES, resultingRules.toArray(new IAccessRule[resultingRules.size()]));
687                                                }
688                                        }                                        
689                                }
690                        }
691                        return result;
692                }
