431                        int style= fPreferenceStore.getBoolean(boldKey) ? SWT.BOLD : SWT.NORMAL;
432 
433                        String italicKey= SemanticHighlightings.getItalicPreferenceKey(semanticHighlighting);
434                        if (fPreferenceStore.getBoolean(italicKey))
435                                style |= SWT.ITALIC;
436 
437                        String strikethroughKey= SemanticHighlightings.getStrikethroughPreferenceKey(semanticHighlighting);
438                        if (fPreferenceStore.getBoolean(strikethroughKey))
439                                style |= TextAttribute.STRIKETHROUGH;

442                        if (fPreferenceStore.getBoolean(underlineKey))
443                                style |= TextAttribute.UNDERLINE;
444 
445                        boolean isEnabled= fPreferenceStore.getBoolean(SemanticHighlightings.getEnabledPreferenceKey(semanticHighlighting));
446 
447                        fHighlightings[i]= new Highlighting(new TextAttribute(fColorManager.getColor(PreferenceConverter.getColor(fPreferenceStore, colorKey)), null, style), isEnabled);

469        /**
470         * Handle the given property change event
471         *
472         * @param event The event
473         */
474        private void handlePropertyChangeEvent(PropertyChangeEvent event) {
475                if (fPreferenceStore == null)
476                        return; // Uninstalled during event notification
477 
478                if (fConfiguration != null)
479                        fConfiguration.handlePropertyChangeEvent(event);
480 
481                if (SemanticHighlightings.affectsEnablement(fPreferenceStore, event)) {
482                        if (isEnabled())
483                                enable();
484                        else
485                                disable();
486                }
487 
488                if (!isEnabled())
489                        return;
490                
491                boolean refreshNeeded= false;
492 
493                for (int i= 0, n= fSemanticHighlightings.length; i < n; i++) {
494                        SemanticHighlighting semanticHighlighting= fSemanticHighlightings[i];
495 
496                        String colorKey= SemanticHighlightings.getColorPreferenceKey(semanticHighlighting);
497                        if (colorKey.equals(event.getProperty())) {
498                                adaptToTextForegroundChange(fHighlightings[i], event);
499                                fPresenter.highlightingStyleChanged(fHighlightings[i]);
500                                refreshNeeded= true;
501                                continue;
502                        }
503 
504                        String boldKey= SemanticHighlightings.getBoldPreferenceKey(semanticHighlighting);
505                        if (boldKey.equals(event.getProperty())) {
506                                adaptToTextStyleChange(fHighlightings[i], event, SWT.BOLD);
507                                fPresenter.highlightingStyleChanged(fHighlightings[i]);
508                                refreshNeeded= true;
509                                continue;
510                        }
511 
512                        String italicKey= SemanticHighlightings.getItalicPreferenceKey(semanticHighlighting);
513                        if (italicKey.equals(event.getProperty())) {
514                                adaptToTextStyleChange(fHighlightings[i], event, SWT.ITALIC);
515                                fPresenter.highlightingStyleChanged(fHighlightings[i]);
516                                refreshNeeded= true;
517                                continue;
518                        }
519 
520                        String strikethroughKey= SemanticHighlightings.getStrikethroughPreferenceKey(semanticHighlighting);
521                        if (strikethroughKey.equals(event.getProperty())) {
522                                adaptToTextStyleChange(fHighlightings[i], event, TextAttribute.STRIKETHROUGH);
523                                fPresenter.highlightingStyleChanged(fHighlightings[i]);
524                                refreshNeeded= true;
525                                continue;
526                        }
527 
528                        String underlineKey= SemanticHighlightings.getUnderlinePreferenceKey(semanticHighlighting);
529                        if (underlineKey.equals(event.getProperty())) {
530                                adaptToTextStyleChange(fHighlightings[i], event, TextAttribute.UNDERLINE);
531                                fPresenter.highlightingStyleChanged(fHighlightings[i]);
532                                refreshNeeded= true;
533                                continue;
534                        }
535 
536                        String enabledKey= SemanticHighlightings.getEnabledPreferenceKey(semanticHighlighting);
537                        if (enabledKey.equals(event.getProperty())) {
538                                adaptToEnablementChange(fHighlightings[i], event);
539                                fPresenter.highlightingStyleChanged(fHighlightings[i]);
540                                refreshNeeded= true;
541                                continue;
542                        }
543                }
544                
545                if (refreshNeeded && fReconciler != null)
546                        fReconciler.refresh();
547        }

549        private void adaptToEnablementChange(Highlighting highlighting, PropertyChangeEvent event) {
550                Object value= event.getNewValue();
551                boolean eventValue;
552                if (value instanceof Boolean)
553                        eventValue= ((Boolean) value).booleanValue();
554                else if (IPreferenceStore.TRUE.equals(value))
555                        eventValue= true;
556                else
557                        eventValue= false;
558                highlighting.setEnabled(eventValue);
559        }

561        private void adaptToTextForegroundChange(Highlighting highlighting, PropertyChangeEvent event) {
562                RGB rgb= null;
563 
564                Object value= event.getNewValue();
565                if (value instanceof RGB)
566                        rgb= (RGB) value;
567                else if (value instanceof String)
568                        rgb= StringConverter.asRGB((String) value);
569 
570                if (rgb != null) {
571 
572                        String property= event.getProperty();
573                        Color color= fColorManager.getColor(property);
574 
575                        if ((color == null || !rgb.equals(color.getRGB())) && fColorManager instanceof IColorManagerExtension) {
576                                IColorManagerExtension ext= (IColorManagerExtension) fColorManager;
577                                ext.unbindColor(property);
578                                ext.bindColor(property, rgb);
579                                color= fColorManager.getColor(property);
580                        }
581 
582                        TextAttribute oldAttr= highlighting.getTextAttribute();
583                        highlighting.setTextAttribute(new TextAttribute(color, oldAttr.getBackground(), oldAttr.getStyle()));
584                }
585        }

587        private void adaptToTextStyleChange(Highlighting highlighting, PropertyChangeEvent event, int styleAttribute) {
588                boolean eventValue= false;
589                Object value= event.getNewValue();
590                if (value instanceof Boolean)
591                        eventValue= ((Boolean) value).booleanValue();
592                else if (IPreferenceStore.TRUE.equals(value))
593                        eventValue= true;
594 
595                TextAttribute oldAttr= highlighting.getTextAttribute();
596                boolean activeValue= (oldAttr.getStyle() & styleAttribute) == styleAttribute;
597 
598                if (activeValue != eventValue)
599                        highlighting.setTextAttribute(new TextAttribute(oldAttr.getForeground(), oldAttr.getBackground(), eventValue ? oldAttr.getStyle() | styleAttribute : oldAttr.getStyle() & ~styleAttribute));
600        }
