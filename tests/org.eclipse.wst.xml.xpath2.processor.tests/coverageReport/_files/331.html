411                private void registerTaglib(String prefix, String uri, IStructuredDocumentRegion anchorStructuredDocumentRegion, CMDocument tld) {
412                        CMNamedNodeMap elements = tld.getElements();
413                        /*
414                         * Go through the CMDocument for any tags that must be marked as
415                         * block tags starting at the anchoring IStructuredDocumentRegion.
416                         * As the document is edited and the IStructuredDocumentRegion
417                         * moved around, the block tag enablement will automatically
418                         * follow it.
419                         */

421                                TLDElementDeclaration ed = (TLDElementDeclaration) elements.item(i);

426                         * Since modifications to StructuredDocumentRegions adjacent to a
427                         * taglib directive can cause that IStructuredDocumentRegion to be
428                         * reported, filter out any duplicated URIs. When the taglib is
429                         * actually modified, a full rebuild will occur and no duplicates
430                         * will/should be found.
431                         */
432                        boolean doTrack = true;
433                        List trackers = getTaglibTrackers();
434                        for (int i = 0; i < trackers.size(); i++) {
435                                TaglibTracker tracker = (TaglibTracker) trackers.get(i);
436                                if (tracker.getPrefix().equals(prefix) && tracker.getURI().equals(uri)) {
437                                        doTrack = false;
438                                }
439                        }

441                                addTaglibTracker(prefix, uri, anchorStructuredDocumentRegion, tld);

709                         * If hasn't been moved into the local table, do so and increment
710                         * the count. A local URI reference can be different depending on
711                         * the file from which it was referenced. Use a computed key to
712                         * keep them straight.
713                         */
714                        Object o = getSharedDocumentCache().get(cacheKey);
715                        if (o != null) {
716                                if (o instanceof TLDCacheEntry) {
717                                        TLDCacheEntry entry = (TLDCacheEntry) o;
718                                        if (_debugCache) {
719                                                System.out.println("TLDCMDocument cache hit on " + cacheKey);
720                                        }
721                                        if (entry != null && entry.modificationStamp != IResource.NULL_STAMP && entry.modificationStamp >= lastModified) {
722                                                doc = entry.document;
723                                                entry.referenceCount++;
724                                        }
725                                        else {
726                                                getSharedDocumentCache().remove(cacheKey);
727                                        }
728                                }
729                                else if (o instanceof Reference) {
730                                        TLDCacheEntry entry = (TLDCacheEntry) ((Reference) o).get();
731                                        if (entry != null) {
732                                                if (entry.modificationStamp != IResource.NULL_STAMP && entry.modificationStamp >= lastModified) {
733                                                        doc = entry.document;
734                                                        entry.referenceCount = 1;
735                                                        getSharedDocumentCache().put(cacheKey, entry);
736                                                }
737                                        }
738                                        else {
739                                                getSharedDocumentCache().remove(cacheKey);
740                                        }
741                                }
742                        }
743                        /* No document was found cached, create a new one and share it */
744                        if (doc == null) {
745                                if (_debugCache) {
746                                        System.out.println("TLDCMDocument cache miss on " + cacheKey);
747                                }
748                                CMDocument document = loadTaglib(reference);
749                                if (document != null) {
750                                        TLDCacheEntry entry = new TLDCacheEntry();
751                                        doc = entry.document = document;
752                                        entry.referenceCount = 1;
753                                        entry.modificationStamp = getModificationStamp(reference);
754                                        getSharedDocumentCache().put(cacheKey, entry);
755                                }
756                        }
757                        if (doc != null) {
758                                getDocuments().put(cacheKey, doc);
759                        }
760                }
761                return doc;
762        }

764        private long getModificationStamp(String reference) {
765                ITaglibRecord record = TaglibIndex.resolve(getCurrentParserPath().toString(), reference, false);
766                long modificationStamp = IResource.NULL_STAMP;
767                if (record != null) {
768                        switch (record.getRecordType()) {
769                                case (ITaglibRecord.TLD) : {
770                                        IFile tldfile = ResourcesPlugin.getWorkspace().getRoot().getFile(((ITLDRecord) record).getPath());
771                                        if (tldfile.isAccessible()) {
772                                                modificationStamp = tldfile.getModificationStamp();
773                                        }
774                                }
775                                        break;
776                                case (ITaglibRecord.JAR) : {
777                                        File jarfile = new File(((IJarRecord) record).getLocation().toOSString());
778                                        if (jarfile.exists()) {
779                                                try {
780                                                        modificationStamp = jarfile.lastModified();
781                                                }
782                                                catch (SecurityException e) {
783                                                        modificationStamp = IResource.NULL_STAMP;
784                                                }
785                                        }
786                                }
787                                        break;
788                                case (ITaglibRecord.TAGDIR) : {
789                                        IFolder tagFolder = ResourcesPlugin.getWorkspace().getRoot().getFolder(((ITagDirRecord) record).getPath());
790                                        if (tagFolder.isAccessible()) {
791                                                IResource[] members;
792                                                try {
793                                                        members = tagFolder.members();
794                                                        for (int i = 0; i < members.length; i++) {
795                                                                modificationStamp = Math.max(modificationStamp, members[i].getModificationStamp());
796                                                        }
797                                                }
798                                                catch (CoreException e) {
799                                                        modificationStamp = IResource.NULL_STAMP;
800                                                }
801                                        }
802                                }
803                                        break;
804                                case (ITaglibRecord.URL) : {
805                                        modificationStamp = IResource.NULL_STAMP;
806                                }
807                                        break;
808                                default :
809                                        break;
810                        }
811                }
812                return modificationStamp;
813        }
