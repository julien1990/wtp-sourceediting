430        private static IProblemLocation[] getHandledProblems(IProblemLocation[] locations, ContributedProcessorDescriptor processor) {
431                // implementation tries to avoid creating a new array
432                boolean allHandled= true;
433                ArrayList res= null;
434                for (int i= 0; i < locations.length; i++) {
435                        IProblemLocation curr= locations[i];
436                        if (processor.canHandleMarkerType(curr.getMarkerType())) {
437                                if (!allHandled) { // first handled problem
438                                        if (res == null) {
439                                                res= new ArrayList(locations.length - i);
440                                        }
441                                        res.add(curr);
442                                }
443                        } else if (allHandled) { 
444                                if (i > 0) { // first non handled problem 
445                                        res= new ArrayList(locations.length - i);
446                                        for (int k= 0; k < i; k++) {
447                                                res.add(locations[k]);
448                                        }
449                                }
450                                allHandled= false;
451                        }
452                }
453                if (allHandled) {
454                        return locations;
455                }
456                if (res == null) {
457                        return null;
458                }
459                return (IProblemLocation[]) res.toArray(new IProblemLocation[res.size()]);
460        }
