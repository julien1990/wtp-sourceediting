617    private void revealElementInEditor(Object elem, Viewer originViewer) {
618        // only allow revealing when the type hierarchy is the active pagae
619        // no revealing after selection events due to model changes
620        if (getSite().getPage().getActivePart() != this) {
621            return;
622        }
623 
624        if (fSelectionProviderMediator.getViewerInFocus() != originViewer) {
625            return;
626        }
627 
628        if (elem instanceof MethodWrapper) {
629            CallLocation callLocation = CallHierarchy.getCallLocation(elem);
630 
631            if (callLocation != null) {
632                IEditorPart editorPart = CallHierarchyUI.isOpenInEditor(callLocation);
633 
634                if (editorPart != null) {
635                    getSite().getPage().bringToTop(editorPart);
636 
637                    if (editorPart instanceof ITextEditor) {
638                        ITextEditor editor = (ITextEditor) editorPart;
639                        editor.selectAndReveal(callLocation.getStart(),
640                            (callLocation.getEnd() - callLocation.getStart()));
641                    }
642                }
643            } else {
644                IEditorPart editorPart = CallHierarchyUI.isOpenInEditor(elem);
645                getSite().getPage().bringToTop(editorPart);
646                EditorUtility.revealInEditor(editorPart,
647                    ((MethodWrapper) elem).getMember());
648            }
649        } else if (elem instanceof IJavaScriptElement) {
650            IEditorPart editorPart = EditorUtility.isOpenInEditor(elem);
651 
652            if (editorPart != null) {
653                //            getSite().getPage().removePartListener(fPartListener);
654                getSite().getPage().bringToTop(editorPart);
655                EditorUtility.revealInEditor(editorPart, (IJavaScriptElement) elem);
656 
657                //            getSite().getPage().addPartListener(fPartListener);
658            }
659        }
