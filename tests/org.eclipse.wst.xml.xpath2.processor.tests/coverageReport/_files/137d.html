446        private static GrayedCheckedModel createGrayedCheckedModel(IAdaptable[] elements, IWorkingSet[] workingSets) {
447                GrayedCheckedModelElement[] result= new GrayedCheckedModelElement[workingSets.length];
448                
449                for (int i= 0; i < workingSets.length; i++) {
450                        IWorkingSet set= workingSets[i];
451                        
452                        int checkCount= 0;
453                        for (int j= 0; j < elements.length; j++) {
454                                IAdaptable adapted= adapt(set, elements[j]);

466        private void updateWorkingSets(IWorkingSet[] newWorkingSets, IWorkingSet[] grayedWorkingSets, IAdaptable[] elements) {
467                HashSet selectedSets= new HashSet(Arrays.asList(newWorkingSets));
468                HashSet grayedSets= new HashSet(Arrays.asList(grayedWorkingSets));
469                IWorkingSet[] workingSets= getAllWorkingSets();
470                
471                for (int i= 0; i < workingSets.length; i++) {
472                        IWorkingSet workingSet= workingSets[i];
473                        if (isValidWorkingSet(workingSet) && !selectedSets.contains(workingSet) && !grayedSets.contains(workingSet)) {
474                                for (int j= 0; j < elements.length; j++) {                                                        
475                                        IAdaptable adapted= adapt(workingSet, elements[j]);
476                                        if (adapted != null && contains(workingSet, adapted)) {
477                                                remove(workingSet, adapted);
478                                        }
479                                }
480                        }
481                }
482 
483                for (int i= 0; i < newWorkingSets.length; i++) {
484                        IWorkingSet set= newWorkingSets[i];
485                        if (isValidWorkingSet(set) && !grayedSets.contains(set)) {
486                                for (int j= 0; j < elements.length; j++) {                                                
487                                        IAdaptable adapted= adapt(set, elements[j]);
488                                        if (adapted != null && !contains(set, adapted)) {
489                                                add(set, adapted);
490                                        }
491                                }
492                        }
493                }
494        }
