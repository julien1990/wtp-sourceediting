318        private int computeJavaAdornmentFlags(IJavaScriptElement element, int renderFlags) {
319                int flags= 0;
320                if (showOverlayIcons(renderFlags) && element instanceof IMember) {
321                        try {
322                                IMember member= (IMember) element;
323                                
324                                if (element.getElementType() == IJavaScriptElement.METHOD && ((IFunction)element).isConstructor())
325                                        flags |= JavaScriptElementImageDescriptor.CONSTRUCTOR;
326                                        
327                                int modifiers= member.getFlags();
328                                if (Flags.isAbstract(modifiers) && confirmAbstract(member))
329                                        flags |= JavaScriptElementImageDescriptor.ABSTRACT;
330                                if (Flags.isFinal(modifiers) || isInterfaceOrAnnotationField(member) || isEnumConstant(member, modifiers))
331                                        flags |= JavaScriptElementImageDescriptor.FINAL;
332                                if (Flags.isStatic(modifiers) || isInterfaceOrAnnotationFieldOrType(member) || isEnumConstant(member, modifiers))
333                                        flags |= JavaScriptElementImageDescriptor.STATIC;
334                                
335                                if (Flags.isDeprecated(modifiers))
336                                        flags |= JavaScriptElementImageDescriptor.DEPRECATED;                        
337                        } catch (JavaScriptModelException e) {
338                                // do nothing. Can't compute runnable adornment or get flags
339                        }
340                }
341                return flags;
342        }
