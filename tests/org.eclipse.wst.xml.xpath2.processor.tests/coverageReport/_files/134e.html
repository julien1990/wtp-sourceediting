352                        Object elem= selElements.get(i);
353                        if (elem instanceof CPListElementAttribute) {
354                                CPListElementAttribute attrib= (CPListElementAttribute) elem;
355                                String key= attrib.getKey();
356                                if (attrib.isBuiltIn()) {
357                                        Object value= null;
358                                        if (key.equals(CPListElement.ACCESSRULES)) {
359                                                value= new IAccessRule[0];
360                                        }
361                                        attrib.setValue(value);
362                                }else {
363                                        removeCustomAttribute(attrib);
364                                }
365                                selElements.remove(i);
366                                if (attrib.getParent().getParentContainer() instanceof CPListElement) { // inside a container: apply changes right away
367                                        CPListElement containerEntry= attrib.getParent();
368                                        HashSet changedAttributes= (HashSet) containerEntriesToUpdate.get(containerEntry);
369                                        if (changedAttributes == null) {
370                                                changedAttributes= new HashSet();
371                                                containerEntriesToUpdate.put(containerEntry, changedAttributes);
372                                        }
373                                        changedAttributes.add(key); // collect the changed attributes
374                                }
375                        }else if(elem instanceof  CPListElement) {
376                                CPListElement listElem = (CPListElement)elem;
377                                if (listElem.getEntryKind()==IIncludePathEntry.CPE_CONTAINER)
378                                {
379                                        JsGlobalScopeContainerInitializer init = listElem.getContainerInitializer();
380                                        init.removeFromProject(fCurrJProject);
381                                }
382                                
383                                
384                        }
385                
386                }
387                if (selElements.isEmpty()) {
388                        fLibrariesList.refresh();
389                        fClassPathList.dialogFieldChanged(); // validate
390                } else {
391                        fLibrariesList.removeElements(selElements);
392                }
393                for (Iterator iter= containerEntriesToUpdate.entrySet().iterator(); iter.hasNext();) {
394                        Map.Entry entry= (Entry) iter.next();
395                        CPListElement curr= (CPListElement) entry.getKey();
396                        HashSet attribs= (HashSet) entry.getValue();
397                        String[] changedAttributes= (String[]) attribs.toArray(new String[attribs.size()]);
398                        IIncludePathEntry changedEntry= curr.getClasspathEntry();
399                        updateContainerEntry(changedEntry, changedAttributes, fCurrJProject, ((CPListElement) curr.getParentContainer()).getPath());
400                }
401        }

403        private boolean canRemove(List selElements) {

405                        return false;
406                }
407                for (int i= 0; i < selElements.size(); i++) {
408                        Object elem= selElements.get(i);
409                        if (elem instanceof CPListElementAttribute) {
410                                CPListElementAttribute attrib= (CPListElementAttribute) elem;
411                                if (attrib.isNonModifiable()) {
412                                        return false;
413                                }
414                                if (attrib.isBuiltIn()) {
415                                        if (attrib.getParent().isInContainer(JavaRuntime.JRE_CONTAINER) && CPListElement.ACCESSRULES.equals(attrib.getKey())) {
416                                                return false; // workaround for 166519 until we have full story
417                                        }
418                                        if (attrib.getKey().equals(CPListElement.ACCESSRULES)) {
419                                                return ((IAccessRule[]) attrib.getValue()).length > 0;
420                                        }
421                                        if (attrib.getValue() == null) {
422                                                return false;
423                                        }
424                                } else {
425                                        if (!canRemoveCustomAttribute(attrib)) {
426                                                return false;
427                                        }
428                                }
429                        } else if (elem instanceof CPListElement) {
430                                CPListElement curr= (CPListElement) elem;
431                                if (curr.getEntryKind()==IIncludePathEntry.CPE_CONTAINER)
432                                        return !curr.isInNonModifiableContainer();
433                                if (curr.getParentContainer() != null) {
434                                        return false;
435                                }
436                        } else { // unknown element
437                                return false;
438                        }
439                }                
440                return true;
441        }
