346                } else if (oldEdit instanceof DeleteEdit) {
347                        DeleteEdit edit= (DeleteEdit) oldEdit;

350                        newEdit= new MultiTextEdit();                        
351                } else {
352                        return null; // not supported
353                }
354                TextEdit[] children= oldEdit.getChildren();
355                for (int i= 0; i < children.length; i++) {
356                        TextEdit shifted= shifEdit(children[i], diff);
357                        if (shifted != null) {
358                                newEdit.addChild(shifted);
359                        }
360                }
361                return newEdit;
362        }
