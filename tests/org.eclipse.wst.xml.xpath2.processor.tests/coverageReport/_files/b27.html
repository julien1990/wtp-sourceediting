378                if (!source15 && methDecl != null && this.valuePositions != -1) {
379                        methScope.problemReporter().javadocUnexpectedTag((int)(this.valuePositions>>>32), (int) this.valuePositions);

385                        this.invalidParameters[i].resolve(methScope, false, false);

389        private void resolveReference(Expression reference, Scope scope) {
390 
391                // Perform resolve
392                int problemCount = scope.referenceContext().compilationResult().problemCount;
393                switch (scope.kind) {
394                        case Scope.METHOD_SCOPE:
395                                reference.resolveType((MethodScope)scope);
396                                break;
397                        case Scope.CLASS_SCOPE:
398                                reference.resolveType((ClassScope)scope);
399                                break;
400                }
401                boolean hasProblems = scope.referenceContext().compilationResult().problemCount > problemCount;
402 
403                // Verify field references
404                boolean source15 = scope.compilerOptions().sourceLevel >= ClassFileConstants.JDK1_5;
405                int scopeModifiers = -1;
406                if (reference instanceof JavadocFieldReference) {
407                        JavadocFieldReference fieldRef = (JavadocFieldReference) reference;
408 
409                        // Verify if this is a method reference
410                        // see bug https://bugs.eclipse.org/bugs/show_bug.cgi?id=51911
411                        if (fieldRef.methodBinding != null) {
412                                // cannot refer to method for @value tag
413                                if (fieldRef.tagValue == JavadocTagConstants.TAG_VALUE_VALUE) {
414                                        if (scopeModifiers == -1) scopeModifiers = scope.getDeclarationModifiers();
415                                        scope.problemReporter().javadocInvalidValueReference(fieldRef.sourceStart, fieldRef.sourceEnd, scopeModifiers);
416                                }
417                                else if (fieldRef.receiverType != null) {
418                                        if (scope.enclosingSourceType().isCompatibleWith(fieldRef.receiverType)) {
419                                                        fieldRef.bits |= ASTNode.SuperAccess;
420                                                }
421                                        fieldRef.methodBinding = scope.findMethod((ReferenceBinding)fieldRef.receiverType, fieldRef.token, new TypeBinding[0], fieldRef);
422                                }
423                        }
424 
425                        // Verify whether field ref should be static or not (for @value tags)
426                        else if (source15 && fieldRef.binding != null && fieldRef.binding.isValidBinding()) {
427                                if (fieldRef.tagValue == JavadocTagConstants.TAG_VALUE_VALUE && !fieldRef.binding.isStatic()) {
428                                        if (scopeModifiers == -1) scopeModifiers = scope.getDeclarationModifiers();
429                                        scope.problemReporter().javadocInvalidValueReference(fieldRef.sourceStart, fieldRef.sourceEnd, scopeModifiers);
430                                }
431                        }
432 
433                        // Verify type references
434                        if (!hasProblems && fieldRef.binding != null && fieldRef.binding.isValidBinding() && fieldRef.receiverType instanceof ReferenceBinding) {
435                                ReferenceBinding resolvedType = (ReferenceBinding) fieldRef.receiverType;
436                                verifyTypeReference(fieldRef, fieldRef.receiver, scope, source15, resolvedType, fieldRef.binding.modifiers);
437                        }
438 
439                        // That's it for field references
440                        return;
441                }
442 
443                // Verify type references
444                if (!hasProblems && (reference instanceof JavadocSingleTypeReference || reference instanceof JavadocQualifiedTypeReference) && reference.resolvedType instanceof ReferenceBinding) {
445                        ReferenceBinding resolvedType = (ReferenceBinding) reference.resolvedType;
446                        verifyTypeReference(reference, reference, scope, source15, resolvedType, resolvedType.modifiers);
447                }
448 
449                // Verify that message reference are not used for @value tags
450                if (reference instanceof JavadocMessageSend) {
451                        JavadocMessageSend msgSend = (JavadocMessageSend) reference;
452 
453                        // tag value
454                        if (source15 && msgSend.tagValue == JavadocTagConstants.TAG_VALUE_VALUE) { // cannot refer to method for @value tag
455                                if (scopeModifiers == -1) scopeModifiers = scope.getDeclarationModifiers();
456                                scope.problemReporter().javadocInvalidValueReference(msgSend.sourceStart, msgSend.sourceEnd, scopeModifiers);
457                        }
458 
459                        // Verify type references
460                        if (!hasProblems && msgSend.binding != null && msgSend.binding.isValidBinding() && msgSend.actualReceiverType instanceof ReferenceBinding) {
461                                ReferenceBinding resolvedType = (ReferenceBinding) msgSend.actualReceiverType;
462                                verifyTypeReference(msgSend, msgSend.receiver, scope, source15, resolvedType, msgSend.binding.modifiers);
463                        }
464                }
465 
466                // Verify that constructor reference are not used for @value tags
467                else if (reference instanceof JavadocAllocationExpression) {
468                        JavadocAllocationExpression alloc = (JavadocAllocationExpression) reference;
469 
470                        // tag value
471                        if (source15 && alloc.tagValue == JavadocTagConstants.TAG_VALUE_VALUE) { // cannot refer to method for @value tag
472                                if (scopeModifiers == -1) scopeModifiers = scope.getDeclarationModifiers();
473                                scope.problemReporter().javadocInvalidValueReference(alloc.sourceStart, alloc.sourceEnd, scopeModifiers);
474                        }
475 
476                        // Verify type references
477                        if (!hasProblems && alloc.binding != null && alloc.binding.isValidBinding() && alloc.resolvedType instanceof ReferenceBinding) {
478                                ReferenceBinding resolvedType = (ReferenceBinding) alloc.resolvedType;
479                                verifyTypeReference(alloc, alloc.type, scope, source15, resolvedType, alloc.binding.modifiers);
480                        }
481                }
482 
483                // Verify that there's no type variable reference
484                // (javadoc does not accept them and this is not a referenced bug or requested enhancement)
485                if (reference.resolvedType != null && reference.resolvedType.isTypeVariable()) {
486                        scope.problemReporter().javadocInvalidReference(reference.sourceStart, reference.sourceEnd);
487                }
488        }

490        /*
491         * Resolve @param tags while method scope
492         */
493        private void resolveParamTags(MethodScope scope, boolean reportMissing, boolean considerParamRefAsUsage) {
494                AbstractMethodDeclaration methodDecl = scope.referenceMethod();
495                int paramTagsSize = this.paramReferences == null ? 0 : this.paramReferences.length;
496 
497                // If no referenced method (field initializer for example) then report a problem for each param tag
498                if (methodDecl == null) {
499                        for (int i = 0; i < paramTagsSize; i++) {
500                                JavadocSingleNameReference param = this.paramReferences[i];
501                                scope.problemReporter().javadocUnexpectedTag(param.tagSourceStart, param.tagSourceEnd);
502                        }
503                        return;
504                }
505 
506                // If no param tags then report a problem for each method argument
507                int argumentsSize = methodDecl.arguments == null ? 0 : methodDecl.arguments.length;
508                if (paramTagsSize == 0) {
509                        if (reportMissing) {
510                                for (int i = 0; i < argumentsSize; i++) {
511                                        Argument arg = methodDecl.arguments[i];
512                                        scope.problemReporter().javadocMissingParamTag(arg.name, arg.sourceStart, arg.sourceEnd, methodDecl.binding.modifiers);
513                                }
514                        }
515                } else {
516                        LocalVariableBinding[] bindings = new LocalVariableBinding[paramTagsSize];
517                        int maxBindings = 0;
518 
519                        // Scan all @param tags
520                        for (int i = 0; i < paramTagsSize; i++) {
521                                JavadocSingleNameReference param = this.paramReferences[i];
522                                param.resolve(scope, true, considerParamRefAsUsage);
523                                if (param.binding != null && param.binding.isValidBinding()) {
524                                        // Verify duplicated tags
525                                        boolean found = false;
526                                        for (int j = 0; j < maxBindings && !found; j++) {
527                                                if (bindings[j] == param.binding) {
528                                                        scope.problemReporter().javadocDuplicatedParamTag(param.token, param.sourceStart, param.sourceEnd, methodDecl.binding.modifiers);
529                                                        found = true;
530                                                }
531                                        }
532                                        if (!found) {
533                                                bindings[maxBindings++] = (LocalVariableBinding) param.binding;
534                                        }
535                                }
536                        }
537 
538                        // Look for undocumented arguments
539                        if (reportMissing) {
540                                for (int i = 0; i < argumentsSize; i++) {
541                                        Argument arg = methodDecl.arguments[i];
542                                        boolean found = false;
543                                        for (int j = 0; j < maxBindings && !found; j++) {
544                                                LocalVariableBinding binding = bindings[j];
545                                                if (arg.binding == binding) {
546                                                        found = true;
547                                                }
548                                        }
549                                        if (!found) {
550                                                scope.problemReporter().javadocMissingParamTag(arg.name, arg.sourceStart, arg.sourceEnd, methodDecl.binding.modifiers);
551                                        }
552                                }
553                        }
554                }
555        }

557        /*
558         * Resolve @param tags for type parameters
559         */
560        private void resolveTypeParameterTags(Scope scope, boolean reportMissing) {
561                int paramTypeParamLength = this.paramTypeParameters == null ? 0 : this.paramTypeParameters.length;
562 
563                // Get declaration infos
564                TypeParameter[] parameters = null;
565                TypeVariableBinding[] typeVariables = null;
566                int modifiers = -1;
567                switch (scope.kind) {
568                        case Scope.METHOD_SCOPE:
569                                AbstractMethodDeclaration methodDeclaration = ((MethodScope)scope).referenceMethod();
570                                // If no referenced method (field initializer for example) then report a problem for each param tag
571                                if (methodDeclaration == null) {
572                                        for (int i = 0; i < paramTypeParamLength; i++) {
573                                                JavadocSingleNameReference param = this.paramReferences[i];
574                                                scope.problemReporter().javadocUnexpectedTag(param.tagSourceStart, param.tagSourceEnd);
575                                        }
576                                        return;
577                                }
578                                parameters = methodDeclaration.typeParameters();
579                                typeVariables = methodDeclaration.binding.typeVariables;
580                                modifiers = methodDeclaration.binding.modifiers;
581                                break;
582                        case Scope.CLASS_SCOPE:
583                                TypeDeclaration typeDeclaration = ((ClassScope) scope).referenceContext;
584                                parameters = typeDeclaration.typeParameters;
585                                typeVariables = typeDeclaration.binding.typeVariables;
586                                modifiers = typeDeclaration.binding.modifiers;
587                                break;
588                }
589 
590                // If no type variables then report a problem for each param type parameter tag
591                if (typeVariables == null || typeVariables.length == 0) {
592                        for (int i = 0; i < paramTypeParamLength; i++) {
593                                JavadocSingleTypeReference param = this.paramTypeParameters[i];
594                                scope.problemReporter().javadocUnexpectedTag(param.tagSourceStart, param.tagSourceEnd);
595                        }
596                        return;
597                }
598 
599                // If no param tags then report a problem for each declaration type parameter
600                if (parameters != null) {
601                        int typeParametersLength = parameters.length;
602                        if (paramTypeParamLength == 0) {
603                                if (reportMissing) {
604                                        for (int i = 0, l=typeParametersLength; i<l; i++) {
605                                                scope.problemReporter().javadocMissingParamTag(parameters[i].name, parameters[i].sourceStart, parameters[i].sourceEnd, modifiers);
606                                        }
607                                }
608 
609                        // Otherwise verify that all param tags match type parameters
610                        } else if (typeVariables.length == typeParametersLength) {
611                                TypeVariableBinding[] bindings = new TypeVariableBinding[paramTypeParamLength];
612 
613                                // Scan all @param tags
614                                for (int i = 0; i < paramTypeParamLength; i++) {
615                                        JavadocSingleTypeReference param = this.paramTypeParameters[i];
616                                        TypeBinding paramBindind = param.internalResolveType(scope);
617                                        if (paramBindind != null && paramBindind.isValidBinding()) {
618                                                if (paramBindind.isTypeVariable()) {
619                                                        // Verify duplicated tags
620                                                        boolean duplicate = false;
621                                                        for (int j = 0; j < i && !duplicate; j++) {
622                                                                if (bindings[j] == param.resolvedType) {
623                                                                        scope.problemReporter().javadocDuplicatedParamTag(param.token, param.sourceStart, param.sourceEnd, modifiers);
624                                                                        duplicate = true;
625                                                                }
626                                                        }
627                                                        if (!duplicate) {
628                                                                bindings[i] = (TypeVariableBinding) param.resolvedType;
629                                                        }
630                                                } else {
631                                                        scope.problemReporter().javadocUndeclaredParamTagName(param.token, param.sourceStart, param.sourceEnd, modifiers);
632                                                }
633                                        }
634                                }
635 
636                                // Look for undocumented type parameters
637                                for (int i = 0; i < typeParametersLength; i++) {
638                                        TypeParameter parameter = parameters[i];
639                                        boolean found = false;
640                                        for (int j = 0; j < paramTypeParamLength && !found; j++) {
641                                                if (parameter.binding == bindings[j]) {
642                                                        found = true;
643                                                        bindings[j] = null;
644                                                }
645                                        }
646                                        if (!found && reportMissing) {
647                                                scope.problemReporter().javadocMissingParamTag(parameter.name, parameter.sourceStart, parameter.sourceEnd, modifiers);
648                                        }
649                                }
650 
651                                // Report invalid param
652                                for (int i=0; i<paramTypeParamLength; i++) {
653                                        if (bindings[i] != null) {
654                                                JavadocSingleTypeReference param = this.paramTypeParameters[i];
655                                                scope.problemReporter().javadocUndeclaredParamTagName(param.token, param.sourceStart, param.sourceEnd, modifiers);
656                                        }
657                                }
658                        }
659                }
660        }

662        /*
663         * Resolve @throws/@exception tags while method scope
664         */
665        private void resolveThrowsTags(MethodScope methScope, boolean reportMissing) {
666                AbstractMethodDeclaration md = methScope.referenceMethod();
667                int throwsTagsLength = this.exceptionReferences == null ? 0 : this.exceptionReferences.length;
668 
669                // If no referenced method (field initializer for example) then report a problem for each throws tag
670                if (md == null) {
671                        for (int i = 0; i < throwsTagsLength; i++) {
672                                TypeReference typeRef = this.exceptionReferences[i];
673                                int start = typeRef.sourceStart;
674                                int end = typeRef.sourceEnd;
675                                if (typeRef instanceof JavadocQualifiedTypeReference) {
676                                        start = ((JavadocQualifiedTypeReference) typeRef).tagSourceStart;
677                                        end = ((JavadocQualifiedTypeReference) typeRef).tagSourceEnd;
678                                } else if (typeRef instanceof JavadocSingleTypeReference) {
679                                        start = ((JavadocSingleTypeReference) typeRef).tagSourceStart;
680                                        end = ((JavadocSingleTypeReference) typeRef).tagSourceEnd;
681                                }
682                                methScope.problemReporter().javadocUnexpectedTag(start, end);
683                        }
684                        return;
685                }
686 
687                // If no throws tags then report a problem for each method thrown exception
688                int boundExceptionLength = (md.binding == null) ? 0 : md.binding.thrownExceptions.length;
689                int thrownExceptionLength = md.thrownExceptions == null ? 0 : md.thrownExceptions.length;
690                if (throwsTagsLength == 0) {
691                        if (reportMissing) {
692                                for (int i = 0; i < boundExceptionLength; i++) {
693                                        ReferenceBinding exceptionBinding = md.binding.thrownExceptions[i];
694                                        if (exceptionBinding != null && exceptionBinding.isValidBinding()) { // flag only valid class name
695                                                int j=i;
696                                                while (j<thrownExceptionLength && exceptionBinding != md.thrownExceptions[j].resolvedType) j++;
697                                                if (j<thrownExceptionLength) {
698                                                        methScope.problemReporter().javadocMissingThrowsTag(md.thrownExceptions[j], md.binding.modifiers);
699                                                }
700                                        }
701                                }
702                        }
703                } else {
704                        int maxRef = 0;
705                        TypeReference[] typeReferences = new TypeReference[throwsTagsLength];
706 
707                        // Scan all @throws tags
708                        for (int i = 0; i < throwsTagsLength; i++) {
709                                TypeReference typeRef = this.exceptionReferences[i];
710                                typeRef.resolve(methScope);
711                                TypeBinding typeBinding = typeRef.resolvedType;
712 
713                                if (typeBinding != null && typeBinding.isValidBinding() && typeBinding.isClass()) {
714                                        // accept only valid class binding
715                                        typeReferences[maxRef++] = typeRef;
716                                }
717                        }
718 
719                        // Look for undocumented thrown exception
720                        for (int i = 0; i < boundExceptionLength; i++) {
721                                ReferenceBinding exceptionBinding = md.binding.thrownExceptions[i];
722                                if (exceptionBinding != null) exceptionBinding = (ReferenceBinding) exceptionBinding.erasure();
723                                boolean found = false;
724                                for (int j = 0; j < maxRef && !found; j++) {
725                                        if (typeReferences[j] != null) {
726                                                TypeBinding typeBinding = typeReferences[j].resolvedType;
727                                                if (exceptionBinding == typeBinding) {
728                                                        found = true;
729                                                        typeReferences[j] = null;
730                                                }
731                                        }
732                                }
733                                if (!found && reportMissing) {
734                                        if (exceptionBinding != null && exceptionBinding.isValidBinding()) { // flag only valid class name
735                                                int k=i;
736                                                while (k<thrownExceptionLength && exceptionBinding != md.thrownExceptions[k].resolvedType) k++;
737                                                if (k<thrownExceptionLength) {
738                                                        methScope.problemReporter().javadocMissingThrowsTag(md.thrownExceptions[k], md.binding.modifiers);
739                                                }
740                                        }
741                                }
742                        }
743 
744                        // Verify additional @throws tags
745                        for (int i = 0; i < maxRef; i++) {
746                                TypeReference typeRef = typeReferences[i];
747                                if (typeRef != null) {
748                                        boolean compatible = false;
749                                        // thrown exceptions subclasses are accepted
750                                        for (int j = 0; j<thrownExceptionLength && !compatible; j++) {
751                                                TypeBinding exceptionBinding = md.thrownExceptions[j].resolvedType;
752                                                if (exceptionBinding != null) {
753                                                        compatible = typeRef.resolvedType.isCompatibleWith(exceptionBinding);
754                                                }
755                                        }
756 
757                                        //  If not compatible only complain on unchecked exception
758                                        if (!compatible && !typeRef.resolvedType.isUncheckedException(false)) {
759                                                methScope.problemReporter().javadocInvalidThrowsClassName(typeRef, md.binding.modifiers);
760                                        }
761                                }
762                        }
763                }
764        }
