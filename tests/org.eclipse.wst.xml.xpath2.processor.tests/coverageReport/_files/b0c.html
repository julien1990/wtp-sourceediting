441        private void reportNLSProblems() {
442                if (this.nlsTags != null || this.stringLiterals != null) {
443                        final int stringLiteralsLength = this.stringLiteralsPtr;
444                        final int nlsTagsLength = this.nlsTags == null ? 0 : this.nlsTags.length;
445                        if (stringLiteralsLength == 0) {
446                                if (nlsTagsLength != 0) {
447                                        for (int i = 0; i < nlsTagsLength; i++) {
448                                                NLSTag tag = this.nlsTags[i];
449                                                if (tag != null) {
450                                                        scope.problemReporter().unnecessaryNLSTags(tag.start, tag.end);
451                                                }
452                                        }
453                                }
454                        } else if (nlsTagsLength == 0) {
455                                // resize string literals
456                                if (this.stringLiterals.length != stringLiteralsLength) {
457                                        System.arraycopy(this.stringLiterals, 0, (stringLiterals = new StringLiteral[stringLiteralsLength]), 0, stringLiteralsLength);
458                                }
459                                Arrays.sort(this.stringLiterals, STRING_LITERAL_COMPARATOR);
460                                for (int i = 0; i < stringLiteralsLength; i++) {
461                                        scope.problemReporter().nonExternalizedStringLiteral(this.stringLiterals[i]);
462                                }
463                        } else {
464                                // need to iterate both arrays to find non matching elements
465                                if (this.stringLiterals.length != stringLiteralsLength) {
466                                        System.arraycopy(this.stringLiterals, 0, (stringLiterals = new StringLiteral[stringLiteralsLength]), 0, stringLiteralsLength);
467                                }
468                                Arrays.sort(this.stringLiterals, STRING_LITERAL_COMPARATOR);
469                                int indexInLine = 1;
470                                int lastLineNumber = -1;
471                                StringLiteral literal = null;
472                                int index = 0;
473                                int i = 0;
474                                stringLiteralsLoop: for (; i < stringLiteralsLength; i++) {
475                                        literal = this.stringLiterals[i];
476                                        final int literalLineNumber = literal.lineNumber;
477                                        if (lastLineNumber != literalLineNumber) {
478                                                indexInLine = 1;
479                                                lastLineNumber = literalLineNumber;
480                                        } else {
481                                                indexInLine++;
482                                        }
483                                        if (index < nlsTagsLength) {
484                                                nlsTagsLoop: for (; index < nlsTagsLength; index++) {
485                                                        NLSTag tag = this.nlsTags[index];
486                                                        if (tag == null) continue nlsTagsLoop;
487                                                        int tagLineNumber = tag.lineNumber;
488                                                        if (literalLineNumber < tagLineNumber) {
489                                                                scope.problemReporter().nonExternalizedStringLiteral(literal);
490                                                                continue stringLiteralsLoop;
491                                                        } else if (literalLineNumber == tagLineNumber) {
492                                                                if (tag.index == indexInLine) {
493                                                                        this.nlsTags[index] = null;
494                                                                        index++;
495                                                                        continue stringLiteralsLoop;
496                                                                } else {
497                                                                        nlsTagsLoop2: for (int index2 = index + 1; index2 < nlsTagsLength; index2++) {
498                                                                                NLSTag tag2 = this.nlsTags[index2];
499                                                                                if (tag2 == null) continue nlsTagsLoop2;
500                                                                                int tagLineNumber2 = tag2.lineNumber;
501                                                                                if (literalLineNumber == tagLineNumber2) {
502                                                                                        if (tag2.index == indexInLine) {
503                                                                                                this.nlsTags[index2] = null;
504                                                                                                continue stringLiteralsLoop;
505                                                                                        } else {
506                                                                                                continue nlsTagsLoop2;
507                                                                                        }
508                                                                                } else {
509                                                                                        scope.problemReporter().nonExternalizedStringLiteral(literal);
510                                                                                        continue stringLiteralsLoop;
511                                                                                }
512                                                                        }
513                                                                        scope.problemReporter().nonExternalizedStringLiteral(literal);
514                                                                        continue stringLiteralsLoop;
515                                                                }
516                                                        } else {
517                                                                scope.problemReporter().unnecessaryNLSTags(tag.start, tag.end);
518                                                                continue nlsTagsLoop;
519                                                        }
520                                                }
521                                        }
522                                        // all nls tags have been processed, so remaining string literals are not externalized
523                                        break stringLiteralsLoop;
524                                }
525                                for (; i < stringLiteralsLength; i++) {
526                                        scope.problemReporter().nonExternalizedStringLiteral(this.stringLiterals[i]);
527                                }
528                                if (index < nlsTagsLength) {
529                                        for (; index < nlsTagsLength; index++) {
530                                                NLSTag tag = this.nlsTags[index];
531                                                if (tag != null) {
532                                                        scope.problemReporter().unnecessaryNLSTags(tag.start, tag.end);
533                                                }
534                                        }
535                                }
536                        }
