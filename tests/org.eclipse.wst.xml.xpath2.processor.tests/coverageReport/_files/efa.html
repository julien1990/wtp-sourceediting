428                /**
429                 * Returns <code>true</code> if <code>subTypeSignature</code>
430                 * describes a type which is a true sub type of the type described by
431                 * <code>superTypeSignature</code>.
432                 * 
433                 * @param subTypeSignature the potential subtype's signature
434                 * @param superTypeSignature the potential supertype's signature
435                 * @return <code>true</code> if the inheritance relationship holds
436                 */
437                private boolean isTrueSubtypeOf(String subTypeSignature, String superTypeSignature) {
438                        // try cheap test first
439                        if (subTypeSignature.equals(superTypeSignature))
440                                return true;
441                        
442                        if (SignatureUtil.isJavaLangObject(subTypeSignature))
443                                return false; // Object has no super types
444                        
445                        if (Signature.getTypeSignatureKind(subTypeSignature) != Signature.BASE_TYPE_SIGNATURE && SignatureUtil.isJavaLangObject(superTypeSignature)) 
446                                return true;
447                        
448                        IJavaScriptProject project= fUnit.getJavaScriptProject();
449                        
450                        try {
451                                
452                                if ((Signature.getTypeSignatureKind(subTypeSignature) & (Signature.TYPE_VARIABLE_SIGNATURE | Signature.CLASS_TYPE_SIGNATURE)) == 0)
453                                        return false;
454                                IType subType= project.findType(SignatureUtil.stripSignatureToFQN(subTypeSignature));
455                                if (subType == null)
456                                        return false;
457                                
458                                if ((Signature.getTypeSignatureKind(superTypeSignature) & (Signature.TYPE_VARIABLE_SIGNATURE | Signature.CLASS_TYPE_SIGNATURE)) == 0)
459                                        return false;
460                                IType superType= project.findType(SignatureUtil.stripSignatureToFQN(superTypeSignature));
461                                if (superType == null)
462                                        return false;
463                                
464                                ITypeHierarchy hierarchy= subType.newSupertypeHierarchy(null);
465                                IType[] types= hierarchy.getAllSupertypes(subType);
466                                
467                                for (int i= 0; i < types.length; i++)
468                                        if (types[i].equals(superType))
469                                                return true;
470                        } catch (JavaScriptModelException e) {
471                                // ignore and return false
472                        }                        
473                        
474                        return false;
475                }

477                /**
478                 * Returns <code>true</code> if <code>signature</code> is a concrete type signature,
479                 * <code>false</code> if it is a type variable.
480                 * 
481                 * @param signature the signature to check
482                 * @param context the context inside which to resolve the type
483                 * @throws JavaScriptModelException if finding the type fails
484                 */
485                private boolean isConcreteType(String signature, IType context) throws JavaScriptModelException {
486                        // Inexpensive check for the variable type first
487                        if (Signature.TYPE_VARIABLE_SIGNATURE == Signature.getTypeSignatureKind(signature))
488                                return false;

490                        // try and resolve otherwise
491                        if (context.isBinary()) {
492                                return fUnit.getJavaScriptProject().findType(SignatureUtil.stripSignatureToFQN(signature)) != null;
493                        } else {
494                                return context.resolveType(SignatureUtil.stripSignatureToFQN(signature)) != null;
495                        }
496                }
