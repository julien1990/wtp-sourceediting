598public char[] computeConstantPoolName(LocalTypeBinding localType) {
599        if (localType.constantPoolName() != null) {
600                return localType.constantPoolName();
601        }
602        // delegates to the outermost enclosing classfile, since it is the only one with a global vision of its innertypes.
603 
604        if (constantPoolNameUsage == null)
605                constantPoolNameUsage = new HashtableOfType();
606 
607        ReferenceBinding outerMostEnclosingType = localType.scope.outerMostClassScope().enclosingSourceType();
608 
609        // ensure there is not already such a local type name defined by the user
610        int index = 0;
611        char[] candidateName;
612        boolean isCompliant15 = compilerOptions().complianceLevel >= ClassFileConstants.JDK1_5;
613        while(true) {
614                if (localType.isMemberType()){
615                        if (index == 0){
616                                candidateName = CharOperation.concat(
617                                        localType.enclosingType().constantPoolName(),
618                                        localType.sourceName,
619                                        '$');
620                        } else {
621                                // in case of collision, then member name gets extra $1 inserted
622                                // e.g. class X { { class L{} new X(){ class L{} } } }
623                                candidateName = CharOperation.concat(
624                                        localType.enclosingType().constantPoolName(),
625                                        '$',
626                                        String.valueOf(index).toCharArray(),
627                                        '$',
628                                        localType.sourceName);
629                        }
630                } else if (localType.isAnonymousType()){
631                        if (isCompliant15) {
632                                // from 1.5 on, use immediately enclosing type name
633                                candidateName = CharOperation.concat(
634                                        localType.enclosingType.constantPoolName(),
635                                        String.valueOf(index+1).toCharArray(),
636                                        '$');
637                        } else {
638                                candidateName = CharOperation.concat(
639                                        outerMostEnclosingType.constantPoolName(),
640                                        String.valueOf(index+1).toCharArray(),
641                                        '$');
642                        }
643                } else {
644                        // local type
645                        if (isCompliant15) {
646                                candidateName = CharOperation.concat(
647                                        CharOperation.concat(
648                                                localType.enclosingType().constantPoolName(),
649                                                String.valueOf(index+1).toCharArray(),
650                                                '$'),
651                                        localType.sourceName);
652                        } else {
653                                candidateName = CharOperation.concat(
654                                        outerMostEnclosingType.constantPoolName(),
655                                        '$',
656                                        String.valueOf(index+1).toCharArray(),
657                                        '$',
658                                        localType.sourceName);
659                        }
660                }
661                if (constantPoolNameUsage.get(candidateName) != null) {
662                        index ++;
663                } else {
664                        constantPoolNameUsage.put(candidateName, localType);
665                        break;
666                }
667        }
668        return candidateName;
669}

700void faultInImports() {
701        if (this.typeOrPackageCache != null)
702                return; // can be called when a field constant is resolved before static imports
703        if (referenceContext.imports == null) {
704                this.typeOrPackageCache = new HashtableOfObject(1);
705                return;
706        }
707 
708        // collect the top level type names if a single type import exists
709        int numberOfStatements = referenceContext.imports.length;
710        HashtableOfType typesBySimpleNames = null;
711        for (int i = 0; i < numberOfStatements; i++) {
712                if ((referenceContext.imports[i].bits & ASTNode.OnDemand) == 0) {
713                        typesBySimpleNames = new HashtableOfType(topLevelTypes.length + numberOfStatements);
714                        for (int j = 0, length = topLevelTypes.length; j < length; j++)
715                                typesBySimpleNames.put(topLevelTypes[j].sourceName, topLevelTypes[j]);
716                        break;
717                }
718        }
719 
720        // allocate the import array, add java.lang.* by default
721        ImportBinding[] defaultImports = getDefaultImports();
722        int numberOfImports = numberOfStatements + defaultImports.length;
723        for (int i = 0; i < numberOfStatements; i++) {
724                ImportReference importReference = referenceContext.imports[i];
725                if (((importReference.bits & ASTNode.OnDemand) != 0) && CharOperation.equals(JAVA_LANG, importReference.tokens)) {
726                        numberOfImports--;
727                        break;
728                }
729        }
730        ImportBinding[] resolvedImports = new ImportBinding[numberOfImports];
731        System.arraycopy(defaultImports, 0, resolvedImports, 0, defaultImports.length);
732        int index = defaultImports.length;
733 
734        // keep static imports with normal imports until there is a reason to split them up
735        // on demand imports continue to be packages & types. need to check on demand type imports for fields/methods
736        // single imports change from being just types to types or fields
737        nextImport : for (int i = 0; i < numberOfStatements; i++) {
738                ImportReference importReference = referenceContext.imports[i];
739                char[][] compoundName = importReference.tokens;
740 
741                // skip duplicates or imports of the current package
742                for (int j = 0; j < index; j++) {
743                        ImportBinding resolved = resolvedImports[j];
744                        if (resolved.onDemand == ((importReference.bits & ASTNode.OnDemand) != 0)) {
745                                if (CharOperation.equals(compoundName, resolved.compoundName)) {
746                                        continue nextImport;
747                                }
748                        }
749                }
750                if ((importReference.bits & ASTNode.OnDemand) != 0) {
751                        if (CharOperation.equals(compoundName, currentPackageName)) {
752                                continue nextImport;
753                        }
754 
755                        Binding importBinding = findImport(compoundName, compoundName.length);
756                        if (!importBinding.isValidBinding()) {
757                                problemReporter().importProblem(importReference, importBinding);
758                                continue nextImport;
759                        }
760                        resolvedImports[index++] = new ImportBinding(compoundName, true, importBinding, importReference);
761                } else {
762                        Binding importBinding = findSingleImport(compoundName);
763                        if (!importBinding.isValidBinding()) {
764                                problemReporter().importProblem(importReference, importBinding);
765                                continue nextImport;
766                        }
767                        if (importBinding instanceof PackageBinding) {
768                                problemReporter().cannotImportPackage(importReference);
769                                continue nextImport;
770                        }
771                        ReferenceBinding conflictingType = null;
772                        if (importBinding instanceof MethodBinding) {
773                                conflictingType = (ReferenceBinding) getType(compoundName, compoundName.length);
774                                if (!conflictingType.isValidBinding())
775                                        conflictingType = null;
776                        }
777                        // collisions between an imported static field & a type should be checked according to spec... but currently not by javac
778                        if (importBinding instanceof ReferenceBinding || conflictingType != null) {
779                                ReferenceBinding referenceBinding = conflictingType == null ? (ReferenceBinding) importBinding : conflictingType;
780                                if (importReference.isTypeUseDeprecated(referenceBinding, this))
781                                        problemReporter().deprecatedType(referenceBinding, importReference);
782 
783                                ReferenceBinding existingType = typesBySimpleNames.get(compoundName[compoundName.length - 1]);
784                                if (existingType != null) {
785                                        // duplicate test above should have caught this case, but make sure
786                                        if (existingType == referenceBinding)
787                                                continue nextImport;
788                                        // either the type collides with a top level type or another imported type
789                                        for (int j = 0, length = topLevelTypes.length; j < length; j++) {
790                                                if (CharOperation.equals(topLevelTypes[j].sourceName, existingType.sourceName)) {
791                                                        problemReporter().conflictingImport(importReference);
792                                                        continue nextImport;
793                                                }
794                                        }
795                                        problemReporter().duplicateImport(importReference);
796                                        continue nextImport;
797                                }
798                                typesBySimpleNames.put(compoundName[compoundName.length - 1], referenceBinding);
799                        } 
800                        resolvedImports[index++] = conflictingType == null
801                                ? new ImportBinding(compoundName, false, importBinding, importReference)
802                                : new ImportConflictBinding(compoundName, importBinding, conflictingType, importReference);
803                }
804        }
805
