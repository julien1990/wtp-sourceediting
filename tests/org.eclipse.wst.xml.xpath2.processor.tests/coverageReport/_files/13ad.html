470        /**
471         * Appends the label for a method to a {@link StringBuffer}. Considers the M_* flags.
472         *         @param method The element to render.
473         * @param flags The rendering flags. Flags with names starting with 'M_' are considered.
474         * @param buf The buffer to append the resulting label to.
475         */                
476        public static void getMethodLabel(IFunction method, long flags, StringBuffer buf) {
477                try {
478                        BindingKey resolvedKey= getFlag(flags, USE_RESOLVED) && method.isResolved() ? new BindingKey(method.getKey()) : null;
479                        String resolvedSig= (resolvedKey != null) ? resolvedKey.toSignature() : null;
480                        
481                        // return type
482                        if (getFlag(flags, M_PRE_RETURNTYPE) && method.exists() && !method.isConstructor()) {
483                                String returnTypeSig= resolvedSig != null ? Signature.getReturnType(resolvedSig) : method.getReturnType();
484                                getTypeSignatureLabel(returnTypeSig, flags, buf);
485                                buf.append(' ');
486                        }
487                        
488                        // qualification
489                        IType declaringType = method.getDeclaringType();
490                        if (getFlag(flags, M_FULLY_QUALIFIED) ) {
491                                if (declaringType!=null && !method.isConstructor())
492                                {
493                                        getTypeLabel(method.getDeclaringType(), T_FULLY_QUALIFIED | (flags & QUALIFIER_FLAGS), buf);
494                                        buf.append('.');
495                                } else        {
496                                                buf.append('[');
497                                                getFileLabel(method, T_FULLY_QUALIFIED | (flags & QUALIFIER_FLAGS), buf);
498                                                buf.append(']');
499                                }
500 
501                        }
502                 
503                                
504                        buf.append(method.getDisplayName());
505                        
506                        // parameters
507                        buf.append('(');
508                        if (getFlag(flags, M_PARAMETER_TYPES | M_PARAMETER_NAMES)) {
509                                String[] types= null;
510                                int nParams= 0;
511                                boolean renderVarargs= false;
512                                if (getFlag(flags, M_PARAMETER_TYPES)&& getFlag(flags,SHOW_TYPE)) {
513                                        if (resolvedSig != null) {
514                                                types= Signature.getParameterTypes(resolvedSig);
515                                        } else {
516                                                types= method.getParameterTypes();
517                                        }
518                                        nParams= types.length;
519                                        renderVarargs= method.exists() && Flags.isVarargs(method.getFlags());
520                                }
521                                String[] names= null;
522                                if (getFlag(flags, M_PARAMETER_NAMES) && method.exists()) {
523                                        names= method.getParameterNames();
524                                        if (types == null) {
525                                                nParams= names.length;
526                                        } else { // types != null
527                                                if (nParams != names.length) {
528                                                        if (resolvedSig != null && types.length > names.length) {
529                                                                // see https://bugs.eclipse.org/bugs/show_bug.cgi?id=99137
530                                                                nParams= names.length;
531                                                                String[] typesWithoutSyntheticParams= new String[nParams];
532                                                                System.arraycopy(types, types.length - nParams, typesWithoutSyntheticParams, 0, nParams);
533                                                                types= typesWithoutSyntheticParams;
534                                                        } else {
535                                                                // https://bugs.eclipse.org/bugs/show_bug.cgi?id=101029
536                                                                // JavaScriptPlugin.logErrorMessage("JavaScriptElementLabels: Number of param types(" + nParams + ") != number of names(" + names.length + "): " + method.getElementName());   //$NON-NLS-1$//$NON-NLS-2$//$NON-NLS-3$
537                                                                names= null; // no names rendered
538                                                        }
539                                                }
540                                        }
541                                }
542                                
543                                for (int i= 0; i < nParams; i++) {
544                                        if (i > 0) {
545                                                buf.append(COMMA_STRING);
546                                        }
547                                        if (types != null) {
548                                                String paramSig= types[i];
549                                                if (renderVarargs && (i == nParams - 1)) {
550                                                        int newDim= Signature.getArrayCount(paramSig) - 1;
551                                                        getTypeSignatureLabel(Signature.getElementType(paramSig), flags, buf);
552                                                        for (int k= 0; k < newDim; k++) {
553                                                                buf.append('[').append(']');
554                                                        }
555                                                        buf.append(ELLIPSIS_STRING);
556                                                } else {
557                                                        getTypeSignatureLabel(paramSig, flags, buf);
558                                                }
559                                        }
560                                        if (names != null) {
561                                                if (types != null) {
562                                                        buf.append(' ');
563                                                }
564                                                buf.append(names[i]);
565                                        }
566                                }
567                        } else {
568                                if (method.getParameterTypes().length > 0) {
569                                        buf.append(ELLIPSIS_STRING);
570                                }
571                        }
572                        buf.append(')');
573                                        
574                        if (getFlag(flags, M_EXCEPTIONS)) {
575                                String[] types;
576                                if (resolvedKey != null) {
577                                        //types= resolvedKey.getThrownExceptions();
578                                        types = new String[0];
579                                } else {
580                                        types= new String[0];
581                                }
582                                if (types.length > 0) {
583                                        buf.append(" throws "); //$NON-NLS-1$
584                                        for (int i= 0; i < types.length; i++) {
585                                                if (i > 0) {
586                                                        buf.append(COMMA_STRING);
587                                                }
588                                                getTypeSignatureLabel(types[i], flags, buf);
589                                        }
590                                }
591                        }
592                        
593                        if (getFlag(flags, M_APP_RETURNTYPE) && method.exists() && !method.isConstructor()) {
594                                if (method.getReturnType()!=null)
595                                {
596                                        buf.append(DECL_STRING);
597                                        String returnTypeSig= resolvedSig != null ? Signature.getReturnType(resolvedSig) : method.getReturnType();
598                                        getTypeSignatureLabel(returnTypeSig, flags, buf);
599                                }
600                        }                        
601 
602                        // category
603                        if (getFlag(flags, M_CATEGORY) && method.exists()) 
604                                getCategoryLabel(method, buf);
605                        
606                        // post qualification
607                        if (getFlag(flags, M_POST_QUALIFIED)) {
608                                buf.append(CONCAT_STRING);
609                                if (declaringType!=null)
610                                        getTypeLabel(method.getDeclaringType(), T_FULLY_QUALIFIED | (flags & QUALIFIER_FLAGS), buf);
611                                else
612                                        getFileLabel(method, T_FULLY_QUALIFIED | (flags & QUALIFIER_FLAGS), buf);
613                        }
614                        
615                } catch (JavaScriptModelException e) {
616                        JavaScriptPlugin.log(e); // NotExistsException will not reach this point
617                }
618        }
