458        protected IPackageFragmentRoot chooseContainer() {
459                IJavaScriptElement initElement= getPackageFragmentRoot();
460                Class[] acceptedClasses= new Class[] { IPackageFragmentRoot.class, IJavaScriptProject.class };
461                TypedElementSelectionValidator validator= new TypedElementSelectionValidator(acceptedClasses, false) {
462                        public boolean isSelectedValid(Object element) {
463                                try {
464                                        if (element instanceof IJavaScriptProject) {
465                                                IJavaScriptProject jproject= (IJavaScriptProject)element;
466                                                IPath path= jproject.getProject().getFullPath();
467                                                return (jproject.findPackageFragmentRoot(path) != null);
468                                        } else if (element instanceof IPackageFragmentRoot) {
469                                                return (((IPackageFragmentRoot)element).getKind() == IPackageFragmentRoot.K_SOURCE);
470                                        }
471                                        return true;
472                                } catch (JavaScriptModelException e) {
473                                        JavaScriptPlugin.log(e.getStatus()); // just log, no UI in validation
474                                }
475                                return false;
476                        }
477                };
478                
479                acceptedClasses= new Class[] { IJavaScriptModel.class, IPackageFragmentRoot.class, IJavaScriptProject.class };
480                ViewerFilter filter= new TypedViewerFilter(acceptedClasses) {
481                        public boolean select(Viewer viewer, Object parent, Object element) {
482                                if (element instanceof IPackageFragmentRoot) {
483                                        try {
484                                                return (((IPackageFragmentRoot)element).getKind() == IPackageFragmentRoot.K_SOURCE);
485                                        } catch (JavaScriptModelException e) {
486                                                JavaScriptPlugin.log(e.getStatus()); // just log, no UI in validation
487                                                return false;
488                                        }
489                                }
490                                return super.select(viewer, parent, element);
491                        }
492                };                
493 
494                StandardJavaScriptElementContentProvider provider= new StandardJavaScriptElementContentProvider();

496                ElementTreeSelectionDialog dialog= new ElementTreeSelectionDialog(getShell(), labelProvider, provider);

500                dialog.setMessage(NewWizardMessages.NewContainerWizardPage_ChooseSourceContainerDialog_description); 
501                dialog.addFilter(filter);
502                dialog.setInput(JavaScriptCore.create(fWorkspaceRoot));
503                dialog.setInitialSelection(initElement);
504                dialog.setHelpAvailable(false);

506                if (dialog.open() == Window.OK) {

510                                return jproject.getPackageFragmentRoot(jproject.getProject());
511                        } else if (element instanceof IPackageFragmentRoot) {
512                                return (IPackageFragmentRoot)element;
513                        }
514                        return null;
515                }
516                return null;
517        }
