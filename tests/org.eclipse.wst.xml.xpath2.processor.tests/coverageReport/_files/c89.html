466protected BuildContext[] notifyParticipants(SourceFile[] unitsAboutToCompile) {
467        BuildContext[] results = new BuildContext[unitsAboutToCompile.length];
468        for (int i = unitsAboutToCompile.length; --i >= 0;)
469                results[i] = new BuildContext(unitsAboutToCompile[i]);
470 
471        // TODO (kent) do we expect to have more than one participant?
472        // and if so should we pass the generated files from the each processor to the others to process?
473        // and what happens if some participants do not expect to be called with only a few files, after seeing 'all' the files?
474        for (int i = 0, l = this.javaBuilder.participants.length; i < l; i++)
475                this.javaBuilder.participants[i].buildStarting(results, this instanceof BatchImageBuilder);
476 
477        SimpleSet uniqueFiles = null;
478        ValidationParticipantResult[] toAdd = null;
479        int added = 0;
480        for (int i = results.length; --i >= 0;) {
481                ValidationParticipantResult result = results[i];
482                if (result == null) continue;
483 
484                IFile[] deletedGeneratedFiles = result.deletedFiles;
485                if (deletedGeneratedFiles != null)
486                        deleteGeneratedFiles(deletedGeneratedFiles);
487 
488                IFile[] addedGeneratedFiles = result.addedFiles;
489                if (addedGeneratedFiles != null) {
490                        for (int j = addedGeneratedFiles.length; --j >= 0;) {
491                                SourceFile sourceFile = findSourceFile(addedGeneratedFiles[j], true);
492                                if (sourceFile == null) continue;
493                                if (uniqueFiles == null) {
494                                        uniqueFiles = new SimpleSet(unitsAboutToCompile.length + 3);
495                                        for (int f = unitsAboutToCompile.length; --f >= 0;)
496                                                uniqueFiles.add(unitsAboutToCompile[f]);
497                                }
498                                if (uniqueFiles.addIfNotIncluded(sourceFile) == sourceFile) {
499                                        ValidationParticipantResult newResult = new BuildContext(sourceFile);
500                                        // is there enough room to add all the addedGeneratedFiles.length ?
501                                        if (toAdd == null) {
502                                                toAdd = new ValidationParticipantResult[addedGeneratedFiles.length];
503                                        } else {
504                                                int length = toAdd.length;
505                                                if (added == length)
506                                                        System.arraycopy(toAdd, 0, toAdd = new ValidationParticipantResult[length + addedGeneratedFiles.length], 0, length);
507                                        }
508                                        toAdd[added++] = newResult;
509                                }
510                        }
511                }
512        }
513 
514        if (added >0 ) {
515                int length = results.length;
516                System.arraycopy(results, 0, results = new BuildContext[length + added], 0 , length);
517                System.arraycopy(toAdd, 0, results, length, added);
518        }
519        return results;
520}

565/**
566 * Creates a marker from each problem and adds it to the resource.
567 * The marker is as follows:
568 *   - its type is T_PROBLEM
569 *   - its plugin ID is the JavaBuilder's plugin ID
570 *         - its message is the problem's message
571 *         - its priority reflects the severity of the problem
572 *         - its range is the problem's range
573 *         - it has an extra attribute "ID" which holds the problem's id
574 *   - it's GENERATED_BY attribute is positioned to JavaBuilder.GENERATED_BY if
575 *     the problem was generated by JDT; else the GENERATED_BY attribute is
576 *     carried from the problem to the marker in extra attributes, if present.
577 */
578protected void storeProblemsFor(SourceFile sourceFile, CategorizedProblem[] problems) throws CoreException {
579        if (sourceFile == null || problems == null || problems.length == 0) return;
580         // once a classpath error is found, ignore all other problems for this project so the user can see the main error
581        // but still try to compile as many source files as possible to help the case when the base libraries are in source
582        if (!this.keepStoringProblemMarkers) return; // only want the one error recorded on this source file
583 
584        IResource resource = sourceFile.resource;
585        IResource container=(resource instanceof IFile)? resource.getParent():resource;
586        if (JavaScriptCore.isReadOnly(container))
587                return;
588        HashSet managedMarkerTypes = JavaModelManager.getJavaModelManager().validationParticipants.managedMarkerTypes();
589        for (int i = 0, l = problems.length; i < l; i++) {
590                CategorizedProblem problem = problems[i];
591                int id = problem.getID();
592 
593                // handle missing classfile situation
594                if (id == IProblem.IsClassPathCorrect) {
595                        String missingClassfileName = problem.getArguments()[0];
596                        if (JavaBuilder.DEBUG)
597                                System.out.println(Messages.bind(Messages.build_incompleteClassPath, missingClassfileName));
598                        boolean isInvalidClasspathError = JavaScriptCore.ERROR.equals(javaBuilder.javaProject.getOption(JavaScriptCore.CORE_INCOMPLETE_CLASSPATH, true));
599                        // insert extra classpath problem, and make it the only problem for this project (optional)
600                        if (isInvalidClasspathError && JavaScriptCore.ABORT.equals(javaBuilder.javaProject.getOption(JavaScriptCore.CORE_JAVA_BUILD_INVALID_CLASSPATH, true))) {
601                                JavaBuilder.removeProblemsAndTasksFor(javaBuilder.currentProject); // make this the only problem for this project
602                                this.keepStoringProblemMarkers = false;
603                        }
604                        IMarker marker = this.javaBuilder.currentProject.createMarker(IJavaScriptModelMarker.JAVASCRIPT_MODEL_PROBLEM_MARKER);
605                        marker.setAttributes(
606                                new String[] {IMarker.MESSAGE, IMarker.SEVERITY, IJavaScriptModelMarker.CATEGORY_ID, IMarker.SOURCE_ID},
607                                new Object[] {
608                                        Messages.bind(Messages.build_incompleteClassPath, missingClassfileName),
609                                        new Integer(isInvalidClasspathError ? IMarker.SEVERITY_ERROR : IMarker.SEVERITY_WARNING),
610                                        new Integer(CategorizedProblem.CAT_BUILDPATH),
611                                        JavaBuilder.SOURCE_ID
612                                }
613                        );
614                        // even if we're not keeping more markers, still fall through rest of the problem reporting, so that offending
615                        // IsClassPathCorrect problem gets recorded since it may help locate the offending reference
616                }
617 
618                String markerType = problem.getMarkerType();
619                boolean managedProblem = false;
620                if (IJavaScriptModelMarker.JAVASCRIPT_MODEL_PROBLEM_MARKER.equals(markerType)
621                                || (managedProblem = managedMarkerTypes.contains(markerType))) {
622                        IMarker marker = resource.createMarker(markerType);
623 
624                        String[] attributeNames = JAVA_PROBLEM_MARKER_ATTRIBUTE_NAMES;
625                        int standardLength = attributeNames.length;
626                        String[] allNames = attributeNames;
627                        int managedLength = managedProblem ? 0 : 1;
628                        String[] extraAttributeNames = problem.getExtraMarkerAttributeNames();
629                        int extraLength = extraAttributeNames == null ? 0 : extraAttributeNames.length;
630                        if (managedLength > 0 || extraLength > 0) {
631                                allNames = new String[standardLength + managedLength + extraLength];
632                                System.arraycopy(attributeNames, 0, allNames, 0, standardLength);
633                                if (managedLength > 0)
634                                        allNames[standardLength] = IMarker.SOURCE_ID;
635                                System.arraycopy(extraAttributeNames, 0, allNames, standardLength + managedLength, extraLength);
636                        }
637 
638                        Object[] allValues = new Object[allNames.length];
639                        // standard attributes
640                        int index = 0;
641                        allValues[index++] = problem.getMessage(); // message
642                        allValues[index++] = problem.isError() ? S_ERROR : S_WARNING; // severity
643                        allValues[index++] = new Integer(id); // ID
644                        allValues[index++] = new Integer(problem.getSourceStart()); // start
645                        allValues[index++] = new Integer(problem.getSourceEnd() + 1); // end
646                        allValues[index++] = new Integer(problem.getSourceLineNumber()); // line
647                        allValues[index++] = Util.getProblemArgumentsForMarker(problem.getArguments()); // arguments
648                        allValues[index++] = new Integer(problem.getCategoryID()); // category ID
649                        // GENERATED_BY attribute for JDT problems
650                        if (managedLength > 0)
651                                allValues[index++] = JavaBuilder.SOURCE_ID;
652                        // optional extra attributes
653                        if (extraLength > 0)
654                                System.arraycopy(problem.getExtraMarkerAttributeValues(), 0, allValues, index, extraLength);
655 
656                        marker.setAttributes(allNames, allValues);
657 
658                        if (!this.keepStoringProblemMarkers) return; // only want the one error recorded on this source file
659                }
660        }
