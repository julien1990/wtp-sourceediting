558        private void updateFilters(IPath oldPath, IPath newPath) {
559                if (oldPath == null)
560                        return;
561                
562                IPath projPath= fNewElement.getJavaProject().getProject().getFullPath();
563                if (projPath.isPrefixOf(oldPath)) {
564                        oldPath= oldPath.removeFirstSegments(projPath.segmentCount()).addTrailingSeparator();
565                }
566                if (projPath.isPrefixOf(newPath)) {
567                        newPath= newPath.removeFirstSegments(projPath.segmentCount()).addTrailingSeparator();
568                }
569                
570                for (Iterator iter= fExistingEntries.iterator(); iter.hasNext();) {
571                        CPListElement element= (CPListElement)iter.next();
572                        IPath elementPath= element.getPath();
573                        if (projPath.isPrefixOf(elementPath)) {
574                                elementPath= elementPath.removeFirstSegments(projPath.segmentCount());
575                                if (elementPath.segmentCount() > 0)
576                                        elementPath= elementPath.addTrailingSeparator();
577                        }
578                        
579                        IPath[] exlusions= (IPath[])element.getAttribute(CPListElement.EXCLUSION);
580                        if (exlusions != null) {
581                                for (int i= 0; i < exlusions.length; i++) {
582                                        if (elementPath.append(exlusions[i]).equals(oldPath)) {
583                                                fModifiedElements.add(element);
584                                                exlusions[i]= newPath.removeFirstSegments(elementPath.segmentCount());
585                                        }
586                                }
587                                element.setAttribute(CPListElement.EXCLUSION, exlusions);
588                        }
589                        
590                        IPath[] inclusion= (IPath[])element.getAttribute(CPListElement.INCLUSION);
591                        if (inclusion != null) {
592                                for (int i= 0; i < inclusion.length; i++) {
593                                        if (elementPath.append(inclusion[i]).equals(oldPath)) {
594                                                fModifiedElements.add(element);
595                                                inclusion[i]= newPath.removeFirstSegments(elementPath.segmentCount());
596                                        }
597                                }
598                                element.setAttribute(CPListElement.INCLUSION, inclusion);
599                        }
600                }
601        }

638        private static StatusInfo validatePathName(String str, IContainer parent) {
639                StatusInfo result= new StatusInfo();
640                result.setOK();
641 
642                IPath parentPath= parent.getFullPath();
643                
644                if (str.length() == 0) {
645                        result.setError(Messages.format(NewWizardMessages.NewSourceFolderWizardPage_error_EnterRootName, parentPath.toString()));
646                        return result;
647                }
648                
649                IPath path= parentPath.append(str);
650 
651                IWorkspaceRoot workspaceRoot= ResourcesPlugin.getWorkspace().getRoot();
652                IStatus validate= workspaceRoot.getWorkspace().validatePath(path.toString(), IResource.FOLDER);
653                if (validate.matches(IStatus.ERROR)) {
654                        result.setError(Messages.format(NewWizardMessages.NewSourceFolderWizardPage_error_InvalidRootName, validate.getMessage())); 
655                        return result;
656                }
657                        
658                IResource res= workspaceRoot.findMember(path);
659                if (res != null) {
660                        if (res.getType() != IResource.FOLDER) {
661                                result.setError(NewWizardMessages.NewSourceFolderWizardPage_error_NotAFolder); 
662                                return result;
663                        }
664                } else {
665                        
666                        URI parentLocation= parent.getLocationURI();
667                        if (parentLocation != null) {
668                                try {
669                                        IFileStore store= EFS.getStore(parentLocation).getChild(str);
670                                        if (store.fetchInfo().exists()) {
671                                                result.setError(NewWizardMessages.NewSourceFolderWizardPage_error_AlreadyExistingDifferentCase); 
672                                                return result;
673                                        }
674                                } catch (CoreException e) {
675                                        // we couldn't create the file store. Ignore the exception
676                                        // since we can't check if the file exist. Pretend that it
677                                        // doesn't.
678                                }
679                        }
680                }
681                
682                return result;
683        }
