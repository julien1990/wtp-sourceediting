671/**
672 * Method doScan.
673 * 
674 * Returns a context region for all of the text from the current position upto the end of input or
675 * to right *before* the first occurence of searchString
676 * 
677 * @param searchString - target string to search for ex.: "-->", "</tagname"
678 * @param requireTailSeparator - whether the target must be immediately followed by whitespace or '>'
679 * @param context - the context of the scanned region if non-zero length
680 * @param exitState - the state to go to if the region was of non-zero length
681 * @param abortState - the state to go to if the searchString was found immediately
682 * @return String - the context found: the desired context on a non-zero length match, the abortContext on immediate success
683 * @throws IOException
684 */
685private final String doScan(String searchString, boolean requireTailSeparator, String searchContext, int exitState, int immediateFallbackState) throws IOException {
686        boolean stillSearching = true;
687        // Disable further block (probably)
688        fIsBlockingEnabled = false;
689        int searchStringLength = searchString.length();
690        int n = 0;
691        char lastCheckChar;
692        int i;
693        boolean same = false;
694        while (stillSearching) {
695                n = 0;
696                // Ensure that enough data from the input exists to compare against the search String.
697                n = yy_advance();
698                while(n != YYEOF && yy_currentPos < searchStringLength)
699                        n = yy_advance();
700                // If the input was too short or we've exhausted the input, stop immediately.
701                if (n == YYEOF) {
702                        stillSearching = false;
703                }
704                else {
705                        same = true;
706                        // Ensure that we've not encountered a complete block (<%%>) that was *shorter* than the closeTagString and
707                        // thus found twice at current-targetLength [since the first scan would have come out this far anyway].
708                        // Check the characters in the target versus the last targetLength characters read from the buffer
709                        // and see if it matches
710                        
711                        // safety check for array accesses (yy_currentPos is the *last* character we can check against)
712                        if(yy_currentPos >= searchStringLength && yy_currentPos <= yy_buffer.length) {
713                                for(i = 0; i < searchStringLength; i++) {
714                                        if(same && fIsCaseSensitiveBlocking)
715                                                same = yy_buffer[i + yy_currentPos - searchStringLength] == searchString.charAt(i);
716                                        else if(same && !fIsCaseSensitiveBlocking)
717                                                same = Character.toLowerCase(yy_buffer[i + yy_currentPos - searchStringLength]) == Character.toLowerCase(searchString.charAt(i));
718                                }
719                        }
720                        // safety check failed; no match is possible right now
721                        else {
722                                same = false;
723                        }
724                        if (same && requireTailSeparator && yy_currentPos < yy_buffer.length) {
725                                // Additional check for close tags to ensure that targetString="</script" doesn't match
726                                // "</scriptS"
727                                lastCheckChar = yy_buffer[yy_currentPos];
728                                // Succeed on "</script>" and "</script "
729                                if(lastCheckChar == '>' || Character.isWhitespace(lastCheckChar))
730                                        stillSearching = false;
731                        }
732                        else {
733                                stillSearching = !same || (yy_currentPos < yy_startRead + searchStringLength);
734                        }
735                }
736        }
737        if (n != YYEOF || same) {
738                // We've stopped short of the end or definitely found a match
739                yy_markedPos = yy_currentPos - searchStringLength;
740                yy_currentPos = yy_markedPos + 1;
741                // If the searchString occurs at the very beginning of what would have
742                // been a Block, resume scanning normally immediately
743                if (yy_markedPos == yy_startRead) {
744                        yybegin(immediateFallbackState);
745                        return primGetNextToken();
746                }
747        }
748        else {
749                // We ran through the rest of the input
750                yy_markedPos = yy_currentPos;
751                yy_currentPos++;
752        }
753        yybegin(exitState);
754        // If the ending occurs at the very beginning of what would have
755        // been a Block, resume scanning normally immediately
756        if(yy_markedPos == yy_startRead)
757                return primGetNextToken();
758        return searchContext;
759}
