478                if (decl == null)

481                IBinding binding= name.resolveBinding();

485                if (decl instanceof SingleVariableDeclaration) {

504        private static boolean canAddFinal(IBinding binding, SimpleName name, ASTNode declNode) {
505                if (!(binding instanceof IVariableBinding))
506                        return false;
507 
508                IVariableBinding varbinding= (IVariableBinding)binding;
509                if (Modifier.isFinal(varbinding.getModifiers()))
510                        return false;
511                
512                ASTNode parent= ASTNodes.getParent(declNode, VariableDeclarationExpression.class);
513                if (parent != null && ((VariableDeclarationExpression)parent).fragments().size() > 1)
514                        return false;
515                
516                if (varbinding.isField() && !Modifier.isPrivate(varbinding.getModifiers())) 
517                        return false;
518                
519                if (varbinding.isParameter()) {
520                        ASTNode varDecl= declNode.getParent();
521                        if (varDecl instanceof FunctionDeclaration) {
522                                FunctionDeclaration declaration= (FunctionDeclaration)varDecl;
523                                if (declaration.getBody() == null)
524                                        return false;
525                        }
526                }
527                
528                return true;
529        }
