345        private int getMax(Integer nrOfTrue, Integer nrOfFalse) {
346            if (nrOfTrue == null)
347                return nrOfFalse.intValue();

349                return nrOfTrue.intValue();
350            if (nrOfTrue.compareTo(nrOfFalse) >= 0)
351                return nrOfTrue.intValue();
352            return nrOfFalse.intValue();
353        }

360        private void updateCombo(Combo combo, Map map, final String[] items) {
361            String[] newItems= new String[items.length];
362            int maxCount= 0, maxStyle= 0;
363                        
364            for(int i = 0; i < items.length; i++) {
365                Integer count= (Integer) map.get(new Integer(i));
366                int val= (count == null) ? 0 : count.intValue();
367                if (val > maxCount) {
368                    maxCount= val;
369                    maxStyle= i;
370                }                
371                newItems[i]= getLabelText(items[i], val, fElements.size()); 
372            }
373            combo.setItems(newItems);
374            combo.setText(newItems[maxStyle]);
375        }

825                                    Messages.format(FormatterMessages.LineWrappingTabPage_error_invalid_value, currentKey), e));

827        }

829        }
