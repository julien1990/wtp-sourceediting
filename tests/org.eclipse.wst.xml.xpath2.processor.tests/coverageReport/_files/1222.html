384                                        newTagElement.fragments().add(newRef);
385                                        insertTabStop(rewrite, newTagElement.fragments(), "throws_tagcomment" + i); //$NON-NLS-1$
386                                        insertThrowsTag(rewrite.getListRewrite(javadoc, JSdoc.TAGS_PROPERTY), exceptions, k, newTagElement);
387 
388                                        addLinkedPosition(rewrite.track(newRef), false, key);
389                                }
390 
391                        } else if (curr instanceof RemoveDescription) {
392                                Name node= (Name) exceptions.get(k);
393 
394                                listRewrite.remove(node, null);
395                                k++;
396 
397                                TagElement tagNode= findThrowsTag(methodDecl, node);
398                                if (tagNode != null) {
399                                        rewrite.remove(tagNode, null);
400                                }
401                        } else if (curr instanceof EditDescription) {
402                                EditDescription desc= (EditDescription) curr;
403 
404                                Name oldNode= (Name) exceptions.get(k);
405 
406                                String type= imports.addImport(desc.type);
407                                ASTNode newNode= ASTNodeFactory.newName(ast, type);
408 
409                                listRewrite.replace(oldNode, newNode, null);
410                                String key= getExceptionTypeGroupId(i);
411                                addLinkedPosition(rewrite.track(newNode), false, key);
412 
413                                k++;
414 
415                                TagElement tagNode= findThrowsTag(methodDecl, oldNode);
416                                if (tagNode != null) {
417                                        ASTNode newRef= ASTNodeFactory.newName(ast, type);
418                                        rewrite.replace((ASTNode) tagNode.fragments().get(0), newRef, null);
419                                        addLinkedPosition(rewrite.track(newRef), false, key);
420                                }
421 
422                        } else if (curr instanceof SwapDescription) {
423                                Name decl1= (Name) exceptions.get(k);
424                                Name decl2= (Name) exceptions.get(((SwapDescription) curr).index);
425 
426                                rewrite.replace(decl1, rewrite.createCopyTarget(decl2), null);
427                                rewrite.replace(decl2, rewrite.createCopyTarget(decl1), null);
428 
429                                k++;
430 
431                                TagElement tagNode1= findThrowsTag(methodDecl, decl1);
432                                TagElement tagNode2= findThrowsTag(methodDecl, decl2);
433                                if (tagNode1 != null && tagNode2 != null) {
434                                        rewrite.replace(tagNode1, rewrite.createCopyTarget(tagNode2), null);
435                                        rewrite.replace(tagNode2, rewrite.createCopyTarget(tagNode1), null);
436                                }
437                        }
438                }
439        }

457        private TagElement insertThrowsTag(ListRewrite tagRewriter, List exceptions, int currentIndex, TagElement newTagElement) {
