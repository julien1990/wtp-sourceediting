644        protected Element getQueryContext() {

651 
652                if (queryByContext.getSelection()) {
653                        ISourceEditingTextTools sett = (ISourceEditingTextTools) structuredTextEditor

661                                        MessageDialog.openInformation(getSite().getShell(),

663                                                        Messages.XPathNavigator_Node_could_not_be_selected);
664                                }
665 
666                                // Go upwards to an element
667                                while ((n != null)
668                                                && !((n instanceof Element) || (n instanceof Document))) {
669                                        n = n.getParentNode();
670                                }
671 
672                                if (n instanceof Document) {
673                                        n = ((Document) n).getDocumentElement();

679                                                        Messages.XPathNavigator_Nothing_selected);

682                                }
683                                return (Element) n;
