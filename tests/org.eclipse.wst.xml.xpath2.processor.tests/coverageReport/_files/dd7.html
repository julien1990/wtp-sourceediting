678        private void insertAt(ASTNode target, Statement declaration) throws JavaScriptModelException {
679                ASTRewrite rewrite= fCURewrite.getASTRewrite();
680                TextEditGroup groupDescription= fCURewrite.createGroupDescription(RefactoringCoreMessages.ExtractTempRefactoring_declare_local_variable);

684                        StructuralPropertyDescriptor locationInParent= target.getLocationInParent();
685                        if (locationInParent == IfStatement.THEN_STATEMENT_PROPERTY
686                                        || locationInParent == IfStatement.ELSE_STATEMENT_PROPERTY
687                                        || locationInParent == ForStatement.BODY_PROPERTY
688                                        || locationInParent == ForInStatement.BODY_PROPERTY
689                                        || locationInParent == EnhancedForStatement.BODY_PROPERTY
690                                        || locationInParent == DoStatement.BODY_PROPERTY
691                                        || locationInParent == WhileStatement.BODY_PROPERTY
692                                        || locationInParent == WithStatement.BODY_PROPERTY
693                                        ) {
694                                Block replacement= rewrite.getAST().newBlock();
695                                ListRewrite replacementRewrite= rewrite.getListRewrite(replacement, Block.STATEMENTS_PROPERTY);

699                                return;
700                        }

702                        parent= parent.getParent();
703                }
704                ListRewrite listRewrite= rewrite.getListRewrite(parent, Block.STATEMENTS_PROPERTY);
705                listRewrite.insertBefore(declaration, target, groupDescription);
706        }
