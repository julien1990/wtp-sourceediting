414private char[] utf8At(int readOffset, int bytesAvailable, byte[] bytes) {
415        int x, y, z;
416        int length = bytesAvailable;
417        char outputBuf[] = new char[bytesAvailable];
418        int outputPos = 0;
419        while (length != 0) {
420                x = bytes[readOffset++] & 0xFF;
421                length--;
422                if ((0x80 & x) != 0) {
423                        y = bytes[readOffset++] & 0xFF;
424                        length--;
425                        if ((x & 0x20) != 0) {
426                                z = bytes[readOffset++] & 0xFF;
427                                length--;
428                                x = ((x & 0x1F) << 12) + ((y & 0x3F) << 6) + (z & 0x3F);
429                        } else {
430                                x = ((x & 0x1F) << 6) + (y & 0x3F);
431                        }
432                }
433                outputBuf[outputPos++] = (char) x;
434        }
435 
436        if (outputPos != bytesAvailable) {
437                System.arraycopy(outputBuf, 0, (outputBuf = new char[outputPos]), 0, outputPos);
438        }
439        return outputBuf;
440}
