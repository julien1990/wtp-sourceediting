368        protected String primGetIndentString(Node node) {
369                IStructuredDocument flatModel = ((IDOMNode) node).getStructuredDocument();
370                int nodeStartOff = ((IDOMNode) node).getStartOffset();
371 
372                int startOff = Math.max(0, nodeStartOff - 100);
373                int endOff = nodeStartOff;
374 
375                try {
376                        String text = flatModel.get(startOff, endOff - startOff);
377 
378                        int inx = text.length() - 1;
379                        if (inx >= 0) {
380                                for (; inx >= 0; inx--) {
381                                        char ch = text.charAt(inx);
382                                        if (Character.isWhitespace(ch) && ch != '\n' && ch != '\r') {
383                                                continue;
384                                        }
385                                        inx++;
386                                        break;
387                                }
388 
389                                return text.substring(inx);
390                        }
391                }
392                catch (BadLocationException ex) {
393                        Logger.logException(ex);
394                }
395                return ""; //$NON-NLS-1$
