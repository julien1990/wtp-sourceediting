888         *

898         *        <code>bound</code> &lt; <code>start</code>, or <code>UNBOUND</code>

904                int token= previousToken(start - 1, bound);
905                if (token == Symbols.TokenIDENT) { // type name
906                        token= previousToken(getPosition(), bound);
907                        while (token == Symbols.TokenOTHER) { // dot of qualification
908                                token= previousToken(getPosition(), bound);

910                                        return false;
911                                token= previousToken(getPosition(), bound);

914                }
