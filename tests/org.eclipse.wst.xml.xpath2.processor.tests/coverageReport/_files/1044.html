436         * Returns a shared compilation unit AST for the given
437         * Java element.
438         * <p>
439         * Clients are not allowed to modify the AST and must
440         * synchronize all access to its nodes.
441         * </p>
442         *
443         * @param je                                the Java element
444         * @param waitFlag                        {@link #WAIT_YES}, {@link #WAIT_NO} or {@link #WAIT_ACTIVE_ONLY}
445         * @param progressMonitor        the progress monitor or <code>null</code>
446         * @return                                        the AST or <code>null</code> if the AST is not available
447         */
448        public JavaScriptUnit getAST(IJavaScriptElement je, WAIT_FLAG waitFlag, IProgressMonitor progressMonitor) {
449                if (je == null)
450                        return null;
451                
452                Assert.isTrue(je.getElementType() == IJavaScriptElement.CLASS_FILE || je.getElementType() == IJavaScriptElement.JAVASCRIPT_UNIT);
453 
454                if (progressMonitor != null && progressMonitor.isCanceled())
455                        return null;
456 
457                boolean isActiveElement;
458                synchronized (this) {
459                        isActiveElement= je.equals(fActiveJavaElement);
460                        if (isActiveElement) {
461                                if (fAST != null) {
462                                        if (DEBUG)
463                                                System.out.println(getThreadName() + " - " + DEBUG_PREFIX + "returning cached AST:" + toString(fAST) + " for: " + je.getElementName()); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
464 
465                                        return fAST;
466                                }
467                                if (waitFlag == WAIT_NO) {
468                                        if (DEBUG)
469                                                System.out.println(getThreadName() + " - " + DEBUG_PREFIX + "returning null (WAIT_NO) for: " + je.getElementName()); //$NON-NLS-1$ //$NON-NLS-2$
470 
471                                        return null;
472 
473                                }
474                        }
475                }
476                if (isActiveElement && isReconciling(je)) {
477                        try {
478                                final IJavaScriptElement activeElement= fReconcilingJavaElement;
479 
480                                // Wait for AST
481                                synchronized (fWaitLock) {
482                                        if (DEBUG)
483                                                System.out.println(getThreadName() + " - " + DEBUG_PREFIX + "waiting for AST for: " + je.getElementName()); //$NON-NLS-1$ //$NON-NLS-2$
484 
485                                        fWaitLock.wait();
486                                }
487 
488                                // Check whether active element is still valid
489                                synchronized (this) {
490                                        if (activeElement == fActiveJavaElement && fAST != null) {
491                                                if (DEBUG)
492                                                        System.out.println(getThreadName() + " - " + DEBUG_PREFIX + "...got AST for: " + je.getElementName()); //$NON-NLS-1$ //$NON-NLS-2$
493 
494                                                return fAST;
495                                        }
496                                }
497                                return getAST(je, waitFlag, progressMonitor);
498                        } catch (InterruptedException e) {
499                                return null; // thread has been interrupted don't compute AST
500                        }
501                } else if (waitFlag == WAIT_NO || (waitFlag == WAIT_ACTIVE_ONLY && !(isActiveElement && fAST == null)))
502                        return null;
503 
504                if (isActiveElement)
505                        aboutToBeReconciled(je);
506 
507                JavaScriptUnit ast= null;
508                try {
509                        ast= createAST(je, progressMonitor);
510                        if (progressMonitor != null && progressMonitor.isCanceled()) {
511                                ast= null;
512                                if (DEBUG)
513                                        System.out.println(getThreadName() + " - " + DEBUG_PREFIX + "Ignore created AST for: " + je.getElementName() + " - operation has been cancelled"); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
514                        }
515                } finally {
516                        if (isActiveElement) {
517                                if (fAST != null) {
518                                        if (DEBUG)
519                                                System.out.println(getThreadName() + " - " + DEBUG_PREFIX + "Ignore created AST for " + je.getElementName() + " - AST from reconciler is newer"); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
520                                        reconciled(fAST, je, null);
521                                } else
522                                        reconciled(ast, je, null);
523                        }
524                }
525 
526                return ast;
527        }

548         */
549        private JavaScriptUnit createAST(final IJavaScriptElement je, final IProgressMonitor progressMonitor) {
550                if (!hasSource(je))
551                        return null;
552                
553                if (progressMonitor != null && progressMonitor.isCanceled())
554                        return null;

556                final ASTParser parser = ASTParser.newParser(SHARED_AST_LEVEL);
557                parser.setResolveBindings(true);
558                parser.setStatementsRecovery(SHARED_AST_STATEMENT_RECOVERY);
559                parser.setBindingsRecovery(SHARED_BINDING_RECOVERY);

562                        return null;

565                        parser.setSource((IJavaScriptUnit)je);
566                else if (je.getElementType() == IJavaScriptElement.CLASS_FILE)

578                                                return;

580                                                System.err.println(getThreadName() + " - " + DEBUG_PREFIX + "creating AST for: " + je.getElementName()); //$NON-NLS-1$ //$NON-NLS-2$
