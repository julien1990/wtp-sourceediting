429        public Point computeSizeHint() {
430                TextPresentation presentation= new TextPresentation();
431                HTML2TextReader reader= new HTML2TextReader(new StringReader(fInputText), presentation);
432                String text;
433                try {
434                        text= reader.getString();
435                } catch (IOException e) {
436                        text= ""; //$NON-NLS-1$
437                }
438 
439                fTextLayout.setText(text);
440                Iterator iter= presentation.getAllStyleRangeIterator();
441                while (iter.hasNext()) {
442                        StyleRange sr= (StyleRange)iter.next();
443                        if (sr.fontStyle == SWT.BOLD)
444                                fTextLayout.setStyle(fBoldStyle, sr.start, sr.start + sr.length - 1);
445                }
446                Rectangle bounds= fTextLayout.getBounds();
447                int width= bounds.width;
448                int height= bounds.height;
449                
450                width += 15; 
451                height += 25; 
452 
453                if (fStatusFieldText != null && fSeparator != null) {
454                        fTextLayout.setText(fStatusFieldText);
455                        Rectangle statusBounds= fTextLayout.getBounds();
456                        Rectangle separatorBounds= fSeparator.getBounds();
457                        width= Math.max(width, statusBounds.width);
458                        height= height + statusBounds.height + separatorBounds.height;
459                }
460 
461                // Apply size constraints
462                if (fMaxWidth != SWT.DEFAULT)
463                        width= Math.min(fMaxWidth, width);
464                if (fMaxHeight != SWT.DEFAULT)
465                        height= Math.min(fMaxHeight, height);
466 
467                // Ensure minimal size
468                width= Math.max(MIN_WIDTH, width);
469                height= Math.max(MIN_HEIGHT, height);
470                
471                return new Point(width, height);
472        }
