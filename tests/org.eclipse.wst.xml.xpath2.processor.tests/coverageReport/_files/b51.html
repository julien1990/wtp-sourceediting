401        public TypeBinding resolveType(BlockScope scope, boolean define, TypeBinding useType) {
402 
403                // for code gen, harm the restrictiveFlag
404                constant = Constant.NotAConstant;
405 
406                this.binding=findBinding(scope);
407                if (define && this.binding instanceof ProblemBinding)
408                {
409                        LocalDeclaration localDeclaration = new LocalDeclaration(this.token,this.sourceEnd,this.sourceEnd);
410                        LocalVariableBinding localBinding=new LocalVariableBinding(localDeclaration,TypeBinding.UNKNOWN,0,false);
411                        localBinding.setConstant(Constant.NotAConstant);
412                    scope.compilationUnitScope().addLocalVariable(localBinding);
413                        this.binding=localBinding;
414                }
415//                this.codegenBinding = this.binding;
416                if (this.binding.isValidBinding()) {
417                        switch (bits & RestrictiveFlagMASK) {
418                                case Binding.FIELD:
419                                case Binding.LOCAL : // =========only variable============
420                                case Binding.VARIABLE : // =========only variable============
421                                case Binding.LOCAL | Binding.TYPE : //====both variable and type============
422                                case Binding.VARIABLE | Binding.TYPE : //====both variable and type============
423                                        if (binding instanceof VariableBinding) {
424                                                VariableBinding variable = (VariableBinding) binding;
425                                                if (binding instanceof LocalVariableBinding) {
426                                                        bits &= ~RestrictiveFlagMASK;  // clear bits
427                                                        bits |= Binding.LOCAL;
428//                                                        if (!variable.isFinal() && (bits & DepthMASK) != 0) {
429//                                                                scope.problemReporter().cannotReferToNonFinalOuterLocal((LocalVariableBinding)variable, this);
430//                                                        }
431                                                        TypeBinding fieldType = variable.type;
432                                                        if (fieldType.isAnonymousType())
433                                                                bits |= Binding.TYPE;
434                                                        if (useType!=null && !(useType.id==T_null ||useType.id==T_any))
435                                                        {
436                                                                if (define)
437                                                                {
438                                                                        fieldType=variable.type=useType;
439                                                                        if (useType.isFunctionType())        // add method binding if function
440                                                                        {

444                                                                                        methodScope.addLocalMethod(methodBinding);
445                                                                                else

447                                                                        }
448                                                                }

450                                                                {

452                                                                                fieldType=variable.type=useType;
453                                                                        else if (!fieldType.isCompatibleWith(useType))
454                                                                                fieldType=variable.type=TypeBinding.ANY;
455                                                                }
456                                                        }
457                                                        if ((this.bits & IsStrictlyAssigned) == 0) {
458                                                                constant = variable.constant();
459                                                                if (fieldType != null)
460                                                                        fieldType = fieldType.capture(scope, this.sourceEnd); // perform capture conversion if read access

462                                                                constant = Constant.NotAConstant;
463                                                        }

478                                                        (((this.bits & IsStrictlyAssigned) == 0)

481                                        }

485                                                return ((MethodBinding)binding).functionTypeBinding;
486                                        }

491                                                bits |= Binding.TYPE;
492                                        }
