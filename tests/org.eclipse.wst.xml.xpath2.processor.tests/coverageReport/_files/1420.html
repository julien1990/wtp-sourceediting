478         * Returns the replacement length of a given completion proposal. The
479         * replacement length is usually the difference between the return values of
480         * <code>proposal.getReplaceEnd</code> and
481         * <code>proposal.getReplaceStart</code>, but this behavior may be
482         * overridden by calling {@link #setReplacementLength(int)}.
483         *
484         * @param proposal the completion proposal to get the replacement length for
485         * @return the replacement length for <code>proposal</code>
486         */
487        protected final int getLength(CompletionProposal proposal) {
488                int start= proposal.getReplaceStart();
489                int end= proposal.getReplaceEnd();
490                int length;
491                if (fUserReplacementLength == -1) {
492                        length= end - start;
493                } else {
494                        length= fUserReplacementLength;
495                        // extend length to begin at start
496                        int behindCompletion= proposal.getCompletionLocation() + 1;
497                        if (start < behindCompletion) {
498                                length+= behindCompletion - start;
499                        }
500                }
501                return length;
502        }

527         * <code>CompletionProposal</code>, or <code>null</code> for proposals
528         * that do not have a declaring type. The return value is <em>not</em>
529         * <code>null</code> for proposals of the following kinds:

538         * <li>PACKAGE_REF (returns the package, but no type)</li>
539         * <li>TYPE_REF</li>
540         * </ul>
541         *
542         * @param proposal the completion proposal to get the declaring type for
543         * @return the type signature of the declaring type, or <code>null</code> if there is none

549                        case CompletionProposal.METHOD_NAME_REFERENCE:

552                        case CompletionProposal.POTENTIAL_METHOD_DECLARATION:
553                        case CompletionProposal.ANONYMOUS_CLASS_DECLARATION:

573                        case CompletionProposal.JSDOC_INLINE_TAG:
574                        case CompletionProposal.JSDOC_PARAM_REF:

576                        default:
577                                Assert.isTrue(false);
578                                return null;
579                }
