430        buffer.append(this.delta == null ? "<null>" : this.delta.toString()); //$NON-NLS-1$
431        return buffer.toString();
432}
433/**
434 * Trims deletion deltas to only report the highest level of deletion
435 */
436private void trimDelta(JavaElementDelta elementDelta) {
437        if (elementDelta.getKind() == IJavaScriptElementDelta.REMOVED) {
438                IJavaScriptElementDelta[] children = elementDelta.getAffectedChildren();
439                for(int i = 0, length = children.length; i < length; i++) {
440                        elementDelta.removeAffectedChild((JavaElementDelta)children[i]);
441                }
442        } else {
443                IJavaScriptElementDelta[] children = elementDelta.getAffectedChildren();
444                for(int i = 0, length = children.length; i < length; i++) {
445                        trimDelta((JavaElementDelta)children[i]);
446                }
447        }
448}
