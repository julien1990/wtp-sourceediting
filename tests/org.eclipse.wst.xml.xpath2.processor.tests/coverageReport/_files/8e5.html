564        /**
565         * Given a line number and column number, returns the corresponding
566         * position in the original source string.
567         * Returns -2 if no line number information is available for this
568         * javaScript unit.
569         * Returns the total size of the source string if <code>line</code>
570         * is greater than the actual number lines in the unit.
571         * Returns -1 if <code>column</code> is less than 0,
572         * or the position of the last character of the line if <code>column</code>
573         * is beyond the legal range, or the given line number is less than one.
574         *
575         * @param line the one-based line number
576         * @param column the zero-based column number
577         * @return the 0-based character position in the source string;
578         * <code>-2</code> if line/column number information is not known
579         * for this javaScript unit or <code>-1</code> the inputs are not valid
580         *  
581         */
582         public int getPosition(int line, int column) {
583                if (this.lineEndTable == null) return -2;
584                if (line < 1 || column < 0) return -1;
585                int length;
586                if ((length = this.lineEndTable.length) == 0) {
587                        if (line != 1) return -1;
588                        return column >= getStartPosition() + getLength() ? -1 : column;
589                }
590                if (line == 1) {
591                        final int endOfLine = this.lineEndTable[0];
592                        return column > endOfLine ? -1 : column;
593                } else if( line > length + 1 ) {
594                        // greater than the number of lines in the source string.
595                        return -1;
596                }
597                // -1 to for one-based to zero-based conversion.
598                // -1, again, to get previous line.
599                final int previousLineOffset = this.lineEndTable[line - 2];
600                 // previousLineOffset + 1 is the first character of the current line
601                final int offsetForLine = previousLineOffset + 1;
602                final int currentLineEnd = line == length + 1 ? getStartPosition() + getLength() - 1 : this.lineEndTable[line-1];
603                if ((offsetForLine + column) > currentLineEnd) {
604                        return -1;
605                } else {
606                        return offsetForLine + column;
607                }
608        }

746         * last character of the end-of-line delimiter. The very last line extends
747         * through the end of the source string and has no line delimiter.
748         * For example, the source string <code>class A\n{\n}</code> has 3 lines
749         * corresponding to inclusive character ranges [0,7], [8,9], and [10,10].
750         * Returns -1 for a character position that does not correspond to any
751         * source line, or -2 if no line number information is available for this
752         * javaScript unit.
753         *
754         * @param position a 0-based character position, possibly
755         *   negative or out of range
756         * @return the 1-based line number, or <code>-1</code> if the character
757         *    position does not correspond to a source line in the original
758         *    source file or <code>-2</code> if line number information is not known for this
759         *    javaScript unit
760         * @see ASTParser
761         *  
762         */

765                int length;

767                        if (position >= getStartPosition() + getLength()) {
768                                return -1;
769                        }
770                        return 1;
771                }
772                int low = 0;
773                if (position < 0) {
774                        // position illegal
775                        return -1;
776                }
777                if (position <= this.lineEndTable[low]) {
778                        // before the first line delimiter
779                        return 1;
780                }
781                // assert position > lineEndTable[low+1]  && low == 0
782                int hi = length - 1;
783                if (position > this.lineEndTable[hi]) {
784                        // position beyond the last line separator
785                        if (position >= getStartPosition() + getLength()) {
786                                // this is beyond the end of the source length
787                                return -1;
788                        } else {
789                                return length + 1;
790                        }
791                }
792                // assert lineEndTable[low]  < position <= lineEndTable[hi]

794

798                        // && 0 <= low < hi <= length - 1

810                                // && 0 <= low < mid < hi <= length - 1
811                                hi = mid;
812                        } else {
813                                // position > lineEndTable[mid]
814                                // assert lineEndTable[mid] < position <= lineEndTable[hi]
815                                // && 0 <= low < mid < hi <= length - 1
816                                low = mid;
817                        }
818                        // in both cases, invariant reachieved with reduced measure
819                }
