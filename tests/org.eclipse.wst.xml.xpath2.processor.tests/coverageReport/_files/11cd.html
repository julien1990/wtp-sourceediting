295                                        return false;
296                                scopeDescription= factory.getWorkingSetScopeDescription(workingSets, includeMask); 
297                                scope= factory.createJavaSearchScope(workingSets, includeMask);
298                                SearchUtil.updateLRUWorkingSets(workingSets);
299                        }
300                }
301                
302                QuerySpecification querySpec= null;
303                if (data.getJavaElement() != null && getPattern().equals(fInitialData.getPattern())) {
304                        if (limitTo == REFERENCES)
305                                SearchUtil.warnIfBinaryConstant(data.getJavaElement(), getShell());
306                        querySpec= new ElementQuerySpecification(data.getJavaElement(), limitTo, scope, scopeDescription);
307                } else {
308                        querySpec= new PatternQuerySpecification(data.getPattern(), searchFor, data.isCaseSensitive(), data.getLimitTo(), scope, scopeDescription);
309                        data.setJavaElement(null);
310                } 
311                
312                JavaSearchQuery textSearchJob= new JavaSearchQuery(querySpec);
313                NewSearchUI.runQueryInBackground(textSearchJob);
314                return true;
315        }
