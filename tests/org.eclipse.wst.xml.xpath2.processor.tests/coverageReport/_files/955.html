495                RefactoringStatus status= super.validateDescriptor();
496                if (fName == null || "".equals(fName)) //$NON-NLS-1$
497                        status.merge(RefactoringStatus.createFatalErrorStatus(DescriptorMessages.RenameResourceDescriptor_no_new_name));
498                if (fJavaElement == null)
499                        status.merge(RefactoringStatus.createFatalErrorStatus(DescriptorMessages.RenameJavaElementDescriptor_no_java_element));
500                else {
501                        final int type= fJavaElement.getElementType();
502                        if (type == IJavaScriptElement.JAVASCRIPT_PROJECT && getProject() != null)
503                                status.merge(RefactoringStatus.createFatalErrorStatus(DescriptorMessages.RenameJavaElementDescriptor_project_constraint));
504                        if (type == IJavaScriptElement.PACKAGE_FRAGMENT_ROOT && fReferences)
505                                status.merge(RefactoringStatus.createFatalErrorStatus(DescriptorMessages.RenameJavaElementDescriptor_reference_constraint));
506                        if (fTextual) {
507                                switch (type) {
508                                        case IJavaScriptElement.PACKAGE_FRAGMENT:
509                                        case IJavaScriptElement.TYPE:
510                                        case IJavaScriptElement.FIELD:
511                                                break;
512                                        default:
513                                                status.merge(RefactoringStatus.createFatalErrorStatus(DescriptorMessages.RenameJavaElementDescriptor_textual_constraint));
514                                }
515                        }
516                        if (fDeprecate) {
517                                switch (type) {
518                                        case IJavaScriptElement.METHOD:
519                                        case IJavaScriptElement.FIELD:
520                                                break;
521                                        default:
522                                                status.merge(RefactoringStatus.createFatalErrorStatus(DescriptorMessages.RenameJavaElementDescriptor_deprecation_constraint));
523                                }
524                        }
525                        if (fDelegate) {
526                                switch (type) {
527                                        case IJavaScriptElement.METHOD:
528                                        case IJavaScriptElement.FIELD:
529                                                break;
530                                        default:
531                                                status.merge(RefactoringStatus.createFatalErrorStatus(DescriptorMessages.RenameJavaElementDescriptor_delegate_constraint));
532                                }
533                        }
534                        if (fRenameGetter || fRenameSetter) {
535                                if (type != IJavaScriptElement.FIELD)
536                                        status.merge(RefactoringStatus.createFatalErrorStatus(DescriptorMessages.RenameJavaElementDescriptor_accessor_constraint));
537                        }
538                        if (fQualified) {
539                                switch (type) {
540                                        case IJavaScriptElement.PACKAGE_FRAGMENT:
541                                        case IJavaScriptElement.TYPE: 
542                                                break;
543                                        default:
544                                                status.merge(RefactoringStatus.createFatalErrorStatus(DescriptorMessages.RenameJavaElementDescriptor_qualified_constraint));
545                                }
546                        }
547                        if (fSimilarDeclarations) {
548                                switch (type) {
549                                        case IJavaScriptElement.TYPE:
550                                                break;
551                                        default:
552                                                status.merge(RefactoringStatus.createFatalErrorStatus(DescriptorMessages.RenameJavaElementDescriptor_similar_constraint));
553                                }
554                        }
555                        if (fHierarchical) {
556                                switch (type) {
557                                        case IJavaScriptElement.PACKAGE_FRAGMENT:
558                                                break;
559                                        default:
560                                                status.merge(RefactoringStatus.createFatalErrorStatus(DescriptorMessages.RenameJavaElementDescriptor_hierarchical_constraint));
561                                }
562                        }
563                }
564                return status;
565        }
