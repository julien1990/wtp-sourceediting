241                                temp = temp.enclosingType();

248        // isDefault()

252        ReferenceBinding declaringClass = enclosingType() == null ? this : enclosingType();

257                if (currentPackage != null && currentPackage != this.fPackage) return false;
258        } while ((currentType = currentType.superclass()) != null);
259        return false;

261/*
262 * Answer true if the receiver is visible to the type provided by the scope.
263 */
264public final boolean canBeSeenBy(Scope scope) {
265        if (isPublic()) return true;
266 
267        SourceTypeBinding invocationType = scope.enclosingSourceType();
268        if (invocationType == this) return true;
269 
270        if (invocationType == null) // static import call
271                return !isPrivate() && scope.getCurrentPackage() == this.fPackage;
272 
273        if (isProtected()) {
274                // answer true if the invocationType is the declaringClass or they are in the same package
275                // OR the invocationType is a subclass of the declaringClass
276                //    AND the invocationType is the invocationType or its subclass
277                //    OR the type is a static method accessed directly through a type
278                //    OR previous assertions are true for one of the enclosing type
279                if (invocationType.fPackage == this.fPackage) return true;
280 
281                ReferenceBinding currentType = invocationType;
282                ReferenceBinding declaringClass = enclosingType(); // protected types always have an enclosing one
283                if (declaringClass == null) return false; // could be null if incorrect top-level protected type
284                // int depth = 0;
285                do {
286                        if (declaringClass == invocationType) return true;
287                        if (declaringClass.isSuperclassOf(currentType)) return true;
288                        // depth++;
289                        currentType = currentType.enclosingType();
290                } while (currentType != null);
291                return false;
292        }
293        if (isPrivate()) {
294                // answer true if the receiver and the invocationType have a common enclosingType
295                // already know they are not the identical type
296                ReferenceBinding outerInvocationType = invocationType;
297                ReferenceBinding temp = outerInvocationType.enclosingType();
298                while (temp != null) {
299                        outerInvocationType = temp;
300                        temp = temp.enclosingType();
301                }
302 
303                ReferenceBinding outerDeclaringClass = (ReferenceBinding)this.erasure();
304                temp = outerDeclaringClass.enclosingType();
305                while (temp != null) {
306                        outerDeclaringClass = temp;
307                        temp = temp.enclosingType();
308                }
309                return outerInvocationType == outerDeclaringClass;
310        }
311 
312        // isDefault()
313        return invocationType.fPackage == this.fPackage;
314}

562public boolean detectAnnotationCycle() {
563        if ((this.tagBits & TagBits.EndAnnotationCheck) != 0) return false; // already checked
564        if ((this.tagBits & TagBits.BeginAnnotationCheck) != 0) return true; // in the middle of checking its methods
565 
566        this.tagBits |= TagBits.BeginAnnotationCheck;
567        MethodBinding[] currentMethods = methods();
568        boolean inCycle = false; // check each method before failing
569        for (int i = 0, l = currentMethods.length; i < l; i++) {
570                TypeBinding returnType = currentMethods[i].returnType.leafComponentType();
571                if (returnType.isAnnotationType() && ((ReferenceBinding) returnType).detectAnnotationCycle()) {
572                        inCycle = true;
573                }
574        }
575        if (inCycle)
576                return true;
577        this.tagBits |= TagBits.EndAnnotationCheck;
578        return false;
579}

679                if (itsInterfaces != null && itsInterfaces != Binding.NO_SUPERINTERFACES) {

684                                int itsLength = itsInterfaces.length;
685                                if (nextPosition + itsLength >= interfacesToVisit.length)
686                                        System.arraycopy(interfacesToVisit, 0, interfacesToVisit = new ReferenceBinding[nextPosition + itsLength + 5], 0, nextPosition);
687                                nextInterface : for (int a = 0; a < itsLength; a++) {
688                                        ReferenceBinding next = itsInterfaces[a];
689                                        for (int b = 0; b < nextPosition; b++)
690                                                if (next == interfacesToVisit[b]) continue nextInterface;
691                                        interfacesToVisit[nextPosition++] = next;
692                                }
693                        }
694                }
695        } while ((currentType = currentType.superclass()) != null);
696 
697        for (int i = 0; i < nextPosition; i++) {
698                currentType = interfacesToVisit[i];
699                if (currentType == otherType) return false;

701                if (match != null && !match.isIntersectingWith(currentType))
702                        return true;

705                if (itsInterfaces != null && itsInterfaces != Binding.NO_SUPERINTERFACES) {
706                        int itsLength = itsInterfaces.length;
707                        if (nextPosition + itsLength >= interfacesToVisit.length)
708                                System.arraycopy(interfacesToVisit, 0, interfacesToVisit = new ReferenceBinding[nextPosition + itsLength + 5], 0, nextPosition);

711                                for (int b = 0; b < nextPosition; b++)
712                                        if (next == interfacesToVisit[b]) continue nextInterface;
713                                interfacesToVisit[nextPosition++] = next;
714                        }
715                }
716        }
717        return false;

818 * Answer true if the receiver type can be assigned to the argument type (right)
819 * In addition to improving performance, caching also ensures there is no infinite regression
820 * since per nature, the compatibility check is recursive through parameterized type arguments (122775)
821 */
822public boolean isCompatibleWith(TypeBinding otherType) {
823 
824        if (otherType == this)
825                return true;
826        if (otherType.id == TypeIds.T_JavaLangObject || otherType.id == TypeIds.T_any)
827                return true;

830                this.compatibleCache = new SimpleLookupTable(3);
831                result = null;
832        } else {
833                result = this.compatibleCache.get(otherType);
834                if (result != null) {
835                        return result == Boolean.TRUE;
836                }
837        }
838        this.compatibleCache.put(otherType, Boolean.FALSE); // protect from recursive call
839        if (isCompatibleWith0(otherType)) {
840                this.compatibleCache.put(otherType, Boolean.TRUE);

846/**
847 * Answer true if the receiver type can be assigned to the argument type (right)
848 */
849private boolean isCompatibleWith0(TypeBinding otherType) {
850        if (otherType == this)
851                return true;
852        if (otherType.id == TypeIds.T_JavaLangObject)
853                return true;
854        // equivalence may allow compatibility with array type through wildcard
855        // bound
856        if (this.isEquivalentTo(otherType))
857                return true;
858        switch (otherType.kind()) {
859                case Binding.WILDCARD_TYPE :
860                        return false; // should have passed equivalence check above if
861                                                        // wildcard
862                case Binding.TYPE_PARAMETER :
863                        // check compatibility with capture of ? super X
864                        if (otherType.isCapture()) {
865                                CaptureBinding otherCapture = (CaptureBinding) otherType;
866                                TypeBinding otherLowerBound;
867                                if ((otherLowerBound = otherCapture.lowerBound) != null) {
868                                        if (otherLowerBound.isArrayType()) return false;
869                                        return this.isCompatibleWith(otherLowerBound);
870                                }
871                        }
872                case Binding.GENERIC_TYPE :
873                case Binding.TYPE :
874                case Binding.PARAMETERIZED_TYPE :
875                case Binding.RAW_TYPE :
876                        switch (this.kind()) {
877                                case Binding.GENERIC_TYPE :
878                                case Binding.PARAMETERIZED_TYPE :
879                                case Binding.RAW_TYPE :
880                                        if (this.erasure() == otherType.erasure())
881                                                return false; // should have passed equivalence check
882                                                                                // above if same erasure
883                        }
884                        ReferenceBinding otherReferenceType = (ReferenceBinding) otherType;
885                        if (otherReferenceType.isInterface()) // could be annotation type
886                                return implementsInterface(otherReferenceType, true);
887                        if (this.isInterface())  // Explicit conversion from an interface
888                                                                                // to a class is not allowed
889                                return false;
890                        if(Arrays.equals(this.compoundName,otherReferenceType.compoundName) && Arrays.equals(this.fileName, otherReferenceType.fileName)) {
891                                return true;
892                        }
893 
894 
895                        if ( otherReferenceType.isSuperclassOf(this))
896                                return true;
897                        return (otherReferenceType.isAnonymousType()  && this.isSuperclassOf(otherReferenceType));
898                case Binding.ARRAY_TYPE:
899                   return this==((ArrayBinding)otherType).referenceBinding;
900                default :
901                        return false;
902        }
903}
