462

465

477        private static int getRandomPoissonValue(int mean) {
478                final int MAX_VALUE= 10;
479 
480                final float random= (float) Math.random();
481                float probability= 0;
482                int i= 0;
483                while (probability < 1 && i < MAX_VALUE) {
484                        probability += getPoissonDistribution(mean, i);
485                        if (random <= probability)
486                                break;
487                        i++;
488                }
489                return i;
490        }

492        private static float getPoissonDistribution(float lambda, int k) {
493                return (float) (Math.exp(-lambda) * Math.pow(lambda, k) / faculty(k));
494        }

496        /**
497         * Returns the faculty of k.
498         *
499         * @param k the <code>int</code> for which to get the faculty
500         * @return the faculty

502        private static final int faculty(int k) {
503                return k == 0
504                        ? 1

506        }
