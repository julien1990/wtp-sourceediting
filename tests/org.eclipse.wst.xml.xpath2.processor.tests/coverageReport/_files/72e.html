699         * @param eid int
700         */
701        CMGroupImpl createContent(int eid) {
702                if (eid == ID_UNKNOWN)
703                        return null;
704 
705                CMGroupImpl content = null;
706                CMNode child = null;
707 
708                switch (eid) {
709                        case Ids.ID_INCLUDE :
710                        case Ids.ID_FORWARD :
711                                // (jsp:param)*
712                                content = new CMGroupImpl(CMGroup.SEQUENCE, 0, CMContentImpl.UNBOUNDED);
713                                child = item(Ids.ID_PARAM);
714                                if (child != null)
715                                        content.appendChild(child);
716                                break;
717                        case Ids.ID_PLUGIN :
718                                // (jsp:params | jsp:fallback)?
719                                content = new CMGroupImpl(CMGroup.CHOICE, 0, 1);
720                                // jsp:params
721                                child = item(Ids.ID_PARAMS);
722                                if (child != null)
723                                        content.appendChild(child);
724                                // jsp:fallback
725                                child = item(Ids.ID_FALLBACK);
726                                if (child != null)
727                                        content.appendChild(child);
728                                break;
729                        case Ids.ID_PARAMS :
730                                // (jsp:param)+
731                                content = new CMGroupImpl(CMGroup.SEQUENCE, 1, CMContentImpl.UNBOUNDED);
732                                child = item(Ids.ID_PARAM);
733                                if (child != null)
734                                        content.appendChild(child);
735                                break;
736                        case Ids.ID_ROOT :
737                                // %Body;
738                                // --> (jsp:text|%Directives;|%Scripts;|%Actions;)*
739                                //     %Directives --> jsp:directive.page|jsp:directive.include
740                                //     %Scripts; --> jsp:scriptlet|jsp:declaration|jsp:expression
741                                //     %Actions --> jsp:useBean|jsp.setProperty|jsp:getProperty
742                                //                  |jsp:include|jsp:forward|jsp:plugin
743                                content = new CMGroupImpl(CMGroup.CHOICE, 0, CMContentImpl.UNBOUNDED);
744                                int validChildren[] = {Ids.ID_TEXT,
745                                // %Directves;
746                                                        Ids.ID_DIRECTIVE_PAGE, Ids.ID_DIRECTIVE_INCLUDE,
747                                                        // %Scripts;
748                                                        Ids.ID_SCRIPTLET, Ids.ID_DECLARATION, Ids.ID_EXPRESSION,
749                                                        // %Actions;
750                                                        Ids.ID_USEBEAN, Ids.ID_SETPROPERTY, Ids.ID_GETPROPERTY, Ids.ID_INCLUDE, Ids.ID_FORWARD, Ids.ID_PLUGIN};
751                                for (int i = 0; i < validChildren.length; i++) {
752                                        child = item(validChildren[i]);
753                                        if (child != null)
754                                                content.appendChild(child);
755                                }
756                                break;
757                }
758 
759                return content;
760        }
