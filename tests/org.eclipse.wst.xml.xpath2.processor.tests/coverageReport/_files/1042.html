211        private String getPathString(IPath fullPath) {
212                if (fBasePath == null || !hasSameDevice(fullPath, fBasePath)) {
213                        return fullPath.toOSString();
214                }
215                int matchingSegments= fBasePath.matchingFirstSegments(fullPath);
216                if (fBasePath.segmentCount() == matchingSegments) {
217                        return getRelativePath(fullPath, matchingSegments);
218                }
219                for (int i= 0; i < fJavaProjects.length; i++) {
220                        IProject proj= fJavaProjects[i].getProject();
221                        IPath projLoc= proj.getLocation();
222                        if (projLoc != null && projLoc.segmentCount() <= matchingSegments && projLoc.isPrefixOf(fullPath)) {
223                                return getRelativePath(fullPath, matchingSegments);
224                        }
225                }
226                IPath workspaceLoc= ResourcesPlugin.getWorkspace().getRoot().getLocation();
227                if (workspaceLoc.segmentCount() <= matchingSegments && workspaceLoc.isPrefixOf(fullPath)) {
228                        return getRelativePath(fullPath, matchingSegments);
229                }                
230                return fullPath.toOSString();
231        }

238                        res.append(PATH_SEPARATOR);
239                        backSegments--;

241                int segCount= fullPath.segmentCount();
242                for (int i= matchingSegments; i < segCount; i++) {
243                        if (i > matchingSegments) {

245                        }
