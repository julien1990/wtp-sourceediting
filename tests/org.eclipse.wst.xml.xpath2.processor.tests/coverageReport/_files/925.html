359        private static int compareImport(char prefix, String qualifier, String name, String curr) {
360                if (curr.charAt(0) != prefix || !curr.endsWith(name)) {
361                        return ImportRewriteContext.RES_NAME_UNKNOWN;
362                }
363 
364                curr= curr.substring(1); // remove the prefix
365 
366                if (curr.length() == name.length()) {
367                        if (qualifier.length() == 0) {
368                                return ImportRewriteContext.RES_NAME_FOUND;
369                        }
370                        return ImportRewriteContext.RES_NAME_CONFLICT;
371                }
372                // at this place: curr.length > name.length
373

375                if (curr.charAt(dotPos) != '.') {
376                        return ImportRewriteContext.RES_NAME_UNKNOWN;
377                }
378                if (qualifier.length() != dotPos || !curr.startsWith(qualifier)) {
379                        return ImportRewriteContext.RES_NAME_CONFLICT;
380                }
381                return ImportRewriteContext.RES_NAME_FOUND;
