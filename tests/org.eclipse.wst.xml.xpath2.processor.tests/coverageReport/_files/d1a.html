337                                        buffer.append("PACKAGE: "); //$NON-NLS-1$
338                                        break;
339                                case TYPE:
340                                        buffer.append("TYPE: "); //$NON-NLS-1$
341                                        break;
342                                case FIELD:
343                                        buffer.append("FIELD: "); //$NON-NLS-1$
344                                        break;
345                                case METHOD:
346                                        buffer.append("METHOD: "); //$NON-NLS-1$
347                                        break;
348                                case ARRAY:
349                                        buffer.append("ARRAY: "); //$NON-NLS-1$
350                                        break;
351                                case LOCAL_VAR:
352                                        buffer.append("LOCAL VAR: "); //$NON-NLS-1$
353                                        break;
354                                case FLAGS:
355                                        buffer.append("MODIFIERS: "); //$NON-NLS-1$
356                                        break;
357                                case WILDCARD:
358                                        buffer.append("WILDCARD: "); //$NON-NLS-1$
359                                        break;
360                                case CAPTURE:
361                                        buffer.append("CAPTURE: "); //$NON-NLS-1$
362                                        break;
363                                case BASE_TYPE:
364                                        buffer.append("BASE TYPE: "); //$NON-NLS-1$
365                                        break;
366                                case END:
367                                        buffer.append("END: "); //$NON-NLS-1$
368                                        break;
369                        }
370                        if (this.index < 0) {
371                                buffer.append("**"); //$NON-NLS-1$
372                                buffer.append(this.source);
373                        } else if (this.index <= this.source.length) {
374                                buffer.append(this.source, 0, this.start);
375                                buffer.append('*');
376                                if (this.start <= this.index) {
377                                        buffer.append(this.source, this.start, this.index - this.start);
378                                        buffer.append('*');
379                                        buffer.append(this.source, this.index, this.source.length - this.index);
380                                } else {
381                                        buffer.append('*');
382                                        buffer.append(this.source, this.start, this.source.length - this.start);
383                                }
384                        } else {
385                                buffer.append(this.source);
386                                buffer.append("**"); //$NON-NLS-1$
387                        }
388                        return buffer.toString();
389                }

533        public void parse(boolean pauseAfterFullyQualifiedName) {
534                try {
535                        if (!this.parsingPaused) {
536                                // fully qualified name
537                                parseFullyQualifiedName();
538                                parseSecondaryType();
539                                if (pauseAfterFullyQualifiedName) {
540                                        this.parsingPaused = true;
541                                        return;
542                                }
543                        }
544                        if (!hasTypeName()) {
545                                consumeKey();
546                                return;
547                        }
548                        consumeTopLevelType();
549                        parseInnerType();
550 
551                        if (this.scanner.isAtParametersStart()) {
552                                this.scanner.skipParametersStart();
553                                if (this.scanner.isAtTypeParameterStart())        {
554                                        // generic type
555                                        parseGenericType();
556                                         // skip ";>"
557                                         this.scanner.skipParametersEnd();
558                                        // local type in generic type
559                                        parseInnerType();
560                                } else if (this.scanner.isAtTypeArgumentStart())
561                                        // parameterized type
562                                        parseParameterizedType(null/*top level type or member type with raw enclosing type*/, false/*no raw*/);
563                                else if (this.scanner.isAtRawTypeEnd())
564                                        // raw type
565                                        parseRawType();
566                        } else {
567                                // non-generic type
568                                consumeNonGenericType();
569                        }
570 
571                        consumeType();
572                        this.scanner.skipTypeEnd();
573 
574                        if (this.scanner.isAtFieldOrMethodStart()) {
575                                switch (this.scanner.nextToken()) {
576                                        case Scanner.FIELD:
577                                                parseField();
578                                                return;
579                                        case Scanner.METHOD:
580                                                parseMethod();
581                                                if (this.scanner.isAtLocalVariableStart()) {
582                                                        parseLocalVariable();
583                                                } else if (this.scanner.isAtTypeVariableStart()) {
584                                                        parseTypeVariable();
585                                                }
586                                                 break;
587                                        default:
588                                                malformedKey();
589                                                return;
590                                }
591                        } else if (this.scanner.isAtTypeVariableStart()) {
592                                parseTypeVariable();
593                        } else if (this.scanner.isAtWildcardStart()) {
594                                parseWildcard();
595                        } else if (this.scanner.isAtTypeWithCaptureStart()) {
596                                parseTypeWithCapture();
597                        }
598 
599                        consumeKey();
600                } catch (IllegalArgumentException e) {
601                        // the given key was illegal
602                }
603        }

632                                                 break;
633                                         case Scanner.BASE_TYPE:

635                                                 this.hasTypeName = false;
636                                                 break;
637                                         default:
638                                                malformedKey();
639                                                return;
640                                }
641                                break;

644                                return;
645                }

828        private void parseWildcard() {
829                if (this.scanner.nextToken() != Scanner.WILDCARD) return;
830                 char[] source = this.scanner.getTokenSource();
831                 if (source.length == 0) {
832                         malformedKey();
833                         return;
834                 }
835                 int kind = -1;
836                 switch (source[0]) {
837                         case '*':
838                                 kind = Wildcard.UNBOUND;
839                                 break;
840                         case '+':
841                                 kind = Wildcard.EXTENDS;
842                                 break;
843                         case '-':
844                                 kind = Wildcard.SUPER;
845                                 break;
846                 }
847                 if (kind == -1) {
848                         malformedKey();
849                         return;
850                 }
851                 if (kind != Wildcard.UNBOUND)
852                         parseWildcardBound();
853                 consumeWildCard(kind);
854        }
