330                                        IBuffer buffer= unit.getBuffer();
331                                        String matchText= buffer.getText(start, length);
332                                        if (fSource.isDefaultPackage()) {
333                                                super.acceptSearchMatch(TypeReference.createSimpleReference(element, accuracy, start, length, insideDocComment, res, matchText));
334                                        } else {
335                                                // assert: matchText doesn't start nor end with comment
336                                                int simpleNameStart= getLastSimpleNameStart(matchText);
337                                                if (simpleNameStart != 0) {
338                                                        super.acceptSearchMatch(TypeReference.createQualifiedReference(element, accuracy, start, length, insideDocComment, res, start + simpleNameStart));
339                                                } else {
340                                                        super.acceptSearchMatch(TypeReference.createSimpleReference(element, accuracy, start, length, insideDocComment, res, matchText));
341                                                }
342                                        }
343                                }
344                        }
345                }

347                private int getLastSimpleNameStart(String reference) {
348                        fScanner.setSource(reference.toCharArray());
349                        int lastIdentifierStart= -1;
350                        try {
351                                int tokenType= fScanner.getNextToken();
352                                while (tokenType != ITerminalSymbols.TokenNameEOF) {
353                                        if (tokenType == ITerminalSymbols.TokenNameIdentifier)
354                                                lastIdentifierStart= fScanner.getCurrentTokenStartPosition();
355                                        tokenType= fScanner.getNextToken();
356                                }
357                        } catch (InvalidInputException e) {
358                                JavaScriptPlugin.log(e);
359                        }
360                        return lastIdentifierStart;
361                }
