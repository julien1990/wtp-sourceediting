342                                if (lastRegion instanceof ITextRegionContainer) {
343                                        ITextRegionContainer container = (ITextRegionContainer) lastRegion;
344                                        container.getRegions().add(region);
345                                        // DW, 4/16/2003 container regions have parent. Probably a
346                                        // better place to set,
347                                        // but for now, will (re)set each time through
348                                        container.setParent(currentNode);
349                                        // DW, 4/16/2003 token regions no longer have parents
350                                        // region.setParent(container);
351                                        region.adjustStart(container.getLength() - region.getStart());
352                                }
353                                currentNode.getLastRegion().adjustLength(region.getLength());
354                                currentNode.adjustLength(region.getLength());
355                        }
356                        else if (type == DOMRegionContext.UNDEFINED && currentNode != null) {
357                                // skip on a very-first region situation as the default
358                                // behavior is good enough
359                                // combine with previous if also undefined
360                                if (currentNode.getLastRegion() != null && currentNode.getLastRegion().getType() == DOMRegionContext.UNDEFINED) {
361                                        currentNode.getLastRegion().adjustLength(region.getLength());
362                                        currentNode.adjustLength(region.getLength());
363                                }
364                                // previous wasn't undefined
365                                else {
366                                        currentNode.addRegion(region);
367                                        currentNode.setLength(region.getStart() + region.getLength() - currentNode.getStart());
368                                        region.adjustStart(-currentNode.getStart());
369                                }
370                                if (region instanceof ITextRegionContainer) {
371                                        ((ITextRegionContainer) region).setParent(currentNode);
372                                }
373                        }
374                        else {
375                                // if an unknown type is the first region in the document,
376                                // ensure that a node exists
377                                if (currentNode == null) {
378                                        currentNode = createStructuredDocumentRegion(type);
379                                        currentNode.setStart(region.getStart());
380                                }
381                                currentNode.addRegion(region);
382                                currentNode.setLength(region.getStart() + region.getLength() - currentNode.getStart());
383                                region.adjustStart(-currentNode.getStart());
384                                // DW, 4/16/2003 token regions no longer have parents
385                                // region.setParent(currentNode);
386                                if (region instanceof ITextRegionContainer) {
387                                        ((ITextRegionContainer) region).setParent(currentNode);
388                                }
389 
390                                if (Debug.debugTokenizer)
391                                        System.out.println(getClass().getName() + " found region of not specifically handled type " + region.getType() + " @ " + region.getStart() + "[" + region.getLength() + "]");//$NON-NLS-4$//$NON-NLS-3$//$NON-NLS-2$//$NON-NLS-1$
392 
393                        }
394 
395                        // these regions also get their own node, so close them cleanly
396                        // NOTE: these regions have new StructuredDocumentRegions created
397                        // for them above; it may
398                        // be more readable if that is handled here as well, but the
399                        // current layout
400                        // ensures that they open StructuredDocumentRegions the same way
401                        if ((type == DOMRegionContext.XML_CONTENT) || (type == DOMRegionContext.XML_CHAR_REFERENCE) || (type == DOMRegionContext.XML_ENTITY_REFERENCE) || (type == DOMJSPRegionContexts.JSP_DECLARATION_OPEN) || (type == DOMJSPRegionContexts.JSP_EXPRESSION_OPEN) || (type == DOMJSPRegionContexts.JSP_SCRIPTLET_OPEN) || (type == DOMJSPRegionContexts.JSP_CONTENT) || (type == DOMJSPRegionContexts.JSP_CLOSE)) {
402                                currentNode.setEnded(true);
403                        }
404                        if (headNode == null && currentNode != null) {
405                                headNode = currentNode;
406                        }
407                }
408                if (currentNode != null) {
409                        fireNodeParsed(currentNode);
410                        currentNode.setPrevious(lastNode);
411                }
412                primReset();
413                return headNode;
414        }
