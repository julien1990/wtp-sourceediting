488        
489        private boolean updateViewerFilters() {
490                ViewerFilter[] installedFilters= fViewer.getFilters();
491                ArrayList viewerFilters= new ArrayList(Arrays.asList(installedFilters));
492                HashSet patterns= new HashSet();
493                
494                boolean hasChange= false;
495                boolean patternChange= false;
496                
497                for (Iterator iterator= fFilterItems.values().iterator(); iterator.hasNext();) {
498                        FilterItem item= (FilterItem) iterator.next();
499                        if (item.descriptor.isCustomFilter()) {
500                                if (item.enabled != item.previouslyEnabled) {
501                                        ViewerFilter filter= item.getFilterInstance(); // only create when changed
502                                        if (filter != null) {
503                                                if (item.enabled) {
504                                                        viewerFilters.add(filter);
505                                                } else {
506                                                        viewerFilters.remove(filter);
507                                                }
508                                                hasChange= true;
509                                        }
510                                }
511                        } else if (item.descriptor.isPatternFilter()) {
512                                if (item.enabled) {
513                                        patterns.add(item.descriptor.getPattern());
514                                }
515                                patternChange |= (item.enabled != item.previouslyEnabled);
516                        }
517                        item.previouslyEnabled= item.enabled;
518                }
519                
520                if (areUserDefinedPatternsEnabled()) {
521                        for (int i= 0; i < fUserDefinedPatterns.length; i++) {
522                                patterns.add(fUserDefinedPatterns[i]);
523                        }
524                }
525                if (!patternChange) { // no pattern change so far, test if the user patterns made a difference
526                        patternChange= hasChanges(patterns, fPreviousPatterns);
527                }
528                
529                fPreviousPatterns= (String[]) patterns.toArray(new String[patterns.size()]);
530                if (patternChange) {
531                        fPatternFilter.setPatterns(fPreviousPatterns);
532                        if (patterns.isEmpty()) {
533                                viewerFilters.remove(fPatternFilter);
534                        } else if (!viewerFilters.contains(fPatternFilter)) {
535                                viewerFilters.add(fPatternFilter);
536                        }
537                        hasChange= true;
538                }
539                if (hasChange) {
540                        fViewer.setFilters((ViewerFilter[]) viewerFilters.toArray(new ViewerFilter[viewerFilters.size()])); // will refresh
541                }
542                return hasChange;
543        }
544 
545        private boolean hasChanges(HashSet patterns, String[] oldPatterns) {
546                HashSet copy= (HashSet) patterns.clone();
547                for (int i= 0; i < oldPatterns.length; i++) {
548                        boolean found= copy.remove(oldPatterns[i]);
549                        if (!found)
550                                return true;
551                }
552                return !copy.isEmpty();
553        }
