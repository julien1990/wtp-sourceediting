552         */
553        protected void fillContentProvider(AbstractContentProvider provider, ItemsFilter itemsFilter, IProgressMonitor progressMonitor) throws CoreException {
554                TypeItemsFilter typeSearchFilter= (TypeItemsFilter) itemsFilter;
555                TypeSearchRequestor requestor= new TypeSearchRequestor(provider, typeSearchFilter);
556                SearchEngine engine= new SearchEngine((WorkingCopyOwner) null);
557                String packPattern= typeSearchFilter.getPackagePattern();
558                progressMonitor.setTaskName(JavaUIMessages.FilteredTypesSelectionDialog_searchJob_taskName);
559                
560                /*
561                 * Setting the filter into match everything mode avoids filtering twice
562                 * by the same pattern (the search engine only provides filtered
563                 * matches). For the case when the pattern is a camel case pattern with
564                 * a terminator, the filter is not set to match everything mode because
565                 * jdt.core's SearchPattern does not support that case.
566                 */ 
567                String typePattern= itemsFilter.getPattern();
568                int matchRule= typeSearchFilter.getMatchRule();
569                if (matchRule == SearchPattern.RULE_CAMELCASE_MATCH) {
570                         // If the pattern is empty, the RULE_BLANK_MATCH will be chosen, so we don't have to check the pattern length
571                        char lastChar= typePattern.charAt(typePattern.length() - 1);
572 
573                        if (lastChar == '<' || lastChar == ' ') {
574                                typePattern= typePattern.substring(0, typePattern.length() - 1);
575                        } else {
576                                typeSearchFilter.setMatchEverythingMode(true);
577                        }
578                } else {
579                        typeSearchFilter.setMatchEverythingMode(true);
580                }
581 
582                try {
583                        engine.searchAllTypeNames(packPattern == null ? null : packPattern.toCharArray(),
584                                        typeSearchFilter.getPackageFlags(), //TODO: https://bugs.eclipse.org/bugs/show_bug.cgi?id=176017
585                                        typePattern.toCharArray(),
586                                        matchRule, //TODO: https://bugs.eclipse.org/bugs/show_bug.cgi?id=176017
587                                        typeSearchFilter.getElementKind(),
588                                        typeSearchFilter.getSearchScope(),
589                                        requestor,
590                                        IJavaScriptSearchConstants.WAIT_UNTIL_READY_TO_SEARCH,
591                                        progressMonitor);
592                } finally {
593                        typeSearchFilter.setMatchEverythingMode(false);
594                }
595        }
