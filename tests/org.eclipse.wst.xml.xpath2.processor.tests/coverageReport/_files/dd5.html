397        private void computeOutput(RefactoringStatus status) {
398                // First find all writes inside the selection.
399                FlowContext flowContext= new FlowContext(0, fMaxVariableId + 1);
400                flowContext.setConsiderAccessMode(true);
401                flowContext.setComputeMode(FlowContext.RETURN_VALUES);
402                FlowInfo returnInfo= new InOutFlowAnalyzer(flowContext).perform(getSelectedNodes());
403                IVariableBinding[] returnValues= returnInfo.get(flowContext, FlowInfo.WRITE | FlowInfo.WRITE_POTENTIAL | FlowInfo.UNKNOWN);
404                
405                // Compute a selection that exactly covers the selected nodes
406                IRegion region= getSelectedNodeRange();
407                Selection selection= Selection.createFromStartLength(region.getOffset(), region.getLength());
408                
409                int counter= 0;
410                flowContext.setComputeMode(FlowContext.ARGUMENTS);
411                FlowInfo argInfo= new InputFlowAnalyzer(flowContext, selection, true).perform(fEnclosingBodyDeclaration);
412                IVariableBinding[] reads= argInfo.get(flowContext, FlowInfo.READ | FlowInfo.READ_POTENTIAL | FlowInfo.UNKNOWN);
413                outer: for (int i= 0; i < returnValues.length && counter <= 1; i++) {
414                        IVariableBinding binding= returnValues[i];
415                        for (int x= 0; x < reads.length; x++) {
416                                if (reads[x] == binding) {
417                                        counter++;
418                                        fReturnValue= binding;
419                                        continue outer;
420                                }
421                        }
422                }
423                switch (counter) {
424                        case 0:
425                                fReturnValue= null;
426                                break;
427                        case 1:
428                                break;
429                        default:
430                                fReturnValue= null;
431                                status.addFatalError(RefactoringCoreMessages.ExtractMethodAnalyzer_assignments_to_local, JavaStatusContext.create(fCUnit, getSelection())); 
432                                return;
433                }
434                List callerLocals= new ArrayList(5);
435                FlowInfo localInfo= new InputFlowAnalyzer(flowContext, selection, false).perform(fEnclosingBodyDeclaration);
436                IVariableBinding[] writes= localInfo.get(flowContext, FlowInfo.WRITE | FlowInfo.WRITE_POTENTIAL | FlowInfo.UNKNOWN);
437                for (int i= 0; i < writes.length; i++) {
438                        IVariableBinding write= writes[i];
439                        if (getSelection().covers(ASTNodes.findDeclaration(write, fEnclosingBodyDeclaration)))
440                                callerLocals.add(write);
441                }
442                fCallerLocals= (IVariableBinding[])callerLocals.toArray(new IVariableBinding[callerLocals.size()]);
443                if (fReturnValue != null && getSelection().covers(ASTNodes.findDeclaration(fReturnValue, fEnclosingBodyDeclaration)))
444                        fReturnLocal= fReturnValue;
445        }

447        private void adjustArgumentsAndMethodLocals() {
448                for (int i= 0; i < fArguments.length; i++) {
449                        IVariableBinding argument= fArguments[i];
450                        // Both arguments and locals consider FlowInfo.WRITE_POTENTIAL. But at the end a variable
451                        // can either be a local of an argument. Fix this based on the compute return type which
452                        // didn't exist when we computed the locals and arguments (see computeInput())
453                        if (fInputFlowInfo.hasAccessMode(fInputFlowContext, argument, FlowInfo.WRITE_POTENTIAL)) {
454                                if (argument != fReturnValue)
455                                        fArguments[i]= null;
456                                // We didn't remove the argument. So we have to remove the local declaration
457                                if (fArguments[i] != null) {
458                                        for (int l= 0; l < fMethodLocals.length; l++) {
459                                                if (fMethodLocals[l] == argument)
460                                                        fMethodLocals[l]= null;                                                
461                                        }
462                                }
463                        }
464                }

473        private IVariableBinding[] compressArray(IVariableBinding[] array) {
474                if (array == null)
475                        return null;
476                int size= 0;
477                for (int i= 0; i < array.length; i++) {
478                        if (array[i] != null)
479                                size++;        
480                }
481                if (size == array.length)
482                        return array;
483                IVariableBinding[] result= new IVariableBinding[size];
484                for (int i= 0, r= 0; i < array.length; i++) {
485                        if (array[i] != null)
486                                result[r++]= array[i];                
487                }
488                return result;

537        public void endVisit(JavaScriptUnit node) {
538                RefactoringStatus status= getStatus();
539                superCall: {
540                        if (status.hasFatalError())
541                                break superCall;
542                        if (!hasSelectedNodes()) {
543                                ASTNode coveringNode= getLastCoveringNode();
544                                if (coveringNode instanceof Block && coveringNode.getParent() instanceof FunctionDeclaration) {
545                                        FunctionDeclaration methodDecl= (FunctionDeclaration)coveringNode.getParent();
546                                        Message[] messages= ASTNodes.getMessages(methodDecl, ASTNodes.NODE_ONLY);
547                                        if (messages.length > 0) {
548                                                status.addFatalError(Messages.format(
549                                                        RefactoringCoreMessages.ExtractMethodAnalyzer_compile_errors, 
550                                                        methodDecl.getName().getIdentifier()), JavaStatusContext.create(fCUnit, methodDecl));
551                                                break superCall;
552                                        }
553                                }
554                                status.addFatalError(RefactoringCoreMessages.ExtractMethodAnalyzer_only_method_body); 
555                                break superCall;
556                        }
557                        fEnclosingBodyDeclaration= (BodyDeclaration)ASTNodes.getParent(getFirstSelectedNode(), BodyDeclaration.class);
558                        if (fEnclosingBodyDeclaration == null || 
559                                        (fEnclosingBodyDeclaration.getNodeType() != ASTNode.FUNCTION_DECLARATION && 
560                                         fEnclosingBodyDeclaration.getNodeType() != ASTNode.INITIALIZER)) {
561                                status.addFatalError(RefactoringCoreMessages.ExtractMethodAnalyzer_only_method_body); 
562                                break superCall;
563                        } else if (ASTNodes.getEnclosingType(fEnclosingBodyDeclaration) == null) {
564                                status.addFatalError(RefactoringCoreMessages.ExtractMethodAnalyzer_compile_errors_no_parent_binding);
565                                break superCall;
566                        } else if (fEnclosingBodyDeclaration.getNodeType() == ASTNode.FUNCTION_DECLARATION) {
567                                fEnclosingMethodBinding= ((FunctionDeclaration)fEnclosingBodyDeclaration).resolveBinding();
568                        }
569                        if (!isSingleExpressionOrStatementSet()) {
570                                status.addFatalError(RefactoringCoreMessages.ExtractMethodAnalyzer_single_expression_or_set); 
571                                break superCall;
572                        }
573                        if (isExpressionSelected()) {
574                                ASTNode expression= getFirstSelectedNode();
575                                if (expression instanceof Name) {
576                                        Name name= (Name)expression;
577                                        if (name.resolveBinding() instanceof ITypeBinding) {
578                                                status.addFatalError(RefactoringCoreMessages.ExtractMethodAnalyzer_cannot_extract_type_reference); 
579                                                break superCall;
580                                        }
581                                        if (name.resolveBinding() instanceof IFunctionBinding) {
582                                                status.addFatalError(RefactoringCoreMessages.ExtractMethodAnalyzer_cannot_extract_method_name_reference); 
583                                        }
584                                        if (name.isSimpleName() && ((SimpleName)name).isDeclaration()) {
585                                                status.addFatalError(RefactoringCoreMessages.ExtractMethodAnalyzer_cannot_extract_name_in_declaration); 
586                                                break superCall;
587                                        }
588                                }
589                                fForceStatic= 
590                                        ASTNodes.getParent(expression, ASTNode.SUPER_CONSTRUCTOR_INVOCATION) != null ||
591                                        ASTNodes.getParent(expression, ASTNode.CONSTRUCTOR_INVOCATION) != null;
592                        }
593                        status.merge(LocalTypeAnalyzer.perform(fEnclosingBodyDeclaration, getSelection()));
594                        computeLastStatementSelected();
595                }
596                super.endVisit(node);
597        }
