791         */
792        public static void flush(List classPathEntries, IJavaScriptProject javaScriptProject, LibrarySuperType superType, IProgressMonitor monitor) throws CoreException, OperationCanceledException {
793                if(superType==null) {
794                        System.out.println("---------------------------------- NULL SUPER TYPE -------------------------"); //$NON-NLS-1$
795                }
796                if (superType != null) {
797                        setProjectSuperType(javaScriptProject, superType);
798                }
799                
800                if (monitor == null) {
801                        monitor= new NullProgressMonitor();
802                }
803                monitor.setTaskName(NewWizardMessages.BuildPathsBlock_operationdesc_java); 
804                monitor.beginTask("", classPathEntries.size() * 4 + 4); //$NON-NLS-1$
805                
806                try {
807                        IProject project= javaScriptProject.getProject();
808                        IPath projPath= project.getFullPath();
809                        
810                        monitor.worked(1);
811                        
812                        IWorkspaceRoot fWorkspaceRoot= JavaScriptPlugin.getWorkspace().getRoot();
813                        
814                        monitor.worked(1);
815                        
816                        if (monitor.isCanceled()) {
817                                throw new OperationCanceledException();
818                        }
819                        
820                        IIncludePathEntry[] classpath= new IIncludePathEntry[classPathEntries.size()];
821                        
822                        int i= 0;
823                        for (Iterator iter= classPathEntries.iterator(); iter.hasNext();) {
824                                CPListElement entry= (CPListElement)iter.next();
825                                classpath[i]= entry.getClasspathEntry();
826                                i++;
827                                
828                                IResource res= entry.getResource();
829                                //1 tick
830                                if (res instanceof IFolder && entry.getLinkTarget() == null && !res.exists()) {
831                                        CoreUtility.createFolder((IFolder)res, true, true, new SubProgressMonitor(monitor, 1));
832                                } else {
833                                        monitor.worked(1);
834                                }
835                                
836                                //3 ticks
837                                if (entry.getEntryKind() == IIncludePathEntry.CPE_SOURCE) {
838                                        monitor.worked(1);
