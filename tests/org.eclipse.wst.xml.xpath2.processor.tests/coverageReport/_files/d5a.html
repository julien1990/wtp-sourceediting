622         *                 Accessibility and return types are not taken into account.

633                        return false;

638                                && m1TypeParams.length != 0) //non-generic m1 can override a generic m2

645

647                        for (int i= 0; i < m1TypeParams.length; i++) {
648                                // loop over m1TypeParams, which is either empty, or equally long as m2TypeParams

656                                return true;

664                                        return false;
665                        }
666                        return true;
667                        
668                } else {
669                        // m1TypeParams.length == m2TypeParams.length == 0  
670                        if (equals(m1Params, m2Params))
671                                return true;
672                        for (int i= 0; i < m1Params.length; i++) {
673                                ITypeBinding m1Param= m1Params[i];
674                                if (m1Param.isRawType())
675                                        m1Param= m1Param.getTypeDeclaration();
676                                if (! (equals(m1Param, m2Params[i].getErasure()))) // can erase m2

680                }
