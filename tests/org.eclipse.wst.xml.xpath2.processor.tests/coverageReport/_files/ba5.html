846public MethodBinding resolveTypesFor(MethodBinding method) {
847        if ((method.modifiers & ExtraCompilerModifiers.AccUnresolved) == 0)
848                return method;
849 
850        if (this.scope!=null && this.scope.compilerOptions().sourceLevel >= ClassFileConstants.JDK1_5) {
851                if ((method.getAnnotationTagBits() & TagBits.AnnotationDeprecated) != 0)
852                        method.modifiers |= ClassFileConstants.AccDeprecated;
853        }
854        if (isViewedAsDeprecated() && !method.isDeprecated())
855                method.modifiers |= ExtraCompilerModifiers.AccDeprecatedImplicitly;
856        if (hasRestrictedAccess())
857                method.modifiers |= ExtraCompilerModifiers.AccRestrictedAccess;
858 
859        Method methodDecl = method.oaaMethod;
860        if (methodDecl == null) return null; // method could not be resolved in previous iteration
861 
862 
863        boolean foundArgProblem = false;
864        Parameter[] arguments = methodDecl.parameters;
865        if (arguments != null) {
866                int size = arguments.length;
867                method.parameters = Binding.NO_PARAMETERS;
868                TypeBinding[] newParameters = new TypeBinding[size];
869                for (int i = 0; i < size; i++) {
870                        Parameter arg = arguments[i];
871                        TypeBinding parameterType = TypeBinding.UNKNOWN;
872                        parameterType = libraryScope.resolveType(arg.dataType) ;
873                        
874                        
875//                        else 
876                        if (parameterType == TypeBinding.VOID) {
877//                                scope.problemReporter().argumentTypeCannotBeVoid(this, methodDecl, arg);
878                                foundArgProblem = true;
879                        } else {
880                                TypeBinding leafType = parameterType.leafComponentType();
881                            if (leafType instanceof ReferenceBinding && (((ReferenceBinding) leafType).modifiers & ExtraCompilerModifiers.AccGenericSignature) != 0)
882                                        method.modifiers |= ExtraCompilerModifiers.AccGenericSignature;
883                                newParameters[i] = parameterType;
884                        }
885                }
886                // only assign parameters if no problems are found
887                if (!foundArgProblem)
888                        method.parameters = newParameters;
889        }
890 
891        boolean foundReturnTypeProblem = false;
892        if (foundArgProblem) {
893                method.parameters = Binding.NO_PARAMETERS; // see 107004
894                // nullify type parameter bindings as well as they have a backpointer to the method binding
895                // (see https://bugs.eclipse.org/bugs/show_bug.cgi?id=81134)
896                return null;
897        }
898        if (foundReturnTypeProblem)
899                return method; // but its still unresolved with a null return type & is still connected to its method declaration
900 
901        method.modifiers &= ~ExtraCompilerModifiers.AccUnresolved;
902        return method;
903}
