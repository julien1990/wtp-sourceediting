412        protected String getAttributeValue(String attrName, IStructuredDocumentRegion sdRegion) {
413                String sdRegionText = fTextToParse.substring(sdRegion.getStartOffset(), sdRegion.getEndOffset());
414                String textRegionText, attrValue = ""; //$NON-NLS-1$

419                        if (nameRegion.getType() == DOMRegionContext.XML_TAG_ATTRIBUTE_NAME) {
420                                textRegionText = sdRegionText.substring(nameRegion.getStart(), nameRegion.getTextEnd());

422                                        while (it.hasNext()) {

425                                                        attrValue = sdRegionText.substring(valueRegion.getStart(), valueRegion.getEnd());
426                                                        break; // inner
427                                                }
428                                        }
429                                        break; // outer
430                                }
431                        }
432                }
433                return StringUtils.stripQuotes(attrValue);
