374                fHighlightingStyles = new HighlightingStyle[semantics.size()];
375                
376                for (int i = 0; i < semantics.size(); i++) {
377                        fHighlightings[i] = ((SemanticContent) semantics.get(i)).getHighlighting();
378                        fHighlightingStyles[i] = createHighlightingStyle(((SemanticContent) semantics.get(i)).getHighlighting());
379                }
380        }

384         * The content type is used in the sorting algorithm. Content types that are more specific
385         * to the file's content type will cause styles to be applied from general to specific
386         */

396                public int compareTo(Object arg0) {
397                        SemanticContent other = (SemanticContent) arg0;
398                        /* Equal weighting for the same types */
399                        if (this.type.equals(other.type))
400                                return 0;
401                        /* Subtypes have more weight than base types */
402                        if (this.type.isKindOf(other.type))
403                                return 1;
404                        return -1;
405                }

412        /**
413         * Creates a highlighting style based on the preferences defined in the semantic highlighting
414         * @param highlighting the semantic highlighting
415         * @return a highlighting style based on the preferences of the semantic highlighting
416         */
417        private HighlightingStyle createHighlightingStyle(ISemanticHighlighting highlighting) {
418                IPreferenceStore store = highlighting.getPreferenceStore();
419                HighlightingStyle highlightingStyle = null;
420                if (store != null) {
421                        int style = getBoolean(store, highlighting.getBoldPreferenceKey()) ? SWT.BOLD : SWT.NORMAL;
422                        
423                        if (getBoolean(store, highlighting.getItalicPreferenceKey()))
424                                style |= SWT.ITALIC;
425                        if (getBoolean(store, highlighting.getStrikethroughPreferenceKey()))
426                                style |= TextAttribute.STRIKETHROUGH;
427                        if (getBoolean(store, highlighting.getUnderlinePreferenceKey()))
428                                style |= TextAttribute.UNDERLINE;
429                        store.addPropertyChangeListener(fHighlightingChangeListener);
430                        boolean isEnabled = getBoolean(store, highlighting.getEnabledPreferenceKey());
431                        String rgbString = getString(store, highlighting.getColorPreferenceKey());
432                        Color color = null;
433                        
434                        if (rgbString != null)
435                                color = EditorUtility.getColor(ColorHelper.toRGB(rgbString));
436                        
437                        highlightingStyle = new HighlightingStyle(new TextAttribute(color, null, style), isEnabled);
438                }
439                return highlightingStyle;
440        }

514        /**
515         * Handles property change events for individual semantic highlightings.
516         * @param event
517         */
518        private void handleHighlightingPropertyChange(PropertyChangeEvent event) {
519                String property = event.getProperty();
520                if (property == null)
521                        return;
522                
523                boolean refreshRequired = false;
524                
525                for (int i = 0; i < fHighlightings.length; i++) {
526                        ISemanticHighlighting highlighting = fHighlightings[i];
527                        
528                        if (property.equals(highlighting.getBoldPreferenceKey())) {
529                                adaptToTextStyleChange(fHighlightingStyles[i], event, SWT.BOLD);
530                                fPresenter.highlightingStyleChanged(fHighlightingStyles[i]);
531                                refreshRequired = true;
532                                continue;
533                        }
534                        
535                        if (property.equals(highlighting.getColorPreferenceKey())) {
536                                adaptToTextForegroundChange(fHighlightingStyles[i], event);
537                                fPresenter.highlightingStyleChanged(fHighlightingStyles[i]);
538                                refreshRequired = true;
539                                continue;
540                        }
541                        
542                        if (property.equals(highlighting.getEnabledPreferenceKey())) {
543                                adaptToEnablementChange(fHighlightingStyles[i], event);
544                                fPresenter.highlightingStyleChanged(fHighlightingStyles[i]);
545                                refreshRequired = true;
546                                continue;
547                        }
548                        
549                        if (property.equals(highlighting.getItalicPreferenceKey())) {
550                                adaptToTextStyleChange(fHighlightingStyles[i], event, SWT.ITALIC);
551                                fPresenter.highlightingStyleChanged(fHighlightingStyles[i]);
552                                refreshRequired = true;
553                                continue;
554                        }
555                        
556                        if (property.equals(highlighting.getStrikethroughPreferenceKey())) {
557                                adaptToTextStyleChange(fHighlightingStyles[i], event, TextAttribute.STRIKETHROUGH);
558                                fPresenter.highlightingStyleChanged(fHighlightingStyles[i]);
559                                refreshRequired = true;
560                                continue;
561                        }
562                        
563                        if (property.equals(highlighting.getUnderlinePreferenceKey())) {
564                                adaptToTextStyleChange(fHighlightingStyles[i], event, TextAttribute.UNDERLINE);
565                                fPresenter.highlightingStyleChanged(fHighlightingStyles[i]);
566                                refreshRequired = true;
567                                continue;
568                        }
569                }
570                
571                if (refreshRequired && fReconciler != null)
572                        fReconciler.refresh();
573        }

575        /**
576         * Handles property changes for enabling and disabling semantic highlighting for
577         * Structured Source Editors
578         * @param event
579         */
580        private void handlePropertyChange(PropertyChangeEvent event) {
581                String property = event.getProperty();
582                if (property == null || !property.equals(EditorPreferenceNames.SEMANTIC_HIGHLIGHTING))
583                        return;
584                
585                Object newValue = event.getNewValue();
586                if (newValue instanceof Boolean) {
587                        if (((Boolean) newValue).booleanValue())
588                                enable();
589                        else
590                                disable();
591                }
592                
593                if (fReconciler != null)
594                        fReconciler.refresh();
595        }

597        private void adaptToEnablementChange(HighlightingStyle highlighting, PropertyChangeEvent event) {
598                Object value = event.getNewValue();
599                boolean eventValue;
600                if (value instanceof Boolean)
601                        eventValue = ((Boolean) value).booleanValue();
602                else if (IPreferenceStore.TRUE.equals(value))
603                        eventValue = true;
604                else
605                        eventValue = false;
606                highlighting.setEnabled(eventValue);
607        }
608 
609        private void adaptToTextForegroundChange(HighlightingStyle highlighting, PropertyChangeEvent event) {
610                RGB rgb = null;
611 
612                Object value = event.getNewValue();
613                if (value instanceof RGB)
614                        rgb= (RGB) value;
615                else if (value instanceof String)
616                        rgb= ColorHelper.toRGB( (String) value);
617 
618                if (rgb != null) {
619                        Color color= EditorUtility.getColor(rgb);
620                        TextAttribute oldAttr= highlighting.getTextAttribute();
621                        highlighting.setTextAttribute(new TextAttribute(color, oldAttr.getBackground(), oldAttr.getStyle()));
622                }
623        }
624        
625        private void adaptToTextStyleChange(HighlightingStyle highlighting, PropertyChangeEvent event, int styleAttribute) {
626                boolean eventValue = false;
627                Object value = event.getNewValue();
628                if (value instanceof Boolean)
629                        eventValue = ((Boolean) value).booleanValue();
630                else if (IPreferenceStore.TRUE.equals(value))
631                        eventValue = true;
632 
633                TextAttribute oldAttr = highlighting.getTextAttribute();
634                boolean activeValue = (oldAttr.getStyle() & styleAttribute) == styleAttribute;
635 
636                if (activeValue != eventValue)
637                        highlighting.setTextAttribute(new TextAttribute(oldAttr.getForeground(), oldAttr.getBackground(), eventValue ? oldAttr.getStyle() | styleAttribute : oldAttr.getStyle() & ~styleAttribute));
638        }
