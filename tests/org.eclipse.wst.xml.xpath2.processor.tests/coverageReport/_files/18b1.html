453 * characters in the given string, return '\0'.

455char _findMnemonic (String string) {

459  do {

462    if (string.charAt (index) != '&') return Character.toLowerCase (string.charAt (index));
463    index++;
464  } while (index < length);
465  return '\0';
466}

675void handleFocus (int type) {
676  if (isDisposed ()) return;
677  switch (type) {
678    case SWT.FocusIn: {
679      if (hasFocus) return;
680      if (getEditable ()) text.selectAll ();
681      hasFocus = true;
682      Shell shell = getShell ();
683      shell.removeListener (SWT.Deactivate, listener);
684      shell.addListener (SWT.Deactivate, listener);
685      Display display = getDisplay ();
686      display.removeFilter (SWT.FocusIn, filter);
687      display.addFilter (SWT.FocusIn, filter);
688      Event e = new Event ();
689      notifyListeners (SWT.FocusIn, e);
690      break;
691    }
692    case SWT.FocusOut: {
693      if (!hasFocus) return;
694      Control focusControl = getDisplay ().getFocusControl ();
695      if (focusControl == arrow || focusControl == list || focusControl == text) return;
696      hasFocus = false;
697      Shell shell = getShell ();
698      shell.removeListener(SWT.Deactivate, listener);
699      Display display = getDisplay ();
700      display.removeFilter (SWT.FocusIn, filter);
701      Event e = new Event ();

878    case SWT.Dispose:

888      handleFocus (SWT.FocusIn);
889      break;
890    }
891    case SWT.MouseUp: {
892      if (event.button != 1) return;

895    }

918      }

922      e.doit = event.doit;
923      e.character = event.character;
924      e.keyCode = event.keyCode;
925      notifyListeners (SWT.Traverse, e);
926      event.doit = e.doit;
927      event.detail = e.detail;

934      e.keyCode = event.keyCode;

936      notifyListeners (SWT.KeyUp, e);

960      e.character = event.character;
961      e.keyCode = event.keyCode;

965      
966    }
967  }
