430        private void removeSelectedWorkingSets() {
431                ISelection selection= fTableViewer.getSelection();
432                if (selection instanceof IStructuredSelection) {
433                        IWorkingSetManager manager= PlatformUI.getWorkbench().getWorkingSetManager();
434                        Iterator iter= ((IStructuredSelection)selection).iterator();
435                        while (iter.hasNext()) {
436                                IWorkingSet workingSet= (IWorkingSet)iter.next();
437                                if (fAddedWorkingSets.contains(workingSet)) {
438                                        fAddedWorkingSets.remove(workingSet);
439                                } else {
440                                        IWorkingSet[] recentWorkingSets= manager.getRecentWorkingSets();
441                                        for (int i= 0; i < recentWorkingSets.length; i++) {
442                                                if (workingSet.equals(recentWorkingSets[i])) {
443                                                        fRemovedMRUWorkingSets.add(workingSet);
444                                                        break;
445                                                }
446                                        }
447                                        fRemovedWorkingSets.add(workingSet);
448                                }
449                                fAllWorkingSets.remove(workingSet);
450                                manager.removeWorkingSet(workingSet);
451                        }
452                        fTableViewer.remove(((IStructuredSelection)selection).toArray());
453                }
454        }
