397                case T_any:
398                        this.resolvedType = TypeBinding.UNKNOWN;
399                        break;
400                case T_function:
401                        this.resolvedType = scope.getJavaLangFunction();
402                        break;
403                default : //error........
404                        this.constant = Constant.NotAConstant;
405                        scope.problemReporter().invalidOperator(this, leftType, rightType);
406                        return null;
407        }
408 
409        // check need for operand cast
410        if (leftIsCast || rightIsCast) {
411                CastExpression.checkNeedForArgumentCasts(scope, operator, operatorSignature, this.left, leftTypeID, leftIsCast, this.right, rightTypeID, rightIsCast);
412        }
413        // compute the constant when valid
414        computeConstant(scope, leftTypeID, rightTypeID);
415        return this.resolvedType;
416}
