542                /**
543                 * The four order criteria as described below - put already used into bit 10, all others into
544                 * bits 0-9, 11-20, 21-30; 31 is sign - always 0
545                 * @param v
546                 * @return the score for <code>v</code>
547                 */
548                private int score(Variable v) {
549                        int variableScore= 100 - v.variableType; // since these are increasing with distance
550                        int subStringScore= getLongestCommonSubstring(v.name, fParamName).length();
551                        // substring scores under 60% are not considered
552                        // this prevents marginal matches like a - ba and false - isBool that will
553                        // destroy the sort order
554                        int shorter= Math.min(v.name.length(), fParamName.length());
555                        if (subStringScore < 0.6 * shorter)
556                                subStringScore= 0;
557 
558                        int positionScore= v.positionScore; // since ???
559                        int matchedScore= v.alreadyMatched ? 0 : 1;
560                        int autoboxingScore= v.isAutoboxingMatch ? 0 : 1;
561 
562                        int score= autoboxingScore << 30 | variableScore << 21 | subStringScore << 11 | matchedScore << 10 | positionScore;
563                        return score;
564                }

568        /**
569         * Determine the best match of all possible type matches.  The input into this method is all
570         * possible completions that match the type of the argument. The purpose of this method is to
571         * choose among them based on the following simple rules:
572         *
573         *         1) Local Variables > Instance/Class Variables > Inherited Instance/Class Variables
574         *
575         *         2) A longer case insensitive substring match will prevail
576         *
577         *  3) Variables that have not been used already during this completion will prevail over
578         *                 those that have already been used (this avoids the same String/int/char from being passed
579         *                 in for multiple arguments)
580         *
581         *         4) A better source position score will prevail (the declaration point of the variable, or
582         *                 "how close to the point of completion?"
583         */
584        private static void orderMatches(List typeMatches, String paramName) {
585                if (typeMatches != null) Collections.sort(typeMatches, new MatchComparator(paramName));

591        private List findProposalsMatchingType(List proposals, Variable parameter) throws JavaScriptModelException {
592

597                // completion engine returns variables in the order they are found -- and we want to find

613         * Returns the longest common substring of two strings.
614         */
615        private static String getLongestCommonSubstring(String first, String second) {
616 
617                String shorter= (first.length() <= second.length()) ? first : second;
618                String longer= shorter == first ? second : first;
619 
620                int minLength= shorter.length();
621 
622                StringBuffer pattern= new StringBuffer(shorter.length() + 2);
623                String longestCommonSubstring= ""; //$NON-NLS-1$
624 
625                for (int i= 0; i < minLength; i++) {
626                        for (int j= i + 1; j <= minLength; j++) {
627                                if (j - i < longestCommonSubstring.length())
628                                        continue;
629 
630                                String substring= shorter.substring(i, j);
631                                pattern.setLength(0);
632                                pattern.append('*');
633                                pattern.append(substring);
634                                pattern.append('*');
635 
636                                StringMatcher matcher= new StringMatcher(pattern.toString(), true, false);
637                                if (matcher.match(longer))
638                                        longestCommonSubstring= substring;
639                        }
640                }
641 
642                return longestCommonSubstring;
643        }
