353                public void selectionChanged(SelectionChangedEvent event) {
354                        /*
355                         * Do not allow selection from other parts to affect selection in
356                         * the text widget if it has focus, or if we're still firing a
357                         * change of selection. Selection events "bouncing" off of other
358                         * parts are all that we can receive if we have focus (since we
359                         * forwarded our selection to the service just a moment ago), and
360                         * only the user should affect selection if we have focus.
361                         */
362 
363                        /* The isFiringSelection check only works if a selection listener */
364                        if (event.getSelection().isEmpty() || fStructuredSelectionProvider.isFiringSelection())
365                                return;
366 
367                        if (getSourceViewer() != null && getSourceViewer().getTextWidget() != null && !getSourceViewer().getTextWidget().isDisposed() && !getSourceViewer().getTextWidget().isFocusControl()) {
368                                int start = -1;
369                                int length = 0;
370                                if (event.getSelection() instanceof IStructuredSelection) {
371                                        ISelection current = getSelectionProvider().getSelection();
372                                        if (current instanceof IStructuredSelection) {
373                                                Object[] currentSelection = ((IStructuredSelection) current).toArray();
374                                                Object[] newSelection = ((IStructuredSelection) event.getSelection()).toArray();
375                                                if (!Arrays.equals(currentSelection, newSelection)) {
376                                                        if (newSelection.length > 0) {
377                                                                /*
378                                                                 * No ordering is guaranteed for multiple
379                                                                 * selection
380                                                                 */
381                                                                Object o = newSelection[0];
382                                                                if (o instanceof IndexedRegion) {
383                                                                        start = ((IndexedRegion) o).getStartOffset();
384                                                                        int end = ((IndexedRegion) o).getEndOffset();
385                                                                        if (newSelection.length > 1) {
386                                                                                for (int i = 1; i < newSelection.length; i++) {
387                                                                                        start = Math.min(start, ((IndexedRegion) newSelection[i]).getStartOffset());
388                                                                                        end = Math.max(end, ((IndexedRegion) newSelection[i]).getEndOffset());
389                                                                                }
390                                                                                length = end - start;
391                                                                        }
392                                                                }
393                                                                else if (o instanceof ITextRegion) {
394                                                                        start = ((ITextRegion) o).getStart();
395                                                                        int end = ((ITextRegion) o).getEnd();
396                                                                        if (newSelection.length > 1) {
397                                                                                for (int i = 1; i < newSelection.length; i++) {
398                                                                                        start = Math.min(start, ((ITextRegion) newSelection[i]).getStart());
399                                                                                        end = Math.max(end, ((ITextRegion) newSelection[i]).getEnd());
400                                                                                }
401                                                                                length = end - start;
402                                                                        }
403                                                                }
404                                                                else if (o instanceof IRegion) {
405                                                                        start = ((IRegion) o).getOffset();
406                                                                        int end = start + ((IRegion) o).getLength();
407                                                                        if (newSelection.length > 1) {
408                                                                                for (int i = 1; i < newSelection.length; i++) {
409                                                                                        start = Math.min(start, ((IRegion) newSelection[i]).getOffset());
410                                                                                        end = Math.max(end, ((IRegion) newSelection[i]).getOffset() + ((IRegion) newSelection[i]).getLength());
411                                                                                }
412                                                                                length = end - start;
413                                                                        }
414                                                                }
415                                                        }
416                                                }
417                                        }
418                                }
419                                else if (event.getSelection() instanceof ITextSelection) {
420                                        start = ((ITextSelection) event.getSelection()).getOffset();
421                                }
422                                if (start > -1) {
423                                        updateRangeIndication(event.getSelection());
424                                        selectAndReveal(start, length);
425                                }
426                        }
427                }
