479  private static List buildInput(XSDSchema xsdSchema)
480  {
481        if ( xsdSchema ==null )
482                return null;
483 
484    List elements = xsdSchema.getElementDeclarations();
485    List attributes = xsdSchema.getAttributeDeclarations();
486    String targetNamespace = xsdSchema.getTargetNamespace();
487 
488    // For safety purpose: We don't append 'attributes' to 'elements'
489    // ArrayStoreException(or similar one) may occur
490    List allItems = new ArrayList(attributes.size() + elements.size());
491    
492    {
493      // getElementDeclarations returns a lot of elements from import
494      // statement, we
495      // only add non-imported elements here. (trung)
496      for (int i = 0; i < elements.size(); i++)
497      {
498        XSDElementDeclaration currentElement = (XSDElementDeclaration) elements.get(i);
499        if (currentElement.getTargetNamespace() != null)
500        {
501          if (currentElement.getTargetNamespace().equals(targetNamespace))
502            allItems.add(currentElement);
503        }
504        else
505        {
506          if (targetNamespace == null)
507            allItems.add(currentElement);
508        }
509      }
510      // getAttributeDeclarations also returns a lot of elements from
511      // import statement, we
512      // only add non-imported elements here. (trung)
513      for (int i = 0; i < attributes.size(); i++)
514      {
515        XSDAttributeDeclaration currentAttribute = (XSDAttributeDeclaration) attributes.get(i);
516        if (currentAttribute.getTargetNamespace() != null)
517        {
518          if (currentAttribute.isGlobal() && currentAttribute.getTargetNamespace().equals(targetNamespace))
519            allItems.add(currentAttribute);
520        }
521        else
522        {
523          if (targetNamespace == null)
524            allItems.add(currentAttribute);
525        }
526      }
527    }
528    return allItems;
529  }
