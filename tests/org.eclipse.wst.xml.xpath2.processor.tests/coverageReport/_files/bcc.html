213 * there is nothing but spaces in between it and the

216public int previousAvailableLineEnd(int position){
217 
218        Parser parser = this.parser();
219        if (parser == null) return position;
220 
221        Scanner scanner = parser.scanner;
222        if (scanner.lineEnds == null) return position;
223 
224        int index = Util.getLineNumber(position, scanner.lineEnds, 0, scanner.linePtr);
225        if (index < 2) return position;
226        int previousLineEnd = scanner.lineEnds[index-2];
227 
228        char[] source = scanner.source;
229        for (int i = previousLineEnd+1; i < position; i++){
230                if (!(source[i] == ' ' || source[i] == '\t')) return position;
231        }
232        return previousLineEnd;
233}
