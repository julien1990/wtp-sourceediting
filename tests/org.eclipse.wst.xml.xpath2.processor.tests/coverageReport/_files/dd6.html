344         */
345        public RefactoringStatus checkParameterNames() {
346                RefactoringStatus result= new RefactoringStatus();
347                for (Iterator iter= fParameterInfos.iterator(); iter.hasNext();) {
348                        ParameterInfo parameter= (ParameterInfo)iter.next();
349                        result.merge(Checks.checkIdentifier(parameter.getNewName()));

351                                ParameterInfo other= (ParameterInfo) others.next();

354                                                RefactoringCoreMessages.ExtractMethodRefactoring_error_sameParameter, 
355                                                other.getNewName()));

357                                }
358                        }
359                        if (parameter.isRenamed() && fUsedNames.contains(parameter.getNewName())) {
360                                result.addError(Messages.format(
361                                        RefactoringCoreMessages.ExtractMethodRefactoring_error_nameInUse, 
362                                        parameter.getNewName()));
363                                return result;
364                        }
365                }
366                return result;
367        }

684        private ASTNode[] createCallNodes(SnippetFinder.Match duplicate) {
685                List result= new ArrayList(2);
686                
687                IVariableBinding[] locals= fAnalyzer.getCallerLocals();
688                for (int i= 0; i < locals.length; i++) {
689                        result.add(createDeclaration(locals[i], null));
690                }
691                
692                FunctionInvocation invocation= fAST.newFunctionInvocation();
693                invocation.setName(fAST.newSimpleName(fMethodName));
694                List arguments= invocation.arguments();
695                for (int i= 0; i < fParameterInfos.size(); i++) {
696                        ParameterInfo parameter= ((ParameterInfo)fParameterInfos.get(i));
697                        arguments.add(ASTNodeFactory.newName(fAST, getMappedName(duplicate, parameter)));
698                }                
699                
700                ASTNode call;                
701                int returnKind= fAnalyzer.getReturnKind();
702                switch (returnKind) {
703                        case ExtractMethodAnalyzer.ACCESS_TO_LOCAL:
704                                IVariableBinding binding= fAnalyzer.getReturnLocal();
705                                if (binding != null) {
706                                        VariableDeclarationStatement decl= createDeclaration(getMappedBinding(duplicate, binding), invocation);
707                                        call= decl;
708                                } else {
709                                        Assignment assignment= fAST.newAssignment();
710                                        assignment.setLeftHandSide(ASTNodeFactory.newName(fAST, 
711                                                        getMappedBinding(duplicate, fAnalyzer.getReturnValue()).getName()));
712                                        assignment.setRightHandSide(invocation);
713                                        call= assignment;
714                                }
715                                break;
716                        case ExtractMethodAnalyzer.RETURN_STATEMENT_VALUE:
717                                ReturnStatement rs= fAST.newReturnStatement();
718                                rs.setExpression(invocation);
719                                call= rs;
720                                break;
721                        default:
722                                call= invocation;
723                }
724                
725                if (call instanceof Expression && !fAnalyzer.isExpressionSelected()) {
726                        call= fAST.newExpressionStatement((Expression)call);
727                }
728                result.add(call);
729                
730                // We have a void return statement. The code looks like
731                // extracted();
732                // return;        
733                if (returnKind == ExtractMethodAnalyzer.RETURN_STATEMENT_VOID && !fAnalyzer.isLastStatementSelected()) {
734                        result.add(fAST.newReturnStatement());
735                }
736                return (ASTNode[])result.toArray(new ASTNode[result.size()]);                
737        }

751        private void replaceDuplicates(CompilationUnitChange result) {
752                int numberOf= getNumberOfDuplicates();
753                if (numberOf == 0 || !fReplaceDuplicates)
754                        return;
755                String label= null;
756                if (numberOf == 1)
757                        label= Messages.format(RefactoringCoreMessages.ExtractMethodRefactoring_duplicates_single, fMethodName); 
758                else
759                        label= Messages.format(RefactoringCoreMessages.ExtractMethodRefactoring_duplicates_multi, fMethodName);

819                                        result.setJavadoc(javadoc);
820                                }
821                        }
822                }

824                return result;

858        private Block createMethodBody(FunctionDeclaration method, ASTNode[] selectedNodes, TextEditGroup substitute) throws BadLocationException, CoreException {
859                Block result= fAST.newBlock();
860                ListRewrite statements= fRewriter.getListRewrite(result, Block.STATEMENTS_PROPERTY);
861                
862                // Locals that are not passed as an arguments since the extracted method only
863                // writes to them
864                IVariableBinding[] methodLocals= fAnalyzer.getMethodLocals();
865                for (int i= 0; i < methodLocals.length; i++) {
866                        if (methodLocals[i] != null) {
867                                result.statements().add(createDeclaration(methodLocals[i], null));
868                        }
869                }
870 
871                for (Iterator iter= fParameterInfos.iterator(); iter.hasNext();) {
872                        ParameterInfo parameter= (ParameterInfo)iter.next();
873                        if (parameter.isRenamed()) {
874                                for (int n= 0; n < selectedNodes.length; n++) {
875                                        SimpleName[] oldNames= LinkedNodeFinder.findByBinding(selectedNodes[n], parameter.getOldBinding());
876                                        for (int i= 0; i < oldNames.length; i++) {
877                                                fRewriter.replace(oldNames[i], fAST.newSimpleName(parameter.getNewName()), null);
878                                        }
879                                }
880                        }
881                }
882                
883                boolean extractsExpression= fAnalyzer.isExpressionSelected();
884                ASTNode[] callNodes= createCallNodes(null);
885                ASTNode replacementNode;
886                if (callNodes.length == 1) {
887                        replacementNode= callNodes[0];
888                } else {
889                        replacementNode= fRewriter.createGroupNode(callNodes);
890                }
891                if (extractsExpression) {
892                        // if we have an expression then only one node is selected.
893                        ITypeBinding binding= fAnalyzer.getExpressionBinding();
894                        if (binding != null && (!binding.isPrimitive() || !"void".equals(binding.getName()))) { //$NON-NLS-1$
895                                ReturnStatement rs= fAST.newReturnStatement();
896                                rs.setExpression((Expression)fRewriter.createMoveTarget(selectedNodes[0]));
897                                statements.insertLast(rs, null);
898                        } else {
899                                ExpressionStatement st= fAST.newExpressionStatement((Expression)fRewriter.createMoveTarget(selectedNodes[0]));
900                                statements.insertLast(st, null);
901                        }
902                        fRewriter.replace(selectedNodes[0], replacementNode, substitute);
903                } else {
904                        if (selectedNodes.length == 1) {
905                                statements.insertLast(fRewriter.createMoveTarget(selectedNodes[0]), substitute);
906                                fRewriter.replace(selectedNodes[0], replacementNode, substitute);
907                        } else {
908                                ListRewrite source= fRewriter.getListRewrite(
909                                        selectedNodes[0].getParent(), 
910                                        (ChildListPropertyDescriptor)selectedNodes[0].getLocationInParent());
911                                ASTNode toMove= source.createMoveTarget(
912                                        selectedNodes[0], selectedNodes[selectedNodes.length - 1],
913                                        replacementNode, substitute);
914                                statements.insertLast(toMove, substitute);
915                        }
916                        IVariableBinding returnValue= fAnalyzer.getReturnValue();
917                        if (returnValue != null) {
918                                ReturnStatement rs= fAST.newReturnStatement();
919                                rs.setExpression(fAST.newSimpleName(getName(returnValue)));
920                                statements.insertLast(rs, null);                                
921                        }
922                }
923                return result;
924        }

962                                final StringTokenizer tokenizer= new StringTokenizer(selection);

964                                        offset= Integer.valueOf(tokenizer.nextToken()).intValue();
965                                if (tokenizer.hasMoreTokens())

968                                        fSelectionStart= offset;
