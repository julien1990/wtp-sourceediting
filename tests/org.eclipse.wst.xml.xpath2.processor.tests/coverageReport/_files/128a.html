353         * @param line the current line under investigation
354         * @param offset the offset of the caret position, relative to the line start.

357        private static int computeAfterTryDoElse(IDocument doc, ITextSelection line, int offset) {
358                // search backward while WS, find 'try', 'do', 'else' in default partition
359                int p= offset + line.getOffset();

361                if (p == -1)
362                        return -1;
363                p--;

365                if (looksLike(doc, p, "try") //$NON-NLS-1$
366                                || looksLike(doc, p, "do")  //$NON-NLS-1$
367                                || looksLike(doc, p, "synchronized")  //$NON-NLS-1$
368                                || looksLike(doc, p, "static")  //$NON-NLS-1$
369                                || looksLike(doc, p, "finally")  //$NON-NLS-1$
370                                || looksLike(doc, p, "else"))  //$NON-NLS-1$

379         *
380         * @param document the document being modified
381         * @param line the current line under investigation
382         * @param offset the offset of the caret position, relative to the line start.
383         * @param partitioning the document partitioning
384         * @return an insert position relative to the line start if <code>line</code> contains a parenthesized expression that can be followed by a block, -1 otherwise
385         */
386        private static int computeAfterParenthesis(IDocument document, ITextSelection line, int offset, String partitioning) {
387                // find the opening parenthesis for every closing parenthesis on the current line after offset
388                // return the position behind the closing parenthesis if it looks like a method declaration
389                // or an expression for an if, while, for, catch statement
390                int pos= offset + line.getOffset();
391                int length= line.getOffset() + line.getLength();
392                int scanTo= scanForward(document, pos, partitioning, length, '}');
393                if (scanTo == -1)
394                        scanTo= length;
395 
396                int closingParen= findClosingParenToLeft(document, pos, partitioning) - 1;
397 
398                while (true) {
399                        int startScan= closingParen + 1;
400                        closingParen= scanForward(document, startScan, partitioning, scanTo, ')');
401                        if (closingParen == -1)
402                                break;
403 
404                        int openingParen= findOpeningParenMatch(document, closingParen, partitioning);
405 
406                        // no way an expression at the beginning of the document can mean anything
407                        if (openingParen < 1)
408                                break;
409 
410                        // only select insert positions for parenthesis currently embracing the caret
411                        if (openingParen > pos)
412                                continue;
413 
414                        if (looksLikeAnonymousClassDef(document, openingParen - 1, partitioning))
415                                return closingParen + 1 - line.getOffset();
416 
417                        if (looksLikeIfWhileForCatch(document, openingParen - 1, partitioning))
418                                return closingParen + 1 - line.getOffset();
419 
420                        if (looksLikeMethodDecl(document, openingParen - 1, partitioning))
421                                return closingParen + 1 - line.getOffset();
422 
423                }
424 
425                return -1;
426        }

852         * From <code>position</code> to the left, eats any whitespace and the first identifier, returning
853         * the position of the first identifier character (in normal read order).
854         * <p>When called on a document with content <code>" some string  "</code> and positionition 13, the
855         * return value will be 6 (the first letter in <code>string</code>).
856         * </p>

859         * @param position the first character position in <code>document</code> to be considered
860         * @param partitioning the document partitioning
861         * @return the smallest character position of an identifier or -1 if none can be found; always &lt;= <code>position</code>
862         */
863        private static int eatIdentToLeft(IDocument document, int position, String partitioning) {
864                if (position < 0)
865                        return -1;
866                Assert.isTrue(position < document.getLength());
867 
868                int p= firstNonWhitespaceBackward(document, position, partitioning, -1);
869                if (p == -1)
870                        return -1;
871 
872                try {
873                        while (p >= 0) {
874 
875                                char ch= document.getChar(p);
876                                if (Character.isJavaIdentifierPart(ch)) {
877                                        p--;
878                                        continue;
879                                }
880 
881                                // length must be > 0
882                                if (Character.isWhitespace(ch) && p != position)
883                                        return p + 1;
884                                else
885                                        return -1;
886 
887                        }
888 
889                        // start of document reached
890                        return 0;
891 
892                } catch (BadLocationException e) {
893                }
894                return -1;

904         * @param partitioning the document partitioning
905         * @return the position of the next Java partition, or the end of <code>line</code>
906         */
907        private static int nextPartitionOrLineEnd(IDocument document, ITextSelection line, int offset, String partitioning) {
908                // run relative to document
909                final int docOffset= offset + line.getOffset();
910                final int eol= line.getOffset() + line.getLength();
911                int nextPartitionPos= eol; // init with line end
912                int validPosition= docOffset;
913 
914                try {
915                        ITypedRegion partition= TextUtilities.getPartition(document, partitioning, nextPartitionPos, true);
916                        validPosition= getValidPositionForPartition(document, partition, eol);
917                        while (validPosition == -1) {
918                                nextPartitionPos= partition.getOffset() - 1;
919                                if (nextPartitionPos < docOffset) {
920                                        validPosition= docOffset;
921                                        break;
922                                }
923                                partition= TextUtilities.getPartition(document, partitioning, nextPartitionPos, false);
924                                validPosition= getValidPositionForPartition(document, partition, eol);
925                        }
926                } catch (BadLocationException e) {
927                }
928 
929                validPosition= Math.max(validPosition, docOffset);
930                // make relative to line
931                validPosition -= line.getOffset();
932                return validPosition;
933        }

936         * Returns a valid insert location (except for whitespace) in <code>partition</code> or -1 if
937         * there is no valid insert location.
938         * An valid insert location is right after any java string or character partition, or at the end
939         * of a java default partition, but never behind <code>maxOffset</code>. Comment partitions or
940         * empty java partitions do never yield valid insert positions.
941         *
942         * @param doc the document being modified
943         * @param partition the current partition
944         * @param maxOffset the maximum offset to consider
945         * @return a valid insert location in <code>partition</code>, or -1 if there is no valid insert location
946         */
947        private static int getValidPositionForPartition(IDocument doc, ITypedRegion partition, int maxOffset) {
948                final int INVALID= -1;
949 
950                if (IJavaScriptPartitions.JAVA_DOC.equals(partition.getType()))
951                        return INVALID;
952                if (IJavaScriptPartitions.JAVA_MULTI_LINE_COMMENT.equals(partition.getType()))
953                        return INVALID;
954                if (IJavaScriptPartitions.JAVA_SINGLE_LINE_COMMENT.equals(partition.getType()))
955                        return INVALID;
956 
957                int endOffset= Math.min(maxOffset, partition.getOffset() + partition.getLength());
958 
959                if (IJavaScriptPartitions.JAVA_CHARACTER.equals(partition.getType()))
960                        return endOffset;
961                if (IJavaScriptPartitions.JAVA_STRING.equals(partition.getType()))
962                        return endOffset;
963                if (IDocument.DEFAULT_CONTENT_TYPE.equals(partition.getType())) {
964                        try {
965                                if (doc.get(partition.getOffset(), endOffset - partition.getOffset()).trim().length() == 0)
966                                        return INVALID;
967                                else
968                                        return endOffset;
969                        } catch (BadLocationException e) {
970                                return INVALID;
971                        }
972                }
973                // default: we don't know anything about the partition - assume valid
974                return endOffset;
975        }
